23 template <typename T>
24 void mpi_broadcast(std::vector<T> &v, communicator c = {}, int root = 0) {
25 size_t s = v.size();
26 broadcast(s, c, root);
27 if (c.rank() != root) v.resize(s);
28 if constexpr (has_mpi_type<T>) {
29 if (s != 0) MPI_Bcast(v.data(), v.size(), mpi_type<T>::get(), root, c.get());
30 } else {
31 for (auto &x : v) broadcast(x, c, root);
32 }
33 }

36 template <typename T>
37 void mpi_reduce_in_place(std::vector<T> &a, communicator c = {}, int root = 0, bool all = false, MPI_Op op = MPI_SUM) {
38 if (a.size() == 0) return; // mpi behaviour not checked in that case.
39 if constexpr (has_mpi_type<T>) {
40 if (!all)
41 MPI_Reduce((c.rank() == root ? MPI_IN_PLACE : a.data()), a.data(), a.size(), mpi_type<T>::get(), op, root, c.get());
42 else
43 MPI_Allreduce(MPI_IN_PLACE, a.data(), a.size(), mpi_type<T>::get(), op, c.get());
44 } else {
45 for (auto &x : a) mpi_reduce_in_place(a, c, root, all);
46 }
47 }

62 template <typename T>
63 using regular_t = typename detail::_regular<std::decay_t<T>>::type;
64
65 template <typename T>
66 std::vector<regular_t<T>> mpi_reduce(std::vector<T> const &a, communicator c = {}, int root = 0, bool all = false, MPI_Op op = MPI_SUM) {
67 size_t s = a.size();
68
69 // For all==true, guarantee that all vectors are of the same size
70 if (all) {
71 auto max_size = mpi_reduce(s, c, root, all, MPI_MAX);
72 if (s != max_size) {
73 std::cerr << "Cannot all_reduce vectors of different sizes\n";
74 std::abort();
75 }
76 }
77
78 if (s == 0) return {}; // nothing to do, and MPI does not like size 0
79 if constexpr (has_mpi_type<T>) {
80 static_assert(std::is_same_v<regular_t<T>, T>, "Internal error");
81 std::vector<T> r(s);
82 if (!all)
83 MPI_Reduce((void *)a.data(), r.data(), s, mpi_type<T>::get(), op, root, c.get());
84 else
85 MPI_Allreduce((void *)a.data(), r.data(), s, mpi_type<T>::get(), op, c.get());
86 return r;
87 } else {
88 std::vector<regular_t<T>> r;
89 r.reserve(s);
90 for (size_t i = 0; i < s; ++i) r.push_back(reduce(a[i], c, root, all, op));
91 return r;
92 }
93 }

98 template <typename T>
99 std::vector<T> mpi_scatter(std::vector<T> const &a, communicator c = {}, int root = 0) {
100
101 auto slow_size = a.size();
102 auto sendcounts = std::vector<int>(c.size());
103 auto displs = std::vector<int>(c.size() + 1, 0);
104 int recvcount = chunk_length(slow_size, c.size(), c.rank());
105 std::vector<T> b(recvcount);
106
107 for (int r = 0; r < c.size(); ++r) {
108 sendcounts[r] = chunk_length(slow_size, c.size(), r);
109 displs[r + 1] = sendcounts[r] + displs[r];
110 }
111
112 if constexpr (has_mpi_type<T>) {
113 MPI_Scatterv((void *)a.data(), &sendcounts[0], &displs[0], mpi_type<T>::get(), (void *)b.data(), recvcount, mpi_type<T>::get(), root, c.get());
114 } else {
115 std::copy(cbegin(a) + displs[c.rank()], cbegin(a) + displs[c.rank() + 1], begin(b));
116 }
117
118 return b;
119 }

122 template <typename T>
123 std::vector<T> mpi_gather(std::vector<T> const &a, communicator c = {}, int root = 0, bool all = false) {
124
125 long size = mpi_reduce(a.size(), c, root, all);
126 std::vector<T> b((all || (c.rank() == root) ? size : 0));
127
128 auto recvcounts = std::vector<int>(c.size());
129 auto displs = std::vector<int>(c.size() + 1, 0);
130 int sendcount = a.size();
131 auto mpi_ty = mpi_type<int>::get();
132 if (!all)
133 MPI_Gather(&sendcount, 1, mpi_ty, &recvcounts[0], 1, mpi_ty, root, c.get());
134 else
135 MPI_Allgather(&sendcount, 1, mpi_ty, &recvcounts[0], 1, mpi_ty, c.get());
136
137 for (int r = 0; r < c.size(); ++r) displs[r + 1] = recvcounts[r] + displs[r];
138
139 if constexpr (has_mpi_type<T>) {
140 if (!all)
141 MPI_Gatherv((void *)a.data(), sendcount, mpi_type<T>::get(), (void *)b.data(), &recvcounts[0], &displs[0], mpi_type<T>::get(), root, c.get());
142 else
143 MPI_Allgatherv((void *)a.data(), sendcount, mpi_type<T>::get(), (void *)b.data(), &recvcounts[0], &displs[0], mpi_type<T>::get(), c.get());
144 } else {
145 if (!all)
146 throw std::runtime_error{"mpi_gather for custom types only implemented with 'all = true'\n"};
147 else {
148 for (int r = 0; r < c.size(); ++r) {
149 for (auto i = displs[r]; i < displs[r + 1]; ++i) {
150 if (c.rank() == r) b[i] = a[i - displs[r]];
151 mpi::broadcast(b[i], c, r);
152 }
153 }
154 }
155 }
156
157 return b;
158 }
