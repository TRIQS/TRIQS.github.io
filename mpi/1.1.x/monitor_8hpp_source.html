37 class monitor {

39 mpi::communicator com;

41 struct future {
42 MPI_Request request{};
43 int value = 0;
44 };
45 std::vector<future> root_futures; // communication of local_stop from the nodes to the root. On root only.
46 MPI_Request req_ibcast{}, req_isent{}; // request for the ibcast and isent. On all nodes.
47
48 int local_stop = 0; // = 1 if the node has requested an emergency stop. Local to the node. (No bool in MPI.)
49 int global_stop = 0; // = 1 if any node has requested an emergency stop. Always the same on all nodes.
50 bool finalized = false; // the finalized() method should be called once,
51 // and can not just be the desctructor since we use the returned value
52
53 public:
55 monitor(mpi::communicator c) : com(c) {
56 if (com.rank() == 0) { // on root
57 // Register an async recv for the variable local_stop from every non-root nodes
58 // The associated send will be issued from each node:
59 // * with value = 1 in case of emergency stop
60 // * with value = 0 during finalization
61 root_futures.resize(c.size() - 1);
62 for (int rank = 1; rank < c.size(); ++rank) { // the index of root_futures is rank - 1 since there is none for rank = 0
63 MPI_Irecv(&(root_futures[rank - 1].value), 1, MPI_INT, rank, 0, MPI_COMM_WORLD, &(root_futures[rank - 1].request));
64 }
65 } else { // not root
66 // Register the receive of the ibcast of global_stop that root will issue (in case of emergency stop or during fianlize)
67 MPI_Ibcast(&global_stop, 1, MPI_INT, 0, MPI_COMM_WORLD, &req_ibcast);
68 }
69 }

74 ~monitor() { finalize_communications(); }

79 void request_emergency_stop() {
80 EXPECTS(!finalized);
81 if (local_stop) { return; } // prevent sending signal multiple times
82 local_stop = 1;
83 if (com.rank() == 0) { // root
84 global_stop = 1;
85 MPI_Ibcast(&global_stop, 1, MPI_INT, 0, MPI_COMM_WORLD, &req_ibcast);
86 } else { // non root
87 MPI_Isend(&local_stop, 1, MPI_INT, 0, 0, MPI_COMM_WORLD, &req_isent);
88 }
89 }
90
92 [[nodiscard]] bool emergency_occured() {
93 if (finalized) return global_stop;
94 if (global_stop or local_stop) return true;
95 if (com.rank() == 0) { // root test whether other nodes have emergency stop and communicates signal
96 root_check_nodes_and_bcast();
97 } else { // other nodes just listen to the root bcast to see if an emergency stop broadcast has occured
98 MPI_Status status;
99 int flag = 0;
100 MPI_Test(&req_ibcast, &flag, &status);
101 // if flag, then global_stop is now what was bcasted by the root
102 }
103 return global_stop;
104 }
105
109 void finalize_communications() {
110 if (finalized) return;
111 if (com.rank() == 0) {
112 // the root is done computing, it just listens to the other nodes and bcast the global_stop until everyone is done.
113 while (root_check_nodes_and_bcast()) { usleep(100); } // 100 us (micro seconds)
114 // all others node have finished
115 // if the root has never emitted the ibcast, we do it and wait it, since we can not cancel it FIXME (why ??).
116 if (!global_stop) { MPI_Ibcast(&global_stop, 1, MPI_INT, 0, MPI_COMM_WORLD, &req_ibcast); }
117 } else {
118 // on non-root node: either Isend was done when local_stop was set (during request_emergency_stop),
119 // or it has to happen now, sending local_stop = 0, i.e, work is done, and fine.
120 if (!local_stop) { MPI_Isend(&local_stop, 1, MPI_INT, 0, 0, MPI_COMM_WORLD, &req_isent); }
121 }
122 // All nodes wait for the ibcast of global_stop to be complete.
123 MPI_Status status;
124 MPI_Wait(&req_ibcast, &status);
125 finalized = true;
126 }
127
128 private:
129 // ROOT ONLY.
130 // looks at all irecv that have arrived, and if they are a stop request, bcast the global stop.
131 // returns true iff there are still running nodes.
132 bool root_check_nodes_and_bcast() {
133 EXPECTS(!finalized);
134 EXPECTS(com.rank() == 0); // root only
135 bool some_nodes_are_still_running = false;
136 for (auto &f : root_futures) {
137 MPI_Status status;
138 int flag = 0;
139 MPI_Test(&(f.request), &flag, &status);
140 if (flag and (not global_stop) and (f.value > 0)) request_emergency_stop(); // the root requires the stop now. It also stops itself...
141 some_nodes_are_still_running |= (flag == 0);
142 }
143 return some_nodes_are_still_running;
144 }
145 };
