112 // Default barrier is implemented as a poll barrier, which only checks each msec's if
113 // all ranks reached the barrier by using a non blocking MPI_Ibarrier combined with MPI_Test.
114 // The default of 1 msec reduces CPU load considerably:
115 // 1 msec ~ 1% cpu load
116 // 10 msec ~ 0.5% cpu load
117 // 100 msec ~ 0.01% cpu load
118 // For very unbalanced load that takes long times to finish, 1000msec is a good choice.
119 // If poll_msec==0 the classic MPI_Barrier will be called.
120 void barrier(long poll_msec = 1) {
121 if (has_env) {
122 if (poll_msec == 0) {
123 MPI_Barrier(_com);
124 } else {
125 MPI_Request req = nullptr;
126 int flag = 0;
127 // non blocking barrier to check which rank is here
128 MPI_Ibarrier(_com, &req);
129 // check each poll_msec via MPI_Test if all ranks reached the barrier
130 do {
131 MPI_Test(&req, &flag, MPI_STATUS_IGNORE);
132 // convert to millisec
133 usleep(poll_msec * 1000);
134 } while (!flag);
135 }
136 }
137 }
138 };

318 template <typename... T>
319 MPI_Datatype get_mpi_type(std::tuple<T...> _tie) {
320 static constexpr int N = sizeof...(T);
321 MPI_Datatype types[N] = {mpi_type<std::remove_reference_t<T>>::get()...};
322
323 int blocklen[N];
324 for (int i = 0; i < N; ++i) { blocklen[i] = 1; }
325 MPI_Aint disp[N];
326 detail::_init_mpi_tuple_displ(std::index_sequence_for<T...>{}, _tie, disp);
327 if (std::any_of(disp, disp + N, [](MPI_Aint i) { return i < 0; })) {
328 std::cerr << "ERROR: Custom mpi types require non-negative displacements\n";
329 std::abort();
330 }
331
332 MPI_Datatype cty = nullptr;
333 MPI_Type_create_struct(N, blocklen, disp, types, &cty);
334 MPI_Type_commit(&cty);
335 return cty;
336 }

355 // variable template that maps the function
356 // for the meaning of +[](...) , cf
357 // https://stackoverflow.com/questions/17822131/resolving-ambiguous-overload-on-function-pointer-and-stdfunction-for-a-lambda
358 template <typename T, T (*F)(T const &, T const &)>
359 MPI_User_function *_map_function = +[](void *in, void *inout, int *len, MPI_Datatype *) {
360 auto *inT = static_cast<T *>(in);
361 auto *inoutT = static_cast<T *>(inout);
362 for (int i = 0; i < *len; ++i, ++inT, ++inoutT) { *inoutT = F(*inoutT, *inT); }
363 };

429 template <typename T>
430 std::enable_if_t<has_mpi_type<T>, T> mpi_reduce(T const &a, communicator c = {}, int root = 0, bool all = false, MPI_Op op = MPI_SUM) {
431 T b;
432 auto d = mpi_type<T>::get();
433 if (!all)
434 // Old mpi implementations may require a non-const void *sendbuf
435 MPI_Reduce(const_cast<T *>(&a), &b, 1, d, op, root, c.get());
436 else
437 MPI_Allreduce(const_cast<T *>(&a), &b, 1, d, op, c.get());
438 return b;
439 }

441 template <typename T>
442 std::enable_if_t<has_mpi_type<T>> mpi_reduce_in_place(T &a, communicator c = {}, int root = 0, bool all = false, MPI_Op op = MPI_SUM) {
443 if (!all)
444 MPI_Reduce((c.rank() == root ? MPI_IN_PLACE : &a), &a, 1, mpi_type<T>::get(), op, root, c.get());
445 else
446 MPI_Allreduce(MPI_IN_PLACE, &a, 1, mpi_type<T>::get(), op, c.get());
447 }
