38 // and then sample the Green function
39 void solver_core::solve(solve_params_t const &solve_params) {
40
41 // http://patorjk.com/software/taag/#p=testall&f=Calvin%20S&t=TRIQS%20inchworm%0A
42 if (solve_params.verbosity > 0)
43 std::cout << "\n"
44 "╔╦╗╦═╗╦╔═╗ ╔═╗ ┬┌┐┌┌─┐┬ ┬┬ ┬┌─┐┬─┐┌┬┐\n"
45 " ║ ╠╦╝║║═╬╗╚═╗ │││││ ├─┤││││ │├┬┘│││\n"
46 " ╩ ╩╚═╩╚═╝╚╚═╝ ┴┘└┘└─┘┴ ┴└┴┘└─┘┴└─┴ ┴\n";
47
48 // Run inchworm to calculate S.u_tau
49 solve_inchworm(solve_params);
50
51 // Sample the Green function S.G_tau
52 solve_green(solve_params);
53 }

55 // Common initilization to any solving scheme:
56 void solver_core::init(solve_params_t const &sp) {
57
58 // Store solve_params
59 last_solve_params = sp;
60
61 // Initialize Delta_tilde
62 if (Delta_tau_ED.has_value())
63 Delta_tau_tilde = Delta_tau - Delta_tau_ED.value();
64 else
65 Delta_tau_tilde = Delta_tau;
66
68 gf_struct_ED = constr_params.gf_struct;
69 for (auto &[bl_name, bl_size] : gf_struct_ED) { bl_size += sp.n_bath_sites_ED; }
70
71 // Determine basis of operators to use
72 fops = fundamental_operator_set{gf_struct_ED};
73
74 for (auto [bl, bl_pair] : enumerate(constr_params.gf_struct)) {
75 auto [bl_name, bl_size] = bl_pair;
76 all_d_ops[bl].clear();
77 all_d_dag_ops[bl].clear();
78 for (auto idx : range(bl_size)) {
79 all_d_ops[bl].push_back({0.0, false, fops[{bl_name, idx}], bl, idx});
80 all_d_dag_ops[bl].push_back({0.0, true, fops[{bl_name, idx}], bl, idx});
81 }
82 }
83
84 if (sp.partition_method == "automatic") {
85 ASSERT(sp.quantum_numbers.empty());
86 ad_imp = {sp.h_imp + h_hyb_ED, fops, create_effective_hyb(constr_params.gf_struct)};
87 } else if (sp.partition_method == "quantum_numbers") {
88 ad_imp = {sp.h_imp + h_hyb_ED, fops, sp.quantum_numbers};
89 } else {
90 TRIQS_RUNTIME_ERROR
91 << "Unknown partition method! Please choose 'automatic' or 'quantum_number' and set solve_params.quantum_numbers accordingly";
92 }
93 }

96 // solve cthyb (no split point + bare propagator):
97 qmc_results_t solver_core::solve_cthyb(solve_params_t const &solve_params, double tau_max) {
98
99 // Initialize solver
100 this->init(solve_params);
101
102 // use_bare_propagator == cthyb (no split point).
103 double tau_split = 0.0;
104 bool use_bare_propagator = true;
105
106 // Initialize the zeroth order frame
107 frame_zeroth_order = make_bare_u_frame(ad_imp, tau_max);
108
109 // Execute cthyb sampling
110 auto res = qmc_step(solve_params, tau_split, tau_max, use_bare_propagator, MODE::PROPAGATOR);
111
112 // Assign the last obtained frame error
113 errs_frame = res.errs_frame;
114
115 // Print results
116 res.print(solve_params.verbosity);
117
118 return res;
119 }

123 // Self consistent solution (one step, with precalculated U(beta) from ED)
124 qmc_results_t solver_core::solve_self_consistently(solve_params_t const &solve_params, u_tau_t const &u_tau_, double tau_split, double tau_max) {
125 // Initialize solver
126 this->init(solve_params);
127
128 // Check that u_tau_ is compatible with ad_imp
129 if (u_tau_.size() != ad_imp.n_subspaces())
130 throw std::runtime_error{
131 fmt::format("u_tau and ad_imp have mismatching number of blocks: {} vs {}", u_tau_.size(), ad_imp.n_subspaces()).c_str()};
132 for (auto bl : range(ad_imp.n_subspaces())) {
133 auto bl_size_u = u_tau_[bl].target_shape()[0];
134 auto bl_size_ad = ad_imp.get_subspace_dim(bl);
135 if (bl_size_u != bl_size_ad)
136 throw std::runtime_error{fmt::format("Propagator block-size for bl={} does not match atomdiag: {} vs {}", bl, bl_size_u, bl_size_ad).c_str()};
137 }
138
139 // precalculated propagator:
140 u_tau = u_tau_;
141
142 // Initialize the interpolator
143 u_interpolator = interpolator_t<scalar_t>(u_tau, u_tau[0].mesh().size());
144
145 // Initialize the zeroth order frame
146 frame_zeroth_order = u_interpolator(tau_max - tau_split) * u_interpolator(tau_split);
147
148 // Execute u_tau self-consistency sampling
149 auto res = qmc_step(solve_params, tau_split, tau_max, false, MODE::PROPAGATOR);
150
151 // Assign the last obtained frame error
152 errs_frame = res.errs_frame;
153
154 // Print results
155 res.print(solve_params.verbosity);
156
157 return res;
158 }

162 void solver_core::solve_inchworm(solve_params_t const &solve_params, bool use_cthyb) {
163
164 // Initialize solver
165 this->init(solve_params);
166
167 // Reset the results
168 container_set::operator=(container_set{});
169
170 // Initialize empty propagator
171 auto u_tau_zero = u_tau_t{{constr_params.beta, Fermion, constr_params.n_tau_inch}, ad_imp.get_subspace_dims()};
172 u_tau_zero() = 0.;
173 u_tau = u_tau_zero;
174 for (auto &ubl : u_tau) {
175 for (int i = 0; i < ubl.target_shape()[0]; ++i) ubl[0](i, i) = 1;
176 }
177 u_tau_by_order = {u_tau};
178
179 if (solve_params.verbosity > 0) std::cout << "\nStarting inchworm calculation of the propagator.. \n";
180
181 // loop on different inchworm steps
182 for (auto n : range(1, constr_params.n_tau_inch)) {
183 if (solve_params.verbosity > 0) std::printf("\n ..step %ld/%d\n", n, constr_params.n_tau_inch - 1);
184
185 if (solve_params.n_tau_inch_stop && n > *solve_params.n_tau_inch_stop) break;
186
187 // define the tau_split and tau_max for this specific inchworm step.
188 //
189 // |===================================|----------------------|__________________|
190 // 0 tau_split tau_max beta
191 //
192 // tau_split < tau_max <= beta
193 //
194 double tau_split = use_cthyb ? 0.0 : u_tau[0].mesh()(n - 1);
195 double tau_max = u_tau[0].mesh()(n);
196
197 // use bare propagator (cthyb) only on the first inchworm iteration:
198 bool use_bare_propagator = (n == 1) or use_cthyb;
199
200 // Initialize the interpolator
201 if (not use_bare_propagator) { u_interpolator = interpolator_t<scalar_t>(u_tau, n); }
202
203 // Initialize the zeroth order frame
204 frame_zeroth_order = use_bare_propagator ? make_bare_u_frame(ad_imp, tau_max) : u_interpolator(tau_max - tau_split) * u_interpolator(tau_split);
205
206 // calculation of the Monte Carlo solution:
207 auto res = qmc_step(solve_params, tau_split, tau_max, use_bare_propagator, MODE::PROPAGATOR);
208
209 // Print results
210 res.print(solve_params.verbosity);
211
212 // Assign the current frame to the propagator u_tau (will be used in the next iteration)
213 set_frame(res.frame, u_tau, n);
214
215 // Assign the last obtained frame error
216 errs_frame = res.errs_frame;
217
218 // Assign u_frame_by_order to the propagator u_tau_by_order
219 if (solve_params.measure_frame_by_order) {
220 if (res.frame_by_order.size() > u_tau_by_order.size()) u_tau_by_order.resize(res.frame_by_order.size(), u_tau_zero);
221 for (auto k : range(res.frame_by_order.size())) set_frame(res.frame_by_order[k], u_tau_by_order[k], n);
222 }
223
224 // Initialize other results
225 if (solve_params.measure_order_histogram) order_histograms.push_back(res.order_histogram);
226
227 // Break out of loop when interrupted by signal
228 if (res.status == 2) break;
229 }
230 }

233
234 void solver_core::solve_green(solve_params_t const &solve_params) {
235
236 // Initialize solver
237 this->init(solve_params);
238
239 // Reset the results
240 container_set::operator=(container_set{});
241
242 if (solve_params.verbosity > 0) std::cout << "\nStarting Green function calculation.. \n";
243
244 // Initialize the interpolator
245 u_interpolator = interpolator_t<scalar_t>(u_tau, constr_params.n_tau_inch);
246
247 double beta = constr_params.beta;
248
249 // Initialize the Green function container
250 G_tau = g_tau_t{{beta, Fermion, constr_params.n_tau_green}, constr_params.gf_struct};
251 auto G_tau_zero = G_tau;
252 G_tau_zero() = 0.;
253
254 // Calculate Tr U(beta)
255 scalar_t Tr_Ubeta = 0.0;
256 for (int bl = 0; bl < u_tau.size(); bl++) Tr_Ubeta += trace(u_tau[bl](beta));
257
258 // Treat n == 0 and n == n_tau -1 seperately
259 frame_t g_frame_n0 = make_bare_g_frame(ad_imp, u_tau, constr_params.gf_struct, 0.0, beta) / Tr_Ubeta;
260 frame_t g_frame_nB = make_bare_g_frame(ad_imp, u_tau, constr_params.gf_struct, beta, beta) / Tr_Ubeta;
261 set_frame(g_frame_n0, G_tau, 0);
262 set_frame(g_frame_nB, G_tau, constr_params.n_tau_green - 1);
263 if (solve_params.measure_frame_by_order) {
264 G_tau_by_order.resize(1, G_tau_zero);
265 set_frame(g_frame_n0, G_tau_by_order[0], 0);
266 set_frame(g_frame_nB, G_tau_by_order[0], constr_params.n_tau_green - 1);
267 }
268
269 // loop on different green function tau values
270 // n == 0 and n == n_tau - 1 already treated
271 for (auto n : range(1, constr_params.n_tau_green - 1)) {
272 if (solve_params.verbosity > 0) std::printf(" ..step %ld/%d\n", n, constr_params.n_tau_green - 2);
273
274 // define the tau_split and tau_max for this specific inchworm step.
275 //
276 // d_dag-================================-d-======================|
277 // 0 tau_split beta
278 //
279 // 0 < tau_split < beta
280 //
281 double tau_split = G_tau[0].mesh()(n);
282
283 // Initialize the zeroth order frame
284 frame_zeroth_order = make_bare_g_frame(ad_imp, u_tau, constr_params.gf_struct, tau_split, beta);
285
286 // calculation of the Monte Carlo solution:
287 auto res = qmc_step(solve_params, tau_split, beta, false, MODE::GREENFUNCTION);
288
289 // Green function results need to be normalized by Tr_Ubeta
290 res.normalize(Tr_Ubeta);
291
292 // Print results
293 res.print(solve_params.verbosity);
294
295 // Assign the current frame to G_tau
296 set_frame(res.frame, G_tau, n);
297
298 // Assign the last obtained frame error
299 errs_frame = res.errs_frame;
300
301 // Assign u_frame_by_order to the propagator u_tau_by_order
302 if (solve_params.measure_frame_by_order) {
303 if (res.frame_by_order.size() > G_tau_by_order.size()) G_tau_by_order.resize(res.frame_by_order.size(), G_tau_zero);
304 for (auto k : range(res.frame_by_order.size())) set_frame(res.frame_by_order[k], G_tau_by_order[k], n);
305 err_frame_by_order = res.err_frame_by_order;
306 }
307
308 // Initialize other results
309 if (solve_params.measure_order_histogram) order_histograms.push_back(res.order_histogram);
310
311 // Break out of loop when interrupted by signal
312 if (res.status == 2) break;
313 }
314 }

318 // One Monte Carlo sampling step (common to all solve scheme above):
319 qmc_results_t solver_core::qmc_step(solve_params_t const &solve_params, double tau_split, double tau_max, bool use_bare_propagator, MODE mode) {
320 params_t params(constr_params, solve_params);
321
322 // Construct the generic Monte-Carlo solver
323 triqs::mc_tools::mc_generic<scalar_t> mc(params.random_name, params.random_seed, params.verbosity);
324
325 // Capture random number generator
326 auto &rng = mc.get_rng();
327
328 // Create frame and Monte-Carlo configuration
329 auto frame = frame_zeroth_order;
330 auto config = config_t(frame, params.gf_struct, {0.0, tau_split});
331
332 // Create Monte-Carlo params
333 qmc_params_t qmc_params{tau_max, tau_split, use_bare_propagator, mode};
334
335 // Initialize the accumulators for the tau difference statistics
336 long n_bl = params.gf_struct.size();
337 std::vector<nda::array<lin_binning<double>, 3>> tau_diff_stat(n_bl);
338 for (auto bl : range(n_bl)) {
339 tau_diff_stat[bl] = nda::array<lin_binning<double>, 3>{params.gf_struct[bl].second, 2, 2};
340 tau_diff_stat[bl] = lin_binning<double>{0.0, -1, 1}; // Lin-binning only
341 }
342
343 // FIXME Reset operator widths manually
344 for (auto bl : range(n_bl)) {
345 for (auto &op : all_d_ops[bl]) {
346 op.left_width = fop_t{}.left_width;
347 op.right_width = fop_t{}.right_width;
348 }
349 for (auto &op : all_d_dag_ops[bl]) {
350 op.left_width = fop_t{}.left_width;
351 op.right_width = fop_t{}.right_width;
352 }
353 }
354
355 // Add moves
356 using moves::KIND;
357 mc.add_move(moves::insert<KIND::Single>{config, frame, qmc_params, *this, rng, tau_diff_stat}, "insert move");
358 mc.add_move(moves::remove<KIND::Single>{config, frame, qmc_params, *this, rng}, "remove move");
359
360 if (params.use_double_insertion) {
361 mc.add_move(moves::insert<KIND::Double>{config, frame, qmc_params, *this, rng, tau_diff_stat}, "double insert move", 0.5);
362 mc.add_move(moves::remove<KIND::Double>{config, frame, qmc_params, *this, rng}, "double remove move", 0.5);
363 mc.add_move(moves::insert<KIND::DoubleEqBl>{config, frame, qmc_params, *this, rng, tau_diff_stat}, "double insert move equal blocks", 0.5);
364 mc.add_move(moves::remove<KIND::DoubleEqBl>{config, frame, qmc_params, *this, rng}, "double remove move equal blocks", 0.5);
365 }
366
367 // Initialize result container
368 std::vector<int> shape_of_frame;
369 if (mode == MODE::PROPAGATOR) {
370 shape_of_frame = ad_imp.get_subspace_dims();
371 } else { // MODE::GREENFUNCTION
372 for (auto const &[blname, blsize] : params.gf_struct) { shape_of_frame.push_back(blsize); }
373 }
374 auto results = qmc_results_t{shape_of_frame};
375
376 // ----- Run the warmup
377 if (params.verbosity > 0) std::printf(" Warming up ...\n");
378 moves::base_move::reweighting_cutoff = 0;
379 moves::base_move::reweighting_coeffs.clear();
380 auto length_cycle = params.length_cycle.value_or(1);
381 int status = mc.warmup(params.n_warmup_cycles, length_cycle, triqs::utility::clock_callback(params.max_time));
382
383 // ----- Define Function for measuring and setting the operator distribution widths
384 bool tau_diff_stat_done = false;
385 auto gather_tau_diff_stat = [&]() {
386 // Gather the tau-diff statistics
387 moves::base_move::gather_tau_diff_stat = true;
388 mc.warmup(params.n_warmup_cycles, length_cycle, triqs::utility::clock_callback(params.max_time));
389 moves::base_move::gather_tau_diff_stat = false;
390
391 // Adjust the operators accordingly
392 long below_threshold_count = 0;
393 for (auto bl : range(n_bl)) {
394 for (auto &op : all_d_ops[bl]) { // FIXME join(all_d_ops[bl], all_d_dag_ops[bl])
395 if (mpi::all_reduce(tau_diff_stat[bl](op.idx, op.dag, 0).count()) > 0)
396 op.left_width = mean_mpi(world, tau_diff_stat[bl](op.idx, op.dag, 0).bins());
397 else
398 ++below_threshold_count;
399
400 if (mpi::all_reduce(tau_diff_stat[bl](op.idx, op.dag, 1).count()) > 0)
401 op.right_width = mean_mpi(world, tau_diff_stat[bl](op.idx, op.dag, 1).bins());
402 else
403 ++below_threshold_count;
404
405 if (params.verbosity > 0) std::cout << op << "\n";
406 }
407 for (auto &op : all_d_dag_ops[bl]) {
408 if (mpi::all_reduce(tau_diff_stat[bl](op.idx, op.dag, 0).count()) > 0)
409 op.left_width = mean_mpi(world, tau_diff_stat[bl](op.idx, op.dag, 0).bins());
410 else
411 ++below_threshold_count;
412
413 if (mpi::all_reduce(tau_diff_stat[bl](op.idx, op.dag, 1).count()) > 0)
414 op.right_width = mean_mpi(world, tau_diff_stat[bl](op.idx, op.dag, 1).bins());
415 else
416 ++below_threshold_count;
417
418 if (params.verbosity > 0) std::cout << op << "\n";
419 }
420 tau_diff_stat_done = true;
421 }
422 if (below_threshold_count > 0 && params.verbosity > 0)
423 fmt::print(" Found {} operators with insertion count below threshold, please raise n_warmup_cycles\n", below_threshold_count);
424 };
425
426 // ----- Run the callibration loop
427 size_t hist_max_idx = 0;
428 if (params.verbosity > 0) {
429 std::printf(" Callibrating ...\n");
430 std::printf(" %-12s| %-12s| %-12s| %-12s| %-12s| %-16s\n", "hist0", "autocorr", "acc insert", "acc remove", "new coeff0",
431 "new length_cycle");
432 }
433 for (int n = 1; status == 0; ++n) {
434 if (params.verbosity > 2) std::cout << "\nCallibration-loop " << n << "\n";
435
436 auto callibration_results = results;
437 mc.add_measure(measures::average_order{params, config, callibration_results}, "measure the average perturbation order");
438 mc.add_measure(measures::order_histogram{params, config, callibration_results}, "measure the perturbation order histogram");
439 mc.add_measure(measures::autocorr{params, config, frame, callibration_results}, "measure the autocorrelation time");
440
441 status = mc.accumulate(params.n_callibration_cycles, length_cycle, triqs::utility::clock_callback(params.max_time));
442 if (status != 0) break;
443 mc.collect_results(world);
444 mc.clear_measures();
445
446 // Update reweighting coefficients based on the perturbation order histogram taking the histogram max as a reference
447 auto const &hist = callibration_results.order_histogram;
448 if (n == 1) { // Set the hist_max_idx and reweighting cutoff on the first iteration
449 hist_max_idx = std::distance(begin(hist), max_element(begin(hist), end(hist)));
450 moves::base_move::reweighting_cutoff = std::max(1ul, hist_max_idx);
451 moves::base_move::reweighting_coeffs.resize(moves::base_move::reweighting_cutoff, 1.0);
452 }
453 for (auto k : range(hist_max_idx)) { // Scale up the weight for orders below hist_max_idx
454 auto hist_ratio = std::max(1.0, hist[hist_max_idx] / std::max(hist[k], 0.5 / params.n_callibration_cycles));
455 moves::base_move::reweighting_coeffs[k] *= hist_ratio;
456 if (config.size() == k) config.imp_weight *= hist_ratio;
457 }
458 if (params.max_prob_zeroth_order < hist[0]) { // Scale down the weight for the zeroth order if necessary
459 auto zero_reweight = 0.95 * std::max(1.0 - hist[0], 0.5 / params.n_callibration_cycles) / hist[0] * params.max_prob_zeroth_order
460 / (1.0 - params.max_prob_zeroth_order);
461 moves::base_move::reweighting_coeffs[0] *= zero_reweight;
462 if (config.size() == 0) config.imp_weight *= zero_reweight;
463 }
464
465 // When not reweighting, auto-deduce cycle length if not set
466 if (hist[0] > 0.7 * hist[hist_max_idx] and hist[0] <= params.max_prob_zeroth_order)
467 length_cycle = params.length_cycle.value_or(1.0 + std::ceil(length_cycle * std::pow(0.1 + callibration_results.auto_corr_time, 0.85)));
468
469 if (params.verbosity > 0) {
470 auto acc_rates = mc.get_acceptance_rates();
471 std::printf(" %-12.3f| %-12.3f| %-12.3f| %-12.3f| %-12.3e| %-16d\n", hist[0], callibration_results.auto_corr_time,
472 acc_rates.at("insert move"), acc_rates.at("remove move"), moves::base_move::reweighting_coeffs[0], length_cycle);
473 }
474
475 // When not reweighting, set operator tau distribution widths
476 if (hist[0] > 0.7 * hist[hist_max_idx] and hist[0] <= params.max_prob_zeroth_order and not tau_diff_stat_done) {
477 gather_tau_diff_stat();
478 continue;
479 }
480
481 // Iterate the callibration until the zeroth order is sampled with finite probability
482 if (hist[0] > 0.7 * hist[hist_max_idx] and hist[0] <= params.max_prob_zeroth_order
483 and (params.length_cycle.has_value() or callibration_results.auto_corr_time < 5.0))
484 break;
485 }
486
487 // Register all measurements
488 mc.add_measure(measures::frame{params, config, frame, results}, "measure the propagator / green function frame");
489 if (params.measure_average_sign) mc.add_measure(measures::average_sign{params, config, results}, "measure the average sign");
490 if (params.measure_average_order) mc.add_measure(measures::average_order{params, config, results}, "measure the average perturbation order");
491 if (params.measure_order_histogram)
492 mc.add_measure(measures::order_histogram{params, config, results}, "measure the perturbation order histogram");
493 if (params.measure_frame_by_order)
494 mc.add_measure(measures::frame_by_order{params, config, frame, results}, "measure the propagator / green function frame by order");
495
496 // Perform QMC run and collect results
497 if (status == 0) {
498 if (params.verbosity > 0) std::printf(" Accumulating ...\n");
499 results.status = mc.accumulate(params.n_cycles, length_cycle, triqs::utility::clock_callback(params.max_time));
500 mc.collect_results(world);
501 if (params.max_order && params.measure_order_histogram && results.order_histogram.size() > *params.max_order
502 && results.order_histogram[*params.max_order] > 0.0)
503 if (world.rank() == 0)
504 std::cout << "WARNING: Maximum perturbation order was sampled with a finite probability of " << results.order_histogram[*params.max_order]
505 << ". Check convergence w.r.t. max_order!\n";
506 }

508 // Post Processing
