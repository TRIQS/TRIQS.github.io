15 void qmc_results_t::normalize(scalar_t normalization_cte) {
16 if (normalization_cte == scalar_t{0.}) TRIQS_RUNTIME_ERROR << "Error in result normalization: Division by zero.";
17 frame /= normalization_cte;
18 weight_zeroth_order /= normalization_cte;
19 for (auto &err_frame_bl : errs_frame) err_frame_bl /= normalization_cte;
20 for (auto &frame_k : frame_by_order) frame_k /= normalization_cte;
21 };

23 void qmc_results_t::print(int verbosity) {

37 for (auto &frame_k : frame_by_order) std::printf(" %.3f, ", norm(frame_k) / norm(frame));

39 }

42 if (verbosity > 4)
43 for (auto Bl : frame) print_matrix(Bl);
44 }
