11 many_body_operator create_effective_hyb(gf_struct_t const &gf_struct) {
12 many_body_operator hyb_effective;
13 for (auto const &[blname, blsize] : gf_struct) {
14 for (auto [a, b] : product_range(blsize, blsize)) { hyb_effective += c_dag(blname, a) * c(blname, b); }
15 }
16 return hyb_effective;
17 }

19 frame_t make_g_frame_from_l_and_r(atom_diag const &ad_imp, gf_struct_t const &gf_struct, u_partial_t const &l, u_partial_t const &r) {
20
21 frame_t g_frame = make_zero_frame(gf_struct);
22
23 // G[bl][tau][i,j] = -<T c[bl][i](tau) cdag[bl][j](0)>
24 for (int bl : range(gf_struct.size())) {
25 auto const &[bl_name, bl_size] = gf_struct[bl];
26
27 for (auto [i, j] : product_range(bl_size, bl_size)) {
28
29 auto l_x_di = l * get_op_block_matrix(ad_imp, bl_name, i, false);
30 auto r_x_djdag = r * get_op_block_matrix(ad_imp, bl_name, j, true);
31 auto prod = make_frame(l_x_di * r_x_djdag);
32
33 g_frame[bl](i, j) -= trace(prod);
34 }
35 }
36
37 return g_frame;
38 }

40 frame_t make_bare_u_frame(atom_diag const &ad, double tau) {
41 auto u_frame = make_zero_frame(ad.get_subspace_dims());
42 for (auto [bl, bl_size] : enumerate(ad.get_subspace_dims()))
43 for (int i : range(bl_size)) u_frame[bl](i, i) = std::exp(-tau * ad.get_eigenvalue(bl, i));
44 return u_frame;
45 }

47 u_tau_t make_ED_propagator(atom_diag const &ad_tot, atom_diag const &ad_imp, atom_diag const &ad_bath, double beta, long n_tau) {
48
49 auto u_tau = u_tau_t{{beta, Fermion, n_tau}, ad_imp.get_subspace_dims()};
50 u_tau() = 0.0;
51
52 double dtau = beta / (n_tau - 1.);
53 for (int i_tau : mpi::chunk(range(n_tau))) {
54 auto u_frame = partial_trace_bath(ad_tot, ad_imp, ad_bath, beta, dtau * i_tau);
55 set_frame(u_frame, u_tau, i_tau);
56 }
57 u_tau = mpi::all_reduce(u_tau);
58
59 return u_tau;
60 }

62 frame_t make_bare_g_frame(atom_diag const &ad_imp, u_tau_t const &u_tau, gf_struct_t const &gf_struct, double tau_split, double beta) {
63 u_partial_t l(u_tau.size()), r(u_tau.size());
64
65 for (int i = 0; i < u_tau.size(); ++i) {
66 l[i] = {i, u_tau[i](beta - tau_split)};
67 r[i] = {i, u_tau[i](tau_split)};
68 }
69
70 return make_g_frame_from_l_and_r(ad_imp, gf_struct, l, r);
71 }

73 u_partial_t get_op_block_matrix(atom_diag const &ad, std::string const &bl_name, long idx, bool op_dag) {
74 auto res = u_partial_t(ad.n_subspaces());
75
76 auto lidx = ad.get_fops()[{bl_name, idx}];
77 for (auto bl_in : range(ad.n_subspaces())) {
78 auto bl_out = (op_dag ? ad.cdag_connection(lidx, bl_in) : ad.c_connection(lidx, bl_in));
79 auto matrix = (op_dag ? ad.cdag_matrix(lidx, bl_in) : ad.c_matrix(lidx, bl_in));
80 res[bl_in] = {bl_out, matrix};
81 }
82
83 return res;
84 }

95 std::pair<uint64_t, uint64_t> split_fs(uint64_t fs, int bit_index) {
96 //std::cout << std::bitset<10>(fs) << "\n";
97 uint64_t fs_left = fs % (1 << bit_index); // All bits left of bit_index
98 uint64_t fs_right = fs >> bit_index; // All bits right of bit_index
99 return {fs_left, fs_right};
100 }

106 std::pair<int, int> fs_to_bl_and_idx(uint64_t fs, atom_diag const &ad) {
107 for (int bl : range(ad.n_subspaces())) {
108 for (auto [idx, fs_i] : enumerate(ad.get_fock_states(bl))) {
109 if (fs == fs_i) return {bl, idx};
110 }
111 }
112 std::printf("error: number not found in the fock states\n");
113 std::abort();
114 }

116 frame_t partial_trace_bath(atom_diag const &ad_tot, atom_diag const &ad_imp, atom_diag const &ad_bath, double beta, double tau) {
117
118 for (auto i : range(ad_imp.get_fops().size())) {
119 if (ad_tot.get_fops().data()[i] != ad_imp.get_fops().data()[i]) {
120 std::printf("error: the first indices of ad_tot should be the same as the one in ad_imp.\n");
121 std::abort();
122 }
123 }
124
125 // Given atom_diag object, calculate e^[-tau * H] in Fockstate Basis
126 auto calc_e_H_fs = [](atom_diag const &ad, double tau) {
127 // e^[-tau * H] in eigenbasis
128 auto e_H_tau = make_bare_u_frame(ad, tau);
129
130 // Rotate to Fockstate Basis
131 auto e_H_tau_fs = e_H_tau;
132 for (auto bl : range(ad.n_subspaces())) {
133 auto const &rot = ad.get_eigensystems()[bl].unitary_matrix;
134 e_H_tau_fs[bl] = rot * e_H_tau[bl] * dagger(rot);
135 }
136 return e_H_tau_fs;
137 };
138 auto e_H_tot_tau_fs = calc_e_H_fs(ad_tot, tau);
139 auto e_H_bath_beta_minus_tau_fs = calc_e_H_fs(ad_bath, beta - tau);
140
141 // The result container
142 frame_t utau_imp = make_zero_frame(ad_imp.get_subspace_dims());
143
144 for (auto [bl, bl_size] : enumerate(ad_tot.get_subspace_dims())) {
145
146 for (auto [i, j] : product_range(bl_size, bl_size)) {
147
148 int nfops_imp = ad_imp.get_fops().size();
149 auto [fs_imp_i, fs_bath_i] = split_fs(ad_tot.get_fock_states(bl)[i], nfops_imp);
150 auto [fs_imp_j, fs_bath_j] = split_fs(ad_tot.get_fock_states(bl)[j], nfops_imp);
151
152 if (fs_bath_i == fs_bath_j) {
153
154 auto [bl_imp_i, i_imp] = fs_to_bl_and_idx(fs_imp_i, ad_imp);
155 auto [bl_imp_j, j_imp] = fs_to_bl_and_idx(fs_imp_j, ad_imp);
156
157 if (bl_imp_i != bl_imp_j) TRIQS_RUNTIME_ERROR << "Block structure of ad_imp incompatible with hybridization";
158
159 auto [bl_bath, i_bath] = fs_to_bl_and_idx(fs_bath_i, ad_bath);
160
161 utau_imp[bl_imp_i](i_imp, j_imp) += e_H_tot_tau_fs[bl](i, j) * e_H_bath_beta_minus_tau_fs[bl_bath](i_bath, i_bath);
162 }
163 }
164 }
165
166 // Rotate to impurity eigenbasis and normalize by Z_bath
167 auto Z_bath = partition_function(ad_bath, beta);
168 for (auto bl_imp : range(ad_imp.n_subspaces())) {
169 auto rot = ad_imp.get_eigensystems()[bl_imp].unitary_matrix;
170 utau_imp[bl_imp] = 1.0 / Z_bath * dagger(rot) * utau_imp[bl_imp] * rot;
171 }
172
173 // Correct for using different reference energies in imp, bath and tot
174 double factor = std::exp((ad_bath.get_gs_energy() + ad_imp.get_gs_energy() - ad_tot.get_gs_energy()) * tau);
175
176 return factor * utau_imp;
177 }
