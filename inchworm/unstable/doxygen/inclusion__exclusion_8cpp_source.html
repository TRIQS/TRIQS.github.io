28 std::vector<set_of_segments_t> combine_segments(std::vector<segment_t> const &segment_list, time_diagram_t const &diagram, bool search_disjoint) {
29
30 auto set_list = std::vector<set_of_segments_t>{};
31 set_list.reserve(1 << (diagram.size() / 3)); // FIXME Improve estimate for set size
32
33 for (auto const &seg : segment_list) { set_list.emplace_back(seg, diagram, segment_list); }
34 if (not set_list.empty()) set_list.pop_back(); // Do not consider the full segment
35
36 // Loop over the (continuously growing) list of sets until we have reached the end
37 for (int set_idx = 0; set_idx != set_list.size(); ++set_idx) {
38
39 for (auto const &additional_segment : segment_list) {
40
41 // Only consider segments that start to the right of the current rightmost segment
42 if (additional_segment.begin < set_list[set_idx].end) continue;
43
44 set_of_segments_t new_set = set_list[set_idx];
45 new_set.append_right(additional_segment);
46
47 // Add set iff disjoint/adjacent when searching for disjoint/adjacent
48 if ((search_disjoint && new_set.disjoint) or (not search_disjoint && new_set.adjacent)) { set_list.emplace_back(std::move(new_set)); }
49 }
50 }
51
52 // If searching adjacent sets, erase all sets with a single segment
53 if (not search_disjoint) set_list.erase(set_list.begin(), set_list.begin() + segment_list.size() - 1);
54
55 //set_list.shrink_to_fit();
56 return set_list;
57 }

64 void calculate_segment(int segment_id,
65 std::vector<segment_t> &segment_list, // not const: modified
66 std::vector<set_of_segments_t> const &list_of_set_of_disjoint_segments,
67 std::vector<set_of_segments_t> const &list_of_set_of_adjacent_segments, hyb_matrix_t const &hyb_mat,
68 time_diagram_t const &diagram, bool verbose) {
69 auto &seg = segment_list[segment_id];
70 EXPECTS(not seg.calculated);
71 seg.calculated = true;
72
73 // Calculate the determinant of the full segment
74 seg.value = hyb_mat.extract_det(range(seg.begin, seg.end));
75
76 // Vertex list for the det calculation. Allocate once and reuse
77 std::vector<int> det_vertices;
78 det_vertices.reserve(seg.size);
79
80 for (auto const &set : list_of_set_of_disjoint_segments) {
81 // FIXME Why do we have to make a distinction here between full segment and smaller? <= ?
82 if ((seg.size < diagram.size()) and (set.begin < seg.begin or seg.end <= set.end)) continue;
83
84 int pos = seg.begin;
85 hyb_scalar_t value = 1.0;
86
87 for (auto const &subseg : set.segs()) {
88 ASSERT(subseg.calculated);
89
90 value *= -subseg.value;
91
92 // Register any non-segment vertices before segment for det calculation
93 for (auto i : range(pos, subseg.begin)) det_vertices.push_back(i);
94 pos = subseg.end;
95
96 // Caution: If pulling the sub-segment to the front of the current segment corresponds to
97 // an odd number of row and column permutations in the segment submatrix we get an additional minus sign
98 // Note: We could move this into the extract_det function
99 if (subseg.size % 4 != 0 && (subseg.begin - seg.begin) % 2 == 1) value *= -1;
100 }
101 for (auto i : range(pos, seg.end)) det_vertices.push_back(i);
102
103 if (det_vertices.size() > 0 && value != 0.0) { value *= hyb_mat.extract_det(det_vertices); }
104 seg.value += value;
105
106 det_vertices.clear();
107 }
108
109 seg.value_without_cuts = seg.value; // This should be the value in Eq (14) !?
110
111 for (auto const &set : list_of_set_of_adjacent_segments) {
112 if (seg.begin == set.begin && seg.end == set.end) {
113 hyb_scalar_t value = 1.0;
114 for (auto const &subseg : set.segs()) {
115 ASSERT(subseg.calculated);
116 value *= -subseg.value_without_cuts;
117 }
118 seg.value -= value;
119 }
120 }
121
122 // ------------ Debug Prints ------------
123 if (verbose) {
124 print_segment(seg, diagram);
125 std::printf("range of vertex: ");
126 for (auto o : range(seg.begin, seg.end)) std::printf("%ld ", o);
127 hyb_mat.print();
128 std::printf("\nsegment[%d]= % 4.8f\n\n", segment_id, hyb_mat.extract_det(range(seg.begin, seg.end)));
129 std::printf(" % 15.8f % 15.8f\n", seg.value, seg.value_without_cuts);
130 }
131 // --------------------------------------
132 }

145 std::vector<segment_t> determine_segments(time_diagram_t const &diagram) {
146 int diag_size = diagram.size();
147
148 // Create sub-segments from smallest to largest, to respect calculation order
149 int segment_id = 0;
150 std::vector<segment_t> seg_list;
151 for (auto size : range(smallest_segment, diag_size - 1, 2))
152 for (auto begin : range(0, diag_size - size + 1)) {
153 auto end = begin + size;
154
155 // Optimization: Do not consider vanishing segments of size four (xoxo & oxox)
156 if constexpr (remove_xoxo)
157 if (size == 4 and diagram.op_list[begin].dag == diagram.op_list[begin + 2].dag) continue;
158
159 // Consider only subsegments that do not contain a split-point
160 if (std::any_of(diagram.split_points.begin(), diagram.split_points.end(), [&](auto &sp) { return begin < sp and sp < end; })) continue;
161
162 // If [begin, end) contains the same number of d and d_dag, add it to the list
163 auto Ndag = std::count_if(diagram.op_list.cbegin() + begin, diagram.op_list.cbegin() + end, [](auto &op) { return op.dag; });
164 if (2 * Ndag == size) seg_list.emplace_back(begin, end, segment_id++);
165 }
166
167 // Finally, add the full segment, which may contain a split-point
168 seg_list.emplace_back(0, diag_size, segment_id++);
169
170 return seg_list;
171 }

173 hyb_scalar_t inclusion_exclusion(time_diagram_t const &diagram, hyb_matrix_t hyb_mat, bool verbose) {
174
175 if (diagram.size() == 0) return 1.0;
176
177 // Ignore segments of lenght 2 by setting necessary matrix elements to zero
178 hyb_mat.optimize_inclusion_exclusion();
179
180 // We need to have at least one split-point between operators for a finite hybridization weight
181 if (diagram.is_trivial) { return 0.0; }
182
183 if (diagram.perturbation_order() == 1) { return hyb_mat.det(); };
184
185 std::vector<segment_t> segment_list = determine_segments(diagram);
186 std::vector<set_of_segments_t> list_of_set_of_disjoint_segments = combine_segments(segment_list, diagram, true /*disjointness*/);
187 std::vector<set_of_segments_t> list_of_set_of_adjacent_segments = combine_segments(segment_list, diagram, false /*disjointness*/);
188
189 for (auto const &seg : segment_list) { // segment_list is sorted w.r.t. segment size
190 calculate_segment(seg.id, segment_list, list_of_set_of_disjoint_segments, list_of_set_of_adjacent_segments, hyb_mat, diagram, verbose);
191 }
192
193 // ------------ Debug Prints ------------
194 if (verbose) {
195 std::printf("\n\n##################\nINCLUSION-EXCLUSION:\n");
196 hyb_mat.print();
197 std::printf("\nnumber of segments = %lu\n\n", segment_list.size());
198 std::printf("list of single segements:\n");
199 print_diag(diagram);
200 for (int j = 0; j < segment_list.size(); j++) {
201 print_segment(segment_list[j], diagram);
202 std::printf("\n");
203 }
204 std::printf("\n\nlist of set of disjoint segments:\n");
205 print_diag(diagram);
206 for (auto const &s : list_of_set_of_disjoint_segments) {
207 print_set(segment_list, s, diagram);
208 std::printf("\n");
209 }
210 std::printf("\n\nlist of set of adjacent segments:\n");
211 print_diag(diagram);
212 for (auto const &s : list_of_set_of_adjacent_segments) {
213 print_set(segment_list, s, diagram);
214 std::printf("\n");
215 }
216 std::printf("\n## diagram = '%s'\n", diagram_string(diagram).c_str());
217 std::printf("kOrder = %d\n", diagram.perturbation_order());
218 std::printf("number of segments = %lu\n", segment_list.size());
219 std::printf("number of adjacent sets = %lu\n", list_of_set_of_adjacent_segments.size());
220 std::printf("number of disjoint sets = %lu\n", list_of_set_of_disjoint_segments.size());
221 }
222 // --------------------------------------
223
224 return segment_list.back().value;
225 }
