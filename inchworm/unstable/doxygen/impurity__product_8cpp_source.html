13 bool has_zero_trace(atom_diag const &ad, diagram::time_diagram_t const &diagram) {
14 if (diagram.size() == 0) return false;
15
16 for (long initial_bl : range(ad.n_subspaces())) {
17 long curr_bl = initial_bl;
18 for (auto const &op : diagram.op_list) {
19 curr_bl = (op.dag ? ad.cdag_connection(op.linear_index, curr_bl) : ad.c_connection(op.linear_index, curr_bl));
20 if (curr_bl == -1) break;
21 }
22 // Any non-void block gives us a finite trace contribution
23 if (curr_bl != -1) return false;
24 }
25 return true;
26 }

28 u_partial_t impurity_product(atom_diag const &ad, diagram::time_diagram_t const &diagram, double tau_max, double tau_min,
29 interpolator_t<scalar_t> const *const u_interpolator_p) {
30 EXPECTS(tau_max > tau_min);
31
32 // Filter out all operators in the time-window [tau_min, tau_max]
33 std::vector<fop_t> op_lst;
34 for (auto const &op : diagram.op_list) {
35 if ((op.tau >= tau_min) and (op.tau <= tau_max)) op_lst.push_back(op);
36 }
37
38 // Helper to calculate u_tau for both inchworm and cthyb case
39 auto u_tau = [&](int bl, double tau) -> matrix_t {
40 if (u_interpolator_p) { // inchworm case, interpolate
41 return (*u_interpolator_p)(bl, tau);
42 } else { // cthyb case
43 auto bl_size = ad.get_subspace_dim(bl);
44 auto res = matrix_t::zeros({bl_size, bl_size});
45 for (auto j : range(bl_size)) res(j, j) = std::exp(-tau * ad.get_eigenvalue(bl, j));
46 return res;
47 }
48 };
49
50 // Calculate full operator product
51 u_partial_t u_partial(ad.n_subspaces());
52 for (long initial_bl : range(ad.n_subspaces())) {
53
54 // Treat the trivial case of zero operators separately
55 if (op_lst.empty()) {
56 u_partial[initial_bl] = {initial_bl, u_tau(initial_bl, tau_max - tau_min)};
57 continue;
58 }
59
60 // Short-circuit if the product contains a void block, i.e. -1
61 long curr_bl = initial_bl;
62 for (auto const &op : op_lst) {
63 curr_bl = (op.dag ? ad.cdag_connection(op.linear_index, curr_bl) : ad.c_connection(op.linear_index, curr_bl));
64 if (curr_bl == -1) break;
65 }
66 if (curr_bl == -1) {
67 u_partial[initial_bl] = {-1, {}};
68 continue; // Skip matrix product
69 }
70
71 // Perform Matrix multiplication
72 // u(tau_max - tau_{n-1}) op_{n-1} u(tau_{n-1}-tau_{n-2}) op_{n-2} ... op_1 u(tau_1 - tau_0) op_0 u(tau_0 - tau_min)
73 auto mat_bl = u_tau(initial_bl, op_lst.front().tau - tau_min);
74 curr_bl = initial_bl;
75 for (auto [i, op] : enumerate(op_lst)) {
76 mat_bl = (op.dag ? ad.cdag_matrix(op.linear_index, curr_bl) : ad.c_matrix(op.linear_index, curr_bl)) * mat_bl;
77 curr_bl = (op.dag ? ad.cdag_connection(op.linear_index, curr_bl) : ad.c_connection(op.linear_index, curr_bl));
78
79 if (i == op_lst.size() - 1)
80 mat_bl = u_tau(curr_bl, tau_max - op.tau) * mat_bl;
81 else
82 mat_bl = u_tau(curr_bl, op_lst[i + 1].tau - op.tau) * mat_bl;
83 }
84 u_partial[initial_bl] = {curr_bl, mat_bl};
85 }
86 return u_partial;
87 }
