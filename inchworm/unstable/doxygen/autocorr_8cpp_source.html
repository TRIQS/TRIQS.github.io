15 void autocorr::accumulate(scalar_t sign) {
16 // We weight the Monte-Carlo by the frobenius norm of the current frame
17 // This importance sampling factor has to be corrected in the measurement
18 scalar_t s = sign / (config.imp_weight);
19
20 log_acc << trace(s * curr_frame);
21 }

23 void autocorr::collect_results(mpi::communicator const &comm) {
24
25 auto errs = std::get<1>(log_acc.mean_errors_and_taus(comm));
26
27 // Debug Prints
28 if (comm.rank() == 0 and verbosity > 1) {
29 std::cout << "errs norm(frame): [";
30 for (auto err : errs) std::cout << err << " ";
31 std::cout << "]\n";
32 }
33
34 // Estimate auto-correlation time
35 results.auto_corr_time = 0.0;
36 if (comm.rank() == 0 && !errs.empty() && errs[0] > 0)
37 results.auto_corr_time = std::max(0.0, tau_estimate_from_errors(errs[int(0.7 * errs.size())], errs[0]));
38 mpi::broadcast(results.auto_corr_time, comm, 0);
39
40 // Reset the accumulator
41 log_acc = {0.0, -1};
42 }
