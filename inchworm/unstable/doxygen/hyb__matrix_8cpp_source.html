15 hyb_matrix_t::hyb_matrix_t(time_diagram_t const &diagram) : diagram{diagram}, size{diagram.perturbation_order()}, mat{size, size} {
16
17 auto hyb_function_dummy = [](double dtau) { return (1.0 / (0.8 * (dtau - 0.5))); };
18 for (auto [i, d] : enumerate(diagram.d_list)) {
19 for (auto [j, d_dag] : enumerate(diagram.d_dag_list)) {
20 mat(i, j) = (0.5 + j - i) * hyb_function_dummy(d_dag.tau - d.tau);
21 if (i < j) mat(i, j) = -mat(i, j);
22 }
23 }
24 }

26 hyb_matrix_t::hyb_matrix_t(time_diagram_t const &diagram, hyb_tau_t const &Delta)
27 : diagram{diagram}, size{diagram.perturbation_order()}, mat{size, size} {
28
29 auto eval_Delta = [&Delta](fop_t const &cdag, fop_t const &c) -> hyb_scalar_t {
30 if (cdag.bl != c.bl) return 0.;
31
32 double dtau = cdag.tau - c.tau;
33 if (dtau >= 0.) {
34 return Delta[c.bl](dtau)(cdag.idx, c.idx);
35 } else {
36 return -Delta[c.bl](Delta[c.bl].mesh().beta() + dtau)(cdag.idx, c.idx);
37 }
38 };
39
40 for (auto [i, d] : enumerate(diagram.d_list)) {
41 for (auto [j, d_dag] : enumerate(diagram.d_dag_list)) { mat(i, j) = eval_Delta(d_dag, d); }
42 }
43 }

45 void hyb_matrix_t::optimize_inclusion_exclusion() {
46 if (smallest_segment == 4) {
47 for (int k = 0; k < diagram.size() - 1; k++) {
48
49 if (std::any_of(begin(diagram.split_points), end(diagram.split_points), [k](int i) { return i == k + 1; })) continue;
50
51 if (not diagram.op_list[k].dag and diagram.op_list[k + 1].dag) { // segment of length 2
52 int it = diagram.op_list[k + 1].order_index;
53 int it_dag = diagram.op_list[k].order_index;
54 mat(it_dag, it) = 0.;
55 } else if (diagram.op_list[k].dag and not diagram.op_list[k + 1].dag) { // segment on length 2
56 int it = diagram.op_list[k].order_index;
57 int it_dag = diagram.op_list[k + 1].order_index;
58 mat(it_dag, it) = 0.;
59 }
60 }
61 }
