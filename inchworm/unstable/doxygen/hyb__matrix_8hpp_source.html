45 template <typename R> hyb_scalar_t hyb_matrix_t::extract_det(R const &list_of_indices) const {
46 EXPECTS(list_of_indices.size() % 2 == 0);
47
48 // Creation of C-Style arrays slightly more performant than sso nda::array
49 int N = list_of_indices.size() / 2;
50 int list_of_d[N], list_of_d_dag[N];
51
52 for (int i = 0, j = 0; auto idx : list_of_indices) {
53 if (diagram.op_list[idx].dag)
54 list_of_d_dag[i++] = diagram.op_list[idx].order_index;
55 else
56 list_of_d[j++] = diagram.op_list[idx].order_index;
57 }
58
59 nda::matrix<hyb_scalar_t, nda::C_layout, nda::sso<1000>> m(N, N);
60 for (int i = 0; i < N; i++) {
61 for (int j = 0; j < N; j++) { m(i, j) = mat(list_of_d[i], list_of_d_dag[j]); }
62 }
63
64 return nda::linalg::det_in_place(m);
65 }
