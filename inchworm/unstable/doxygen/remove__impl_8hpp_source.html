13 template <KIND Kind> double remove<Kind>::remove_single_op_pair(long bl, config_t &config) {
14
15 if (config.size(bl) == 0) return 0;
16
17 long idx = rng(config.size(bl));
18 long idx_dag = rng(config.size(bl));
19
20 fop_t d = config.d_bl_list[bl][idx];
21 fop_t d_dag = config.d_dag_bl_list[bl][idx_dag];
22
23 double prop_prob = 1.0 / config.size(bl) / config.size(bl);
24
25 // Never accept removal from trivial configurations of first order
26 // Note: Special insertion rule will make sure that 1st order is non-trivial
27 if (params.tau_split > 0.0 && config.size() == 1)
28 if ((d.tau < params.tau_split && d_dag.tau < params.tau_split) || (d.tau > params.tau_split && d_dag.tau > params.tau_split)) return 0;
29
30 if (not config.try_erase(bl, idx_dag, idx)) return 0;
31
32 long bl_size = gf_struct[bl].second;
33 double inv_prop_prob = get_time_prop_prob(d, d_dag, config, params.tau_split, params.tau_max) / bl_size / bl_size;
34
35 return inv_prop_prob / prop_prob;
36 }
37
38 template <KIND Kind> double remove<Kind>::try_config_update(config_t &config) {
39
40 long n_bl = gf_struct.size();
41
42 if constexpr (Kind == KIND::Single) {
43
44 return remove_single_op_pair(rng(n_bl), config);
45
46 } else { // Double Insert
47
48 long bl1 = rng(n_bl);
49 long bl2 = (Kind == KIND::DoubleEqBl) ? bl1 : rng(n_bl);
50
51 if (bl1 == bl2 and config.size(bl1) < 2) return 0;
52
53 double t_ratio1 = remove_single_op_pair(bl1, config);
54 if (t_ratio1 == 0.0) return 0.0;
55
56 double t_ratio2 = remove_single_op_pair(bl2, config);
57
58 return t_ratio1 * t_ratio2;
59 }
60 }
