21 time_diagram_t::time_diagram_t(std::vector<fop_t> const &d_list_, std::vector<fop_t> const &d_dag_list_, std::vector<double> const &split_times,
22 int verbose)
23 : op_list(2 * d_list_.size()), d_list{d_list_}, d_dag_list{d_dag_list_} {
24 EXPECTS(d_list.size() == d_dag_list.size());
25
26 int order = d_list.size();
27 if (order == 0) return;
28
29 // Initialize the time-ordered list of all operators
30 for (int i = 0; i < order; i++) {
31 op_list[2 * i].tau = d_dag_list[i].tau;
32 op_list[2 * i].dag = true;
33 op_list[2 * i].linear_index = d_dag_list[i].linear_index;
34 op_list[2 * i].bl = d_dag_list[i].bl;
35 op_list[2 * i].idx = d_dag_list[i].idx;
36 op_list[2 * i].order_index = i;
37
38 op_list[2 * i + 1].tau = d_list[i].tau;
39 op_list[2 * i + 1].dag = false;
40 op_list[2 * i + 1].linear_index = d_list[i].linear_index;
41 op_list[2 * i + 1].bl = d_list[i].bl;
42 op_list[2 * i + 1].idx = d_list[i].idx;
43 op_list[2 * i + 1].order_index = i;
44 }
45
46 std::vector<int> ivec(op_list.size());
47 std::iota(begin(ivec), end(ivec), 0);
48 std::sort(begin(ivec), end(ivec), [this](int i, int j) { return op_list[i] < op_list[j]; });
49 _sign = find_parity(ivec);
50
51 std::sort(op_list.begin(), op_list.end(), std::less<>{});
52 for (int i = 0; i < op_list.size() - 1; i++) EXPECTS(op_list[i].tau != op_list[i + 1].tau);
53
54 // For each split_time determine the op_list index of the operator to the right (i.e. the split point)
55 if (verbose > 3) std::printf("split points:\n");
56 split_points.reserve(split_times.size());
57 for (auto s_time : split_times) {
58 int i = 0;
59 for (; i < op_list.size(); i++) {
60 EXPECTS(s_time != op_list[i].tau);
61 if (s_time < op_list[i].tau) break;
62 }
63 split_points.push_back(i);
64 if (verbose > 3) std::printf("%d % 4.3f\n", i, s_time);
65 }
66
67 // Diagram is trivial if no split_times are found between the smallest and largest operator time
68 is_trivial = std::all_of(cbegin(split_points), cend(split_points), [&](int i) { return i == 0 or i == op_list.size(); });
69 if (is_trivial and verbose > 3) std::printf("diagram is trivial\n");
70
71 // For each d and d_dag, find its position in op_list (inverse table of order_index)
72 for (int i = 0; i < op_list.size(); i++) {
73 if (op_list[i].dag)
74 pos_d_dag.push_back(i);
75 else
76 pos_d.push_back(i);
77 }
78 }
