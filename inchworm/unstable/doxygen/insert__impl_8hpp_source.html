14 template <KIND Kind> double insert<Kind>::insert_single_op_pair(long bl, config_t &config) {
15
16 long bl_size = gf_struct[bl].second;
17
18 auto d = solver.all_d_ops[bl][rng(bl_size)];
19 auto d_dag = solver.all_d_dag_ops[bl][rng(bl_size)];
20
21 auto [d_tau, d_dag_tau] = gen_op_times(rng, d, d_dag, config, params.tau_split, params.tau_max);
22 d.tau = d_tau;
23 d_dag.tau = d_dag_tau;
24
25 last_d = d;
26 last_d_dag = d_dag;
27
28 double prop_prob = get_time_prop_prob(d, d_dag, config, params.tau_split, params.tau_max) / bl_size / bl_size;
29
30 if (not config.try_insert(d_dag, d)) return 0.0;
31
32 double inv_prop_prob = 1.0 / config.size(bl) / config.size(bl);
33
34 return inv_prop_prob / prop_prob;
35 }

37 template <KIND Kind> double insert<Kind>::try_config_update(config_t &config) {
38
39 long n_bl = gf_struct.size();
40
41 if constexpr (Kind == KIND::Single) {
42
43 return insert_single_op_pair(rng(n_bl), config);
44
45 } else { // Double Insert
46
47 long bl1 = rng(n_bl);
48 long bl2 = (Kind == KIND::DoubleEqBl) ? bl1 : rng(n_bl);
49
50 double t_ratio1 = insert_single_op_pair(bl1, config);
51 if (t_ratio1 == 0.0) return 0.0;
52
53 double t_ratio2 = insert_single_op_pair(bl2, config);
54
55 return t_ratio1 * t_ratio2;
56 }
57 }

59 template <KIND Kind> scalar_t insert<Kind>::accept() {
60
61 // Gather tau statistic during warmup
62 if (gather_tau_diff_stat) {
63 for (auto const &op : {last_d, last_d_dag}) {
64 double tau_diff = cyclic_difference(op, prop_config, params.tau_max);
65 if (tau_diff < params.tau_max / 2.0) // left_width
66 tau_diff_stat[op.bl](op.idx, op.dag, 0) << tau_diff;
67 else // right_width
68 tau_diff_stat[op.bl](op.idx, op.dag, 1) << params.tau_max - tau_diff;
69 }
70 }
71
72 return base_move::accept();
73 }
