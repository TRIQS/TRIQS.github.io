19 scalar_t base_move::attempt() {
20
21 prop_config = config;
22
23 // ------ Generate the new configuration and diagram -------
24
25 auto t_ratio = try_config_update(prop_config);
26 if (t_ratio == 0.0) return 0.0; // Check if move has failed
27 auto prop_pert_order = prop_config.size();
28 if (prop_pert_order > solver.last_solve_params->max_order.value_or(prop_pert_order)) return 0.0;
29 auto diagram = diagram::time_diagram_t{prop_config, {params.tau_split}};
30
31 // We need to have at least one split-point between operators for a finite hybridization weight
32 if (not params.use_bare_propagator and diagram.size() > 0 and diagram.is_trivial) return 0.0;
33
34 // Quick-check for vanishing impurity trace
35 if (params.mode == MODE::PROPAGATOR and has_zero_trace(solver.ad_imp, diagram)) return 0.0;
36
37 // ------ Calculate the hybridization weight -------
38
39 auto hyb_mat = diagram::hyb_matrix_t(diagram, solver.Delta_tau_tilde);
40 prop_config.sign = diagram.sign();
41
42 if (params.use_bare_propagator)
43 prop_config.hyb_weight = hyb_mat.det();
44 else
45#ifdef PROPER_ENUMERATION
46 prop_config.hyb_weight = diagram::proper_enum(diagram, hyb_mat);
47#else
48 prop_config.hyb_weight = diagram::inclusion_exclusion(diagram, hyb_mat);
49#endif
50
51 // ------ Calculate the impurity frame weight -------
52
53 if (prop_config.size() == 0) { // Use precalculated zeroth-order frame if possible
54 prop_frame = solver.frame_zeroth_order;
55
56 } else if (params.mode == MODE::PROPAGATOR) {
57 if (params.use_bare_propagator) {
58 prop_frame = make_frame(impurity_product(solver.ad_imp, diagram, params.tau_max, 0));
59 } else {
60 prop_frame = make_frame(impurity_product(solver.ad_imp, diagram, params.tau_max, params.tau_split, &solver.u_interpolator)
61 * impurity_product(solver.ad_imp, diagram, params.tau_split, 0, &solver.u_interpolator));
62 }
63
64 } else { // MODE::GREENFUNCTION
65 EXPECTS(not params.use_bare_propagator);
66
67 prop_frame = make_zero_frame(gf_struct);
68
69 // Calculate -Tr[imp_prod(beta, tau) * c(tau) * imp_prod(tau, 0) * cdag(0)]
70 // for all combinations of fundamental operator flavors
71 auto l = impurity_product(solver.ad_imp, diagram, params.tau_max, params.tau_split, &solver.u_interpolator);
72 auto r = impurity_product(solver.ad_imp, diagram, params.tau_split, 0, &solver.u_interpolator);
73 prop_frame = make_g_frame_from_l_and_r(solver.ad_imp, gf_struct, l, r);
74
75 // Account for the sign due to the additional operator insertions
76 auto const &ops = diagram.op_list;
77 int nop_r = std::count_if(begin(ops), end(ops), [tau_split = params.tau_split](auto const &op) { return tau_split > op.tau; });
78 if (nop_r % 2 == 1) { prop_frame *= -1; }
79 }
80
81 prop_config.imp_weight = norm(prop_frame);
82
83 // Reweight perturbation orders below the reweighting_cutoff to guarantee
84 // that the zeroth order is sampled properly for normalization purposes
85 if (prop_pert_order < reweighting_cutoff) prop_config.imp_weight *= reweighting_coeffs[prop_pert_order];
86
87 // ------ Calculate overall weight ratio -------
88
89 auto sign_ratio = prop_config.sign / config.sign;
90 auto w_hyb_ratio = prop_config.hyb_weight / config.hyb_weight;
91 auto w_imp_ratio = prop_config.imp_weight / config.imp_weight;
92
93 auto ratio = sign_ratio * t_ratio * w_imp_ratio * w_hyb_ratio;
94
95 // ------ Debugging Information -------
96
97#ifdef INCHWORM_DEBUG_PRINTS
98 //if(rng.preview() >= std::min(1.0, std::abs(ratio))) return ratio;
99 std::printf("\n\n====== Try %s ======\n", name().c_str());
100 print_configuration(diagram);
101 print(prop_frame);
102 hyb_mat.print();
103 std::printf("\n\nhyb.det()=% 4.7e \n", hyb_mat.det());
104 std::printf("\n\nsign= %d w_hyb=% 4.7e w_imp=% 4.7e old_w_hyb=% 4.7e old_w_imp=% 4.7e \n", prop_config.sign, prop_config.hyb_weight,
105 prop_config.imp_weight, config.hyb_weight, config.imp_weight);
106 std::printf("\n\nsign_ratio= %d w_hyb_ratio=% 4.7e w_imp_ratio=% 4.7e t_ratio=% 4.7e\n", sign_ratio, w_hyb_ratio, w_imp_ratio, t_ratio);
107 std::printf("\n\nratio=% 4.7e\n", ratio);
108 if (not params.use_bare_propagator) {
109 std::printf("proper_enum w.hyb =% 4.7f \n", diagram::proper_enum(diagram, hyb_mat));
110 std::printf("inclusion_exclusion w.hyb =% 4.7f \n", diagram::inclusion_exclusion(diagram, hyb_mat));
111 }
112 getchar();
113#endif
114
115 return ratio;
116 }
