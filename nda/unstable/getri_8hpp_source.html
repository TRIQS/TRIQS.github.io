30 [[nodiscard]] int getri(M &&m, basic_array<int, 1, C_layout, Algebra, CP> &ipiv) {
31 using M_t = std::decay_t<M>;
32 static_assert(is_regular_or_view_v<M_t>, "getrf: M must be a matrix, matrix_view, array or array_view of rank 2");
33 static_assert(M_t::rank == 2, "M must be of rank 2");
34 static_assert(is_blas_lapack_v<typename M_t::value_type>, "Matrices must have the same element type and it must be double, complex ...");
35
36 EXPECTS(ipiv.size() >= std::min(m.extent(0), m.extent(1)));
37
38 using T = typename M_t::value_type;
39 int info = 0;
40 std::array<T, 2> work1{0, 0}; // always init for MSAN and clang-tidy ...
41
42 // first call to get the optimal lwork
43 f77::getri(get_n_rows(m), m.data(), get_ld(m), ipiv.data(), work1.data(), -1, info);
44 int lwork;
45 if constexpr (is_complex_v<T>)
46 lwork = std::round(std::real(work1[0])) + 1;
47 else
48 lwork = std::round(work1[0]) + 1;
49
50 array<T, 1> work(lwork);
51
52#if defined(__has_feature)
53#if __has_feature(memory_sanitizer)
54 work = 0;
55#endif
56#endif
57
58 // second call to do the job
59 f77::getri(get_n_rows(m), m.data(), get_ld(m), ipiv.data(), work.data(), lwork, info);
60 return info;
61 }
