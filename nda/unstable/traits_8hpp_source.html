135 // ---------------------- Guarantees at compile time for some optimization --------------------------------
136
137 enum class layout_prop_e : uint64_t {
138 none = 0x0,
139 strided_1d = 0x1,
140 smallest_stride_is_one = 0x2,
141 contiguous = strided_1d | smallest_stride_is_one
142 };
143
144 // Whether we can degrade the property. It is a partial order.
145 // contiguous -> strided_1d -> none
146 // contiguous -> smallest_stride_is_one -> none
148 inline constexpr bool layout_property_compatible(layout_prop_e from, layout_prop_e to) {
149 if (from == layout_prop_e::contiguous) return true;
150 if (from == layout_prop_e::none) return (to == layout_prop_e::none);
151 return ((to == layout_prop_e::none) or (to == from));
152 }
153
154 // operator for the layout_prop_e
155 constexpr layout_prop_e operator|(layout_prop_e a, layout_prop_e b) { return layout_prop_e(uint64_t(a) | uint64_t(b)); }
156
157 constexpr layout_prop_e operator&(layout_prop_e a, layout_prop_e b) { return layout_prop_e{uint64_t(a) & uint64_t(b)}; }
158 //bool operator|=(layout_prop_e & a, layout_prop_e b) { return a = layout_prop_e(uint64_t(a) | uint64_t(b));}
159
160 inline constexpr bool has_strided_1d(layout_prop_e x) { return uint64_t(x) & uint64_t(layout_prop_e::strided_1d); }
161 inline constexpr bool has_smallest_stride_is_one(layout_prop_e x) { return uint64_t(x) & uint64_t(layout_prop_e::smallest_stride_is_one); }
162 inline constexpr bool has_contiguous(layout_prop_e x) { return has_strided_1d(x) and has_smallest_stride_is_one(x); }
163
164 // FIXME : I need a NONE for stride_order. For the scalars ...
165 struct layout_info_t {
166 uint64_t stride_order = 0;
167 layout_prop_e prop = layout_prop_e::none;
168 };
169
170 // Combining layout_info
171 constexpr layout_info_t operator&(layout_info_t a, layout_info_t b) {
172 if (a.stride_order == b.stride_order)
173 return layout_info_t{a.stride_order, layout_prop_e(uint64_t(a.prop) & uint64_t(b.prop))};
174 else
175 return layout_info_t{uint64_t(-1), layout_prop_e::none}; // -1 is undefined stride_order, it corresponds to no permutation
176 }
