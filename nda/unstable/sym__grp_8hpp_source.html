26 struct operation {

32 operation operator*(operation const &x) { return operation{bool(sgn xor x.sgn), bool(cc xor x.cc)}; }

34 template <typename T>
35 T operator()(T const &x) const {
36 if (sgn) return cc ? -conj(x) : -x;
37 return cc ? conj(x) : x;

116 template <typename H>
117 requires(NdaInitFunc<H, A>)
118 void init(A &x, H const &init_func, bool parallel = false) const {
119
120 if (parallel) {
121 // reset input array to allow for mpi reduction
122 x() = 0.0;
123
124#pragma omp parallel
125 for (auto const &sym_class : itertools::omp_chunk(mpi::chunk(sym_classes))) {
126 auto idx = x.indexmap().to_idx(sym_class[0].first);
127 auto ref_val = init_func(idx);
128 std::apply(x, idx) = ref_val;
129 for (auto const &[lin_idx, op] : sym_class) { std::apply(x, x.indexmap().to_idx(lin_idx)) = op(ref_val); }
130 }
131
132 // distribute data among all ranks
133 x = mpi::all_reduce(x);
134
135 } else {
136 for (auto const &sym_class : sym_classes) {
137 auto idx = x.indexmap().to_idx(sym_class[0].first);
138 auto ref_val = init_func(idx);
139 std::apply(x, idx) = ref_val;
140 for (auto const &[lin_idx, op] : sym_class) { std::apply(x, x.indexmap().to_idx(lin_idx)) = op(ref_val); }
141 }
142 }

153 std::pair<double, arr_idx_t> symmetrize(A &x) const {
154
155 double max_diff = 0.0;
156 auto max_idx = arr_idx_t{};
157
158 for (auto const &sym_class : sym_classes) {
159 get_value_t<A> ref_val = 0.0;
160
161 for (auto const &[lin_idx, op] : sym_class) { ref_val += op(std::apply(x, x.indexmap().to_idx(lin_idx))); }
162
163 ref_val /= sym_class.size();
164
165 for (auto const &[lin_idx, op] : sym_class) {
166 auto mapped_val = op(ref_val);
167 auto mapped_idx = x.indexmap().to_idx(lin_idx);
168 auto current_val = std::apply(x, mapped_idx);
169 auto diff = std::abs(mapped_val - current_val);
170
171 if (diff > max_diff) {
172 max_diff = diff;
173 max_idx = mapped_idx;
174 };
175
176 std::apply(x, mapped_idx) = mapped_val;
177 }
178 }
179
180 return std::pair{max_diff, max_idx};
181 }

194 sym_grp(A const &x, std::vector<F> const &sym_list, long const max_length = 0) {
195
196 // array to check whether index has been sorted into a symmetry class already
197 array<bool, ndims> checked(x.shape());
198 checked() = false;
199
200 // initialize data array (we have as many elements as in the original nda array)
201 data.reserve(x.size());
202
203 for_each(checked.shape(), [&checked, &sym_list, max_length, this](auto... i) {
204 if (not checked(i...)) {
205 operation op;
206 checked(i...) = true; // this index is now checked and defines a new symmetry class
207 auto idx = std::array{i...};
208 auto class_start = data.end(); // the class is added to the end of the data list
209 data.emplace_back(checked.indexmap()(i...), op); // every class is initialized by one representative with op = identity
210
211 // apply all symmetries to current index and generate the symmetry class
212 auto class_size = iterate(idx, op, checked, sym_list, max_length) + 1;
213
214 sym_classes.emplace_back(class_start, class_size);
215 }
216 });
217 }
218
219 private:
220 // implementation of the actual symmetry reduction algorithm
221 long long iterate(std::array<long, static_cast<std::size_t>(get_rank<A>)> const &idx, operation const &op, array<bool, ndims> &checked,
222 std::vector<F> const &sym_list, long const max_length, long excursion_length = 0) {
223
224 // initialize the local segment_length to 0 (we have not advanced to a new member of the symmetry class so far)
225 long long segment_length = 0;
226
227 for (auto const &sym : sym_list) {
228 // apply the symmetry
229 auto [idxp, opp] = sym(idx);
230 opp = opp * op;
231
232 // check if index is valid, reset excursion_length iff we start from valid & unchecked index
233 if (is_valid(checked, idxp)) {
234
235 // check if index has been used already
236 if (not std::apply(checked, idxp)) {
237 std::apply(checked, idxp) = true; // this index is now checked
238
239 // add new member to symmetry class and increment the segment_length
240 data.emplace_back(std::apply(checked.indexmap(), idxp), opp);
241 segment_length += iterate(idxp, opp, checked, sym_list, max_length) + 1;
242 }
243
244 // if index is invalid, increment excursion length and keep going (segment_length is not incremented)
245 } else if (excursion_length < max_length) {
246 segment_length += iterate(idxp, opp, checked, sym_list, max_length, ++excursion_length);
247 }
248 }
249
250 // return the final value of the local segment_length, which will be added
251 // to the segment_length higher up in the recursive call tree
252 return segment_length;
253 }
