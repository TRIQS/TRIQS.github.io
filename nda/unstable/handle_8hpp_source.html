100 // ------------------ HEAP -------------------------------------
101
102 template <typename T, typename Allocator>
103 struct handle_heap {
104 private:
105 T *_data = nullptr; // Pointer to the start of the memory block
106 size_t _size = 0; // Size of the memory block. Invariant: size > 0 iif data != 0
107
108 // In case we need to share the memory
109 mutable std::shared_ptr<void> sptr;
110
111 using blk_t = std::pair<T *, size_t>;
112
113 // code for destructor
114 static void destruct(blk_t b) noexcept {
115
116 static_assert(std::is_nothrow_destructible_v<T>, "nda::mem::handle requires the value_type to have a non-throwing constructor");
117 auto [data, size] = b;
118
119 if (data == nullptr) return;
120
121 // If needed, call the T destructors
122 if constexpr (!std::is_trivial_v<T>) {
123 for (size_t i = 0; i < size; ++i) data[i].~T();
124 }
125
126 // Deallocate the memory block
127 allocator_singleton<Allocator>::deallocate({(char *)data, size * sizeof(T)});
128 }
129
130 // a deleter for the data in the sptr
131 static void deleter(void *p) noexcept { destruct(*((blk_t *)p)); }
132
133 public:
134 std::shared_ptr<void> get_sptr() const {
135 if (not sptr) sptr.reset(new blk_t{_data, _size}, deleter);
136 return sptr;
137 }
138
139 using value_type = T;
140
141 ~handle_heap() noexcept {
142 // if the data is not in the shared_ptr, we delete it, otherwise the shared_ptr will take care of it
143 if (not sptr and not(is_null())) destruct({_data, _size});
144 }
145
146 handle_heap() = default;
147
148 handle_heap(handle_heap &&x) noexcept {
149 _data = x._data;
150 _size = x._size;
151 sptr = std::move(x.sptr);
152 x._data = nullptr;
153 x._size = 0;
154 }
155
156 handle_heap &operator=(handle_heap &&x) noexcept {
157 if (not sptr and not(is_null())) destruct({_data, _size});
158 _data = x._data;
159 _size = x._size;
160 sptr = std::move(x.sptr);
161 x._data = nullptr;
162 x._size = 0;
163 return *this;
164 }
165
166 handle_heap &operator=(handle_heap const &x) {
167 *this = handle_heap{x};
168 return *this;
169 }
170
171 // Set up a memory block of the correct size without initializing it
172 handle_heap(long size, do_not_initialize_t) {
173 if (size == 0) return; // no size -> null handle
174 auto b = allocator_singleton<Allocator>::allocate(size * sizeof(T)); //, alignof(T));
175 ASSERT(b.ptr != nullptr);
176 _data = (T *)b.ptr;
177 _size = size;
178 }
179
180 // Set up a memory block of the correct size without initializing it
181 handle_heap(long size, init_zero_t) {
182 if (size == 0) return; // no size -> null handle
183 auto b = allocator_singleton<Allocator>::allocate_zero(size * sizeof(T)); //, alignof(T));
184 ASSERT(b.ptr != nullptr);
185 _data = (T *)b.ptr;
186 _size = size;
187 }
188
189 // Construct a new block of memory of given size and init if needed.
190 handle_heap(long size) {
191 if (size == 0) return; // no size -> null handle
192
193 mem::blk_t b;
194 if constexpr (is_complex_v<T> && init_dcmplx)
195 b = allocator_singleton<Allocator>::allocate_zero(size * sizeof(T));
196 else
197 b = allocator_singleton<Allocator>::allocate(size * sizeof(T));
198
199 ASSERT(b.ptr != nullptr);
200 _data = (T *)b.ptr;
201 _size = size;
202
203 // Call placement new except for complex types
204 if constexpr (!std::is_trivial_v<T> and !is_complex_v<T>) {
205 for (size_t i = 0; i < size; ++i) new (_data + i) T();
206 }
207 }
208
209 // Construct by making a clone of the data
210 handle_heap(handle_heap const &x) : handle_heap(x.size(), do_not_initialize) {
211 if (is_null()) return; // nothing to do for null handle
212 if constexpr (std::is_trivially_copyable_v<T>) {
213 std::memcpy(_data, x.data(), x.size() * sizeof(T));
214 } else {
215 for (size_t i = 0; i < _size; ++i) new (_data + i) T(x[i]); // placement new
216 }
217 }
218
219 // Construct by making a clone of the data. same code
220 handle_heap(handle_shared<T> const &x) : handle_heap(x.size(), do_not_initialize) {
221 if (is_null()) return; // nothing to do for null handle
222 if constexpr (std::is_trivially_copyable_v<T>) {
223 std::memcpy(_data, x.data(), x.size() * sizeof(T));
224 } else {
225 for (size_t i = 0; i < _size; ++i) new (_data + i) T(x[i]); // placement new
226 }
227 }
228
229 T &operator[](long i) noexcept { return _data[i]; }
230 T const &operator[](long i) const noexcept { return _data[i]; }
231
232 bool is_null() const noexcept {
233#ifdef NDA_DEBUG
234 // Check the Invariants in Debug Mode
235 EXPECTS((_data == nullptr) == (_size == 0));
236#endif
237 return _data == nullptr;
238 }
239
240 // A const-handle does not entail T const data
241 T *data() const noexcept { return _data; }
242
243 long size() const noexcept { return _size; }
244 };

254

257
258 public:

260
261 //

263
264 T &operator[](long i) noexcept { return data()[i]; }
265 T const &operator[](long i) const noexcept { return data()[i]; }
266
267 handle_stack() = default;
268
269 ~handle_stack() noexcept {
270 if constexpr (!std::is_trivial_v<T>) {
271 for (size_t i = 0; i < Size; ++i) data()[i].~T();
272 }
273 }
274
275 handle_stack(handle_stack &&x) noexcept { operator=(x); } // no move makes a copy, we are on stack
276
277 handle_stack &operator=(handle_stack &&x) noexcept {
278 operator=(x);
279 return *this;
280 }
281
282 handle_stack(long /*size*/) : handle_stack{} {
283 // Call placement new except for complex types
284 if constexpr (!std::is_trivial_v<T> and !is_complex_v<T>) {
285 for (size_t i = 0; i < Size; ++i) new (data() + i) T();
286 }
287 }
288
289 handle_stack(long /*size*/, do_not_initialize_t) {}
290
291 // Set up a memory block of the correct size without initializing it
292 handle_stack(long /*size*/, init_zero_t) {
293 static_assert(std::is_scalar_v<T> or is_complex_v<T>, "Internal Error");
294 for (size_t i = 0; i < Size; ++i) data()[i] = 0;
295 }
296
297 handle_stack &operator=(handle_stack const &x) {
298 for (size_t i = 0; i < Size; ++i) new (data() + i) T(x[i]); // placement new
299 return *this;
300 }
301
302 // Construct by making a clone of the data
303 handle_stack(handle_stack const &x) noexcept { // if an exception occurs in T construction, so be it, we terminate
304 operator=(x);
305 }
306
307 static constexpr bool is_null() noexcept { return false; }
308 static constexpr long size() noexcept { return Size; }
309 };

313 template <typename T, size_t Size>
314 //struct alignas(alignof(T)) handle_sso {
315 struct handle_sso {
316 static_assert(Size > 0, "Size =0 makes no sense here");
317 static_assert(std::is_copy_constructible_v<T>,
318 "nda::mem::handle_sso requires the value_type to be copy constructible, or it can not even move (it is on stack)");
319 static_assert(std::is_nothrow_destructible_v<T>, "nda::mem::handle requires the value_type to have a non-throwing constructor");
320
321 private:
322 std::array<char, sizeof(T) * Size> buffer; //
323
324 T *_data = nullptr; // Pointer to the start of the memory block
325 size_t _size = 0; // Size of the memory block. Invariant: size > 0 iif data != 0
326
327 public:
328 using value_type = T;
329
330 bool on_heap() const { return _size > Size; }
331
332 bool is_null() const noexcept {
333#ifdef NDA_DEBUG
334 EXPECTS((_data == nullptr) == (_size == 0));
335#endif
336 return _data == nullptr;
337 }
338
339 T *data() const noexcept { return _data; }
340
341 T &operator[](long i) noexcept { return _data[i]; }
342 T const &operator[](long i) const noexcept { return _data[i]; }
343
344 long size() const noexcept { return _size; }
345
346 // Caution! We need to provide a user-defined constructor (over =default)
347 // to avoid value initialization of the buffer
348 handle_sso(){};
349
350 private:
351 void clean() noexcept {
352 if (is_null()) return;
353 if constexpr (!std::is_trivial_v<T>) {
354 for (size_t i = 0; i < _size; ++i) data()[i].~T();
355 // STACK for (size_t i = 0; i < Size; ++i) data()[i].~T();
356 }
357 if (on_heap()) mem::mallocator::deallocate({(char *)_data, _size * sizeof(T)});
358 _data = nullptr;
359 }
360
361 public:
362 ~handle_sso() noexcept { clean(); }
363
364 handle_sso(handle_sso &&x) noexcept {
365 _size = x._size;
366 if (on_heap()) { // heap path
367 _data = x._data;
368 } else { // stack path. We MUST copy
369 if (_size != 0) { // if crucial to maintain invariant
370 _data = (T *)buffer.data();
371 for (size_t i = 0; i < _size; ++i) new (data() + i) T(x[i]);
372 }
373 //for (size_t i = 0; i < Size; ++i) new (data() + i) T(x[i]);
374 }
375 x._data = nullptr; // steal data
376 x._size = 0; // maintain invariant
377 }
378
379 handle_sso &operator=(handle_sso &&x) noexcept {
380 clean();
381 _size = x._size;
382 if (on_heap()) {
383 _data = x._data;
384 } else {
385 if (_size != 0) { // if crucial to maintain invariant
386 _data = (T *)buffer.data();
387 for (size_t i = 0; i < _size; ++i) new (_data + i) T(x[i]); // placement new
388 }
389 }
390 x._data = nullptr; // steal data
391 x._size = 0; // maintain invariant
392 return *this;
393 }
394
395 // Set up a memory block of the correct size without initializing it
396 handle_sso(long size, do_not_initialize_t) {
397 if (size == 0) return; // no size -> null handle
398 _size = size;
399 if (not on_heap()) {
400 _data = (T *)buffer.data();
401 } else {
402 mem::blk_t b;
403 b = mem::mallocator::allocate(size * sizeof(T));
404 ASSERT(b.ptr != nullptr);
405 _data = (T *)b.ptr;
406 }
407 }
408
409 // Same but call calloc with malloc ...
410 handle_sso(long size, init_zero_t) {
411 static_assert(std::is_scalar_v<T> or is_complex_v<T>, "Internal Error");
412 if (size == 0) return; // no size -> null handle
413 _size = size;
414 if (not on_heap()) {
415 _data = (T *)buffer.data();
416 for (size_t i = 0; i < _size; ++i) data()[i] = 0;
417 } else {
418 auto b = mem::mallocator::allocate_zero(size * sizeof(T)); //, alignof(T));
419 ASSERT(b.ptr != nullptr);
420 _data = (T *)b.ptr;
421 }
422 }
423
424 // Copy data
425 handle_sso(handle_sso const &x) : handle_sso(x.size(), do_not_initialize) {
426 for (size_t i = 0; i < _size; ++i) new (_data + i) T(x[i]); // placement new
427 }
428
429 handle_sso &operator=(handle_sso const &x) noexcept {
430 clean();
431 _size = x._size;
432 if (_size == 0) return *this;
433 if (on_heap()) {
434 mem::blk_t b;
435 b = mem::mallocator::allocate(_size * sizeof(T));
436 ASSERT(b.ptr != nullptr);
437 _data = (T *)b.ptr;
438 } else {
439 _data = (T *)buffer.data();
440 }
441 for (size_t i = 0; i < _size; ++i) new (_data + i) T(x[i]);
442
443 return *this;
444 }
445
446 // Construct a new block of memory of given size and init if needed.
447 handle_sso(long size) {
448 if (size == 0) return; // no size -> null handle
449 _size = size; // NOt needed in the Stack path if type is trivial ??
450
451 if (not on_heap()) {
452 _data = (T *)buffer.data();
453 _size = size;
454 } else {
455 mem::blk_t b;
456 if constexpr (is_complex_v<T> && init_dcmplx)
457 b = mem::mallocator::allocate_zero(size * sizeof(T));
458 else
459 b = mem::mallocator::allocate(size * sizeof(T));
460 ASSERT(b.ptr != nullptr);
461 _data = (T *)b.ptr;
462 _size = size;
463 }
464
465 // Call placement new except for complex types
466 if constexpr (!std::is_trivial_v<T> and !is_complex_v<T>) {
467 for (size_t i = 0; i < size; ++i) new (_data + i) T();
468 }
469 }
