22 * Iterator on a rectangular grid (in C traversal order)

25 // -------------------------------
26 // Rank >1 : general case
27 template <int Rank>
28 class grid_iterator {
29 long stri = 0;
30 long pos = 0;
31 long offset = 0;
32 grid_iterator<Rank - 1> it_begin, it_end, it;
33
34 public:
35 using iterator_category = std::forward_iterator_tag;
36 using value_type = long;
37 using difference_type = std::ptrdiff_t;
38 using pointer = long *;
39 using reference = long &;
40
41 grid_iterator() = default;
42
43 grid_iterator(long const *lengths, long const *strides, bool at_end)
44 : stri(strides[0]),
45 pos(at_end ? lengths[0] : 0),
46 offset(pos * stri),
47 it_begin(lengths + 1, strides + 1, false),
48 it_end(lengths + 1, strides + 1, true),
49 it(it_begin) {} // NB always it_begin, even if at_end. The end iterator correspond to pos = (length -1) + 1, but it is at its *beginning*
50
51 [[nodiscard]] std::array<long, Rank> indices() { return stdutil::front_append(it.indices(), pos); }
52
53 [[nodiscard]] long operator*() const { return offset + *it; }
54 long operator->() const { return operator*(); }
55
56 bool operator==(grid_iterator const &other) const { return ((other.pos == pos) and (other.it == it)); }
57 bool operator!=(grid_iterator const &other) const { return not operator==(other); }
58 grid_iterator &operator++() {
59 ++it;
60 if (it == it_end) { //FIXME [[unlikely]]
61 ++pos;
62 offset += stri;
63 it = it_begin;
64 }
65 return *this;
66 }
67
68 grid_iterator operator++(int) {
69 auto c = *this;
70 ++(*this);
71 return c;
72 }
73 };

89
90 grid_iterator() = default;
91 grid_iterator(long const *lengths, long const *strides, bool at_end) : stri(strides[0]), pos(at_end ? lengths[0] : 0), offset(pos * stri) {}
92
93 [[nodiscard]] std::array<long, 1> indices() { return {pos}; }
94
95 [[nodiscard]] long operator*() const { return offset; }
96 long operator->() const { return operator*(); }
97
98 bool operator==(grid_iterator const &other) const { return (other.pos == pos); }
99 bool operator!=(grid_iterator const &other) const { return (other.pos != pos); }
100
101 grid_iterator &operator++() {
102 offset += stri;
103 ++pos;
104 return *this;
105 }
106
107 grid_iterator &operator--() {
108 offset -= stri;
109 --pos;
110 return *this;
111 }
112
113 grid_iterator &operator+=(std::ptrdiff_t n) {
114 offset += n * stri;
115 pos += n;
116 return *this;
117 }
118
119 // We do not implement the full LegacyRandomAccessIterator here, just what is needed for the array_iterator below.
120 friend grid_iterator operator+(grid_iterator it, std::ptrdiff_t n) { return it += n; }
121 friend std::ptrdiff_t operator-(grid_iterator const &it1, grid_iterator const &it2) { return it1.pos - it2.pos; }
122 friend bool operator<(grid_iterator const &it1, grid_iterator const &it2) { return it1.pos < it2.pos; }
123 friend bool operator>(grid_iterator const &it1, grid_iterator const &it2) { return it1.pos > it2.pos; }
124 };

128 // The iterator for the array and array_view container.
129 // Makes an iterator of rank Rank on a pointer of type T.
130 // e.g. for a strided_1d, we use Rank == 1, whatever the real array is
131 // T can be const
132 template <int Rank, typename T, typename Pointer>
133 class array_iterator {
134 T *data = nullptr;
135 std::array<long, Rank> len, stri;
136 grid_iterator<Rank> iter;
137
138 public:
139 using iterator_category = std::forward_iterator_tag;
140 using value_type = T;
141 using difference_type = std::ptrdiff_t;
142 using pointer = T *;
143 using reference = T &;
144
145 array_iterator() = default;
146 array_iterator(array_iterator const &) = default;
147
148 array_iterator(std::array<long, Rank> const &lengths, std::array<long, Rank> const &strides, T *start, bool at_end)
149 : data(start), len(lengths), stri(strides), iter(len.data(), stri.data(), at_end) {}
150
151 [[nodiscard]] std::array<long, Rank> indices() { return iter.indices(); }
152
153 [[nodiscard]] value_type &operator*() const { return ((Pointer)data)[*iter]; }
154 value_type &operator->() const { return operator*(); }
155
156 array_iterator &operator++() {
157 ++iter;
158 return *this;
159 }
160
161 array_iterator operator++(int) {
162 auto c = *this;
163 ++iter;
164 return c;
165 }
166
167 bool operator==(array_iterator const &other) const { return (other.iter == iter); }
168 bool operator!=(array_iterator const &other) const { return (!operator==(other)); }
169 };
