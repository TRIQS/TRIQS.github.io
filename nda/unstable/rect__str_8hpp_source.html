108 template <typename T>
109 auto peel_string(int pos, T const &x) const {
110 if constexpr (not std::is_constructible_v<std::string, T>)
111 return x;
112 else {
113 auto const &sind = get_string_indices();
114 auto const &idx = sind[pos];
115 auto it = std::find(idx.begin(), idx.end(), x);
116 if (it == idx.end())
117 NDA_RUNTIME_ERROR << "Calling array with string. Key " << x << " at position " << pos << " does not match the indices " << sind;
118 return it - idx.begin(); // 1d array, have LegacyRandomAccessIterator const iterator
119 }

133 // ---------------- Slice -------------------------

135 private:
136 template <typename... T, auto... Is>
137 FORCEINLINE decltype(auto) slice_impl(std::index_sequence<Is...>, T const &... x) const {
138 auto const [offset, idxm2] = base_t::slice(peel_string(Is, x)...);
139 using new_rect_str_t = typename details::rect_str_from_base<std::decay_t<decltype(idxm2)>>::type;
140
141 if (not s_indices) return std::make_pair(offset, new_rect_str_t{idxm2});
142
143 // slice the indices. Not optimized, but simple
144 auto const &current_ind = get_string_indices();
145 ind_t ind2((not argument_is_allowed_for_call<T> + ...)); // number of arg which are range, ellipsis, i.e. not arg of simple call
146
147 auto l = [p = 0, &current_ind, &ind2](int n, auto const &y) mutable -> void {
148 using U = std::decay_t<decltype(y)>;
149 if constexpr (not argument_is_allowed_for_call<U>) { ind2[p++] = current_ind[n](y); }
150 };
151 (l(Is, x), ...);
152
153 return std::make_pair(offset, new_rect_str_t{idxm2, ind2});
154 }

169 template <uint64_t Permutation>
170 auto transpose() const {
171
172 auto idxm2 = base_t::template transpose<Permutation>();
173
174 using new_rect_str_t = typename details::rect_str_from_base<std::decay_t<decltype(idxm2)>>::type;
175 if (not s_indices) return new_rect_str_t{idxm2};
176
177 ind_t ind2(s_indices->size());
178 static constexpr std::array<int, Rank> permu = decode<Rank>(Permutation);
179 for (int u = 0; u < Rank; ++u) { ind2[permu[u]] = (*s_indices)[u]; }
180 return new_rect_str_t{idxm2, ind2};
181 }
