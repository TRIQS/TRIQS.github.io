24 class gelss_worker {
25 // cf. Notation in https://math.stackexchange.com/questions/772039/how-does-the-svd-solve-the-least-squares-problem

28 long M, N;

30 // The matrix to be decomposed by SVD

54 gelss_worker(matrix_const_view<T> _A) : M(_A.extent(0)), N(_A.extent(1)), A(_A), s_vec(std::min(M, N)) {
55
56 if (N > M) NDA_RUNTIME_ERROR << "ERROR: Matrix A for linear least square procedure cannot have more columns than rows";
57
58 matrix<T, F_layout> A_FL{_A};
59 matrix<T, F_layout> U(M, M);
60 matrix<T, F_layout> VT(N, N);
61
62 // Calculate the SVD A = U * Diag(S_vec) * VT
63 gesvd(A_FL, s_vec, U, VT);
64
65 // Calculate the matrix V * Diag(S_vec)^{-1} * UT for the least square procedure
66 matrix<double, F_layout> S_inv(N, M);
67 S_inv = 0.;
68 for (int i : range(std::min(M, N))) S_inv(i, i) = 1.0 / s_vec(i);
69 V_x_InvS_x_UT = dagger(VT) * S_inv * dagger(U);
70
71 // Read off U_Null for defining the error of the least square procedure
72 if (N < M) UT_NULL = dagger(U)(range(N, M), range(M));
73 }
74
76 std::pair<matrix<T>, double> operator()(matrix_const_view<T> B, std::optional<long> /*inner_matrix_dim*/ = {}) const {
77 using std::sqrt;
78 double err = 0.0;
79 if (M != N) {
80 std::vector<double> err_vec;
81 for (int i : range(B.shape()[1])) err_vec.push_back(frobenius_norm(UT_NULL * B(range(), range(i, i + 1))) / sqrt(B.shape()[0]));
82 err = *std::max_element(err_vec.begin(), err_vec.end());
83 }
84 return std::make_pair(V_x_InvS_x_UT * B, err);
85 }
86 };

112 // Solve the least-square problem that minimizes || A * x - B ||_2 given A and B with a real-valued vector x
113 std::pair<matrix<dcomplex>, double> operator()(matrix_const_view<dcomplex> B, std::optional<long> inner_matrix_dim = {}) const {
114
115 if (not inner_matrix_dim.has_value()) NDA_RUNTIME_ERROR << "Inner matrix dimension required for hermitian least square fitting\n";
116 long d = *inner_matrix_dim;
117
118 // Construction of an inner 'adjoint' matrix by performing the following steps
119 // * reshape B from (M, M1) to (M, N, d, d)
120 // * for each M and N take the adjoint matrix (d, d)
121 // * reshape to (M, M)
122 auto inner_adjoint = [d](auto &M) {
123 auto idx_map = M.indexmap();
124 auto l = idx_map.lengths();
125 //auto s = idx_map.strides();
126
127 NDA_ASSERT2(l[1] % (d * d) == 0, "ERROR in hermitian least square fitting: Data shape incompatible with given dimension");
128 long N = l[1] / (d * d);
129
130 // We reshape the Matrix into a dim=4 array and swap the two innermost indices
131
132 // FIXME OLD CODE SUPRRESS AFTER PORTING

137 //return matrix<dcomplex>{matrix<dcomplex>::layout_t{l, s}, std::move(arr_dag).storage()};
138
139 // FIXME C++20 remove encode
140 array<dcomplex, 4> arr_dag = conj(permuted_indices_view<encode(std::array{0, 1, 3, 2})>(reshaped_view(M, std::array{l[0], N, d, d})));
141
142 return matrix<dcomplex>{reshape(std::move(arr_dag), l)}; // move into a matrix
143 };

145 // Solve the enlarged system vstack(A, A*) * x = vstack(B, B_dag)
146 matrix<dcomplex> B_dag = inner_adjoint(B);
147 auto [x, err] = _lss_matrix(vstack(B, B_dag));
