31 template <ArrayOfRank<2> M>
32 auto trace(M const &m) {
33 static_assert(get_rank<M> == 2, "trace: array must have rank two");
34 EXPECTS(m.shape()[0] == m.shape()[1]);
35 auto r = get_value_t<M>{};
36 auto d = m.shape()[0];
37 for (int i = 0; i < d; ++i) r += m(i, i);
38 return r;
39 }

51 template <MemoryArrayOfRank<2> M>
52 ArrayOfRank<1> auto diagonal(M &m) {
53 long dim = std::min(m.shape()[0], m.shape()[1]);
54 long stride = stdutil::sum(m.indexmap().strides());
55 using vector_view_t = basic_array_view<std::remove_reference_t<decltype(*m.data())>, 1, C_stride_layout, 'V', nda::default_accessor, nda::borrowed>;
56 return vector_view_t{C_stride_layout::mapping<1>{{dim}, {stride}}, m.data()};
57 }

70 template <ArrayOfRank<2> A, ArrayOfRank<2> B>
71 requires(std::same_as<get_value_t<A>, get_value_t<B>>) // NB the get_value_t gets rid of const if any
72 matrix<get_value_t<A>> vstack(A const &a, B const &b)
73 {
74 static_assert(get_rank<A> == 2, "vstack: first argument must have rank two");
75 static_assert(get_rank<B> == 2, "vstack: second argument must have rank two");
76 EXPECTS_WITH_MESSAGE(a.shape()[1] == b.shape()[1],
77 "vstack. The second dimension of the two matrices must be equal but \n a is of shape " + to_string(a.shape())
78 + " b is of shape" + to_string(b.shape()));
79 // Impl. Concept only ! A, B can be expression template, e.g.
80
81 auto [n, q] = a.shape();
82 auto p = b.shape()[0];
83 auto _ = range::all;
84
85 matrix<get_value_t<A>> res(n + p, q);
86 res(range(0, n), _) = a;
87 res(range(n, n + p), _) = b;
88 return res;
89 }
