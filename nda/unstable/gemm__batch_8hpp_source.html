39 template <bool VBATCH = false, Matrix X, Matrix Y, MemoryMatrix C>
40 requires((MemoryMatrix<X> or is_conj_array_expr<X>) and //
41 (MemoryMatrix<Y> or is_conj_array_expr<Y>) and //
42 have_same_value_type_v<X, Y, C> and is_blas_lapack_v<get_value_t<X>>) //
43 void gemm_batch(get_value_t<X> alpha, std::vector<X> const &vx, std::vector<Y> const &vy, get_value_t<X> beta, std::vector<C> &vc) {
44
45 EXPECTS(vx.size() == vy.size() and vx.size() == vc.size());
46 if (vx.empty()) return;
47 int batch_count = vx.size();
48
49 // FIXMEOP : move in tools : reepeat in every function
50 auto to_mat = []<typename Z>(Z &z) -> auto & {
51 if constexpr (is_conj_array_expr<Z>)
52 return std::get<0>(z.a);
53 else
54 return z;
55 };
56 auto &a0 = to_mat(vx[0]);
57 auto &b0 = to_mat(vy[0]);
58 auto &c0 = vc[0];
59
60 static constexpr bool conj_A = is_conj_array_expr<X>;
61 static constexpr bool conj_B = is_conj_array_expr<Y>;
62
63 using A = decltype(a0);
64 using B = decltype(b0);
65 static_assert(mem::have_compatible_addr_space<A, B, C>, "Matrices must have same memory address space");
66
67 // c is in C order: compute the transpose of the product in Fortran order
68 if constexpr (has_C_layout<C>) {
69
70 auto map_transpose = [](auto &v) {
71 auto vT = std::vector<std::decay_t<decltype(transpose(v[0]))>>{};
72 vT.reserve(v.size());
73 std::transform(v.begin(), v.end(), std::back_inserter(vT), [](auto &x) { return transpose(x); });
74 return vT;
75 };
76 auto vcT = map_transpose(vc);
77 gemm_batch<VBATCH>(alpha, map_transpose(vy), map_transpose(vx), beta, vcT);
78 return;
79 } else { // c is in Fortran order
80
81 // For operations on the device, use unified memory for vector of ints or ptrs
82 auto constexpr vec_adr_spc = []() { return mem::on_host<C> ? mem::Host : mem::Unified; }();
83
84 // Convert the vector of matrices into the associated vector of pointers
85 auto get_ptrs = [&to_mat]<typename V>(V &v) {
86 EXPECTS(std::all_of(v.begin(), v.end(),
87 [&v, &to_mat](auto &z) { return (VBATCH or z.shape() == v[0].shape()) and to_mat(z).indexmap().min_stride() == 1; }));
88 using value_t = get_value_t<typename V::value_type>;
89 using ptr_t = std::conditional_t<std::is_const_v<V>, value_t const *, value_t *>;
90 auto v_ptrs = vector<ptr_t, heap<vec_adr_spc>>(v.size());
91 std::transform(v.begin(), v.end(), v_ptrs.begin(), [&to_mat](auto &z) { return to_mat(z).data(); });
92 return v_ptrs;
93 };
94 auto a_ptrs = get_ptrs(vx);
95 auto b_ptrs = get_ptrs(vy);
96 auto c_ptrs = get_ptrs(vc);
97
98 char op_a = get_op<conj_A, /*transpose =*/has_C_layout<A>>;
99 char op_b = get_op<conj_B, /*transpose =*/has_C_layout<B>>;
100
101 if constexpr (VBATCH) {
102
103 // Create vectors of size 'batch_count + 1' as required by Magma
104 nda::vector<int, heap<vec_adr_spc>> vm(batch_count + 1), vk(batch_count + 1), vn(batch_count + 1), vlda(batch_count + 1),
105 vldb(batch_count + 1), vldc(batch_count + 1);
106
107 for (auto i : range(batch_count)) {
108 auto &ai = to_mat(vx[i]);
109 auto &bi = to_mat(vy[i]);
110 auto &ci = vc[i];
111
112 EXPECTS(ai.extent(1) == bi.extent(0));
113 EXPECTS(ai.extent(0) == ci.extent(0));
114 EXPECTS(bi.extent(1) == ci.extent(1));
115
116 vm[i] = ai.extent(0);
117 vk[i] = ai.extent(1);
118 vn[i] = bi.extent(1);
119
120 vlda[i] = get_ld(ai);
121 vldb[i] = get_ld(bi);
122 vldc[i] = get_ld(ci);
123 }
124
125 if constexpr (mem::have_device_compatible_addr_space<A, B, C>) {
126#if defined(NDA_HAVE_DEVICE)
127 device::gemm_vbatch(op_a, op_b, vm.data(), vn.data(), vk.data(), alpha, a_ptrs.data(), vlda.data(), b_ptrs.data(), vldb.data(), beta,
128 c_ptrs.data(), vldc.data(), batch_count);
129#else
130 compile_error_no_gpu();
131#endif
132 } else {
133 f77::gemm_vbatch(op_a, op_b, vm.data(), vn.data(), vk.data(), alpha, a_ptrs.data(), vlda.data(), b_ptrs.data(), vldb.data(), beta,
134 c_ptrs.data(), vldc.data(), batch_count);
135 }
136 } else {
137
138 EXPECTS(a0.extent(1) == b0.extent(0));
139 EXPECTS(a0.extent(0) == c0.extent(0));
140 EXPECTS(b0.extent(1) == c0.extent(1));
141
142 auto [m, k] = a0.shape();
143 auto n = b0.extent(1);
144
145 if constexpr (mem::have_device_compatible_addr_space<A, B, C>) {
146#if defined(NDA_HAVE_DEVICE)
147 device::gemm_batch(op_a, op_b, m, n, k, alpha, a_ptrs.data(), get_ld(a0), b_ptrs.data(), get_ld(b0), beta, c_ptrs.data(), get_ld(c0),
148 batch_count);
149#else
150 compile_error_no_gpu();
151#endif
152 } else {
153 f77::gemm_batch(op_a, op_b, m, n, k, alpha, a_ptrs.data(), get_ld(a0), b_ptrs.data(), get_ld(b0), beta, c_ptrs.data(), get_ld(c0),
154 batch_count);
155 }
156 }
157 }
158 }

180
181 EXPECTS(x.shape()[0] == y.shape()[0] and y.shape()[0] == c.shape()[0]);
182 auto to_arr = []<typename Z>(Z &z) -> auto & {
183 if constexpr (is_conj_array_expr<Z>)
184 return std::get<0>(z.a);
185 else
186 return z;
187 };
188 auto a = to_arr(x);
189 auto b = to_arr(y);
190
191 static constexpr bool conj_A = is_conj_array_expr<X>;
192 static constexpr bool conj_B = is_conj_array_expr<Y>;
193
194 using A = decltype(a);
195 using B = decltype(b);
196 static_assert(mem::have_compatible_addr_space<A, B, C>, "Arrays must have same memory address space");
197
198 auto _ = nda::range::all;
199 auto a0 = a(0, _, _);
200 auto b0 = b(0, _, _);
201 auto c0 = c(0, _, _);
202 EXPECTS(a0.extent(1) == b0.extent(0));
203 EXPECTS(a0.extent(0) == c0.extent(0));
204 EXPECTS(b0.extent(1) == c0.extent(1));
205

208 EXPECTS(b.indexmap().min_stride() == 1);

211 // c is in C order: compute the transpose of the product in Fortran order
212 if constexpr (has_C_layout<C>) {
213 gemm_batch_strided(alpha, transposed_view<1, 2>(y), transposed_view<1, 2>(x), beta, transposed_view<1, 2>(std::forward<C>(c)));
214 return;
215 } else { // c is in Fortran order
216 char op_a = get_op<conj_A, /*transpose =*/has_C_layout<A>>;
217 char op_b = get_op<conj_B, /*transpose =*/has_C_layout<B>>;
218 auto [m, k] = a0.shape();
219 auto n = b0.extent(1);
220
221 if constexpr (mem::have_device_compatible_addr_space<A, B, C>) {
222#if defined(NDA_HAVE_DEVICE)
223 device::gemm_batch_strided(op_a, op_b, m, n, k, alpha, a.data(), get_ld(a0), a.indexmap().strides()[0], b.data(), get_ld(b0), b.strides()[0],
224 beta, c.data(), get_ld(c0), c.indexmap().strides()[0], a.extent(0));

231 }
