68 template <typename RealType = double, std::integral Int, auto Rank>
69 auto rand(std::array<Int, Rank> const &shape) requires(std::is_floating_point_v<RealType>) {
70 // For Rank == 0 we should return a scalar
71 if constexpr (Rank == 0) {
72 auto static gen = std::mt19937(std::random_device{}());
73 auto static dist = std::uniform_real_distribution<>(0.0, 1.0);
74 return dist(gen);
75 } else {
76 return array<RealType, Rank>::rand(shape);
77 }
78 }

200 // --------------- operator == ---------------------
201
203 template <Array A, Array B>
204 bool operator==(A const &a, B const &b) {
205 // FIXME not implemented in clang .. readd when done for better error message
206#ifndef __clang__
207 static_assert(std::equality_comparable_with<get_value_t<A>, get_value_t<B>>, "A == B is only defined when their element can be compared");
208#endif
209 if (a.shape() != b.shape()) return false;
210 bool r = true;
211 nda::for_each(a.shape(), [&](auto &&...x) { r &= (a(x...) == b(x...)); });
212 return r;
213 }

229 template <Array A, typename F>
230 void clef_auto_assign(A &&a, F &&f) {
231 nda::for_each(a.shape(), [&a, &f](auto &&...x) {
232 if constexpr (clef::is_function<std::decay_t<decltype(f(x...))>>) {
233 clef_auto_assign(a(x...), f(x...));
234 } else {
235 a(x...) = f(x...);
236 }
237 });
238 }
