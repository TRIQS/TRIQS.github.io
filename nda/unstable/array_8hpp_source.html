36 template <typename T, size_t R>
37 std::string to_string(std::array<T, R> const &a) {
38 std::stringstream fs;
39 fs << "(";
40 for (int i = 0; i < R; ++i) fs << (i == 0 ? "" : " ") << a[i];
41 fs << ")";
42 return fs.str();
43 }

95 template <typename T, typename U, size_t R>
96 constexpr std::array<T, R> make_std_array(std::array<U, R> const &a) {
97 static_assert(std::is_constructible_v<T, U>, "make_std_array : T must be constructible from U, Cf doc");
98 std::array<T, R> result = make_initialized_array<R>(T{});
99 for (int u = 0; u < R; ++u) result[u] = a[u];

207 template <typename T, size_t R1, size_t R2>
208 constexpr std::array<T, R1 + R2> join(std::array<T, R1> const &a1, std::array<T, R2> const &a2) {
209 std::array<T, R1 + R2> res = make_initialized_array<R1 + R2>(T{});
210 for (int i = 0; i < R1; ++i) res[i] = a1[i];
211 for (int i = 0; i < R2; ++i) res[R1 + i] = a2[i];
212 return res;
213 }

221 // ------------- sum --------------------------------------
222 template <typename T, size_t R>
223 constexpr auto sum(std::array<T, R> const &a) {
224 if constexpr (R == 0)
225 return T{};
226 else {
227 auto res = a[0];
228 for (int i = 1; i < R; ++i) res += a[i];
229 return res;

239 // ------------- product --------------------------------------
240 template <typename T, size_t R>
241 constexpr auto product(std::array<T, R> const &a) {
242 static_assert(R > 0, "Product of elements in std::array<T, R> only defined for R > 0");
243 auto res = a[0];
244 for (int i = 1; i < R; ++i) res *= a[i];
245 return res;

256 // ------------- dot --------------------------------------

258 constexpr auto dot_product(std::array<T, R> const &a1, std::array<U, R> const &a2) {
259 if constexpr (R == 0)
260 return T{};
261 else {
262 auto res = a1[0] * a2[0];
263 for (int i = 1; i < R; ++i) res += a1[i] * a2[i];
264 return res;
