24 template <bool star, typename X, typename Y>
25 auto _dot_impl(X const &x, Y const &y) {
26 EXPECTS(x.shape() == y.shape());
27 long N = x.shape()[0];
28
29 auto _conj = [](auto z) __attribute__((always_inline)) {
30 if constexpr (star and is_complex_v<decltype(z)>) {
31 return std::conj(z);
32 } else
33 return z;
34 };
35
36 if constexpr (has_layout_smallest_stride_is_one<X> and has_layout_smallest_stride_is_one<Y>) {
37 if constexpr (is_regular_or_view_v<X> and is_regular_or_view_v<Y>) {
38 auto *__restrict px = x.data();
39 auto *__restrict py = y.data();
40 auto res = _conj(px[0]) * py[0];
41 for (size_t i = 1; i < N; ++i) { res += _conj(px[i]) * py[i]; }
42 return res;
43 } else {
44 auto res = _conj(x(_linear_index_t{0})) * y(_linear_index_t{0});
45 for (long i = 1; i < N; ++i) { res += _conj(x(_linear_index_t{i})) * y(_linear_index_t{i}); }
46 return res;
47 }
48 } else {
49 auto res = _conj(x(0)) * y(0);
50 for (long i = 1; i < N; ++i) { res += _conj(x(i)) * y(i); }
51 return res;
52 }
53 }
