109// I keep this public for the call of gf, which has to be reinterpreted as matrix
110// I can construct a matrix at once. Of course, the optimizer may eliminate the copy of handle and idx_map
111// but I prefer to keep it shorter here.

115template <char ResultAlgebra, bool SelfIsRvalue, typename Self, typename... T>
116FORCEINLINE static decltype(auto) call(Self &&self, T const &... x) noexcept(has_no_boundcheck) {
117
118 using r_v_t = std::conditional_t<std::is_const_v<std::remove_reference_t<Self>>, ValueType const, ValueType>;
119
120 if constexpr (clef::is_any_lazy<T...>) return clef::make_expr_call(std::forward<Self>(self), x...);
121
122 // () returns a full view
123 else if constexpr (sizeof...(T) == 0) {
124 return basic_array_view<r_v_t, Rank, Layout, Algebra, AccessorPolicy, OwningPolicy>{self.lay, self.sto};
125 }
126
127 else {
128 static_assert(((layout_t::template argument_is_allowed_for_call_or_slice<T> + ...) > 0),
129 "Slice arguments must be convertible to range, Ellipsis, or long (or string if the layout permits it)");
130
131 static constexpr int n_args_long = (layout_t::template argument_is_allowed_for_call<T> + ...);
132
133 // case 1 : all arguments are long, we simply compute the offset
134 if constexpr (n_args_long == rank) { // no range, simply compute the linear position. There may be an ellipsis, but it is of zero length !
135 long offset = self.lay(x...); // compute the offset
136 if constexpr (is_view or not SelfIsRvalue) //
137 return AccessorPolicy::template accessor<ValueType>::access(self.sto.data(),
138 offset); // We return a REFERENCE here. Ok since underlying array is still alive
139 else //
140 return ValueType{self.sto[offset]}; // We return a VALUE here, the array is about be destroyed.
141 }
142 // case 2 : we have to make a slice
143 else {
144 // Static rank
145 auto const [offset, idxm] = self.lay.slice(x...);
146
147 using r_layout_t = std::decay_t<decltype(idxm)>;
148
149 static constexpr char newAlgebra = (ResultAlgebra == 'M' and (r_layout_t::rank() == 1) ? 'V' : ResultAlgebra);
150
151 using r_view_t =
152 // FIXME basic_array_view<r_v_t, r_layout_t::rank(),
153 basic_array_view<ValueType, r_layout_t::rank(), typename details::layout_to_policy<r_layout_t>::type, newAlgebra, AccessorPolicy,
154 OwningPolicy>;
155
156 return r_view_t{std::move(idxm), {self.sto, offset}};
157 }
158 }

236template <typename Iterator>
237[[nodiscard]] auto make_iterator(bool at_end) const noexcept {
238 if constexpr (iterator_rank == Rank) {
239 // FIXME : remove optimziation becasue of a TRIQS bug
240 return Iterator{indexmap().lengths(), indexmap().strides(), sto.data(), at_end};
241 if constexpr (layout_t::is_stride_order_C())
242 return Iterator{indexmap().lengths(), indexmap().strides(), sto.data(), at_end};
243 else
244 // general case. In C order, no need to spend time applying the identity permutation
245 // the new length used by the iterator is length[ stride_order[0]], length[ stride_order[1]], ...
246 // since stride_order[0] is the slowest, it will traverse the memory in sequential order
247 return Iterator{nda::permutations::apply(layout_t::stride_order, indexmap().lengths()),
248 nda::permutations::apply(layout_t::stride_order, indexmap().strides()), sto.data(), at_end};
249 } else // 1d iteration
250 return Iterator{std::array<long, 1>{size()}, std::array<long, 1>{indexmap().min_stride()}, sto.data(), at_end};
251}

318void assign_from_ndarray(RHS const &rhs) { // FIXME noexcept {
319
320#ifdef NDA_ENFORCE_BOUNDCHECK
321 if (this->shape() != rhs.shape())
322 NDA_RUNTIME_ERROR << "Size mismatch:"
323 << "\n LHS.shape() = " << this->shape() << "\n RHS.shape() = " << rhs.shape();
324#endif

326 // FIXME : what if RHS has no memory ??
327 // firt if SPECIFIC to container ...
328 // check call -> _linear_inde ?
329 // general case if RHS is not a scalar (can be isp, expression...)
330 static_assert(std::is_assignable_v<value_type &, get_value_t<RHS>>, "Assignment impossible for the type of RHS into the type of LHS");
331
332 // If LHS and RHS are both 1d strided order or contiguous, and have the same stride order
333 // we can make a 1d loop
334 if constexpr ((get_layout_info<self_t>.stride_order == get_layout_info<RHS>.stride_order) // same stride order and both contiguous ...
335 and has_layout_strided_1d<self_t> and has_layout_strided_1d<RHS>) {
336
337 static_assert(!std::is_reference_v<RHS>, "W?");
338 //static_assert(is_regular_or_view_v<RHS>, "oops");
339 // In general, has_layout_strided_1d is FALSE by default
340 // VALID ALSO FOR EXPRESSION !!!
341 long L = size();
342 for (long i = 0; i < L; ++i) (*this)(_linear_index_t{i}) = rhs(_linear_index_t{i});
343 } else {
344 auto l = [this, &rhs](auto const &... args) { (*this)(args...) = rhs(args...); };
345 nda::for_each(shape(), l);
346 }
347}

351template <typename Scalar>
352void fill_with_scalar(Scalar const &scalar) noexcept {
353 // we make a special implementation if the array is 1d strided or contiguous
354 if constexpr (has_layout_strided_1d<self_t>) { // possibly contiguous
355 const long L = size();
356 auto *__restrict const p = data(); // no alias possible here !
357 if constexpr (has_contiguous_layout<self_t>) {
358 for (long i = 0; i < L; ++i) p[i] = scalar;
359 } else {
360 const long stri = indexmap().min_stride();
361 const long Lstri = L * stri;
362 for (long i = 0; i < Lstri; i += stri) p[i] = scalar;
363 }
364 } else {
365 for (auto &x : *this) x = scalar;
366 }
367}
