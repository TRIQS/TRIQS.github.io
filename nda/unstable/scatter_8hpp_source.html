19template <nda::Array A>
20struct mpi::lazy<mpi::tag::scatter, A> {
21
22 using value_type = typename std::decay_t<A>::value_type;
23 using const_view_type = decltype(std::declval<const A>()());
24
25 const_view_type rhs; // the rhs array
26 mpi::communicator c; // mpi comm
27 const int root; //
28 const bool all;
29
31 [[nodiscard]] auto shape() const {
32 auto dims = rhs.shape();
33 long slow_size = dims[0];
34 mpi::broadcast(slow_size, c, root);
35 dims[0] = mpi::chunk_length(slow_size, c.size(), c.rank());
36 return dims;
37 }
38
40 template <nda::Array T>
41 void invoke(T &&target) const {
42 if (not target.is_contiguous()) NDA_RUNTIME_ERROR << "mpi operations require contiguous target.data() to be contiguous";
43
44 static_assert(std::decay_t<A>::layout_t::stride_order_encoded == std::decay_t<T>::layout_t::stride_order_encoded,
45 "Array types for rhs and target have incompatible stride order");
46
47 if (not mpi::has_env) {
48 target = rhs;
49 return;
50 }
51
52 auto sha = shape(); // WARNING : Keep this out of any if condition (shape USES MPI) !
53 resize_or_check_if_view(target, sha);
54
55 auto slow_size = rhs.extent(0);
56 auto slow_stride = rhs.indexmap().strides()[0];
57 auto sendcounts = std::vector<int>(c.size());
58 auto displs = std::vector<int>(c.size() + 1, 0);
59 int recvcount = mpi::chunk_length(slow_size, c.size(), c.rank()) * slow_stride;
60 auto D = mpi::mpi_type<value_type>::get();
61
62 for (int r = 0; r < c.size(); ++r) {
63 sendcounts[r] = mpi::chunk_length(slow_size, c.size(), r) * slow_stride;
64 displs[r + 1] = sendcounts[r] + displs[r];
65 }
66
67 MPI_Scatterv((void *)rhs.data(), &sendcounts[0], &displs[0], D, (void *)target.data(), recvcount, D, root, c.get());
68 }
69};
