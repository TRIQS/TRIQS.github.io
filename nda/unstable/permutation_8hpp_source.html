26 // Compact representation of std::array<int,N> for N< 16
27 // indeed, we can not yet (C++20 ?) template array, array_view on a std::array ...
28 // when the elements are also <16/
29 // As a bit pattern.
30 // since N <= 16, we use 4 bits per N, for a maximum of 16* 4 = 64 bits
31 // the remaining bits are 0.
32 // the two following constexpr functions encode and decode it from the binary representation
33 // to std::array<int, Rank>
34 template <size_t Rank>
35 constexpr std::array<int, Rank> decode(uint64_t binary_representation) {
36 auto result = stdutil::make_initialized_array<Rank>(0);
37 for (int i = 0; i < int(Rank); ++i) result[i] = (binary_representation >> (4 * i)) & 0b1111ull;
38 return result;
39 }
40
41 template <size_t Rank>
42 constexpr uint64_t encode(std::array<int, Rank> const &a) {
43 uint64_t result = 0;
44 for (int i = 0; i < int(Rank); ++i) result += (a[i] << (4 * i));
45 return result;
46 }

52 template <std::integral Int, size_t Rank>
53 constexpr bool is_valid(std::array<Int, Rank> const &permutation) {
54 auto idx_counts = stdutil::make_initialized_array<Rank>(0);
55 for (auto idx : permutation) {
56 if (idx_counts[idx] > 0) return false;
57 idx_counts[idx] = 1;
58 }
59 return true;
60 }

63 template <std::integral Int, size_t Rank>
64 constexpr std::array<Int, Rank> compose(std::array<Int, Rank> const &p1, std::array<Int, Rank> const &p2) {
65 EXPECTS(is_valid(p1));
66 EXPECTS(is_valid(p2));
67 auto result = stdutil::make_initialized_array<Rank>(0);
68 for (int u = 0; u < Rank; ++u) { result[u] = p1[p2[u]]; }
69 return result;
70 }

110 template <size_t Rank>
111 constexpr std::array<int, Rank> transposition(int i, int j) {
112 auto r = identity<Rank>();
113 r[i] = j;
114 r[j] = i;
115 return r;

118 // cyclic permutation, p times. Forward
119 // n = 1 : 0 1 2 3 ----> 3 0 1 2
120 // P[n] = (Rank + n - p) % Rank
121 // 4 + 0 -1 = 3, 4 + 1 -1 = 0, 4 +2 -1 = 1, etc...
122 // if pos < Rank, the cycle is partial
123 // cycle<5> (1, 3) --> 2 0 1 3 4
124 // pos ==0 --> identity
125 template <size_t Rank>
126 constexpr std::array<int, Rank> cycle(int p, int pos = Rank) {
127 auto result = stdutil::make_initialized_array<Rank>(0);
128 for (int i = 0; i < Rank; ++i) result[i] = (i < pos ? (pos + i - p) % pos : i);
129 return result;
130 }
