59// Arrays are close
60template <typename X, typename Y>
61::testing::AssertionResult array_are_close(X const &x1, Y const &y1, double precision = 1.e-10) {
62 nda::array<nda::get_value_t<X>, nda::get_rank<X>> x = x1;
63 nda::array<nda::get_value_t<X>, nda::get_rank<X>> y = y1;
64
65 if (x.shape() != y.shape())
66 return ::testing::AssertionFailure() << "Comparing two arrays of different size "
67 << "\n X = " << x << "\n Y = " << y;
68
69 // both x, y are contiguous, I check with basic tools instead of max_element(abs(x - y))
70 if (x.size() == 0) return ::testing::AssertionSuccess();
71 auto xx = make_regular(x);
72 auto yy = make_regular(y);
73 auto maxdiff = max_element(abs(make_regular(xx - yy)));
74 if (maxdiff < precision)
75 return ::testing::AssertionSuccess();
76 else
77 return ::testing::AssertionFailure() << "max_element(abs(x-y)) = " << maxdiff << "\n X = " << x << "\n Y = " << y;
78}
79
80#define EXPECT_ARRAY_NEAR(X, ...) EXPECT_TRUE(array_are_close(X, __VA_ARGS__))
