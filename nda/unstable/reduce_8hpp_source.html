20// Models ArrayInitializer concept
21template <nda::Array A>
22struct mpi::lazy<mpi::tag::reduce, A> {
23
24 using value_type = typename std::decay_t<A>::value_type;
25 using const_view_type = decltype(std::declval<const A>()());
26
27 const_view_type rhs; // the rhs array
28 mpi::communicator c; // mpi comm
29 const int root; //
30 const bool all;
31 const MPI_Op op;
32
34 [[nodiscard]] auto shape() const { return rhs.shape(); }
35
37 template <nda::Array T>
38 void invoke(T &&target) const {
39 if (not target.is_contiguous()) NDA_RUNTIME_ERROR << "mpi operations require contiguous target.data() to be contiguous";
40
41 static_assert(std::decay_t<A>::layout_t::stride_order_encoded == std::decay_t<T>::layout_t::stride_order_encoded,
42 "Array types for rhs and target have incompatible stride order");
43
44 if (not mpi::has_env) {
45 target = rhs;
46 return;
47 }
48
49 if constexpr (not mpi::has_mpi_type<value_type>) {
50 target = nda::map([this](auto const &x) { return mpi::reduce(x, this->c, this->root, this->all, this->op); })(rhs);
51 } else {
52
53 // some checks.
54 bool in_place = (target.data() == rhs.data());
55 auto sha = shape();
56 if (in_place) {
57 if (rhs.size() != target.size()) NDA_RUNTIME_ERROR << "mpi reduce of array : same pointer to data start, but different number of elements !";
58 } else { // check no overlap
59 if ((c.rank() == root) || all) resize_or_check_if_view(target, sha);
60 if (std::abs(target.data() - rhs.data()) < rhs.size()) NDA_RUNTIME_ERROR << "mpi reduce of array : overlapping arrays !";
61 }
62
63 void *v_p = (void *)target.data();
64 void *rhs_p = (void *)rhs.data();
65 auto rhs_n_elem = rhs.size();
66 auto D = mpi::mpi_type<value_type>::get();
67
68 if (!all) {
69 if (in_place)
70 MPI_Reduce((c.rank() == root ? MPI_IN_PLACE : rhs_p), rhs_p, rhs_n_elem, D, op, root, c.get());
71 else
72 MPI_Reduce(rhs_p, v_p, rhs_n_elem, D, op, root, c.get());
73 } else {
74 if (in_place)
75 MPI_Allreduce(MPI_IN_PLACE, rhs_p, rhs_n_elem, D, op, c.get());
76 else
77 MPI_Allreduce(rhs_p, v_p, rhs_n_elem, D, op, c.get());
78 }
79 }
80 }
81};
