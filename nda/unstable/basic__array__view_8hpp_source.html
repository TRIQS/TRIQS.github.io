211 template <typename T, int R, typename L, char A, typename AP, typename OP>
212 void rebind(basic_array_view<T, R, L, A, AP, OP> v) noexcept {
213 static_assert(R == Rank, "One can only rebind a view to a view of same rank");
214 static_assert(std::is_same_v<std::remove_const_t<T>, std::remove_const_t<ValueType>>, "Type must be the same, except maybe const");
215 static constexpr bool same_type = std::is_same_v<T, ValueType>;
216
217 static_assert(same_type or is_const, "One can not rebind a view of T onto a view of const T. It would discard the const qualifier");
218 if constexpr (same_type) {
219 // FIXME Error message in layout error !
220 lay = v.lay;
221 sto = v.sto;
222 } else if constexpr (is_const) {
223 // the last if is always trivially true BUT in case of an error in the static_assert above,
224 // it improves the error message by not compiling the = afterwards
225 lay = layout_t{v.indexmap()};
226 sto = storage_t{v.storage()};
227 }
228 }
