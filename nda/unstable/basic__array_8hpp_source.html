144 template <ArrayOfRank<Rank> A>
145 basic_array(A const &a) noexcept //
146 requires(HasValueTypeConstructibleFrom<A, value_type>)
147
148 : lay(a.shape()), sto{lay.size(), mem::do_not_initialize} {
149 static_assert(std::is_constructible_v<value_type, get_value_t<A>>,
150 "Can not construct the array. ValueType can not be constructed from the value_type of the argument");
151 if constexpr (std::is_trivial_v<ValueType> or mem::is_complex_v<ValueType>) {
152 // simple type. the initialization was not necessary anyway.
153 // we use the assign, including the optimization (1d strided, contiguous) possibly
154 assign_from_ndarray(a);
155 } else {
156 // in particular ValueType may or may not be default constructible
157 // so we do not init memory, and make the placement new now, directly with the value returned by a
158 nda::for_each(lay.lengths(), [&](auto const &...is) { new (sto.data() + lay(is...)) ValueType{a(is...)}; });
159 }
160 }

177 template <typename L>
178 static auto shape_from_init_list(std::initializer_list<L> const &l) noexcept {
179 const auto [min, max] =
180 std::minmax_element(std::begin(l), std::end(l), [](auto &&x, auto &&y) { return shape_from_init_list(x) == shape_from_init_list(y); });
181 EXPECTS_WITH_MESSAGE(shape_from_init_list(*min) == shape_from_init_list(*max), "initializer list not rectangular !");
182 return stdutil::front_append(shape_from_init_list(*max), long(l.size()));
183 }
