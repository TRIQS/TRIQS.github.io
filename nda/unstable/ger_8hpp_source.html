82 template <ArrayOrScalar A, ArrayOrScalar B>
83 auto outer_product(A const &a, B const &b) {
84
85 if constexpr (Scalar<A> or Scalar<B>) {
86 return a * b;
87 } else {
88 static_assert(has_contiguous_layout<A> and has_contiguous_layout<B>);
89 auto res = zeros<get_value_t<A>>(stdutil::join(a.shape(), b.shape()));
90
91 auto a_vec = reshaped_view(a, std::array{a.size()});
92 auto b_vec = reshaped_view(b, std::array{b.size()});
93 auto mat = reshaped_view(res, std::array{a.size(), b.size()});
94 ger(1.0, a_vec, b_vec, mat);
95
96 return res;
97 }
