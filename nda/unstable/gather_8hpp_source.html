18// Models ArrayInitializer concept
19template <nda::Array A>
20struct mpi::lazy<mpi::tag::gather, A> {
21
22 using value_type = typename std::decay_t<A>::value_type;
23 using const_view_type = decltype(std::declval<const A>()());
24
25 const_view_type rhs; // the rhs array
26 mpi::communicator c; // mpi comm
27 const int root; //
28 const bool all;
29
31 [[nodiscard]] auto shape() const {
32 auto dims = rhs.shape();
33 long slow_size = dims[0];
34 if (!all) {
35 dims[0] = mpi::reduce(slow_size, c, root); // valid only on root
36 if (c.rank() != root) dims[0] = 1; // valid only on root
37 } else
38 dims[0] = mpi::all_reduce(slow_size, c); // in this case, it is valid on all nodes
39
40 return dims;
41 }
42
44 template <nda::Array T>
45 void invoke(T &&target) const {
46 if (not target.is_contiguous()) NDA_RUNTIME_ERROR << "mpi operations require contiguous target.data() to be contiguous";
47
48 static_assert(std::decay_t<A>::layout_t::stride_order_encoded == std::decay_t<T>::layout_t::stride_order_encoded,
49 "Array types for rhs and target have incompatible stride order");
50
51 if (not mpi::has_env) {
52 target = rhs;
53 return;
54 }
55
56 auto recvcounts = std::vector<int>(c.size());
57 auto displs = std::vector<int>(c.size() + 1, 0);
58 int sendcount = rhs.size();
59 auto D = mpi::mpi_type<value_type>::get();
60
61 auto sha = shape(); // WARNING : Keep this out of the if condition (shape USES MPI) !
62 if (all || (c.rank() == root)) resize_or_check_if_view(target, sha);
63
64 void *v_p = target.data();
65 const void *rhs_p = rhs.data();
66
67 auto mpi_ty = mpi::mpi_type<int>::get();
68 if (!all)
69 MPI_Gather(&sendcount, 1, mpi_ty, &recvcounts[0], 1, mpi_ty, root, c.get());
70 else
71 MPI_Allgather(&sendcount, 1, mpi_ty, &recvcounts[0], 1, mpi_ty, c.get());
72
73 for (int r = 0; r < c.size(); ++r) displs[r + 1] = recvcounts[r] + displs[r];
74
75 if (!all)
76 MPI_Gatherv((void *)rhs_p, sendcount, D, v_p, &recvcounts[0], &displs[0], D, root, c.get());
77 else
78 MPI_Allgatherv((void *)rhs_p, sendcount, D, v_p, &recvcounts[0], &displs[0], D, c.get());
79 }
80};
