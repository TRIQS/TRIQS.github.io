78 // a placeholder is an empty struct, labelled by an int.
79 template <int N>
80 struct placeholder {
81 static constexpr int index = N;
82 template <typename RHS>
83 pair<N, RHS> operator=(RHS &&rhs) const { // NOLINT It is correct here to return a pair, not the placeholder& as suggested by clang-tidy
84 return {std::forward<RHS>(rhs)};
85 }
86 template <typename... T>
87 expr<tags::function, placeholder, expr_storage_t<T>...> operator()(T &&...x) const {
88 return {tags::function{}, *this, std::forward<T>(x)...};
89 }
90 template <typename T>
91 expr<tags::subscript, placeholder, expr_storage_t<T>> operator[](T &&x) const {
92 return {tags::subscript{}, *this, std::forward<T>(x)};
93 }
94 };

371 template <typename Tag, typename... Args>

381 // the evaluator for an expression
382 template <typename Tag, typename... Childs, typename... Pairs>
383 struct evaluator<expr<Tag, Childs...>, Pairs...> {
384 static constexpr bool is_lazy = (evaluator<Childs, Pairs...>::is_lazy or ...);
385
386 template <size_t... Is>
387 [[nodiscard]] FORCEINLINE decltype(auto) eval_impl(std::index_sequence<Is...>, expr<Tag, Childs...> const &ex, Pairs const &...pairs) const {
388 // if constexpr(is_lazy)
389 // return {Tag(), eval(std::get<Is>(ex.childs), pairs...)...};
390
391 return op_dispatch<Tag>(std::integral_constant<bool, is_lazy>{}, eval(std::get<Is>(ex.childs), pairs...)...);
392 }
393
394 [[nodiscard]] FORCEINLINE decltype(auto) operator()(expr<Tag, Childs...> const &ex, Pairs const &...pairs) const {
395 return eval_impl(std::make_index_sequence<sizeof...(Childs)>(), ex, pairs...);
396 }
397 };

409 template <typename F>
410 struct apply_on_each_leaf_impl {
411 F f;
412
413 private:
414 template <typename ChildTuple, size_t... Is>
415 void _apply_this_on_each(std::index_sequence<Is...>, ChildTuple const &child_tuple) {
416 ((*this)(std::get<Is>(child_tuple)), ...);
417 }
418
419 public:
420 template <typename Tag, typename... T>
421 FORCEINLINE void operator()(expr<Tag, T...> const &ex) {
422 _apply_this_on_each(std::make_index_sequence<sizeof...(T)>{}, ex.childs);
423 }
424 template <typename T>
425 FORCEINLINE void operator()(T const &x) CLEF_requires(!is_any_lazy<T>) {
426 f(x);
427 }
428 template <typename T>
429 FORCEINLINE void operator()(std::reference_wrapper<T> const &x) CLEF_requires(!is_any_lazy<T>) {
430 f(x.get());
431 }
432 };

434 template <typename F, typename Expr>
435 FORCEINLINE void apply_on_each_leaf(F &&f, Expr const &ex) {
436 auto impl = apply_on_each_leaf_impl<F>{std::forward<F>(f)};
437 impl(ex);
