31 template <Array A, typename F, typename R>
32 auto fold(F f, A const &a, R r) {
33 decltype(f(r, get_value_t<A>{})) r2 = r;
34 // to take into account that f may be double,double -> double, while one passes 0 (an int...)
35 // R = int, R2= double in such case, and the result will be a double, or narrowing will occur
36 nda::for_each(a.shape(), [&a, &r2, &f](auto &&... args) { r2 = f(r2, a(args...)); });
37 return r2;
38 }

48 template <Array A, typename F>
49 auto fold(F f, A const &a) {
50 return fold(std::move(f), a, get_value_t<A>{});
51 }

57 template <Array A>
58 bool any(A const &a) {
59 static_assert(std::is_same_v<get_value_t<A>, bool>, "OOPS");
60 return fold([](bool r, auto const &x) -> bool { return r or bool(x); }, a, false);

68 return fold([](bool r, auto const &x) -> bool { return r and bool(x); }, a, true);

111 template <ArrayOfRank<2> A>
112 double frobenius_norm(A const &a) {
113 return std::sqrt(fold(
114 [](double r, auto const &x) -> double {
115 auto ab = std::abs(x);
116 return r + ab * ab;
117 },
118 a, double(0)));
119 }
