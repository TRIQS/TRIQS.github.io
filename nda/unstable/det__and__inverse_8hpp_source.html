24 template <typename A>
25 bool is_matrix_square(A const &a, bool print_error = false) {
26 bool r = (a.shape()[0] == a.shape()[1]);
27 if (not r and print_error)
28 std::cerr << "Error non-square matrix. Dimensions are :(" << a.shape()[0] << "," << a.shape()[1] << ")\n " << std::endl;
29 return r;
30 }

32 // ---------- Determinant -------------------------
33
34 template <typename M>
35 auto determinant_in_place(M &m) requires(is_matrix_or_view_v<M>) {
36 using value_t = get_value_t<M>;
37 static_assert(std::is_convertible_v<value_t, double> or std::is_convertible_v<value_t, std::complex<double>>,
38 "determinant requires a matrix of values that can be implicitly converted to double or std::complex<double>");
39 static_assert(not std::is_const_v<M>, "determinant_in_place can not be const. It destroys its argument");
40
41 if(m.empty()) return value_t{1};
42
43 if(m.extent(0) != m.extent(1))
44 NDA_RUNTIME_ERROR << "Error in determinant. Matrix is not square but has shape " << m.shape();
45 const int dim = m.extent(0);
46
47 // Calculate the LU decomposition using lapack getrf
48 basic_array<int, 1, C_layout, 'A', sso<100>> ipiv(dim);
49 int info = lapack::getrf(m, ipiv); // it is ok to be in C order. Lapack compute the inverse of the transpose.
50 if (info < 0) NDA_RUNTIME_ERROR << "Error in determinant. Info lapack is " << info;
51
52 // Calculate the determinant from the LU decomposition
53 auto det = value_t{1};
54 int n_flips = 0;
55 for (int i = 0; i < dim; i++){
56 det *= m(i, i);
57 // Count the number of column interchanges performed by getrf
58 if(ipiv(i) != i + 1) ++n_flips;
59 }
60
61 return ((n_flips % 2 == 1) ? -det : det);
62 }

64 template <typename M>
65 auto determinant(M const &m) {
66 auto m_copy = make_regular(m);
67 return determinant_in_place(m_copy);
68 }

70 // ---------- inverse -------------------------

72 template <typename T, typename L, typename AP, typename OP>
73 void inverse_in_place(basic_array_view<T, 2, L, 'M', AP, OP> a) {
74 EXPECTS(is_matrix_square(a, true));
75 if(a.empty()) return;
76 array<int, 1> ipiv(a.extent(0));
77 int info = lapack::getrf(a, ipiv); // it is ok to be in C order. Lapack compute the inverse of the transpose.
78 if (info != 0) NDA_RUNTIME_ERROR << "Inverse/Det error : matrix is not invertible. Step 1. Lapack error : " << info;
79 info = lapack::getri(a, ipiv);
80 if (info != 0) NDA_RUNTIME_ERROR << "Inverse/Det error : matrix is not invertible. Step 2. Lapack error : " << info;
81 } // namespace nda

83 template <typename T, typename L, typename CP>
84 void inverse_in_place(basic_array<T, 2, L, 'M', CP> &a) {
85 inverse_in_place(a());

88 template <Array A>
89 auto inverse(A const &a) requires(get_algebra<A> == 'M') {
90 static_assert(get_rank<A> == 2, "inverse: array must have rank two");
91 EXPECTS(is_matrix_square(a, true));
92 auto r = make_regular(a);
93 inverse_in_place(r);
94 return r;
95 }
