27 template <AddressSpace DestAdrSp, AddressSpace SrcAdrSp>
28 void memcpy(void *dest, void const *src, size_t count) {
29 check_adr_sp_valid<DestAdrSp, SrcAdrSp>();
30 static_assert(nda::have_device == nda::have_cuda, "Adjust function for new device types");
31
32 if constexpr (DestAdrSp == Host && SrcAdrSp == Host) {
33 std::memcpy(dest, src, count);
34 } else { // Device or Unified
35 device_error_check(cudaMemcpy(dest, src, count, cudaMemcpyDefault), "CudaMemcpy");
36 }
37 }

39 template <AddressSpace DestAdrSp, AddressSpace SrcAdrSp>
40 void memcpy2D(void *dest, size_t dpitch, const void *src, size_t spitch, size_t width, size_t height) {
41 EXPECTS(width <= dpitch && width <= spitch);
42 check_adr_sp_valid<DestAdrSp, SrcAdrSp>();
43 static_assert(nda::have_device == nda::have_cuda, "Adjust function for new device types");
44
45 if constexpr (DestAdrSp == Host && SrcAdrSp == Host) {
46 auto *desti = static_cast<unsigned char *>(dest);
47 auto *srci = static_cast<const unsigned char *>(src);
48 for (size_t i = 0; i < height; ++i, desti += dpitch, srci += spitch) std::memcpy(desti, srci, width);
49 } else if (nda::have_device) {
50 device_error_check(cudaMemcpy2D(dest, dpitch, src, spitch, width, height, cudaMemcpyDefault), "CudaMemcpy2D");
51 }
52 }
