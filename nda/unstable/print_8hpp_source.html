40 template <typename A>
41 std::ostream &operator<<(std::ostream &out, A const &a) requires(is_regular_or_view_v<A>) {
42
43 if constexpr (A::rank == 1) {
44 out << "[";
45 auto const &len = a.indexmap().lengths();
46 for (size_t i = 0; i < len[0]; ++i) out << (i > 0 ? "," : "") << a(i);
47 out << "]";
48 }
49
50 if constexpr (A::rank == 2) {
51 auto const &len = a.indexmap().lengths();
52 out << "\n[";
53 for (size_t i = 0; i < len[0]; ++i) {
54 out << (i == 0 ? "[" : " [");
55 for (size_t j = 0; j < len[1]; ++j) out << (j > 0 ? "," : "") << a(i, j);
56 out << "]" << (i == len[0] - 1 ? "" : "\n");
57 }
58 out << "]";
59 }
60
61 // FIXME : not very pretty, do better here, but that was the arrays way
62 if constexpr (A::rank > 2) {
63 out << "[";
64 for (size_t j = 0; j < a.size(); ++j) out << (j > 0 ? "," : "") << a.data()[j];
65 out << "]";
66 }
67
68 return out;
69 }
