19 // Notations for this file
20 //
21 // N : rank of the original idx_map
22 // P : rank of the resulting idx_map
23 // Q : number of arguments of the slice.
24 //
25 // n : 0, ..., N : index the dimensions of the original idx_map
26 // p : 0, ..., P : index the dimensions of the resulting idx_map
27 // q : 0, ..., Q : index the arguments
28 //
29 // NB : N - Q is the length of the ellipsis, 0 if no ellipsis in the argument.
30 //
31 // Arguments are of the form e.g. for N = 6 :
32 // long, long , Ellipsis, long , Range
33 // with Ellipsis expanded :
34 // long, long, RangeAll, RangeAll, long, Range
35 //
36 // p are the indices of the non long arguments (after ellipsis expansion)
37
38 // We compute compile time map : p -> n, p -> q
39 // example :
40 // Args = long, long, ellipsis, long, range
41 // q 0 1 2 3 4
42 // Expanded Args = long, long, range, range, range, long, range
43 // n 0 1 2 3 4 5 6
44 // e_pos = 2, e_pos + e_len = 5
45 // n will be 0 1 2 3 4 5 6
46 // q will be 0 1 2 2 2 3 4
47 // p -> n : 2 3 4 6
48 // p -> q : 2 2 2 4
49
50 // Case of ellipsis of zero size
51 // Args = long, long, range, ellipsis, long, range
52 // q 0 1 2 3 4 5
53 // Expanded Args = long, long, range, long, range
54 // n 0 1 2 3 4
55 // e_pos = 2, e_len = 0
56 // n will be 0 1 2 3 4
57 // q will be 0 1 2 4 5
58 // p -> n : 2 4
59 // p -> q : 2 5

65 constexpr int ellipsis_position(std::index_sequence<Is...>) {
66 // We know that there is at most one ellipsis.
67 int r = ((std::is_same_v<Args, ellipsis> ? int(Is) + 1 : 0) + ...); // position + 1 or 0
68 return (r == 0 ? 128 : r - 1);
69 }

78 // ------------ Relation n -> q, given the position and length of the ellipsis -----------
79 // e_pos : ellipsis position
80 // e_len : ellipsis length
81 // return q
82 //
83 constexpr int q_of_n(int n, int e_pos, int e_len) {
84 if (n < e_pos) return n; // if no ellipsis, e_pos is 128 = infty
85 if (n < (e_pos + e_len)) // in the ellipsis
86 return e_pos;
87 else
88 return n - (e_len - 1); // ok if e_len ==0 : n+1
89 }

91 // ------------- The map p-> n -------------------------
92 // args_is_range : for each q, True iif the args is a range, or range_all or ellipsis
93 // e_pos : ellipsis position
94 // e_len : ellipsis length
95 // return the map p-> n as a std::array : result[p] = n
96 // for each index p, what is its position in the original idx_map
97 //
98 template <int N, int P, size_t Q> // Q is a size_t for matching std::array
99 constexpr std::array<int, P> n_of_p_map(std::array<bool, Q> const &args_is_range, int e_pos, int e_len) {
100 auto result = stdutil::make_initialized_array<P>(0);
101 int p = 0;
102 for (int n = 0; n < N; ++n) {
103 int q = q_of_n(n, e_pos, e_len);
104 if (args_is_range[q]) result[p++] = n;
105 }
106 if (p != P) throw std::logic_error("Internal Error");
107 return result;
108 }

110 // ------------- The map p-> q -------------------------
111 // same as before except that it returns p-> q instead of p-> n
112 //
113 template <int N, int P, size_t Q>
114 constexpr std::array<int, P> q_of_p_map(std::array<bool, Q> const &args_is_range, int e_pos, int e_len) {
115 auto result = stdutil::make_initialized_array<P>(0);
116 int p = 0;
117 for (int n = 0; n < N; ++n) {
118 int q = q_of_n(n, e_pos, e_len);
119 if (args_is_range[q]) result[p++] = q;
120 }
121 if (p != P) throw std::logic_error("Internal Error");
122 return result;
123 }

125 // ------------- The map n->p -------------------------
126 // n_of_p : the map n->p
127 // return the (pseudo) inverse map
128 // n -> p or -1 if n is the index of a long argument
129 template <size_t N, size_t P>
130 constexpr std::array<int, N> p_of_n_map(std::array<int, P> const &n_of_p) {
131 auto result = stdutil::make_initialized_array<N>(-1);
132 for (size_t p = 0; p < P; ++p) result[n_of_p[p]] = p;
133 return result;
134 }

136 // -------------- Slice the stride_order -----------------------
137 // stride_order : the permutation stride_order. stride_order[0] : slowest, etc...
138 // n_of_p : the map p-> n
139 // return : the new stride_order of the sliced map
140 template <size_t P, size_t N>
141 constexpr std::array<int, P> sliced_mem_stride_order(std::array<int, N> const &stride_order_in, std::array<int, P> const &n_of_p) {
142 // quick short cut : does it really impact compilation time ?
143 //if (stride_order_in == 0) return 0; // quick decision C-> C
144 auto stride_order = stdutil::make_initialized_array<P>(0);
145 auto p_of_n = p_of_n_map<N>(n_of_p); // reverse the map
146 // traverse the n in the order given by stride_order_in and keep them if they are associated to a p
147 for (int i = 0, ip = 0; i < N; ++i) { // i : index of the n
148 int n = stride_order_in[i]; // n traverses the N in the order of the stride_order. Slowest first.
149 int p = p_of_n[n]; // n->p or -1 is n is a long argument
150 if (p != -1) stride_order[ip++] = p; // if p is fine, it is the next
151 }
152 return stride_order;
153 }

155 // -------------- Slice the stride_order info flags-----------
156 // P : as before
157 // has_only_rangeall_and_long : as the name says...
158 // args_is_range_all : for each q, True iif the args is a range_all or an ellipsis [NO range here !]
159 // stride_order : the stride_order of idx_map to be slided
160 // Nlast : position, in q, of the argument corresponding to the fastest stride
161 // layout_prop : to be sliced
162 //
163 template <size_t Q, size_t N>
164 constexpr layout_prop_e slice_layout_prop(int P, bool has_only_rangeall_and_long, std::array<bool, Q> const &args_is_range_all,
165 std::array<int, N> const &stride_order, layout_prop_e layout_prop, int e_pos, int e_len) {
166
167 // if we have some ranges, we give up
168 if (not has_only_rangeall_and_long) {
169 if (P == 1) // rank one is special. Alyways at least 1d strided
170 return layout_prop_e::strided_1d;
171 else
172 return layout_prop_e::none;
173 }
174 // count the number of times 1 0 appears args_in_range
175 // we must traverse in the order of the stride_order ! (in memory order, slowest to fastest)
176 int n_1_blocks = 0;
177 bool previous_arg_is_rangeall = false;
178 for (int i_n = 0; i_n < int(N); ++i_n) {
179 int q = q_of_n(stride_order[i_n], e_pos, e_len);
180 bool arg_is_rangeall = args_is_range_all[q];
181 if (arg_is_rangeall and (not previous_arg_is_rangeall)) ++n_1_blocks;
182 previous_arg_is_rangeall = arg_is_rangeall;
183 }
184 // in mem order. e.g. (long, all, all, long) or (all, all, long), but not (all, long, all)
185 bool rangeall_are_grouped_in_memory = (n_1_blocks <= 1);
186 bool last_is_rangeall = previous_arg_is_rangeall;
187
188 if (has_contiguous(layout_prop) and rangeall_are_grouped_in_memory and last_is_rangeall) return layout_prop_e::contiguous;
189 if (has_strided_1d(layout_prop) and rangeall_are_grouped_in_memory) return layout_prop_e::strided_1d;
190 if (has_smallest_stride_is_one(layout_prop) and last_is_rangeall) return layout_prop_e::smallest_stride_is_one;
191
192 return layout_prop_e::none;
193 }

195 // ------------ Small pieces of code for the fold in functions below, with dispatch on type --------------------------------------
196 // offset
197 // first arg : the n-th argument (after expansion of the ellipsis)
198 // second arg : s_n : stride[n] of the idx_map
199 FORCEINLINE long get_offset(long R, long s_n) { return R * s_n; }
200 FORCEINLINE long get_offset(range const &R, long s_n) { return R.first() * s_n; }
201 FORCEINLINE long get_offset(range::all_t, long) { return 0; }
202
203 // length. Same convention
204 // second arg : l_n : length[n] of the idx_map
205 FORCEINLINE long get_l(range const &R, long l_n) {
206 return ((R.last() == -1 ? l_n : R.last()) - R.first() + R.step() - 1) / R.step(); // python behaviour
207 }
208 FORCEINLINE long get_l(range::all_t, long l_n) { return l_n; }
209
210 // strides
211 FORCEINLINE long get_s(range const &R, long s_n) { return s_n * R.step(); }
212 FORCEINLINE long get_s(range::all_t, long s_n) { return s_n; }

219 // ----------------------------- slice of index map : implementation function ----------------------------------------------
220
221 // Ns, Ps, Qs : sequence indices for size N, P, Q
222 // IdxMap : type of the indexmap idx
223 // Arg : arguments of the slice
224 // returns : a pair: (offset, new sliced stride_order)
225 //
226 // FIXME : Q only needed, not Qs
227 template <size_t... Ps, size_t... Ns, size_t... Qs, typename IdxMap, typename... Args>
228 FORCEINLINE auto slice_stride_order_impl(std::index_sequence<Ps...>, std::index_sequence<Ns...>, std::index_sequence<Qs...>, IdxMap const &idxm,
229 Args const &... args) {
230
231#ifdef NDA_ENFORCE_BOUNDCHECK
232 details::assert_in_bounds(idxm.rank(), idxm.lengths().data(), args...);
233#endif
234
235 static_assert(IdxMap::rank() == sizeof...(Ns), "Internal error");
236 static_assert(sizeof...(Args) == sizeof...(Qs), "Internal error");
237
238 static constexpr int N = sizeof...(Ns);
239 static constexpr int P = sizeof...(Ps);
240 static constexpr int Q = sizeof...(Qs);
241 static constexpr int e_len = N - Q + 1; // len of ellipsis : how many ranges are missing
242 static constexpr int e_pos = ellipsis_position<Args...>();
243
244 // Pattern of the arguments. 1 for a range/range_all/ellipsis, 0 for long
245 static constexpr std::array<bool, Q> args_is_range{(std::is_same_v<Args, range> or std::is_base_of_v<range::all_t, Args>)...};
246 static constexpr std::array<bool, Q> args_is_range_all{(std::is_base_of_v<range::all_t, Args>)...};
247
248 static constexpr std::array<int, P> n_of_p = n_of_p_map<N, P>(args_is_range, e_pos, e_len);
249 static constexpr std::array<int, P> q_of_p = q_of_p_map<N, P>(args_is_range, e_pos, e_len);
250
251 static_assert(n_of_p.size() == P, "Oops"); // sanity check
252 static_assert(q_of_p.size() == P, "Oops");
253
254 auto argstie = std::tie(args...);
255
256 // PRINT(bitsP, Qn_of_p_map(bitsP, 0, e_pos, e_len)); PRINT(e_pos); PRINT(e_len);
257
258 // Compute the offset of the pointer
259 long offset = (get_offset(std::get<q_of_n(Ns, e_pos, e_len)>(argstie), std::get<Ns>(idxm.strides())) + ... + 0);
260
261 // Compute the new len and strides
262 std::array<long, P> len{get_l(std::get<q_of_p[Ps]>(argstie), std::get<n_of_p[Ps]>(idxm.lengths()))...};
263 std::array<long, P> str{get_s(std::get<q_of_p[Ps]>(argstie), std::get<n_of_p[Ps]>(idxm.strides()))...};
264
265 // Compute the new static_extents
266 static constexpr std::array<int, P> new_static_extents{(args_is_range_all[q_of_p[Ps]] ? IdxMap::static_extents[n_of_p[Ps]] : 0)...};
267
268 // The new Stride Order
269 static constexpr std::array<int, P> mem_stride_order = sliced_mem_stride_order(IdxMap::stride_order, n_of_p);
270
271 // Compute the new layout_prop
272 static constexpr bool has_only_rangeall_and_long = ((std::is_constructible_v<long, Args> or std::is_base_of_v<range::all_t, Args>)and...);
273
274 static constexpr layout_prop_e li =
275 slice_layout_prop(P, has_only_rangeall_and_long, args_is_range_all, IdxMap::stride_order, IdxMap::layout_prop, e_pos, e_len);
276
277 static constexpr uint64_t new_static_extents_encoded = encode(new_static_extents);
278 static constexpr uint64_t mem_stride_order_encoded = encode(mem_stride_order);
279 return std::make_pair(offset, idx_map<P, new_static_extents_encoded, mem_stride_order_encoded, li>{len, str});
280 }

282 // ----------------------------- slice of index map ----------------------------------------------
283
284 template <int R, uint64_t SE, uint64_t SO, layout_prop_e LP, typename... T>
285 FORCEINLINE decltype(auto) slice_stride_order(idx_map<R, SE, SO, LP> const &idxm, T const &... x) {
286
287 static constexpr int n_args_ellipsis = ((std::is_same_v<T, ellipsis>)+...);
288 static constexpr int n_args_long = (std::is_constructible_v<long, T> + ...); // any T I can construct a long from
289
290 static_assert(n_args_ellipsis <= 1, "At most one ellipsis argument is authorized");
291 static_assert((sizeof...(T) <= R + 1), "Incorrect number of arguments in array call ");
292 static_assert((n_args_ellipsis == 1) or (sizeof...(T) == R), "Incorrect number of arguments in array call ");
293
294 return slice_stride_order_impl(std::make_index_sequence<R - n_args_long>{}, std::make_index_sequence<R>{},
295 std::make_index_sequence<sizeof...(T)>{}, idxm, x...);
296 }
