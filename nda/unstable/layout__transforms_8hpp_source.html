28 template <typename T, int R, typename L, char Algebra, typename AccessorPolicy, typename OwningPolicy, typename NewLayoutType>
29 auto map_layout_transform(basic_array_view<T, R, L, Algebra, AccessorPolicy, OwningPolicy> a, NewLayoutType const &new_layout) {
30 using layout_policy = typename details::layout_to_policy<NewLayoutType>::type;
31 return basic_array_view<T, NewLayoutType::rank(), layout_policy, (NewLayoutType::rank() == R ? Algebra : 'A'), AccessorPolicy, OwningPolicy>{
32 new_layout, a.storage()};
33 }

57 template <typename T, int R, typename L, char Algebra, typename ContainerPolicy, std::integral Int, auto newRank>
58 auto reshape(basic_array<T, R, L, Algebra, ContainerPolicy> &&a, std::array<Int, newRank> const &new_shape) {
59 using layout_t = typename L::template mapping<newRank>;
60 EXPECTS_WITH_MESSAGE(a.size() == (std::accumulate(new_shape.cbegin(), new_shape.cend(), 1, std::multiplies<>{})),
61 "Reshape : the new shape has a incorrect number of elements");
62 return map_layout_transform(std::move(a), layout_t{new_shape});
63 }

149 template <int N, auto R>
150 constexpr std::array<int, R + N> complete_stride_order_with_fast(std::array<int, R> const &a) {
151 auto r = stdutil::make_initialized_array<R + N>(0);
152 for (int i = 0; i < R; ++i) r[i] = a[i];
153 for (int i = 0; i < N; ++i) r[R + i] = R + i;
154 return r;
155 }

158 // Take an array or view and add N dimensions of size 1 in the fastest indices
159 template <int N, typename A>
160 auto reinterpret_add_fast_dims_of_size_one(A &&a) requires(nda::is_regular_or_view_v<std::decay_t<A>>) {
161
162 auto const &lay = a.indexmap();
163 using lay_t = std::decay_t<decltype(lay)>;
164
165 static constexpr uint64_t new_stride_order_encoded = encode(impl::complete_stride_order_with_fast<N>(lay_t::stride_order));
166 // (lay_t::stride_order_encoded == 0 ? 0 : encode(impl::complete_stride_order_with_fast<N>(lay_t::stride_order)));
167
168 static constexpr uint64_t new_static_extents_encoded = encode(stdutil::join(lay_t::static_extents, stdutil::make_initialized_array<N>(0)));
169 // (lay_t::static_extents_encoded == 0 ? 0 : encode(stdutil::join(lay_t::static_extents, stdutil::make_initialized_array<N>(0))));
170
171 using new_lay_t = idx_map<get_rank<A> + N, new_static_extents_encoded, new_stride_order_encoded, lay_t::layout_prop>;
172
173 auto shap1111 = stdutil::make_initialized_array<N>(1l);
174 return map_layout_transform(std::forward<A>(a), new_lay_t{stdutil::join(lay.lengths(), shap1111), stdutil::join(lay.strides(), shap1111)});
175 }
