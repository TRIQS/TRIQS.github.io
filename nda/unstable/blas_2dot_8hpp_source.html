26 template <typename X, typename Y>
27 requires((Scalar<X> or MemoryVector<X>) and (Scalar<Y> or MemoryVector<X>))
28 auto dot(X const &x, Y const &y) {
29 if constexpr (Scalar<X> or Scalar<Y>) {
30 return x * y;
31 } else {
32 static_assert(have_same_value_type_v<X, Y>, "Vectors must have same value type");
33 static_assert(mem::have_compatible_addr_space<X, Y>, "Vectors must have compatible memory address space");
34 static_assert(is_blas_lapack_v<get_value_t<X>>, "Vectors hold value_type incompatible with blas");
35
36 EXPECTS(x.shape() == y.shape());
37
38 if constexpr (mem::have_device_compatible_addr_space<X, Y>) {
39#if defined(NDA_HAVE_DEVICE)
40 return device::dot(x.size(), x.data(), x.indexmap().strides()[0], y.data(), y.indexmap().strides()[0]);
41#else
42 compile_error_no_gpu();
43 return get_value_t<X>(0);
44#endif
45 } else {
46 return f77::dot(x.size(), x.data(), x.indexmap().strides()[0], y.data(), y.indexmap().strides()[0]);
47 }
48 }
49 }

52 template <typename X, typename Y>
53 requires((Scalar<X> or MemoryVector<X>) and (Scalar<Y> or MemoryVector<X>))
54 auto dotc(X const &x, Y const &y) {
55 if constexpr (Scalar<X> or Scalar<Y>) {
56 return conj(x) * y;
57 } else {
58 static_assert(have_same_value_type_v<X, Y>, "Vectors must have same value type");
59 static_assert(mem::have_compatible_addr_space<X, Y>, "Vectors must have same memory address space");
60 static_assert(is_blas_lapack_v<get_value_t<X>>, "Vectors hold value_type incompatible with blas");
61
62 EXPECTS(x.shape() == y.shape());
63
64 if constexpr (!is_complex_v<get_value_t<X>>) {
65 return dot(x, y);
66 } else if constexpr (mem::have_device_compatible_addr_space<X, Y>) {
67#if defined(NDA_HAVE_DEVICE)
68 return device::dotc(x.size(), x.data(), x.indexmap().strides()[0], y.data(), y.indexmap().strides()[0]);
69#else
70 compile_error_no_gpu();
71 return get_value_t<X>(0);
72#endif
73 } else {
74 return f77::dotc(x.size(), x.data(), x.indexmap().strides()[0], y.data(), y.indexmap().strides()[0]);
75 }

81 template <bool star, typename X, typename Y>
82 auto _dot_impl(X const &x, Y const &y) {
83 EXPECTS(x.shape() == y.shape());
84 long N = x.shape()[0];
85
86 auto _conj = [](auto z) __attribute__((always_inline)) {
87 if constexpr (star and is_complex_v<decltype(z)>) {
88 return std::conj(z);
89 } else
90 return z;
91 };
92
93 if constexpr (has_layout_smallest_stride_is_one<X> and has_layout_smallest_stride_is_one<Y>) {
94 if constexpr (is_regular_or_view_v<X> and is_regular_or_view_v<Y>) {
95 auto *__restrict px = x.data();
96 auto *__restrict py = y.data();
97 auto res = _conj(px[0]) * py[0];
98 for (size_t i = 1; i < N; ++i) { res += _conj(px[i]) * py[i]; }
99 return res;
100 } else {
101 auto res = _conj(x(_linear_index_t{0})) * y(_linear_index_t{0});
102 for (long i = 1; i < N; ++i) { res += _conj(x(_linear_index_t{i})) * y(_linear_index_t{i}); }
103 return res;
104 }
105 } else {
106 auto res = _conj(x(0)) * y(0);
107 for (long i = 1; i < N; ++i) { res += _conj(x(i)) * y(i); }
108 return res;
109 }
110 }
