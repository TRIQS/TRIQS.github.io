41 inline size_t round_to_align(size_t s) {
42 size_t a = alignof(std::max_align_t);
43 return ((s + a - 1) / a) * a; // s = 3a +2 --> (4a + 2)/a = 4 --> 4a
44 }

66 // ------------------------- Bucket allocator ----------------------------
67 //
68 //
69 template <int ChunkSize>
70 class bucket {
71 static constexpr int TotalChunkSize = 64 * ChunkSize;
72 std::unique_ptr<char[]> _start = std::make_unique<char[]>(TotalChunkSize); // NOLINT
73 char *p = _start.get();
74 uint64_t flags = uint64_t(-1);
75
76 public:
77#ifdef NDA_USE_ASAN
78 bucket() { __asan_poison_memory_region(p, TotalChunkSize); }
79 ~bucket() { __asan_unpoison_memory_region(p, TotalChunkSize); }
80#else
81 bucket() = default;
82#endif
83 bucket(bucket const &) = delete;
84 bucket(bucket &&) = default;
85 bucket &operator=(bucket const &) = delete;
86 bucket &operator=(bucket &&) = default;
87
88 blk_t allocate(size_t s) noexcept {
89 // FIXME not here ! in the handle
90 //auto n = round_to_align(s);
91 //if (n > ChunkSize) std::abort();
92 if (flags == 0) std::abort();
93 int pos = __builtin_ctzll(flags);
94 flags &= ~(1ull << pos);
95 blk_t b{p + pos * ChunkSize, s};
96#ifdef NDA_USE_ASAN
97 __asan_unpoison_memory_region(b.ptr, ChunkSize);
98#endif
99 return b;
100 }
101
102 blk_t allocate_zero(size_t s) noexcept {
103 auto blk = allocate(s);
104 std::memset(blk.ptr, 0, s);
105 return blk;
106 }
107
108 void deallocate(blk_t b) noexcept {
109#ifdef NDA_USE_ASAN
110 __asan_poison_memory_region(b.ptr, ChunkSize);
111#endif
112 int pos = (b.ptr - p) / ChunkSize;
113 flags |= (1ull << pos);
114 }
115
116 [[nodiscard]] bool is_full() const noexcept { return flags == 0; }
117 [[nodiscard]] bool empty() const noexcept { return flags == uint64_t(-1); }
118
119 [[nodiscard]] const char *data() const noexcept { return p; }
120
121 [[nodiscard]] bool owns(blk_t b) const noexcept { return b.ptr >= p and b.ptr < p + TotalChunkSize; }
122 };

124 // ------------------------- Multiple bucket allocator ----------------------------
125 //
126 //
127 template <int ChunkSize>
128 class multiple_bucket {
129
130 using b_t = bucket<ChunkSize>;
131 std::vector<b_t> bu_vec; // an ordered vector of buckets
132 typename std::vector<b_t>::iterator bu; // current bucket in use
133
134 // find the next bucket with some space. Possibly allocating new ones.
135 [[gnu::noinline]] void find_non_full_bucket() {
136 bu = std::find_if(bu_vec.begin(), bu_vec.end(), [](auto const &b) { return !b.is_full(); });
137 if (bu != bu_vec.end()) return;
138
139 // insert a new bucket ordered. Position is defined by data (NB : which is NOT affected by the move)
140 b_t b;
141 auto insert_position = std::upper_bound(bu_vec.begin(), bu_vec.end(), b, [](auto const &B, auto const &B2) { return B.data() < B2.data(); });
142 bu = bu_vec.insert(insert_position, std::move(b));
143
144 //for (auto const &bb : bu_vec) TRIQS_PRINT((void *)bb.data());
145 //TRIQS_PRINT("---------------");
146 }
147
148 public:
149 multiple_bucket() : bu_vec(1), bu(bu_vec.begin()) {}
150 multiple_bucket(multiple_bucket const &) = delete;
151 multiple_bucket(multiple_bucket &&) = delete;
152 multiple_bucket &operator=(multiple_bucket const &) = delete;
153 multiple_bucket &operator=(multiple_bucket &&) = delete;
154
155 blk_t allocate(size_t s) {
156 //[[unlikely]]
157 if ((bu == bu_vec.end()) or (bu->is_full())) find_non_full_bucket();
158 return bu->allocate(s);
159 }
160
161 blk_t allocate_zero(size_t s) noexcept {
162 auto blk = allocate(s);
163 std::memset(blk.ptr, 0, s);
164 return blk;
165 }
166
167 void deallocate(blk_t b) noexcept {
168 //[[likely]]
169 if (bu != bu_vec.end() and bu->owns(b)) {
170 bu->deallocate(b);
171 return;
172 }
173 bu = std::lower_bound(bu_vec.begin(), bu_vec.end(), b.ptr, [](auto const &B, auto p) { return B.data() <= p; });
174 --bu;
175 EXPECTS_WITH_MESSAGE((bu != bu_vec.end()), "Fatal Logic Error in allocator. Not in bucket. \n");
176 EXPECTS_WITH_MESSAGE((bu->owns(b)), "Fatal Logic Error in allocator. \n");
177 bu->deallocate(b);
178 if (!bu->empty()) return;
179 if (bu_vec.size() <= 1) return;
180 bu_vec.erase(bu);
181 bu = bu_vec.end();
182 }

187 // ------------------------- segregator allocator ----------------------------
188 //
189 // Dispatch according to size to two allocators
190 //
191 template <size_t Threshold, typename A, typename B>
192 class segregator {
193 A small;
194 B big;
195
196 public:
197 segregator() = default;
198 segregator(segregator const &) = delete;
199 segregator(segregator &&) = default;
200 segregator &operator=(segregator const &) = delete;
201 segregator &operator=(segregator &&) = default;
202
203 blk_t allocate(size_t s) { return s <= Threshold ? small.allocate(s) : big.allocate(s); }
204 blk_t allocate_zero(size_t s) { return s <= Threshold ? small.allocate_zero(s) : big.allocate_zero(s); }
205
206 void deallocate(blk_t b) noexcept { return b.s <= Threshold ? small.deallocate(b) : big.deallocate(b); }
207 [[nodiscard]] bool owns(blk_t b) const noexcept { return small.owns(b) or big.owns(b); }
208 };

210 // ------------------------- dress allocator with leak_checking ----------------------------
211 template <typename A>
212 class leak_check : A {
213
214 long memory_used = 0;
215
216 public:
217 leak_check() = default;
218 leak_check(leak_check const &) = delete;
219 leak_check(leak_check &&) = default;
220 leak_check &operator=(leak_check const &) = delete;

222
223 ~leak_check() {
224 if (!empty()) {
225#ifndef NDEBUG
226 std::cerr << "Allocator : MEMORY LEAK of " << memory_used << " bytes\n";
227 std::abort();
228#endif
229 }
230 }
231
232 [[nodiscard]] bool empty() const { return (memory_used == 0); }
233
234 blk_t allocate(size_t s) {
235 blk_t b = A::allocate(s);
236 memory_used = memory_used + b.s;
237 // std::cerr<< "Allocating "<< b.s << "Total = "<< memory_used << "\n";
238 return b;
239 }
240
241 blk_t allocate_zero(size_t s) {
242 blk_t b = A::allocate_zero(s);
243 memory_used = memory_used + b.s;
244 // std::cerr<< "Allocating "<< b.s << "Total = "<< memory_used << "\n";
245 return b;
246 }
247
248 void deallocate(blk_t b) noexcept {
249 memory_used -= b.s;
250 // std::cerr<< "Deallocating "<< b.s << "Total = "<< memory_used << "\n";
251
252 if (memory_used < 0) {
253#ifndef NDEBUG
254 std::cerr << "Allocator : memory_used <0 : " << memory_used << " b.s = " << b.s << " b.ptr = " << (void *)b.ptr;
255 std::abort();
256#endif
257 }
258 A::deallocate(b);
259 }
260
261 [[nodiscard]] bool owns(blk_t b) const noexcept { return A::owns(b); }
262
263 [[nodiscard]] long get_memory_used() const noexcept { return memory_used; }
264 };
