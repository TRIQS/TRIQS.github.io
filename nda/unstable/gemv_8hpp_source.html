21 // make the generic version for non lapack types or more complex types
22 // largely suboptimal
23 template <typename A, typename B, typename Out>
24 void gemv_generic(typename A::value_type alpha, A const &a, B const &b, typename A::value_type beta, Out &c) {
25 EXPECTS(a.extent(1) == b.extent(0));
26 EXPECTS(a.extent(0) == c.extent(0));
27 c() = 0;
28 for (int i = 0; i < a.extent(0); ++i) {
29 for (int k = 0; k < a.extent(1); ++k) c(i) += alpha * a(i, k) * b(k);
30 c(i) += beta * c(i);
31 }
32 }
