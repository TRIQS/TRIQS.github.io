40 double norm(A const &x, double p = 2.0) {
41 // Scalar check: Can't move to template constraint, get_value_t not generically implemented
42 static_assert(Scalar<get_value_t<A>>, "norm only works for arrays with scalar values");
43
44 if (p == 2.0) [[likely]] {
45 if constexpr (MemoryArray<A>)
46 return std::sqrt(std::real(nda::blas::dotc(x, x)));
47 else
48 return norm(make_regular(x));
49 } else if (p == 1.0) {
50 return sum(abs(x));
51 } else if (p == 0.0) {
52 // return std::count_if(a.begin(), a.end(), [](S s) { return s != S{0}; }); Fails for nda::expr
53 long count = 0;
54 for (long i = 0; i < x.size(); ++i) {
55 if (x(i) != get_value_t<A>{0}) ++count;
56 }
57 return double(count);
58 } else if (p == std::numeric_limits<double>::infinity()) {
59 return max_element(abs(x));
60 } else if (p == -std::numeric_limits<double>::infinity()) {
61 return min_element(abs(x));
62 } else {
63 return std::pow(sum(pow(abs(x), p)), 1.0 / p);
64 }
65 }
