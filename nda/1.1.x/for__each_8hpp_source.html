44 // return the i th index in Strider
45 template <int R>
46 constexpr int index_from_stride_order(uint64_t StrideOrder, int i) {
47 if (StrideOrder == 0) return i; // default C order
48 auto stride_order = decode<R>(StrideOrder); // FIXME C++20
49 return stride_order[i];
50 }

54 template <int I, int R, uint64_t StaticExtents, std::integral Int = long>
55 long get_extent(std::array<Int, R> const &l) {
56 if constexpr (StaticExtents == 0)
57 return l[I]; // quick exit, no computation of
58 else {
59 constexpr auto static_extents = decode<R>(StaticExtents); // FIXME C++20
60 if constexpr (static_extents[I] == 0)
61 return l[I];
62 else
63 return static_extents[I];
64 }
65 }

69 template <int I, uint64_t StaticExtents, uint64_t StrideOrder, typename F, size_t R, std::integral Int = long>
70 FORCEINLINE void for_each_static_impl(std::array<Int, R> const &idx_lengths, F &&f) {
71 if constexpr (I == R)
72 f();
73 else {
74 static constexpr int J = detail::index_from_stride_order<R>(StrideOrder, I);
75 const long imax = detail::get_extent<J, R, StaticExtents>(idx_lengths);
76 for (long i = 0; i < imax; ++i) {
77 for_each_static_impl<I + 1, StaticExtents, StrideOrder>(
78 idx_lengths,
79 [ i, &f ](auto &&... x)
80// Great: clang and gcc want the lambda mutable and attribute in a different order !:
81#ifdef __clang__
82 __attribute__((always_inline)) mutable
83#else
84 mutable __attribute__((always_inline))
85#endif
86 { return f(i, x...); }); // mutable since f itself can be mutable !
87 }
88 }
89 }
