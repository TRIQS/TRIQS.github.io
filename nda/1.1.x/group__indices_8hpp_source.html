25 // -- is_partition_of_indices --
26 //
27 // checks if the groups are a partition of the indices [0,R[
28 // where R is the dimension of the array being regrouped
29 // it will throw (compiler stop) if index is out of bounds
30 template <size_t R, size_t... Rs>
31 constexpr bool is_partition_of_indices(std::array<int, Rs> const &... grps) {
32
33 // check we have exactly R indices ...
34 // FIXME in C++20, we can have constexpr string, hence have more informative error message.
35 auto Rstot = (Rs + ...);
36 if (Rstot < R) throw "Too few indices in the group. Should be the rank";
37 if (Rstot > R) throw "Too many indices in the group. Should be the rank";
38
39 // we go over each indices, in each group and count them
40 std::array<int, R> count = stdutil::make_initialized_array<R>(0); // FIXME : not necessary in C++20
41 std::array<int, R> correct_answer = stdutil::make_initialized_array<R>(1); // FIXME : not necessary in C++20
42
43 auto l = [&](auto &&grp) mutable -> void {
44 for (int u = 0; u < grp.size(); ++u) {
45 if (grp[u] < 0) throw "Negative index !";
46 if (grp[u] >= R) throw "Index larger than the rank !!";
47 count[grp[u]]++;
48 }
49 };
50 (l(grps), ...); // execute all lambdas
51 return (count == correct_answer);
52 }

56 // --- stride_order_of_grouped_idx_map ---
57 //
58 // Given a stride_order and some groups, returns the stride_order of the grouped idx_map
59 // compile time only
60 template <size_t R, size_t... Rs>
61 constexpr std::array<int, sizeof...(Rs)> stride_order_of_grouped_idx_map(std::array<int, R> const &stride_order,
62 std::array<int, Rs> const &... grps) {
63 // stride_order is permutation which by definition is such that
64 // stride_order[0] is the slowest index, stride_order[1] the next, stride_order[Rank-1] the fastest.
65
66 // We need to work with the inverse permutation mem_pos.
67 // For each index k, mem_pos[k] gives the position of index k in the memory layout, 0 for slowest to R-1 for fastest.
68 // e.g. so by definition mem_pos[ stride_order[0]] = 0, mem_pos[ stride_order[1]] = 1, etc...
69
70 // For each group, we look at the mem_pos of its indices.
71 // They must be consecutive to be regrouped.
72 // We select their min, it will give us the mem_pos of the regrouped view, which we will then invert to get the stride_order.
73
74 auto mem_pos = permutations::inverse(stride_order);
75
76 // Find the minimum memory position of the indices in this group
77 // Throw (compile time, i.e. stop compilation) if the indices are not consecutive in memory in this group
78 auto min_mem_pos = [&mem_pos](auto &&grp) {
79 // m = minimum, M = maximum
80 int m = R, M = 0;
81 for (int idx: grp) {
82 int v = mem_pos[idx];
83 if (v > M) M = v;
84 if (v < m) m = v;
85 }
86 bool idx_are_consecutive_in_memory = (M - m + 1 == grp.size());
87 if (!idx_are_consecutive_in_memory)
88 throw "Indices are not consecutive in memory"; // FIXME : can I have a better error message at compile time ??
89 return m;
90 };
91
92 // The Number of groups <-> Dimension of the returned idx_map.
93 constexpr int Ngrps = sizeof...(Rs);
94
95 // An array containing the minimal memory position for each group
96 std::array<int, Ngrps> min_mem_positions{min_mem_pos(grps)...};
97
98 // The problem is that they are not consecutive numbers, they run from 0 to R -1, not Ngrps -1
99 // We compress them back to [0, Ngrps[ by counting how many there are before each of them
100 std::array<int, Ngrps> mem_pos_out = stdutil::make_initialized_array<Ngrps>(0); // FIXME : not necessary in C++20
101 for (int u = 0; u < Ngrps; ++u) {
102 for (int i = 0; i < Ngrps; ++i) {
103 if (min_mem_positions[i] < min_mem_positions[u]) ++mem_pos_out[u];
104 }
105 }
106
107 // The new stride_order is the inverse of mem_pos_out
108 return permutations::inverse(mem_pos_out);
109 }

141 template <int Rank, uint64_t StaticExtents, uint64_t StrideOrder, layout_prop_e LayoutProp, typename... IdxGrps>
142 auto group_indices_layout(idx_map<Rank, StaticExtents, StrideOrder, LayoutProp> const &idxm, IdxGrps...) {
143
144 static_assert(LayoutProp == layout_prop_e::contiguous, "Not implemented for non contiguous arrays");
145
146 // decoded StrideOrder
147 static constexpr auto stride_order = idx_map<Rank, StaticExtents, StrideOrder, LayoutProp>::stride_order;
148
149 static_assert(detail::is_partition_of_indices<Rank>(IdxGrps::as_std_array...),
150 "The indices provided in the groups are not a partition of the indices of the array, i.e. [0, Rank[");
151
153 auto total_len_of_a_grp = [&idxm](auto &&grp) {
154 auto ll = std::accumulate(grp.begin(), grp.end(), 1, [&idxm](long l, long u) { return l * idxm.lengths()[u]; });
155 return ll;
156 };
157
158 auto min_stride_of_a_grp = [&idxm](auto &&grp) {
159 return std::accumulate(grp.begin(), grp.end(), idxm.size(), [&idxm](long s, long u) { return std::min(s, idxm.strides()[u]); });
160 };
161
162 static constexpr int new_rank = sizeof...(IdxGrps);
163 std::array<long, new_rank> new_extents{total_len_of_a_grp(IdxGrps::as_std_array)...};
164 std::array<long, new_rank> new_strides{min_stride_of_a_grp(IdxGrps::as_std_array)...};
165
166 // new layout using the new stride_order
167 using new_layout_t =
168 idx_map<new_rank, 0, encode(detail::stride_order_of_grouped_idx_map(stride_order, IdxGrps::as_std_array...)), layout_prop_e::contiguous>;
169
170 return new_layout_t{new_extents, new_strides};
171 }
