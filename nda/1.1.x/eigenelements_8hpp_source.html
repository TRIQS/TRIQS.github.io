20 template <typename M>
21 // dispatch the implementation of invoke for T = double or complex
22 auto _eigen_element_impl(M &&m, char compz) {
23
24 EXPECTS((not m.empty()));
25 EXPECTS(is_matrix_square(m, true));
26 EXPECTS(m.indexmap().is_contiguous());
27
28 int dim = m.extent(0);
29
30 using T = typename std::decay_t<M>::value_type;
31
32 array<double, 1> ev(dim);
33 int lwork = 64 * dim;
34 array<T, 1> work(lwork);
35 array<double, 1> work2(is_complex_v<T> ? lwork : 0);
36
37#if defined(__has_feature)
38#if __has_feature(memory_sanitizer)
39 work2 = 0;
40 work = 0;
41 ev = 0;
42#endif
43#endif
44
45 int info = 0;
46 if constexpr (not is_complex_v<T>) {
47 lapack::f77::syev(compz, 'U', dim, m.data(), dim, ev.data(), work.data(), lwork, info);
48 } else {
49 lapack::f77::heev(compz, 'U', dim, m.data(), dim, ev.data(), work.data(), lwork, work2.data(), info);
50 }
51 if (info) NDA_RUNTIME_ERROR << "Diagonalization error";
52 return ev;
53 }
