24 // ------- CallableWithLongs<A, R> ----------
25 // A can be called with R longs
26
27#ifndef __clang__
28 // clang does not have an implementation yet of "Lambda in unevaluated context"
29 // we make a workaround below
30 // clang-format off
31 template <typename A, int R>
32 concept CallableWithLongs = requires(A const &a) {
33 {
34 []<auto... Is>(std::index_sequence<Is...>, auto const &aa) -> decltype(aa(long(Is)...)) {return aa(long(Is)...);} // a lambda to call aa(Is...)
35 (std::make_index_sequence<R>{}, a) // apply it
36 };
37 // the expression a(long(Is)...) should compile, with a pack of R longs
38 // the concept says nothing about the return type of this function
39 // NB decltype is crucial for the concept to work, to use SFINAE.
40 // if not present, the concept will fail to compile for an A for which the a(Is...) it not well formed.
41 };
42// clang-format on
43#else
44 namespace detail {
45 template <auto... Is, typename A>
46 auto call_on_R_longs(std::index_sequence<Is...>, A const &a) -> decltype(a((long{Is})...)); // no impl needed
47 }
48
49 template <typename A, int R>
50 concept CallableWithLongs = requires(A const &a) {
51 {detail::call_on_R_longs(std::make_index_sequence<R>{}, a)};
52 };
53
54#endif

83
85 template <typename A>
86 concept Array = requires(A const &a) {
87
88 // A has a shape() which returns an array<long, R>
89 // its length is the rank, as deduced by get_rank
90 { a.shape() } -> StdArrayOfLong;
91
92 // Contract: Expects the size to be equal to the product of the elements in shape()
93 { a.size() } -> std::same_as<long>;
94
95 // a(0,0,0,0... R times) returns something, which is of type value_type by definition
96 requires CallableWithLongs<A, get_rank<A>>;
97};

100template <typename A>
101concept MemoryArray = Array<A> && requires(A &a) {
102
103 // We can acquire the pointer to the underlying data
104 { a.data() } -> std::same_as<std::conditional_t<std::is_const_v<A>, const get_value_t<A>, get_value_t<A>> *>;
105
106 // Exposes the memory stride for each dimension
107 { a.indexmap().strides() } -> StdArrayOfLong;
108};
