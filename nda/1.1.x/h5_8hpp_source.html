58 template <typename A>
59 h5::char_buf to_char_buf(A const &v) requires(is_regular_or_view_v<A>) {
60 static_assert(A::rank == 1, "H5 for array<string, N> for N>1 not implemented");
61 size_t s = 0;
62 for (auto &x : v) s = std::max(s, x.size() + 1);
63 auto len = v_t{size_t(v.size()), s};
64
65 std::vector<char> buf;
66 buf.resize(v.size() * s, 0x00);
67 size_t i = 0;
68 for (auto &x : v) {
69 strcpy(&buf[i * s], x.c_str());
70 ++i;
71 }
72 return {buf, len};
73 }

75 template <typename A>
76 void from_char_buf(h5::char_buf const &cb, A &v) requires(is_regular_or_view_v<A>) {
77 static_assert(A::rank == 1, "H5 for array<string, N> for N>1 not implemented");
78 v.resize(cb.lengths[0]);
79 auto len_string = cb.lengths[1];
80
81 size_t i = 0;
82 for (auto &x : v) {
83 x = "";
84 x.append(&cb.buffer[i * len_string]);
85 ++i;
86 }
87 }

91 template <typename A>
92 void h5_write(h5::group g, std::string const &name, A const &a) requires(is_regular_or_view_v<A>) {
93
94 // Properly treat arrays with non-standard memory layout
95 if constexpr (not std::decay_t<A>::layout_t::is_stride_order_C()) {
96 using h5_arr_t = nda::array<typename A::value_type, A::rank>;
97 auto a_c_layout = h5_arr_t{a.shape()};
98 a_c_layout() = a;
99 h5_write(g, name, a_c_layout);
100 return;
101 }
102
103 // first case array of string
104 if constexpr (std::is_same_v<typename A::value_type, std::string>) { // special case of string. Like vector of string
105
106 h5_write(g, name, h5_detail::to_char_buf(a));
107
108 } else if constexpr (is_scalar_v<typename A::value_type>) { // FIXME : register types as USER DEFINED hdf5 types
109
110 static constexpr bool is_complex = is_complex_v<typename A::value_type>;
111 h5_detail::write(g, name, h5::hdf5_type<get_value_t<A>>(), (void *)(a.data()), A::rank, is_complex, a.indexmap().lengths().data(),
112 a.indexmap().strides().data(), a.size());
113
114 } else { // generic unknown type to hdf5
115 auto g2 = g.create_group(name);
116 h5_write(g2, "shape", a.shape());
117 auto make_name = [](auto i0, auto... is) { return (std::to_string(i0) + ... + ("_" + std::to_string(is))); };
118 nda::for_each(a.shape(), [&](auto... is) { h5_write(g2, make_name(is...), a(is...)); });
119 }
120 }

122 template <typename A>
123 void h5_read(h5::group g, std::string const &name, A &a) requires(is_regular_or_view_v<A>) {
124
125 // If array is not C-strided, read into array with default layout and copy
126 if constexpr (not std::decay_t<A>::layout_t::is_stride_order_C()) {
127 static_assert(is_regular_v<A>, "Cannot read into an array_view to an array with non C-style memory layout");
128 using h5_arr_t = nda::array<typename A::value_type, A::rank>;
129 auto a_c_layout = h5_arr_t{};
130 h5_read(g, name, a_c_layout);
131 a.resize(a_c_layout.shape());
132 a() = a_c_layout;
133 return;
134 }
135
136 // Special case array<string>, store as char buffer
137 if constexpr (std::is_same_v<typename A::value_type, std::string>) {
138 h5::char_buf cb;
139 h5_read(g, name, cb);
140 h5_detail::from_char_buf(cb, a);
141
142 } else if constexpr (is_scalar_v<typename A::value_type>) { // FIXME : register types as USER DEFINED hdf5 types
143
144 static constexpr bool is_complex = is_complex_v<typename A::value_type>;
145
146 auto lt = h5::array_interface::get_h5_lengths_type(g, name);
147
148 // Allow to read non-complex data into array<complex>
149 if (is_complex && !lt.has_complex_attribute) {
150 array<double, A::rank> tmp;
151 h5_read(g, name, tmp);
152 a = tmp;
153 return;
154 }
155
156 int rank_in_file = lt.rank() - (is_complex ? 1 : 0);
157 if (rank_in_file != A::rank)
158 NDA_RUNTIME_ERROR << " h5 read of nda::array : incorrect rank. In file: " << rank_in_file << " In memory " << A::rank;
159 std::array<long, A::rank> L;
160 for (int u = 0; u < A::rank; ++u) L[u] = lt.lengths[u]; // NB : correct for complex
161
162 if constexpr (is_regular_v<A>) {
163 a.resize(L);
164 } else {
165 if (a.shape() != L)
166 NDA_RUNTIME_ERROR << "Error trying to read from an hdf5 file to a view. Dimension mismatch"
167 << "\n in file : " << L << "\n in view : " << a.shape();
168 }
169
170 h5::array_interface::h5_array_view v{h5::hdf5_type<get_value_t<A>>(), (void *)(a.data()), A::rank, is_complex};
171 for (int u = 0; u < A::rank; ++u) {
172 v.slab.count[u] = L[u];
173 v.L_tot[u] = L[u];
174 }
175 h5::array_interface::read(g, name, v, lt);
176
177 } else { // generic unknown type to hdf5
178 auto g2 = g.open_group(name);
179
180 // Reshape if necessary
181 std::array<long, A::rank> h5_shape;
182 h5_read(g2, "shape", h5_shape);
183 if (a.shape() != h5_shape) a.resize(h5_shape);
184
185 // Read using appropriate h5_read implementation
186 auto make_name = [](auto i0, auto... is) { return (std::to_string(i0) + ... + ("_" + std::to_string(is))); };
187 nda::for_each(a.shape(), [&](auto... is) { h5_read(g2, make_name(is...), a(is...)); });
188 }
189 }
