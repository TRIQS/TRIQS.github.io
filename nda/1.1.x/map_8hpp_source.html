38 // a lazy expression of f(A...) where f is the function to map and A the arrays
39 // e.g. f is sqrt(x) and there is one A, or f is min(x,y) and there are 2 A
40 template <typename F, typename... A>
41 struct expr_call {
42
43 F f;
44 std::tuple<const A...> a; // tuple of array (ref) on which f is applied
45
46 private: // FIXME C++20 lambda implementation details
47 template <size_t... Is, typename... Args>
48 [[gnu::always_inline]] [[nodiscard]] auto _call(std::index_sequence<Is...>, Args const &... args) const {
49 // a priori, we just need to return
50 // f(std::get<Is>(a)(args...)...)
51 // but if a(args) is a view, we can not always use f.
52 // It works if f is well written, i.e. it takes a temporary and forwards it, like
53 // f(auto &&x) { return ...}
54 // but if the user's function is badly written, like (auto const & x) { ...}
55 // which is quite natural for a simple function, then the temporary view will be dangling.
56 // So we take care of this case explicitly
57 if constexpr (is_regular_or_view_v<decltype(std::get<0>(a)(args...))>) { // we have views, same for all a...
58 return mapped<F>{f}(std::get<Is>(a)(args...)...);
59 } else {
60 return f(std::get<Is>(a)(args...)...);
61 }
62 }
63 template <size_t... Is, typename Args>
64 [[gnu::always_inline]] auto _call_bra(std::index_sequence<Is...>, Args const &args) const {
65 return f(std::get<Is>(a)[args]...);
66 }
67
68 public:
69 template <typename... Args>
70 auto operator()(Args const &... args) const {
71 return _call(std::make_index_sequence<sizeof...(A)>{}, args...);
72 }
73
74 // vector interface
75 template <typename Args>
76 auto operator[](Args &&args) const {
77 return _call_bra(std::make_index_sequence<sizeof...(A)>{}, args);
78 }
79
80 // FIXME copy needed for the && case only. Overload ?
81 [[nodiscard]] auto shape() const { return std::get<0>(a).shape(); }
82
83 [[nodiscard]] long size() const { return a.size(); }
84 };
