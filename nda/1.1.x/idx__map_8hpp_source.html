200 template <std::integral Int>
201 [[nodiscard]] static bool is_stride_order_valid(Int *lenptr, Int *strptr) {
202 auto dims_to_check = std::vector<int>{};
203 dims_to_check.reserve(Rank);
204 for (auto dim : stride_order)
205 if (lenptr[dim] > 1) dims_to_check.push_back(dim);
206
207 for (int n = 1; n < dims_to_check.size(); ++n)
208 if (strptr[dims_to_check[n - 1]] < strptr[dims_to_check[n]]) return false;
209
210 return true;
211 }
212
213 [[nodiscard]] bool is_stride_order_valid() const { return is_stride_order_valid(len.data(), str.data()); }

221 EXPECTS(is_stride_order_valid());
222 if constexpr (not layout_property_compatible(P, LayoutProp)) {
223 if constexpr (has_contiguous(LayoutProp)) {
224 EXPECTS_WITH_MESSAGE(
225 idxm.is_contiguous(),
226 "Failed check of contiguity. Constructing a contiguous layout from another layout which was not guaranteed to be contiguous at compile time. The check fails so your program is incorrect");
227 }
228 if constexpr (has_strided_1d(LayoutProp)) {
229 EXPECTS_WITH_MESSAGE(

231 "Failed check of quasi-contiguity (1d-strided). Constructing a contiguous layout from another layout which was not guaranteed to be quasi-contiguous at compile time. The check fails so your program is incorrect");

334 // call implementation
335 template <typename... Args, size_t... Is>
336 [[nodiscard]] FORCEINLINE long call_impl(std::index_sequence<Is...>, Args... args) const noexcept {
337 static constexpr int e_pos = ((std::is_same_v<Args, ellipsis> ? int(Is) + 1 : 0) + ...) - 1; // position ellipsis + 1 or 0
338
339 if constexpr (e_pos == -1) { // common case, no ellipsis
340 if constexpr (smallest_stride_is_one)
341 return (myget<true, Is>(args) + ...);
342 else
343 return ((args * std::get<Is>(str)) + ...);
344 } else {
345 // there is an empty ellipsis to skip
346 return (myget<smallest_stride_is_one, (Is < e_pos ? Is : Is - 1)>(args) + ...);
347 }
348 }

399 template <uint64_t Permutation>
400 auto transpose() const {
401 // Denoting this as A, an indexmap, calling it returns the linear index given by
402 //
403 // A(i_k) = sum_k i_k * S[k] (1)
404 //
405 // where S[k] denotes the strides.
406 //
407 // 1- S' : strides of A'
408 // A'(i_k) = sum_k i_{P[k]} * S[k] = sum_k i_k * S[P{^-1}[k]]
409 // so
410 // S'[k] = S[P{^-1}[k]] (2)
411 // i.e. apply (inverse(P), S) or apply_inverse directly.
412 //
413 // 2- L' : lengths of A'
414 // if L[k] is the k-th length, then because of the definition of A', i.e. A'(i_k) = A(i_{P[k]})
415 // i_q in the lhs A is at position q' such that P[q'] = q (A'(i0 i1 i2...) = A( i_P0 i_P1 i_P2....)
416 // hence L'[q] = L[q'] = L[P^{-1}[q]]
417 // same for static length
418 //
419 // 3- stride_order: denoted in this paragraph as Q (and Q' for A').
420 // by definition Q is a permutation such that Q[0] is the slowest index, Q[Rank -1] the fastest
421 // hence S[Q[k]] is a strictly decreasing sequence (as checked by strides_compatible_to_stride_order)
422 // we want therefore Q' the permutation that will sort the S', i.e.
423 // S'[Q'[k]] = S[Q[k]]
424 // using (2), we have S[P{^-1}[Q'[k]]] = S[Q[k]]
425 // so the permutation Q' is such that P{^-1}Q' = Q or Q' = PQ (as permutation product/composition).
426 // NB : Q and P are permutations, so the operation must be a composition, not an apply (apply applies a P to any set, like L, S, not only a permutation)
427 // even though they are all std::array in the code ...
428 //
429 static constexpr std::array<int, Rank> permu = decode<Rank>(Permutation);
430 static constexpr std::array<int, Rank> new_stride_order = permutations::compose(permu, stride_order);
431 static constexpr std::array<int, Rank> new_static_extents = permutations::apply_inverse(permu, static_extents);
432
433 // FIXME
434 // Compute the new layout_prop of the new view
435 // NB : strided_1d property is preserved, but smallest_stride_is_one is not
436 static constexpr layout_prop_e new_layout_prop =
437 layout_prop_e::none; // BUT FIX (has_strided_1d(layout_prop) ? layout_prop_e::strided_1d : layout_prop_e::none);
438
439 return idx_map<Rank, encode(new_static_extents), encode(new_stride_order), new_layout_prop>{permutations::apply_inverse(permu, lengths()),
440 permutations::apply_inverse(permu, strides())};
441 }
