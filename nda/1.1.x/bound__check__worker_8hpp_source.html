19 // A worker that checks all arguments and gather potential errors in error_code
20 struct bound_check_worker {
21 long const *lengths{}; // length of input slice
22 uint32_t error_code = 0;
23 int ellipsis_loss = 0;
24 int N = 0;
25
26 void f(long key) {
27 bool pb = ((key < 0) or (key >= lengths[N]));
28 if (pb) error_code += 1ul << N; // binary code
29 ++N;
30 }
31
32 void f(range::all_t) { ++N; }
33 void f(range) { ++N; }
34 void f(ellipsis) { N += ellipsis_loss + 1; }
35
36 void g(std::stringstream &fs, long key) {
37 if (error_code & (1ull << N)) fs << "argument " << N << " = " << key << " is not within [0," << lengths[N] << "[\n";
38 N++;
39 }
40 void g(std::stringstream &, range) { ++N; }
41 void g(std::stringstream &, range::all_t) { ++N; }
42 void g(std::stringstream &, ellipsis) { N += ellipsis_loss + 1; }
43 };
44
45 template <typename... Args>
46 void assert_in_bounds(int rank, long const *lengths, Args const &... args) {
47 bound_check_worker w{lengths};
48 w.ellipsis_loss = rank - sizeof...(Args); // len of ellipsis : how many ranges are missing
49 (w.f(args), ...); // folding with , operator ...
50 if (!w.error_code) return;
51 w.N = 0;
52 std::stringstream fs;
53 (w.g(fs, args), ...); // folding with , operator ...
54 //std::cerr << " key out of domain \n" + fs.str()<<std::endl;
55 //EXPECTS_WITH_MESSAGE(false, "Stopping");
56 throw std::runtime_error(" key out of domain \n" + fs.str());
57 }
