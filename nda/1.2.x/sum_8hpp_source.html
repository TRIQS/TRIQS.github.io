38 // sum a function f on a domain D, using a simple foreach
39 template <typename F, typename D>
40 auto sum_f_domain_impl(F const &f, D const &d)
41 requires(not is_clef_expression<F, D>)
42 {
43 auto it = d.begin();
44 auto ite = d.end();
45 if (it == ite) NDA_RUNTIME_ERROR << "Sum over an empty domain";
46 auto res = make_regular(f(*it));
47 ++it;
48 for (; it != ite; ++it) res = res + f(*it);
49 return res;
50 }
51
52 CLEF_MAKE_FNT_LAZY(sum_f_domain_impl);
53
54 // sum( expression, i = domain)
55 template <typename Expr, int N, typename D>
56 decltype(auto) sum(Expr const &f, clef::pair<N, D> const &d) {
57 return sum_f_domain_impl(make_function(f, clef::placeholder<N>()), d.rhs);
58 }
59 // warning : danger here : if the d is a temporary, the domain MUST be moved in case the Expr
60 // is still lazy after eval, or we will obtain a dangling reference.
61 template <typename Expr, int N, typename D>
62 decltype(auto) sum(Expr const &f, clef::pair<N, D> &&d) {
63 return sum_f_domain_impl(make_function(f, clef::placeholder<N>()), std::move(d.rhs));
64 }

66 // two or more indices : sum recursively
67 template <typename Expr, typename A0, typename A1, typename... A>
68 auto sum(Expr const &f, A0 &&a0, A1 &&a1, A &&...a) {
69 return sum(sum(f, std::forward<A0>(a0)), std::forward<A1>(a1), std::forward<A>(a)...);
