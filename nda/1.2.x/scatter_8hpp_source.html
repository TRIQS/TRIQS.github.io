24template <nda::Array A>
25struct mpi::lazy<mpi::tag::scatter, A> {
26
27 using value_type = typename std::decay_t<A>::value_type;
28 using const_view_type = decltype(std::declval<const A>()());
29
30 const_view_type rhs; // the rhs array
31 mpi::communicator c; // mpi comm
32 const int root; //
33 const bool all;
34
36 [[nodiscard]] auto shape() const {
37 auto dims = rhs.shape();
38 long slow_size = dims[0];
39 mpi::broadcast(slow_size, c, root);
40 dims[0] = mpi::chunk_length(slow_size, c.size(), c.rank());
41 return dims;
42 }
43
45 template <nda::Array T>
46 void invoke(T &&target) const {
47 if (not target.is_contiguous()) NDA_RUNTIME_ERROR << "mpi operations require contiguous target.data() to be contiguous";
48
49 static_assert(std::decay_t<A>::layout_t::stride_order_encoded == std::decay_t<T>::layout_t::stride_order_encoded,
50 "Array types for rhs and target have incompatible stride order");
51
52 if (not mpi::has_env) {
53 target = rhs;
54 return;
55 }
56
57 auto sha = shape(); // WARNING : Keep this out of any if condition (shape USES MPI) !
58 resize_or_check_if_view(target, sha);
59
60 auto slow_size = rhs.extent(0);
61 auto slow_stride = rhs.indexmap().strides()[0];
62 auto sendcounts = std::vector<int>(c.size());
63 auto displs = std::vector<int>(c.size() + 1, 0);
64 int recvcount = mpi::chunk_length(slow_size, c.size(), c.rank()) * slow_stride;
65 auto D = mpi::mpi_type<value_type>::get();
66
67 for (int r = 0; r < c.size(); ++r) {
68 sendcounts[r] = mpi::chunk_length(slow_size, c.size(), r) * slow_stride;
69 displs[r + 1] = sendcounts[r] + displs[r];
70 }
71
72 MPI_Scatterv((void *)rhs.data(), &sendcounts[0], &displs[0], D, (void *)target.data(), recvcount, D, root, c.get());
73 }
74};
