229 template <typename T, int R, typename L, char A, typename AP, typename OP>
230 void rebind(basic_array_view<T, R, L, A, AP, OP> v) noexcept {
231 static_assert(R == Rank, "One can only rebind a view to a view of same rank");
232 static_assert(std::is_same_v<std::remove_const_t<T>, std::remove_const_t<ValueType>>, "Type must be the same, except maybe const");
233 static constexpr bool same_type = std::is_same_v<T, ValueType>;
234
235 static_assert(same_type or is_const, "One can not rebind a view of T onto a view of const T. It would discard the const qualifier");
236 if constexpr (same_type) {
237 // FIXME Error message in layout error !
238 lay = v.lay;
239 sto = v.sto;
240 } else if constexpr (is_const) {
241 // the last if is always trivially true BUT in case of an error in the static_assert above,
242 // it improves the error message by not compiling the = afterwards
243 lay = layout_t{v.indexmap()};
244 sto = storage_t{v.storage()};
245 }
246 }
