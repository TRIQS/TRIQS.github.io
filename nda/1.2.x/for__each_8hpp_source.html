48 // return the i th index in Strider
49 template <int R>
50 constexpr int index_from_stride_order(uint64_t StrideOrder, int i) {
51 if (StrideOrder == 0) return i; // default C order
52 auto stride_order = decode<R>(StrideOrder); // FIXME C++20
53 return stride_order[i];
54 }

58 template <int I, int R, uint64_t StaticExtents, std::integral Int = long>
59 long get_extent(std::array<Int, R> const &l) {
60 if constexpr (StaticExtents == 0)
61 return l[I]; // quick exit, no computation of
62 else {
63 constexpr auto static_extents = decode<R>(StaticExtents); // FIXME C++20
64 if constexpr (static_extents[I] == 0)
65 return l[I];
66 else
67 return static_extents[I];
68 }

73 template <int I, uint64_t StaticExtents, uint64_t StrideOrder, typename F, size_t R, std::integral Int = long>
74 FORCEINLINE void for_each_static_impl(std::array<Int, R> const &idx_lengths, F &&f) {
75 if constexpr (I == R)
76 f();
77 else {
78 static constexpr int J = details::index_from_stride_order<R>(StrideOrder, I);
79 const long imax = details::get_extent<J, R, StaticExtents>(idx_lengths);
80 for (long i = 0; i < imax; ++i) {
81 for_each_static_impl<I + 1, StaticExtents, StrideOrder>(
82 idx_lengths,
83 [ i, &f ](auto &&...x)
84// Great: clang and gcc want the lambda mutable and attribute in a different order !:
85#ifdef __clang__
86 __attribute__((always_inline)) mutable
87#else
88 mutable __attribute__((always_inline))
89#endif
90 { return f(i, x...); }); // mutable since f itself can be mutable !
91 }
92 }
93 }
