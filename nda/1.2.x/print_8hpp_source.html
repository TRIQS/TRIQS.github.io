46 // array
47 template <typename A>
48 std::ostream &operator<<(std::ostream &out, A const &a)
49 requires(is_regular_or_view_v<A>)
50 {
51
52 if constexpr (A::rank == 1) {
53 out << "[";
54 auto const &len = a.indexmap().lengths();
55 for (size_t i = 0; i < len[0]; ++i) out << (i > 0 ? "," : "") << a(i);
56 out << "]";
57 }
58
59 if constexpr (A::rank == 2) {
60 auto const &len = a.indexmap().lengths();
61 out << "\n[";
62 for (size_t i = 0; i < len[0]; ++i) {
63 out << (i == 0 ? "[" : " [");
64 for (size_t j = 0; j < len[1]; ++j) out << (j > 0 ? "," : "") << a(i, j);
65 out << "]" << (i == len[0] - 1 ? "" : "\n");
66 }
67 out << "]";
68 }
69
70 // FIXME : not very pretty, do better here, but that was the arrays way
71 if constexpr (A::rank > 2) {
72 out << "[";
73 for (bool first = true; auto &v : a) {
74 out << (first ? "" : ",") << v;
75 first = false;
76 }
77 out << "]";
78 }
79
80 return out;
81 }
