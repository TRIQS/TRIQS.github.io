24 // A general function that maps any transform of the layout onto the basic_array_view
25 template <MemoryArray A, typename NewLayoutType>
26 auto map_layout_transform(A &&a, NewLayoutType const &new_layout) {
27 using A_t = std::remove_reference_t<A>;
28 using layout_policy = typename details::layout_to_policy<NewLayoutType>::type;
29 static constexpr auto algebra = (NewLayoutType::rank() == get_rank<A> ? get_algebra<A> : 'A');
30 if constexpr (is_regular_v<A> and !std::is_reference_v<A>) { // basic_array rvalue
31 using container_policy_t = typename A_t::container_policy_t;
32 return basic_array<typename A_t::value_type, NewLayoutType::rank(), layout_policy, algebra, container_policy_t>{new_layout,
33 std::forward<A>(a).storage()};
34 } else {
35 using value_t = std::conditional_t<std::is_const_v<A_t>, const typename A_t::value_type, typename A_t::value_type>;
36 using accessor_policy = typename get_view_t<A>::accessor_policy_t;
37 using owning_policy = typename get_view_t<A>::owning_policy_t;
38 return basic_array_view<value_t, NewLayoutType::rank(), layout_policy, algebra, accessor_policy, owning_policy>{new_layout, a.storage()};
39 }
40 }

44 template <MemoryArray A, std::integral Int, auto newRank>
45 auto reshape(A &&a, std::array<Int, newRank> const &new_shape) {
46 using layout_t = typename std::decay_t<A>::layout_policy_t::template mapping<newRank>;
47 EXPECTS_WITH_MESSAGE(a.size() == (std::accumulate(new_shape.cbegin(), new_shape.cend(), Int{1}, std::multiplies<>{})),
48 "Reshape : the new shape has a incorrect number of elements");
49 EXPECTS_WITH_MESSAGE(a.indexmap().is_contiguous(), "reshape only works with contiguous data");
50 return map_layout_transform(std::forward<A>(a), layout_t{stdutil::make_std_array<long>(new_shape)});
51 }

89 template <typename A>
90 auto transpose(A &&a)
91 requires(MemoryMatrix<A> or is_instantiation_of_v<expr_call, A>)
92 {
93 if constexpr (MemoryMatrix<A>) {
94 return permuted_indices_view<encode(std::array{1, 0})>(std::forward<A>(a));
95 } else { // expr_call
96 static_assert(std::tuple_size_v<decltype(a.a)> == 1, "Cannot transpose expr_call with more than one array argument");
97 return map(a.f)(transpose(std::get<0>(std::forward<A>(a).a)));
98 }

123 template <int N, auto R>
124 constexpr std::array<int, R + N> complete_stride_order_with_fast(std::array<int, R> const &a) {
125 auto r = stdutil::make_initialized_array<R + N>(0);
126 for (int i = 0; i < R; ++i) r[i] = a[i];
127 for (int i = 0; i < N; ++i) r[R + i] = R + i;
128 return r;
129 }

132 // Take an array or view and add N dimensions of size 1 in the fastest indices
133 template <int N, typename A>
134 auto reinterpret_add_fast_dims_of_size_one(A &&a)
135 requires(nda::is_regular_or_view_v<A>)
136 {
137
138 auto const &lay = a.indexmap();
139 using lay_t = std::decay_t<decltype(lay)>;
140
141 static constexpr uint64_t new_stride_order_encoded = encode(impl::complete_stride_order_with_fast<N>(lay_t::stride_order));
142 // (lay_t::stride_order_encoded == 0 ? 0 : encode(impl::complete_stride_order_with_fast<N>(lay_t::stride_order)));
143
144 static constexpr uint64_t new_static_extents_encoded = encode(stdutil::join(lay_t::static_extents, stdutil::make_initialized_array<N>(0)));
145 // (lay_t::static_extents_encoded == 0 ? 0 : encode(stdutil::join(lay_t::static_extents, stdutil::make_initialized_array<N>(0))));
146
147 using new_lay_t = idx_map<get_rank<A> + N, new_static_extents_encoded, new_stride_order_encoded, lay_t::layout_prop>;
148
149 auto shap1111 = stdutil::make_initialized_array<N>(1l);
150 return map_layout_transform(std::forward<A>(a), new_lay_t{stdutil::join(lay.lengths(), shap1111), stdutil::join(lay.strides(), shap1111)});
151 }
