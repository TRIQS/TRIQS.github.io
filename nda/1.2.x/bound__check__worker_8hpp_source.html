23 // -------------------- bound_check_worker ---------------------

25 // A worker that checks all arguments and gather potential errors in error_code
26 struct bound_check_worker {
27 long const *lengths{}; // length of input slice
28 uint32_t error_code = 0;
29 int ellipsis_loss = 0;
30 int N = 0;
31
32 void f(long key) {
33 if ((key < 0) or (key >= lengths[N])) error_code += 1ul << N; // binary code
34 ++N;
35 }
36
37 void f(range::all_t) { ++N; }
38 void f(range const &r) {
39 if (r.size() > 0) {
40 auto first_idx = r.first();
41 auto last_idx = first_idx + (r.size() - 1) * r.step();
42 if (first_idx < 0 or first_idx >= lengths[N] or last_idx < 0 or last_idx >= lengths[N]) error_code += 1ul << N;
43 }
44 ++N;
45 }
46 void f(ellipsis) { N += ellipsis_loss + 1; }
47
48 void g(std::stringstream &fs, long key) {
49 if (error_code & (1ull << N)) fs << "argument " << N << " = " << key << " is not within [0," << lengths[N] << "[\n";
50 N++;
51 }
52 void g(std::stringstream &, range) { ++N; }
53 void g(std::stringstream &, range::all_t) { ++N; }
54 void g(std::stringstream &, ellipsis) { N += ellipsis_loss + 1; }
55 };

57 template <typename... Args>
58 void assert_in_bounds(int rank, long const *lengths, Args const &...args) {
59 bound_check_worker w{lengths};
60 w.ellipsis_loss = rank - sizeof...(Args); // len of ellipsis : how many ranges are missing
61 (w.f(args), ...); // folding with , operator ...
62 if (!w.error_code) return;
63 w.N = 0;
64 std::stringstream fs;
65 (w.g(fs, args), ...); // folding with , operator ...
66 //std::cerr << " key out of domain \n" + fs.str()<<std::endl;
67 //EXPECTS_WITH_MESSAGE(false, "Stopping");
68 throw std::runtime_error(" key out of domain \n" + fs.str());
69 }
