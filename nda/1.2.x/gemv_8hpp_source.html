27 void gemv_generic(get_value_t<A> alpha, A const &a, B const &b, get_value_t<A> beta, Out &c) {
28 EXPECTS(a.extent(1) == b.extent(0));
29 EXPECTS(a.extent(0) == c.extent(0));
30 c() = 0;
31 for (int i = 0; i < a.extent(0); ++i) {
32 for (int k = 0; k < a.extent(1); ++k) c(i) += alpha * a(i, k) * b(k);
33 c(i) += beta * c(i);
34 }
35 }
