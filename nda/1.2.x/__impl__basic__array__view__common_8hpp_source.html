122// I keep this public for the call of gf, which has to be reinterpreted as matrix
123// I can construct a matrix at once. Of course, the optimizer may eliminate the copy of handle and idx_map
124// but I prefer to keep it shorter here.
125// FIXME : consider to make this private, and use a make_matrix (...) on the resut in gf operator()

128template <char ResultAlgebra, bool SelfIsRvalue, typename Self, typename... T>
129FORCEINLINE static decltype(auto) call(Self &&self, T const &...x) noexcept(has_no_boundcheck) {
130
131 using r_v_t = std::conditional_t<std::is_const_v<std::remove_reference_t<Self>>, ValueType const, ValueType>;
132
133 if constexpr (clef::is_any_lazy<T...>) return clef::make_expr_call(std::forward<Self>(self), x...);
134
135 // () returns a full view
136 else if constexpr (sizeof...(T) == 0) {
137 return basic_array_view<r_v_t, Rank, LayoutPolicy, Algebra, AccessorPolicy, OwningPolicy>{self.lay, self.sto};
138 }
139
140 else {
141 static_assert(((layout_t::template argument_is_allowed_for_call_or_slice<T> + ...) > 0),
142 "Slice arguments must be convertible to range, Ellipsis, or long (or string if the layout permits it)");
143
144 static constexpr int n_args_long = (layout_t::template argument_is_allowed_for_call<T> + ...);
145
146 // case 1 : all arguments are long, we simply compute the offset
147 if constexpr (n_args_long == rank) { // no range, simply compute the linear position. There may be an ellipsis, but it is of zero length !
148 long offset = self.lay(x...); // compute the offset
149 if constexpr (is_view or not SelfIsRvalue) //
150 return AccessorPolicy::template accessor<ValueType>::access(self.sto.data(),
151 offset); // We return a REFERENCE here. Ok since underlying array is still alive
152 else //
153 return ValueType{self.sto[offset]}; // We return a VALUE here, the array is about be destroyed.
154 }
155 // case 2 : we have to make a slice
156 else {
157 // Static rank
158 auto const [offset, idxm] = self.lay.slice(x...);
159
160 using r_layout_t = std::decay_t<decltype(idxm)>;
161
162 static constexpr char newAlgebra = (ResultAlgebra == 'M' and (r_layout_t::rank() == 1) ? 'V' : ResultAlgebra);
163
164 using r_view_t =
165 // FIXME basic_array_view<r_v_t, r_layout_t::rank(),
166 basic_array_view<ValueType, r_layout_t::rank(), typename details::layout_to_policy<r_layout_t>::type, newAlgebra, AccessorPolicy,
167 OwningPolicy>;
168
169 return r_view_t{std::move(idxm), {self.sto, offset}};
170 }
171 }

249template <typename Iterator>
250[[nodiscard]] auto make_iterator(bool at_end) const noexcept {
251 if constexpr (iterator_rank == Rank) {
252 if constexpr (layout_t::is_stride_order_C())
253 return Iterator{indexmap().lengths(), indexmap().strides(), sto.data(), at_end};
254 else
255 // general case. In C order, no need to spend time applying the identity permutation
256 // the new length used by the iterator is length[ stride_order[0]], length[ stride_order[1]], ...
257 // since stride_order[0] is the slowest, it will traverse the memory in sequential order
258 return Iterator{nda::permutations::apply(layout_t::stride_order, indexmap().lengths()),
259 nda::permutations::apply(layout_t::stride_order, indexmap().strides()), sto.data(), at_end};
260 } else // 1d iteration
261 return Iterator{std::array<long, 1>{size()}, std::array<long, 1>{indexmap().min_stride()}, sto.data(), at_end};
262}

330template <typename RHS>
331void assign_from_ndarray(RHS const &rhs) { // FIXME noexcept {
332
333#ifdef NDA_ENFORCE_BOUNDCHECK
334 if (this->shape() != rhs.shape())
335 NDA_RUNTIME_ERROR << "Size mismatch:"
336 << "\n LHS.shape() = " << this->shape() << "\n RHS.shape() = " << rhs.shape();
337#endif
338
339 static_assert(std::is_assignable_v<value_type &, get_value_t<RHS>>, "Assignment impossible for the type of RHS into the type of LHS");
340
341 static constexpr bool both_in_memory = MemoryArray<self_t> and MemoryArray<RHS>;
342 static constexpr bool same_stride_order = get_layout_info<self_t>.stride_order == get_layout_info<RHS>.stride_order;
343
344 if constexpr (both_in_memory and same_stride_order) {
345 if (rhs.empty()) return;
346
347 static constexpr bool both_1d_strided = has_layout_strided_1d<self_t> and has_layout_strided_1d<RHS>;
348
349 if constexpr (mem::on_host<self_t, RHS> and both_1d_strided) { // -> vectorizable host copy
350 for (long i = 0; i < size(); ++i) (*this)(_linear_index_t{i}) = rhs(_linear_index_t{i});
351 return;
352 } else if constexpr (!mem::on_host<self_t, RHS> and have_same_value_type_v<self_t, RHS>) {
353 // Check for block-layout and use mem::memcpy2D if possible
354 auto bl_layout_dst = get_block_layout(*this);
355 auto bl_layout_src = get_block_layout(rhs);
356 if (bl_layout_dst && bl_layout_src) {
357 auto [n_bl_dst, bl_size_dst, bl_str_dst] = *bl_layout_dst;
358 auto [n_bl_src, bl_size_src, bl_str_src] = *bl_layout_src;
359
360 if (n_bl_dst * bl_size_dst != n_bl_src * bl_size_src) NDA_RUNTIME_ERROR << "Incompatible block sizes in assign_from_ndarray";
361 // If either destination or source consist of a single block we can chunk it up to make the layouts compatible
362 if (n_bl_dst == 1 && n_bl_src > 1) {
363 n_bl_dst = n_bl_src;
364 bl_size_dst /= n_bl_src;
365 bl_str_dst = bl_size_dst;
366 }
367 if (n_bl_src == 1 && n_bl_dst > 1) {
368 n_bl_src = n_bl_dst;
369 bl_size_src /= n_bl_dst;
370 bl_str_src = bl_size_src;
371 }
372
373 // Copy only if block-layouts are compatible, otherwise continue to fallback
374 if (n_bl_dst == n_bl_src && bl_size_dst == bl_size_src) {
375 mem::memcpy2D<mem::get_addr_space<self_t>, mem::get_addr_space<RHS>>((void *)data(), bl_str_dst * sizeof(value_type), (void *)rhs.data(),
376 bl_str_src * sizeof(value_type), bl_size_src * sizeof(value_type),
377 n_bl_src);
378 return;
379 }
380 }
381 }
382 }
383 if constexpr (mem::on_device<self_t> || mem::on_device<RHS>)
384 NDA_RUNTIME_ERROR << "Fallback to elementwise assignment not implemented for arrays on the GPU";
385 // Fallback to elementwise assignment
386 auto l = [this, &rhs](auto const &...args) { (*this)(args...) = rhs(args...); };
387 nda::for_each(shape(), l);
388}

392template <typename Scalar>
393void fill_with_scalar(Scalar const &scalar) noexcept {
394 // we make a special implementation if the array is 1d strided or contiguous
395 if constexpr (has_layout_strided_1d<self_t>) { // possibly contiguous
396 const long L = size();
397 auto *__restrict const p = data(); // no alias possible here !
398 if constexpr (has_contiguous_layout<self_t>) {
399 for (long i = 0; i < L; ++i) p[i] = scalar;
400 } else {
401 const long stri = indexmap().min_stride();
402 const long Lstri = L * stri;
403 for (long i = 0; i < Lstri; i += stri) p[i] = scalar;
404 }
405 } else {
406 for (auto &x : *this) x = scalar;
407 }
408}
