31 // Check if a type is a conjugate array expression
32 template <typename A>
33 static constexpr bool is_conj_array_expr = false;
34 template <MemoryArray A>
35 static constexpr bool is_conj_array_expr<expr_call<conj_f, A>> = true;
36 template <typename A>
37 requires(!std::is_same_v<A, std::remove_cvref_t<A>>)
38 static constexpr bool is_conj_array_expr<A> = is_conj_array_expr<std::remove_cvref_t<A>>;

40 // ==== Layout Checks (Fortran/C) for both MemoryMatrix and conj(MemoryMatrix)
41
42 template <Array A>
43 requires(MemoryArray<A> or is_conj_array_expr<A>)
44 static constexpr bool has_F_layout = []() {
45 if constexpr (is_conj_array_expr<A>)
46 return has_F_layout<decltype(std::get<0>(std::declval<A>().a))>;
47 else
48 return std::remove_cvref_t<A>::is_stride_order_Fortran();
49 }();
50
51 template <Array A>
52 requires(MemoryArray<A> or is_conj_array_expr<A>)
53 static constexpr bool has_C_layout = []() {
54 if constexpr (is_conj_array_expr<A>)
55 return has_C_layout<decltype(std::get<0>(std::declval<A>().a))>;
56 else
57 return std::remove_cvref_t<A>::is_stride_order_C();
58 }();

60 // Determine the blas matrix operation tag ('N','T','C') based on the bools for conjugation and transposition
61 template <bool conj, bool transpose>
62 const char get_op = []() {
63 static_assert(!(conj and not transpose), "Cannot use conjugate of a matrix in blas operations. Please perform operation before call");
64 if constexpr (conj and transpose)
65 return 'C';
66 else if constexpr (transpose)
67 return 'T';
68 else // !conj and !transpose
69 return 'N';
70 }();

72 // LDA in lapack jargon
73 template <MemoryMatrix A>
74 int get_ld(A const &a) {
75 return a.indexmap().strides()[has_F_layout<A> ? 1 : 0];
76 }
77
78 // column number in lapack jargon
79 template <MemoryMatrix A>
80 int get_ncols(A const &a) {
81 return a.shape()[has_F_layout<A> ? 1 : 0];
82 }
