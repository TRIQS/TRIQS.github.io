159 template <ArrayOfRank<Rank> A>
160 basic_array(A const &a) noexcept //
161 requires(HasValueTypeConstructibleFrom<A, value_type>)
162
163 : lay(a.shape()), sto{lay.size(), mem::do_not_initialize} {
164 static_assert(std::is_constructible_v<value_type, get_value_t<A>>,
165 "Can not construct the array. ValueType can not be constructed from the value_type of the argument");
166 if constexpr (std::is_trivial_v<ValueType> or is_complex_v<ValueType>) {
167 // simple type. the initialization was not necessary anyway.
168 // we use the assign, including the optimization (1d strided, contiguous) possibly
169 assign_from_ndarray(a);
170 } else {
171 // in particular ValueType may or may not be default constructible
172 // so we do not init memory, and make the placement new now, directly with the value returned by a
173 nda::for_each(lay.lengths(), [&](auto const &...is) { new (sto.data() + lay(is...)) ValueType{a(is...)}; });
174 }
175 }

191 template <typename L>
192 static auto shape_from_init_list(std::initializer_list<L> const &l) noexcept {
193 const auto [min, max] =
194 std::minmax_element(std::begin(l), std::end(l), [](auto &&x, auto &&y) { return shape_from_init_list(x) == shape_from_init_list(y); });
195 EXPECTS_WITH_MESSAGE(shape_from_init_list(*min) == shape_from_init_list(*max), "initializer list not rectangular !");
196 return stdutil::front_append(shape_from_init_list(*max), long(l.size()));
197 }

201 basic_array(std::initializer_list<ValueType> const &l) noexcept //
202 requires(Rank == 1)
203 : lay(std::array<long, 1>{long(l.size())}), sto{lay.size(), mem::do_not_initialize} {
204 long i = 0;
205 // We can not assume that ValueType is default constructible. As before, we do not initialize,
206 // and use placement new
207 // https://godbolt.org/z/Lwic2o. Same code as = for basic type
208 // Alternative : if constexpr (std::is_trivial_v<ValueType> or is_complex<ValueType>::value) for (auto const &x : l) *(sto.data() + lay(i++)) = x;
209 for (auto const &x : l) { new (sto.data() + lay(i++)) ValueType{x}; }
210 }

213 basic_array(std::initializer_list<std::initializer_list<ValueType>> const &l2) noexcept //
214 requires(Rank == 2)
215 : lay(shape_from_init_list(l2)), sto{lay.size(), mem::do_not_initialize} {
216 long i = 0, j = 0;
217 for (auto const &l1 : l2) {
218 for (auto const &x : l1) { new (sto.data() + lay(i, j++)) ValueType{x}; } // cf dim1
219 j = 0;
220 ++i;
221 }
222 }

225 basic_array(std::initializer_list<std::initializer_list<std::initializer_list<ValueType>>> const &l3) noexcept //
226 requires(Rank == 3)
227 : lay(shape_from_init_list(l3)), sto{lay.size(), mem::do_not_initialize} {
228 long i = 0, j = 0, k = 0;
229 static_assert(Rank == 3, "?");
230 for (auto const &l2 : l3) {
231 for (auto const &l1 : l2) {
232 for (auto const &x : l1) { new (sto.data() + lay(i, j, k++)) ValueType{x}; } // cf dim1
233 k = 0;
234 ++j;
235 }
236 j = 0;
237 ++i;
238 }

310 template <std::integral Int = long>
311 static basic_array rand(std::array<Int, Rank> const &shape)
312 requires(std::is_floating_point_v<ValueType> or nda::is_complex_v<ValueType>)
313 {
314
315 auto static gen = std::mt19937(std::random_device{}());
316 auto static dist = std::uniform_real_distribution<>(0.0, 1.0);
317
318 auto res = basic_array{shape};
319 if constexpr (nda::is_complex_v<ValueType>)
320 for (auto &x : res) x = dist(gen) + 1i * dist(gen);
321 else
322 for (auto &x : res) x = dist(gen);
323
324 return res;
325 }
