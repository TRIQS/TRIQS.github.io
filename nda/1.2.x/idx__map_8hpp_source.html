208 template <std::integral Int>
209 [[nodiscard]] static bool is_stride_order_valid(Int *lenptr, Int *strptr) {
210 auto dims_to_check = std::vector<int>{};
211 dims_to_check.reserve(Rank);
212 for (auto dim : stride_order)
213 if (lenptr[dim] > 1) dims_to_check.push_back(dim);
214
215 for (int n = 1; n < dims_to_check.size(); ++n)
216 if (std::abs(strptr[dims_to_check[n - 1]]) < std::abs(strptr[dims_to_check[n]])) return false;
217
218 return true;
219 }

221 [[nodiscard]] bool is_stride_order_valid() const { return is_stride_order_valid(len.data(), str.data()); }

348 // call implementation
349 template <typename... Args, size_t... Is>
350 [[nodiscard]] FORCEINLINE long call_impl(std::index_sequence<Is...>, Args... args) const noexcept {
351 static constexpr int e_pos = ((std::is_same_v<Args, ellipsis> ? int(Is) + 1 : 0) + ...) - 1; // position ellipsis + 1 or 0
352
353 if constexpr (e_pos == -1) { // common case, no ellipsis
354 if constexpr (smallest_stride_is_one)
355 return (myget<true, Is>(args) + ...);
356 else
357 return ((args * std::get<Is>(str)) + ...);
358 } else {
359 // there is an empty ellipsis to skip
360 return (myget<smallest_stride_is_one, (Is < e_pos ? Is : Is - 1)>(args) + ...);
361 }
362 }

392 std::array<long, Rank> to_idx(long lin_idx) const {
393
394 // compute stride residues starting from slowest index
395 std::array<long, Rank> residues;
396 residues[0] = lin_idx;
397
398 for (auto i : range(1, Rank)) {
399 residues[i] = residues[i - 1] % str[stride_order[i - 1]];
400 } // residues[Rank - 1] is now the value of the fastest index times the corresponding stride
401
402 // convert residues to indices, ordered from slowest to fastest
403 std::array<long, Rank> idx;
404 idx[Rank - 1] = residues[Rank - 1] / str[stride_order[Rank - 1]];
405
406 for (auto i : range(Rank - 2, -1, -1)) { idx[i] = (residues[i] - residues[i + 1]) / str[stride_order[i]]; }
407
408 return permutations::apply_inverse(stride_order, idx);
409 }

442 template <uint64_t Permutation>
443 auto transpose() const {
444 // Denoting this as A, an indexmap, calling it returns the linear index given by
445 //
446 // A(i_k) = sum_k i_k * S[k] (1)
447 //
448 // where S[k] denotes the strides.
449 //
450 // 1- S' : strides of A'
451 // A'(i_k) = sum_k i_{P[k]} * S[k] = sum_k i_k * S[P{^-1}[k]]
452 // so
453 // S'[k] = S[P{^-1}[k]] (2)
454 // i.e. apply (inverse(P), S) or apply_inverse directly.
455 //
456 // 2- L' : lengths of A'
457 // if L[k] is the k-th length, then because of the definition of A', i.e. A'(i_k) = A(i_{P[k]})
458 // i_q in the lhs A is at position q' such that P[q'] = q (A'(i0 i1 i2...) = A( i_P0 i_P1 i_P2....)
459 // hence L'[q] = L[q'] = L[P^{-1}[q]]
460 // same for static length
461 //
462 // 3- stride_order: denoted in this paragraph as Q (and Q' for A').
463 // by definition Q is a permutation such that Q[0] is the slowest index, Q[Rank -1] the fastest
464 // hence S[Q[k]] is a strictly decreasing sequence (as checked by strides_compatible_to_stride_order)
465 // we want therefore Q' the permutation that will sort the S', i.e.
466 // S'[Q'[k]] = S[Q[k]]
467 // using (2), we have S[P{^-1}[Q'[k]]] = S[Q[k]]
468 // so the permutation Q' is such that P{^-1}Q' = Q or Q' = PQ (as permutation product/composition).
469 // NB : Q and P are permutations, so the operation must be a composition, not an apply (apply applies a P to any set, like L, S, not only a permutation)
470 // even though they are all std::array in the code ...
471 //
472 static constexpr std::array<int, Rank> permu = decode<Rank>(Permutation);
473 static constexpr std::array<int, Rank> new_stride_order = permutations::compose(permu, stride_order);
474 static constexpr std::array<int, Rank> new_static_extents = permutations::apply_inverse(permu, static_extents);
475
476 return idx_map<Rank, encode(new_static_extents), encode(new_stride_order), LayoutProp>{permutations::apply_inverse(permu, lengths()),
477 permutations::apply_inverse(permu, strides())};
478 }
