27 // -------------------------------
28 // Rank >1 : general case
29 template <int Rank>
30 class grid_iterator {
31 long stri = 0;
32 long pos = 0;
33 long offset = 0;
34 grid_iterator<Rank - 1> it_begin, it_end, it;
35
36 public:
37 using iterator_category = std::forward_iterator_tag;
38 using value_type = long;
39 using difference_type = std::ptrdiff_t;
40 using pointer = long *;
41 using reference = long &;
42
43 grid_iterator() = default;
44
45 grid_iterator(long const *lengths, long const *strides, bool at_end)
46 : stri(strides[0]),
47 pos(at_end ? lengths[0] : 0),
48 offset(pos * stri),
49 it_begin(lengths + 1, strides + 1, false),
50 it_end(lengths + 1, strides + 1, true),
51 it(it_begin) {} // NB always it_begin, even if at_end. The end iterator correspond to pos = (length -1) + 1, but it is at its *beginning*
52
53 [[nodiscard]] std::array<long, Rank> indices() { return stdutil::front_append(it.indices(), pos); }
54
55 [[nodiscard]] long operator*() const { return offset + *it; }
56 long operator->() const { return operator*(); }
57
58 bool operator==(grid_iterator const &other) const { return ((other.pos == pos) and (other.it == it)); }
59 bool operator!=(grid_iterator const &other) const { return not operator==(other); }
60 grid_iterator &operator++() {
61 ++it;
62 if (it == it_end) { //FIXME [[unlikely]]
63 ++pos;
64 offset += stri;
65 it = it_begin;
66 }
67 return *this;
68 }
69
70 grid_iterator operator++(int) {
71 auto c = *this;
72 ++(*this);
73 return c;
74 }
75 };

95 [[nodiscard]] std::array<long, 1> indices() { return {pos}; }
96
97 [[nodiscard]] long operator*() const { return offset; }
98 long operator->() const { return operator*(); }
99
100 bool operator==(grid_iterator const &other) const { return (other.pos == pos); }
101 bool operator!=(grid_iterator const &other) const { return (other.pos != pos); }
102
103 grid_iterator &operator++() {
104 offset += stri;
105 ++pos;
106 return *this;
107 }
108
109 grid_iterator &operator--() {
110 offset -= stri;
111 --pos;
112 return *this;
113 }
114
115 grid_iterator &operator+=(std::ptrdiff_t n) {
116 offset += n * stri;
117 pos += n;
118 return *this;
119 }
120
121 // We do not implement the full LegacyRandomAccessIterator here, just what is needed for the array_iterator below.
122 friend grid_iterator operator+(grid_iterator it, std::ptrdiff_t n) { return it += n; }
123 friend std::ptrdiff_t operator-(grid_iterator const &it1, grid_iterator const &it2) { return it1.pos - it2.pos; }
124 friend bool operator<(grid_iterator const &it1, grid_iterator const &it2) { return it1.pos < it2.pos; }
125 friend bool operator>(grid_iterator const &it1, grid_iterator const &it2) { return it1.pos > it2.pos; }
126 };

130 // The iterator for the array and array_view container.
131 // Makes an iterator of rank Rank on a pointer of type T.
132 // e.g. for a strided_1d, we use Rank == 1, whatever the real array is
133 // T can be const
134 template <int Rank, typename T, typename Pointer>
135 class array_iterator {
136 T *data = nullptr;
137 std::array<long, Rank> len, stri;
138 grid_iterator<Rank> iter;
139
140 public:
141 using iterator_category = std::forward_iterator_tag;
142 using value_type = T;
143 using difference_type = std::ptrdiff_t;
144 using pointer = T *;
145 using reference = T &;
146
147 array_iterator() = default;
148 array_iterator(std::array<long, Rank> const &lengths, std::array<long, Rank> const &strides, T *start, bool at_end)
149 : data(start), len(lengths), stri(strides), iter(len.data(), stri.data(), at_end) {}
150
151 [[nodiscard]] std::array<long, Rank> indices() { return iter.indices(); }
152
153 [[nodiscard]] value_type &operator*() const { return ((Pointer)data)[*iter]; }
154 value_type &operator->() const { return operator*(); }
155
156 array_iterator &operator++() {
157 ++iter;
158 return *this;
159 }
160
161 array_iterator operator++(int) {
162 auto c = *this;
163 ++iter;
164 return c;
165 }
166
167 bool operator==(array_iterator const &other) const { return (other.iter == iter); }
168 bool operator!=(array_iterator const &other) const { return (!operator==(other)); }
169 };
