46 // a lazy expression of f(A...) where f is the function to map and A the arrays
47 // e.g. f is sqrt(x) and there is one A, or f is min(x,y) and there are 2 A
48 template <typename F, Array... A>
49 struct expr_call {
50
51 F f;
52 std::tuple<const A...> a; // tuple of array (ref) on which f is applied
53
54 private: // FIXME C++20 lambda implementation details
55 template <size_t... Is, typename... Args>
56 [[gnu::always_inline]] [[nodiscard]] auto _call(std::index_sequence<Is...>, Args const &...args) const {
57 // In the case that (args...) invokes a slice on the array
58 // we need to return an call_expr on the resulting view
59 if constexpr ((is_range_or_ellipsis<Args> or ... or false)) {
60 return mapped<F>{f}(std::get<Is>(a)(args...)...);
61 } else {
62 return f(std::get<Is>(a)(args...)...);
63 }
64 }
65 template <size_t... Is, typename Args>
66 [[gnu::always_inline]] auto _call_bra(std::index_sequence<Is...>, Args const &args) const {
67 return f(std::get<Is>(a)[args]...);
68 }
69
70 public:
71 template <typename... Args>
72 auto operator()(Args const &...args) const {
73 return _call(std::make_index_sequence<sizeof...(A)>{}, args...);
74 }
75
76 // vector interface
77 template <typename Args>
78 auto operator[](Args &&args) const {
79 return _call_bra(std::make_index_sequence<sizeof...(A)>{}, args);
80 }
81
82 // FIXME copy needed for the && case only. Overload ?
83 [[nodiscard]] auto shape() const { return std::get<0>(a).shape(); }
84
85 [[nodiscard]] long size() const { return a.size(); }
86 };

98 return {f, {std::forward<A0>(a0), std::forward<A>(a)...}};
