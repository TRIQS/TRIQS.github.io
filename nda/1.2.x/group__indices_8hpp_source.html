27 // -- is_partition_of_indices --
28 //
29 // checks if the groups are a partition of the indices [0,R[
30 // where R is the dimension of the array being regrouped
31 // it will throw (compiler stop) if index is out of bounds
32 template <size_t R, size_t... Rs>
33 constexpr bool is_partition_of_indices(std::array<int, Rs> const &...grps) {
34
35 // check we have exactly R indices ...
36 // FIXME in C++20, we can have constexpr string, hence have more informative error message.
37 auto Rstot = (Rs + ...);
38 if (Rstot < R) throw "Too few indices in the group. Should be the rank";
39 if (Rstot > R) throw "Too many indices in the group. Should be the rank";
40
41 // we go over each indices, in each group and count them
42 std::array<int, R> count = stdutil::make_initialized_array<R>(0); // FIXME : not necessary in C++20
43 std::array<int, R> correct_answer = stdutil::make_initialized_array<R>(1); // FIXME : not necessary in C++20
44
45 auto l = [&](auto &&grp) mutable -> void {
46 for (int u = 0; u < grp.size(); ++u) {
47 if (grp[u] < 0) throw "Negative index !";
48 if (grp[u] >= R) throw "Index larger than the rank !!";
49 count[grp[u]]++;
50 }
51 };
52 (l(grps), ...); // execute all lambdas
53 return (count == correct_answer);
54 }

58 // --- stride_order_of_grouped_idx_map ---
59 //
60 // Given a stride_order and some groups, returns the stride_order of the grouped idx_map
61 // compile time only
62 template <size_t R, size_t... Rs>
63 constexpr std::array<int, sizeof...(Rs)> stride_order_of_grouped_idx_map(std::array<int, R> const &stride_order,
64 std::array<int, Rs> const &...grps) {
65 // stride_order is permutation which by definition is such that
66 // stride_order[0] is the slowest index, stride_order[1] the next, stride_order[Rank-1] the fastest.
67
68 // We need to work with the inverse permutation mem_pos.
69 // For each index k, mem_pos[k] gives the position of index k in the memory layout, 0 for slowest to R-1 for fastest.
70 // e.g. so by definition mem_pos[ stride_order[0]] = 0, mem_pos[ stride_order[1]] = 1, etc...
71
72 // For each group, we look at the mem_pos of its indices.
73 // They must be consecutive to be regrouped.
74 // We select their min, it will give us the mem_pos of the regrouped view, which we will then invert to get the stride_order.
75
76 auto mem_pos = permutations::inverse(stride_order);
77
78 // Find the minimum memory position of the indices in this group
79 // Throw (compile time, i.e. stop compilation) if the indices are not consecutive in memory in this group
80 auto min_mem_pos = [&mem_pos](auto &&grp) {
81 // m = minimum, M = maximum
82 int m = R, M = 0;
83 for (int idx : grp) {
84 int v = mem_pos[idx];
85 if (v > M) M = v;
86 if (v < m) m = v;
87 }
88 bool idx_are_consecutive_in_memory = (M - m + 1 == grp.size());
89 if (!idx_are_consecutive_in_memory)
90 throw "Indices are not consecutive in memory"; // FIXME : can I have a better error message at compile time ??
91 return m;
92 };
93
94 // The Number of groups <-> Dimension of the returned idx_map.
95 constexpr int Ngrps = sizeof...(Rs);
96
97 // An array containing the minimal memory position for each group
98 std::array<int, Ngrps> min_mem_positions{min_mem_pos(grps)...};
99
100 // The problem is that they are not consecutive numbers, they run from 0 to R -1, not Ngrps -1
101 // We compress them back to [0, Ngrps[ by counting how many there are before each of them
102 std::array<int, Ngrps> mem_pos_out = stdutil::make_initialized_array<Ngrps>(0); // FIXME : not necessary in C++20
103 for (int u = 0; u < Ngrps; ++u) {
104 for (int i = 0; i < Ngrps; ++i) {
105 if (min_mem_positions[i] < min_mem_positions[u]) ++mem_pos_out[u];
106 }
107 }
108
109 // The new stride_order is the inverse of mem_pos_out
110 return permutations::inverse(mem_pos_out);
111 }

143 template <int Rank, uint64_t StaticExtents, uint64_t StrideOrder, layout_prop_e LayoutProp, typename... IdxGrps>
144 auto group_indices_layout(idx_map<Rank, StaticExtents, StrideOrder, LayoutProp> const &idxm, IdxGrps...) {
145
146 static_assert(LayoutProp == layout_prop_e::contiguous, "Not implemented for non contiguous arrays");
147
148 // decoded StrideOrder
149 static constexpr auto stride_order = idx_map<Rank, StaticExtents, StrideOrder, LayoutProp>::stride_order;
150
151 static_assert(details::is_partition_of_indices<Rank>(IdxGrps::as_std_array...),
152 "The indices provided in the groups are not a partition of the indices of the array, i.e. [0, Rank[");
153
155 auto total_len_of_a_grp = [&idxm](auto &&grp) {
156 auto ll = std::accumulate(grp.begin(), grp.end(), 1L, [&idxm](long l, long u) { return l * idxm.lengths()[u]; });
157 return ll;
158 };
159
160 auto min_stride_of_a_grp = [&idxm](auto &&grp) {
161 return std::accumulate(grp.begin(), grp.end(), idxm.size(), [&idxm](long s, long u) { return std::min(s, idxm.strides()[u]); });
162 };
163
164 static constexpr int new_rank = sizeof...(IdxGrps);
165 std::array<long, new_rank> new_extents{total_len_of_a_grp(IdxGrps::as_std_array)...};
166 std::array<long, new_rank> new_strides{min_stride_of_a_grp(IdxGrps::as_std_array)...};
167
168 // new layout using the new stride_order
169 using new_layout_t =
170 idx_map<new_rank, 0, encode(details::stride_order_of_grouped_idx_map(stride_order, IdxGrps::as_std_array...)), layout_prop_e::contiguous>;
171
172 return new_layout_t{new_extents, new_strides};
173 }
