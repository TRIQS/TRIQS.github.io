53 static std::array<long, Rank> make_shape_from_string_indices(ind_t const &str_indices) {
54 if (str_indices.size() != Rank) NDA_RUNTIME_ERROR << "String indices are not of the correct size";
55 std::array<long, Rank> sha;
56 for (int i = 0; i < Rank; ++i) sha[i] = str_indices[i].size();
57 return sha;
58 }

61 // ---------------- Accessors -------------------------
62
63 nda::array<nda::array<std::string, 1>, 1> const &get_string_indices() const {
64 if (not s_indices) {
65 auto ind = ind_t(Rank);
66 for (int i = 0; i < Rank; ++i) {
67 auto a = nda::array<std::string, 1>(this->lengths()[i]);
68 for (int j = 0; j < a.size(); ++j) a(j) = std::to_string(j);
69 ind(i) = std::move(a);
70 }
71 s_indices = std::make_shared<ind_t>(std::move(ind));
72 }
73 return *s_indices;
74 }

124 // ---------------- Call operator -------------------------
125 private:
126 template <typename T>
127 auto peel_string(int pos, T const &x) const {
128 if constexpr (not std::is_constructible_v<std::string, T>)
129 return x;
130 else {
131 auto const &sind = get_string_indices();
132 auto const &idx = sind[pos];
133 auto it = std::find(idx.begin(), idx.end(), x);
134 if (it == idx.end())
135 NDA_RUNTIME_ERROR << "Calling array with string. Key " << x << " at position " << pos << " does not match the indices " << sind;
136 return it - idx.begin(); // 1d array, have LegacyRandomAccessIterator const iterator
137 }
138 }

151 // ---------------- Slice -------------------------
152
153 private:
154 template <typename... T, auto... Is>
155 FORCEINLINE decltype(auto) slice_impl(std::index_sequence<Is...>, T const &...x) const {
156 auto const [offset, idxm2] = base_t::slice(peel_string(Is, x)...);
157 using new_rect_str_t = typename details::rect_str_from_base<std::decay_t<decltype(idxm2)>>::type;
158
159 if (not s_indices) return std::make_pair(offset, new_rect_str_t{idxm2});
160
161 // slice the indices. Not optimized, but simple
162 auto const &current_ind = get_string_indices();
163 ind_t ind2((not argument_is_allowed_for_call<T> + ...)); // number of arg which are range, ellipsis, i.e. not arg of simple call
164
165 auto l = [p = 0, &current_ind, &ind2](int n, auto const &y) mutable -> void {
166 using U = std::decay_t<decltype(y)>;
167 if constexpr (not argument_is_allowed_for_call<U>) { ind2[p++] = current_ind[n](y); }
168 };
169 (l(Is, x), ...);
170
171 return std::make_pair(offset, new_rect_str_t{idxm2, ind2});
172 }
173
174 public:
175 template <typename... Args>
176 auto slice(Args const &...args) const {
177 return slice_impl(std::make_index_sequence<sizeof...(args)>{}, args...);
178 }

180 // ---------------- Comparison -------------------------
181
182 bool operator==(rect_str const &x) const { return base_t::operator==(x) and (!s_indices or !x.s_indices or (*s_indices == *(x.s_indices))); }
183 bool operator!=(rect_str const &x) { return !(operator==(x)); }
184
185 // ---------------- Transposition -------------------------
186
187 template <uint64_t Permutation>
188 auto transpose() const {
189
190 auto idxm2 = base_t::template transpose<Permutation>();
191
192 using new_rect_str_t = typename details::rect_str_from_base<std::decay_t<decltype(idxm2)>>::type;
193 if (not s_indices) return new_rect_str_t{idxm2};
194
195 ind_t ind2(s_indices->size());
196 static constexpr std::array<int, Rank> permu = decode<Rank>(Permutation);
197 for (int u = 0; u < Rank; ++u) { ind2[permu[u]] = (*s_indices)[u]; }
198 return new_rect_str_t{idxm2, ind2};
199 }
