33 template <ArrayOfRank<2> M>
34 auto trace(M const &m) {
35 static_assert(get_rank<M> == 2, "trace: array must have rank two");
36 EXPECTS(m.shape()[0] == m.shape()[1]);
37 auto r = get_value_t<M>{};
38 auto d = m.shape()[0];
39 for (int i = 0; i < d; ++i) r += m(i, i);
40 return r;
41 }

53 template <MemoryArrayOfRank<2> M>
54 ArrayOfRank<1> auto diagonal(M &m) {
55 long dim = std::min(m.shape()[0], m.shape()[1]);
56 long stride = stdutil::sum(m.indexmap().strides());
57 using vector_view_t =
58 basic_array_view<std::remove_reference_t<decltype(*m.data())>, 1, C_stride_layout, 'V', nda::default_accessor, nda::borrowed<>>;
59 return vector_view_t{C_stride_layout::mapping<1>{{dim}, {stride}}, m.data()};

63 template <typename V>
64 requires(std::ranges::contiguous_range<V> or ArrayOfRank<V, 1>)
65 ArrayOfRank<2> auto diag(V const &v) {
66 if constexpr (std::ranges::contiguous_range<V>) {
67 return diag(nda::basic_array_view{v});
68 } else {
69 auto m = matrix<std::remove_const_t<typename V::value_type>>::zeros({v.size(), v.size()});
70 diagonal(m) = v;
71 return m;
72 }

78 template <ArrayOfRank<2> A, ArrayOfRank<2> B>
79 requires(std::same_as<get_value_t<A>, get_value_t<B>>) // NB the get_value_t gets rid of const if any
80 matrix<get_value_t<A>> vstack(A const &a, B const &b) {
81 static_assert(get_rank<A> == 2, "vstack: first argument must have rank two");
82 static_assert(get_rank<B> == 2, "vstack: second argument must have rank two");
83 EXPECTS_WITH_MESSAGE(a.shape()[1] == b.shape()[1],
84 "vstack. The second dimension of the two matrices must be equal but \n a is of shape " + to_string(a.shape())
85 + " b is of shape" + to_string(b.shape()));
86 // Impl. Concept only ! A, B can be expression template, e.g.
87
88 auto [n, q] = a.shape();
89 auto p = b.shape()[0];
90 auto _ = range::all;
91
92 matrix<get_value_t<A>> res(n + p, q);
93 res(range(0, n), _) = a;
94 res(range(n, n + p), _) = b;
95 return res;
96 }
