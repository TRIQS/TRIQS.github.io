40 template <MemoryVector X, MemoryVector Y, MemoryMatrix M>
41 requires(have_same_value_type_v<X, Y, M> and mem::have_compatible_addr_space<X, Y, M> and is_blas_lapack_v<get_value_t<X>>)
42 void ger(get_value_t<X> alpha, X const &x, Y const &y, M &&m) {
43
44 EXPECTS(m.extent(0) == x.extent(0));
45 EXPECTS(m.extent(1) == y.extent(0));
46 // Must be lapack compatible
47 EXPECTS(m.indexmap().min_stride() == 1);
48
49 // if in C, we need to call fortran with transposed matrix
50 if (has_C_layout<M>) {
51 ger(alpha, y, x, transpose(m));
52 return;
53 }
54
55 if constexpr (mem::have_device_compatible_addr_space<X, Y, M>) {
56#if defined(NDA_HAVE_DEVICE)
57 device::ger(m.extent(0), m.extent(1), alpha, x.data(), x.indexmap().strides()[0], y.data(), y.indexmap().strides()[0], m.data(), get_ld(m));
58#else
59 compile_error_no_gpu();
60#endif
61 } else {
62 f77::ger(m.extent(0), m.extent(1), alpha, x.data(), x.indexmap().strides()[0], y.data(), y.indexmap().strides()[0], m.data(), get_ld(m));
63 }
64 }

80 template <ArrayOrScalar A, ArrayOrScalar B>
81 auto outer_product(A const &a, B const &b) {
82
83 if constexpr (Scalar<A> or Scalar<B>) {
84 return a * b;
85 } else {
86 if (not a.is_contiguous()) NDA_RUNTIME_ERROR << "First argument to outer_product call has non-contiguous layout";
87 if (not b.is_contiguous()) NDA_RUNTIME_ERROR << "Second argument to outer_product call has non-contiguous layout";
88 auto res = zeros<get_value_t<A>, mem::common_addr_space<A, B>>(stdutil::join(a.shape(), b.shape()));
89
90 auto a_vec = reshape(a, std::array{a.size()});
91 auto b_vec = reshape(b, std::array{b.size()});
92 auto mat = reshape(res, std::array{a.size(), b.size()});
93 ger(1.0, a_vec, b_vec, mat);
94
95 return res;
96 }
97 }
