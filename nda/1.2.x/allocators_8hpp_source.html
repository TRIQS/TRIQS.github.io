52 // ------------------------- Malloc allocator ----------------------------
53 //
54 // Allocates simply with malloc
55 //
56 template <AddressSpace AdrSp = Host>
57 class mallocator {
58 public:
59 mallocator() = default;
60 mallocator(mallocator const &) = delete;
61 mallocator(mallocator &&) = default;
62 mallocator &operator=(mallocator const &) = delete;
63 mallocator &operator=(mallocator &&) = default;
64
65 static constexpr auto address_space = AdrSp;
66
67 static blk_t allocate(size_t s) noexcept {
68 return {(char *)malloc<AdrSp>(s), s}; // NOLINT
69 }
70 static blk_t allocate_zero(size_t s) noexcept {
71 if constexpr (AdrSp == mem::Host) {
72 return {(char *)std::calloc(s, 1 /*byte*/), s}; // NOLINT
73 } else {
74 char *ptr = (char *)malloc<AdrSp>(s);
75 memset<AdrSp>(ptr, 0, s);
76 return {ptr, s};
77 }
78 }
79
80 static void deallocate(blk_t b) noexcept {
81 free<AdrSp>((void *)b.ptr); // NOLINT
82 }
83 };

85 // ------------------------- Bucket allocator ----------------------------
86 //
87 //
88 template <int ChunkSize>
89 class bucket {
90 static constexpr int TotalChunkSize = 64 * ChunkSize;
91 std::unique_ptr<char[]> _start = std::make_unique<char[]>(TotalChunkSize); // NOLINT
92 char *p = _start.get();
93 uint64_t flags = uint64_t(-1);
94
95 public:
96 static constexpr auto address_space = Host;
97
98#ifdef NDA_USE_ASAN
99 bucket() { __asan_poison_memory_region(p, TotalChunkSize); }
100 ~bucket() { __asan_unpoison_memory_region(p, TotalChunkSize); }
101#else
102 bucket() = default;
103#endif
104 bucket(bucket const &) = delete;
105 bucket(bucket &&) = default;
106 bucket &operator=(bucket const &) = delete;
107 bucket &operator=(bucket &&) = default;
108
109 blk_t allocate(size_t s) noexcept {
110 // FIXME not here ! in the handle
111 //auto n = round_to_align(s);
112 //if (n > ChunkSize) std::abort();
113 if (flags == 0) std::abort();
114 int pos = __builtin_ctzll(flags);
115 flags &= ~(1ull << pos);
116 blk_t b{p + pos * ChunkSize, s};
117#ifdef NDA_USE_ASAN
118 __asan_unpoison_memory_region(b.ptr, ChunkSize);
119#endif
120 return b;
121 }
122
123 blk_t allocate_zero(size_t s) noexcept {
124 auto blk = allocate(s);
125 std::memset(blk.ptr, 0, s);
126 return blk;
127 }
128
129 void deallocate(blk_t b) noexcept {
130#ifdef NDA_USE_ASAN
131 __asan_poison_memory_region(b.ptr, ChunkSize);
132#endif
133 int pos = (b.ptr - p) / ChunkSize;
134 flags |= (1ull << pos);
135 }
136
137 [[nodiscard]] bool is_full() const noexcept { return flags == 0; }
138 [[nodiscard]] bool empty() const noexcept { return flags == uint64_t(-1); }
139
140 [[nodiscard]] const char *data() const noexcept { return p; }
141
142 [[nodiscard]] bool owns(blk_t b) const noexcept { return b.ptr >= p and b.ptr < p + TotalChunkSize; }
143 };

145 // ------------------------- Multiple bucket allocator ----------------------------
146 //
147 //
148 template <int ChunkSize>
149 class multi_bucket {
150
151 using b_t = bucket<ChunkSize>;
152 std::vector<b_t> bu_vec; // an ordered vector of buckets
153 typename std::vector<b_t>::iterator bu; // current bucket in use
154
155 // find the next bucket with some space. Possibly allocating new ones.
156 [[gnu::noinline]] void find_non_full_bucket() {
157 bu = std::find_if(bu_vec.begin(), bu_vec.end(), [](auto const &b) { return !b.is_full(); });
158 if (bu != bu_vec.end()) return;
159
160 // insert a new bucket ordered. Position is defined by data (NB : which is NOT affected by the move)
161 b_t b;
162 auto insert_position = std::upper_bound(bu_vec.begin(), bu_vec.end(), b, [](auto const &B, auto const &B2) { return B.data() < B2.data(); });
163 bu = bu_vec.insert(insert_position, std::move(b));
164
165 //for (auto const &bb : bu_vec) TRIQS_PRINT((void *)bb.data());
166 //TRIQS_PRINT("---------------");
167 }
168
169 public:
170 static constexpr auto address_space = Host;
171
172 multi_bucket() : bu_vec(1), bu(bu_vec.begin()) {}
173 multi_bucket(multi_bucket const &) = delete;
174 multi_bucket(multi_bucket &&) = delete;
175 multi_bucket &operator=(multi_bucket const &) = delete;
176 multi_bucket &operator=(multi_bucket &&) = delete;
177
178 blk_t allocate(size_t s) noexcept {
179 //[[unlikely]]
180 if ((bu == bu_vec.end()) or (bu->is_full())) find_non_full_bucket();
181 return bu->allocate(s);
182 }
183
184 blk_t allocate_zero(size_t s) noexcept {
185 auto blk = allocate(s);
186 std::memset(blk.ptr, 0, s);
187 return blk;
188 }
189
190 void deallocate(blk_t b) noexcept {
191 //[[likely]]
192 if (bu != bu_vec.end() and bu->owns(b)) {
193 bu->deallocate(b);
194 return;
195 }
196 bu = std::lower_bound(bu_vec.begin(), bu_vec.end(), b.ptr, [](auto const &B, auto p) { return B.data() <= p; });
197 --bu;
198 EXPECTS_WITH_MESSAGE((bu != bu_vec.end()), "Fatal Logic Error in allocator. Not in bucket. \n");
199 EXPECTS_WITH_MESSAGE((bu->owns(b)), "Fatal Logic Error in allocator. \n");
200 bu->deallocate(b);
201 if (!bu->empty()) return;
202 if (bu_vec.size() <= 1) return;
203 bu_vec.erase(bu);
204 bu = bu_vec.end();
205 }

210 // ------------------------- segregator allocator ----------------------------
211 //
212 // Dispatch according to size to two allocators
213 //
214 template <size_t Threshold, Allocator A, Allocator B>
215 class segregator {
216
217 A small;
218 B big;
219
220 public:
221 static_assert(A::address_space == B::address_space);
222 static constexpr auto address_space = A::address_space;
223
224 segregator() = default;
225 segregator(segregator const &) = delete;
226 segregator(segregator &&) = default;
227 segregator &operator=(segregator const &) = delete;
228 segregator &operator=(segregator &&) = default;
229
230 blk_t allocate(size_t s) noexcept { return s <= Threshold ? small.allocate(s) : big.allocate(s); }
231 blk_t allocate_zero(size_t s) noexcept { return s <= Threshold ? small.allocate_zero(s) : big.allocate_zero(s); }
232
233 void deallocate(blk_t b) noexcept { return b.s <= Threshold ? small.deallocate(b) : big.deallocate(b); }
234 [[nodiscard]] bool owns(blk_t b) const noexcept { return small.owns(b) or big.owns(b); }
235 };

264 blk_t b = A::allocate(s);
265 memory_used = memory_used + b.s;
266 // std::cerr<< "Allocating "<< b.s << "Total = "<< memory_used << "\n";
267 return b;
268 }
269
270 blk_t allocate_zero(size_t s) {
271 blk_t b = A::allocate_zero(s);
272 memory_used = memory_used + b.s;
273 // std::cerr<< "Allocating "<< b.s << "Total = "<< memory_used << "\n";
274 return b;
275 }
276
277 void deallocate(blk_t b) noexcept {
278 memory_used -= b.s;
279 // std::cerr<< "Deallocating "<< b.s << "Total = "<< memory_used << "\n";
280
281 if (memory_used < 0) {
282#ifndef NDEBUG
283 std::cerr << "Allocator : memory_used <0 : " << memory_used << " b.s = " << b.s << " b.ptr = " << (void *)b.ptr;
284 std::abort();
285#endif
286 }
287 A::deallocate(b);
288 }
289
290 [[nodiscard]] bool owns(blk_t b) const noexcept { return A::owns(b); }
291
