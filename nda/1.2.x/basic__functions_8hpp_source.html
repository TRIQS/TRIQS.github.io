29 template <typename T, mem::AddressSpace AdrSp = mem::Host, std::integral Int, auto Rank>
30 auto zeros(std::array<Int, Rank> const &shape) {
31 static_assert(AdrSp != mem::None);
32
33 // For Rank == 0 we should return the underlying scalar_t
34 if constexpr (Rank == 0)
35 return T{};
36 else if constexpr (AdrSp == mem::Host)
37 return array<T, Rank>::zeros(shape);
38 else
39 return cuarray<T, Rank>::zeros(shape);
40 }

94 template <typename RealType = double, std::integral Int, auto Rank>
95 auto rand(std::array<Int, Rank> const &shape)
96 requires(std::is_floating_point_v<RealType>)
97 {
98 // For Rank == 0 we should return a scalar
99 if constexpr (Rank == 0) {
100 auto static gen = std::mt19937(std::random_device{}());
101 auto static dist = std::uniform_real_distribution<>(0.0, 1.0);
102 return dist(gen);
103 } else {
104 return array<RealType, Rank>::rand(shape);
105 }
106 }

289 template <Array A, Array B>
290 bool operator==(A const &a, B const &b) {
291 // FIXME not implemented in clang .. readd when done for better error message
292#ifndef __clang__
293 static_assert(std::equality_comparable_with<get_value_t<A>, get_value_t<B>>, "A == B is only defined when their element can be compared");
294#endif
295 if (a.shape() != b.shape()) return false;
296 bool r = true;
297 nda::for_each(a.shape(), [&](auto &&...x) { r &= (a(x...) == b(x...)); });
298 return r;
299 }

319 clef_auto_assign(a(x...), f(x...));

321 a(x...) = f(x...);

349 template <MemoryArray A>
350 std::optional<std::tuple<int, int, int>> get_block_layout(A const &a) {
351
352 EXPECTS(!a.empty());
353
354 auto const &l = a.indexmap().lengths();
355 auto const &s = a.indexmap().strides();
356 auto const &i = a.indexmap().stride_order;
357
358 int data_size = l[i[0]] * s[i[0]];
359 int block_size = data_size;
360 int block_str = data_size;
361 int n_blocks = 1;
362
363 for (auto n : range(A::rank)) {
364 auto inner_size = (n == A::rank - 1) ? 1 : s[i[n + 1]] * l[i[n + 1]];
365 if (s[i[n]] != inner_size) {
366 if (block_size < data_size) // second strided dimension
367 return {};
368 // found a strided dimension with (assumed) contiguous inner blocks
369 n_blocks = a.size() / inner_size;
370 block_size = inner_size;
371 block_str = s[i[n]];
372 }
373 }
374 ASSERT(n_blocks * block_size == a.size());
375 ASSERT(n_blocks * block_str == l[i[0]] * s[i[0]]);
376 return std::make_tuple(n_blocks, block_size, block_str);
377 }
