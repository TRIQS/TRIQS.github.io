63 template <typename T, Allocator alloc_t = mallocator<>>
64 struct handle_heap {
65 private:
66 T *_data = nullptr; // Pointer to the start of the memory block
67 size_t _size = 0; // Size of the memory block. Invariant: size > 0 iif data != 0
68
69#ifndef NDA_DEBUG_LEAK_CHECK
70 static inline alloc_t allocator;
71#else
72 static inline leak_check<alloc_t> allocator;
73#endif
74
75 // In case we need to share the memory
76 mutable std::shared_ptr<void> sptr;
77
78 using blk_T_t = std::pair<T *, size_t>;
79
80 // code for destructor
81 static void destruct(blk_T_t b) noexcept {
82
83 static_assert(std::is_nothrow_destructible_v<T>, "nda::mem::handle requires the value_type to have a non-throwing constructor");
84 auto [data, size] = b;
85
86 if (data == nullptr) return;
87
88 // If needed, call the T destructors
89 if constexpr (alloc_t::address_space == Host and !(std::is_trivial_v<T> or nda::is_complex_v<T>)) {
90 for (size_t i = 0; i < size; ++i) data[i].~T();
91 }
92
93 // Deallocate the memory block
94 allocator.deallocate({(char *)data, size * sizeof(T)});
95 }
96
97 // a deleter for the data in the sptr
98 static void deleter(void *p) noexcept { destruct(*((blk_T_t *)p)); }
99
100 public:
101 std::shared_ptr<void> get_sptr() const {
102 if (not sptr) sptr.reset(new blk_T_t{_data, _size}, deleter);
103 return sptr;
104 }
105
106 using value_type = T;
107 static constexpr auto address_space = alloc_t::address_space;
108
109 ~handle_heap() noexcept {
110 // if the data is not in the shared_ptr, we delete it, otherwise the shared_ptr will take care of it
111 if (not sptr and not(is_null())) destruct({_data, _size});
112 }
113
114 handle_heap() = default;
115
116 handle_heap(handle_heap &&x) noexcept {
117 _data = x._data;
118 _size = x._size;
119 sptr = std::move(x.sptr);
120 x._data = nullptr;
121 x._size = 0;
122 }
123
124 handle_heap &operator=(handle_heap &&x) noexcept {
125 if (not sptr and not(is_null())) destruct({_data, _size});
126 _data = x._data;
127 _size = x._size;
128 sptr = std::move(x.sptr);
129 x._data = nullptr;
130 x._size = 0;
131 return *this;
132 }
133
134 handle_heap &operator=(handle_heap const &x) {
135 *this = handle_heap{x};
136 return *this;
137 }
138
139 template <Allocator alloc_t_other>
140 handle_heap &operator=(handle_heap<T, alloc_t_other> const &x) {
141 *this = handle_heap{x};
142 return *this;
143 }

145 // Set up a memory block of the correct size without initializing it
146 handle_heap(long size, do_not_initialize_t) {
147 if (size == 0) return; // no size -> null handle
148 auto b = allocator.allocate(size * sizeof(T)); //, alignof(T));
149 ASSERT(b.ptr != nullptr);
150 _data = (T *)b.ptr;
151 _size = size;
152 }
153
154 // Set up a memory block of the correct size without initializing it
155 handle_heap(long size, init_zero_t) {
156 if (size == 0) return; // no size -> null handle
157 auto b = allocator.allocate_zero(size * sizeof(T)); //, alignof(T));
158 ASSERT(b.ptr != nullptr);
159 _data = (T *)b.ptr;
160 _size = size;
161 }
162
163 // Construct a new block of memory of given size and init if needed.
164 handle_heap(long size) {
165 if (size == 0) return; // no size -> null handle
166
167 blk_t b;
168 if constexpr (is_complex_v<T> && init_dcmplx)
169 b = allocator.allocate_zero(size * sizeof(T));
170 else
171 b = allocator.allocate(size * sizeof(T));
172
173 ASSERT(b.ptr != nullptr);
174 _data = (T *)b.ptr;
175 _size = size;
176
177 // Call placement new except for complex types
178 if constexpr (!std::is_trivial_v<T> and !is_complex_v<T>) {
179 for (size_t i = 0; i < size; ++i) new (_data + i) T();
180 }
181 }
182
183 explicit handle_heap(handle_heap const &x) : handle_heap(x.size(), do_not_initialize) {
184 if (is_null()) return; // nothing to do for null handle
185 if constexpr (std::is_trivially_copyable_v<T>) {
186 memcpy<address_space, address_space>(_data, x.data(), x.size() * sizeof(T));
187 } else {
188 for (size_t i = 0; i < _size; ++i) new (_data + i) T(x[i]); // placement new
189 }
190 }
191
192 // Copy data from another owning handle
193 template <OwningHandle<value_type> H>
194 explicit handle_heap(H const &x) : handle_heap(x.size(), do_not_initialize) {
195 if (is_null()) return; // nothing to do for null handle
196 if constexpr (std::is_trivially_copyable_v<T>) {
197 memcpy<address_space, H::address_space>((void *)_data, (void *)x.data(), _size * sizeof(T));
198 } else {
199 static_assert(address_space == H::address_space,
200 "Constructing from handle of different address space requires trivially copyable value_type");
201 for (size_t i = 0; i < _size; ++i) new (_data + i) T(x[i]); // placement new
202 }
203 }
204
205 T &operator[](long i) noexcept { return _data[i]; }
206 T const &operator[](long i) const noexcept { return _data[i]; }
207
208 bool is_null() const noexcept {
209#ifdef NDA_DEBUG
210 // Check the Invariants in Debug Mode
211 EXPECTS((_data == nullptr) == (_size == 0));
212#endif
213 return _data == nullptr;
214 }
215
216 // A const-handle does not entail T const data
217 T *data() const noexcept { return _data; }
218
219 long size() const noexcept { return _size; }
220 };
