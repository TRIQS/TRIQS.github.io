26 /*
27 * Write an array or a view into an hdf5 file
28 * The HDF5 exceptions will be caught and rethrown as std::runtime_error
29 *
30 * @param g The h5 group
31 * @param name The name of the hdf5 array in the file/group where the stack will be stored
32 * @param a The array to be stored
33 * @param slice Optional slice (tuple of range, range::all_t, ellipsis or integer types) to limit which data to write (assumes existing dataset in File!)
34 * @tparam A The type of the array/matrix/vector, etc..
35 * @tparam IRs The types in the slice tuple, i.e. integer, index ranges, range::all_t or ellipsis
36 */

61 template <MemoryArrayOfRank<1> A>
62 h5::char_buf to_char_buf(A const &v) {
63 size_t s = 0;
64 for (auto &x : v) s = std::max(s, x.size() + 1);
65 auto len = h5::v_t{size_t(v.size()), s};
66
67 std::vector<char> buf;
68 buf.resize(v.size() * s, 0x00);
69 size_t i = 0;
70 for (auto &x : v) {
71 strcpy(&buf[i * s], x.c_str());
72 ++i;
73 }
74 return {buf, len};
75 }

77 template <MemoryArrayOfRank<1> A>
78 void from_char_buf(h5::char_buf const &cb, A &v) {
79 v.resize(cb.lengths[0]);
80 auto len_string = cb.lengths[1];
81
82 size_t i = 0;
83 for (auto &x : v) {
84 x = "";
85 x.append(&cb.buffer[i * len_string]);
86 ++i;
87 }
88 }

106 template <size_t NDim, typename... IRs>
107 auto hyperslab_and_shape_from_slice(std::tuple<IRs...> const &slice, std::vector<h5::hsize_t> const &lengths, bool is_complex) {
108
109 static constexpr auto size_of_slice = sizeof...(IRs);
110
111 static constexpr auto ellipsis_count = (std::is_same_v<IRs, ellipsis> + ... + 0);
112 static_assert(ellipsis_count < 2, "Can only provide single ellipsis in slicing");
113 static constexpr auto has_ellipsis = (ellipsis_count == 1);
114 static constexpr auto ellipsis_position = [&]<size_t... Is>(std::index_sequence<Is...>) {
115 if constexpr (has_ellipsis) return ((std::is_same_v<IRs, ellipsis> * Is) + ... + 0);
116 return size_of_slice;
117 }(std::index_sequence_for<IRs...>{});
118
119 static constexpr auto integer_count = (std::integral<IRs> + ... + 0);
120 static constexpr auto range_count = size_of_slice - integer_count - ellipsis_count;
121 static constexpr auto ellipsis_width = NDim - range_count;
122
123 static_assert((has_ellipsis && range_count <= NDim) || range_count == NDim,
124 "Array rank does not match the number of non-trivial slice dimensions");
125
126 static constexpr auto slab_rank = NDim + integer_count;
127 auto rank_in_file = lengths.size() - is_complex;
128 if (slab_rank != rank_in_file)
129 NDA_RUNTIME_ERROR << "Error in h5 slice: Incorrect slice rank. In file: " << rank_in_file << " Rank of slice: " << size_of_slice;
130
131 auto hsl = h5::array_interface::hyperslab(slab_rank, is_complex);
132 auto shape = std::array<long, NDim>{};
133 [&, m = 0]<size_t... Is>(std::index_sequence<Is...>) mutable {
134 (
135 [&]<typename IR>(size_t n, IR const &ir) mutable {
136 if (n > ellipsis_position) n += (ellipsis_width - 1);
137 if constexpr (std::integral<IR>) {
138 hsl.offset[n] = ir;
139 hsl.count[n] = 1;
140 } else if constexpr (std::is_same_v<IR, nda::ellipsis>) {
141 for (auto k : range(n, n + ellipsis_width)) {
142 hsl.count[k] = lengths[k];
143 shape[m++] = lengths[k];
144 }
145 } else if constexpr (std::is_same_v<IR, nda::range>) {
146 hsl.offset[n] = ir.first();
147 hsl.stride[n] = ir.step();
148 hsl.count[n] = ir.size();
149 shape[m++] = ir.size();
150 } else {
151 static_assert(std::is_same_v<IR, nda::range::all_t>);
152 hsl.count[n] = lengths[n];
153 shape[m++] = lengths[n];
154 }
155 }(Is, std::get<Is>(slice)),
156 ...);
157 }(std::make_index_sequence<size_of_slice>{});
158 return std::make_pair(hsl, shape);
159 }

161 template <MemoryArray A, typename... IRs>
162 void h5_write(h5::group g, std::string const &name, A const &a, std::tuple<IRs...> const &slice) {
163
164 // If array is not in C-order or not contiguous
165 // copy into array with default layout and write
166 if (not a.indexmap().is_stride_order_C() or not a.indexmap().is_contiguous()) {
167 using h5_arr_t = nda::array<get_value_t<A>, A::rank>;
168 auto a_c_layout = h5_arr_t{a.shape()};
169 a_c_layout() = a;
170 h5_write(g, name, a_c_layout, slice);
171 return;
172 }
173
174 constexpr int size_of_slice = sizeof...(IRs);
175 constexpr bool slicing = (size_of_slice > 0);
176
177 // Special case of string, write as char buffer
178 if constexpr (std::is_same_v<typename A::value_type, std::string>) {
179 static_assert(!slicing, "Slicing not supported in h5_write for array of string");
180 h5_write(g, name, h5_details::to_char_buf(a));
181
182 } else if constexpr (is_scalar_v<typename A::value_type>) {
183
184 static constexpr bool is_complex = is_complex_v<typename A::value_type>;
185
186 if constexpr (slicing) {
187 // Dataset must already exist for the sliced h5_write
188 auto lt = h5::array_interface::get_h5_lengths_type(g, name);
189 if (is_complex != lt.has_complex_attribute)
190 NDA_RUNTIME_ERROR << "Error in sliced h5_write. Existing dataset and array must both be either complex or real";
191 auto const [sl, sh] = hyperslab_and_shape_from_slice<A::rank>(slice, lt.lengths, is_complex);
192 if (sh != a.shape())
193 NDA_RUNTIME_ERROR << "Error in sliced h5_write. Shape of slice and Array shape incompatible"
194 << "\n shape of slice : " << sh << "\n array : " << a.shape();
195
196 auto rank_in_file = lt.rank() - is_complex;
197 h5::array_interface::h5_array_view v{h5::hdf5_type<get_value_t<A>>(), (void *)(a.data()), rank_in_file, is_complex};
198 v.slab.count = sl.count;
199 v.L_tot = sl.count;
200
201 h5::array_interface::write_slice(g, name, v, lt, sl);
202 } else {
203
204 auto [L_tot, strides_h5] = h5::array_interface::get_L_tot_and_strides_h5(a.indexmap().strides().data(), A::rank, a.size());
205
206 h5::array_interface::h5_array_view v{h5::hdf5_type<get_value_t<A>>(), (void *)a.data(), A::rank, is_complex};
207 for (int u = 0; u < A::rank; ++u) { // size of lhs may be size of the rhs vector + 1 if complex. Can not simply use =
208 v.slab.count[u] = a.shape()[u];
209 v.slab.stride[u] = strides_h5[u];
210 v.L_tot[u] = L_tot[u];
211 }
212 h5::array_interface::write(g, name, v, true);
213 }
214
215 } else { // generic unknown type to hdf5
216
217 static_assert(!slicing, "Slicing not supported in generic h5_write");
218
219 auto g2 = g.create_group(name);
220 h5_write(g2, "shape", a.shape());
221 auto make_name = [](auto i0, auto... is) { return (std::to_string(i0) + ... + ("_" + std::to_string(is))); };
222 nda::for_each(a.shape(), [&](auto... is) { h5_write(g2, make_name(is...), a(is...)); });
223 }
224 }

226 template <MemoryArray A, typename... IRs>
227 void h5_read(h5::group g, std::string const &name, A &a, std::tuple<IRs...> const &slice) {
228
229 // If array is not in C-order or not contiguous
230 // read into array with default layout and copy
231 constexpr bool is_stride_order_C = std::decay_t<A>::is_stride_order_C();
232 static_assert(is_stride_order_C or is_regular_v<A>, "Cannot read into an array_view to an array with non C-style memory layout");
233 if (not is_stride_order_C or not a.indexmap().is_contiguous()) {
234 using h5_arr_t = nda::array<typename A::value_type, A::rank>;
235 auto a_c_layout = h5_arr_t{};
236 h5_read(g, name, a_c_layout, slice);
237 if constexpr (is_regular_v<A>) a.resize(a_c_layout.shape());
238 a() = a_c_layout;
239 return;
240 }
241
242 constexpr int size_of_slice = sizeof...(IRs);
243 constexpr bool slicing = (size_of_slice > 0);
244
245 // Special case array<string>, store as char buffer
246 if constexpr (std::is_same_v<typename A::value_type, std::string>) {
247 static_assert(!slicing, "Slicing not supported in h5_read for array of string");
248 h5::char_buf cb;
249 h5_read(g, name, cb);
250 h5_details::from_char_buf(cb, a);
251
252 } else if constexpr (is_scalar_v<typename A::value_type>) {
253
254 static constexpr bool is_complex = is_complex_v<typename A::value_type>;
255
256 auto lt = h5::array_interface::get_h5_lengths_type(g, name);
257
258 // Allow to read non-complex data into array<complex>
259 if constexpr (is_complex) {
260 if (!lt.has_complex_attribute) {
261 array<double, A::rank> tmp;
262 h5_read(g, name, tmp);
263 a = tmp;
264 return;
265 }
266 }
267
268 std::array<long, A::rank> shape{};
269 auto slice_slab = h5::array_interface::hyperslab{};
270
271 if constexpr (slicing) {
272 auto const [sl, sh] = hyperslab_and_shape_from_slice<A::rank>(slice, lt.lengths, is_complex);
273 slice_slab = sl;
274 shape = sh;
275 } else {
276 for (int u = 0; u < A::rank; ++u) shape[u] = lt.lengths[u]; // NB : correct for complex
277 }
278
279 if constexpr (is_regular_v<A>) {
280 a.resize(shape);
281 } else {
282 if (a.shape() != shape)
283 NDA_RUNTIME_ERROR << "Error trying to read from an hdf5 file to a view. Dimension mismatch"
284 << "\n in file : " << shape << "\n in view : " << a.shape();
285 }
286
287 auto rank_in_file = lt.rank() - is_complex;
288 if (!slicing && rank_in_file != A::rank)
289 NDA_RUNTIME_ERROR << "Error in h5_read: Rank mismatch. Array has rank " << A::rank << " while Dataset has rank " << rank_in_file;
290 h5::array_interface::h5_array_view v{h5::hdf5_type<get_value_t<A>>(), (void *)(a.data()), rank_in_file, is_complex};
291 if constexpr (slicing) {
292 v.slab.count = slice_slab.count;
293 v.L_tot = slice_slab.count;
294 } else {
295 for (int u = 0; u < A::rank; ++u) {
296 v.slab.count[u] = shape[u];
297 v.L_tot[u] = shape[u];
298 }
299 }
300 h5::array_interface::read(g, name, v, lt, slice_slab);
301
302 } else { // generic unknown type to hdf5
303
304 static_assert(!slicing, "Slicing not supported in generic h5_read");
305
306 auto g2 = g.open_group(name);
307
308 // Reshape if necessary
309 std::array<long, A::rank> h5_shape;
310 h5_read(g2, "shape", h5_shape);
311 if (a.shape() != h5_shape) a.resize(h5_shape);
312
313 // Read using appropriate h5_read implementation
314 auto make_name = [](auto i0, auto... is) { return (std::to_string(i0) + ... + ("_" + std::to_string(is))); };
315 nda::for_each(a.shape(), [&](auto... is) { h5_read(g2, make_name(is...), a(is...)); });
316 }
317 }
