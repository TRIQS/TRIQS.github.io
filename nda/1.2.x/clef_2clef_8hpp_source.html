415 template <typename F>
416 struct apply_on_each_leaf_impl {
417 F f;
418
419 private:
420 template <typename ChildTuple, size_t... Is>
421 void _apply_this_on_each(std::index_sequence<Is...>, ChildTuple const &child_tuple) {
422 ((*this)(std::get<Is>(child_tuple)), ...);
423 }
424
425 public:
426 template <typename Tag, typename... T>
427 FORCEINLINE void operator()(expr<Tag, T...> const &ex) {
428 _apply_this_on_each(std::make_index_sequence<sizeof...(T)>{}, ex.childs);

431 FORCEINLINE void operator()(T const &x) CLEF_requires(!is_any_lazy<T>) {
432 f(x);
433 }
434 template <typename T>
435 FORCEINLINE void operator()(std::reference_wrapper<T> const &x) CLEF_requires(!is_any_lazy<T>) {
436 f(x.get());
437 }
438 };
