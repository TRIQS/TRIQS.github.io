22 template <typename M>
23 // dispatch the implementation of invoke for T = double or complex
24 auto _eigen_element_impl(M &&m, char compz) {
25
26 EXPECTS((not m.empty()));
27 EXPECTS(is_matrix_square(m, true));
28 EXPECTS(m.indexmap().is_contiguous());
29
30 int dim = m.extent(0);
31
32 using T = typename std::decay_t<M>::value_type;
33
34 array<double, 1> ev(dim);
35 int lwork = 64 * dim;
36 array<T, 1> work(lwork);
37 array<double, 1> work2(is_complex_v<T> ? lwork : 0);
38
39#if defined(__has_feature)
40#if __has_feature(memory_sanitizer)
41 work2 = 0;
42 work = 0;
43 ev = 0;
44#endif
45#endif
46
47 int info = 0;
48 if constexpr (not is_complex_v<T>) {
49 lapack::f77::syev(compz, 'U', dim, m.data(), dim, ev.data(), work.data(), lwork, info);
50 } else {
51 lapack::f77::heev(compz, 'U', dim, m.data(), dim, ev.data(), work.data(), lwork, work2.data(), info);
52 }
53 if (info) NDA_RUNTIME_ERROR << "Diagonalization error";
54 return ev;
55 }
