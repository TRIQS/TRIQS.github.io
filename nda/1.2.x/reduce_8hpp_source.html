24// Models ArrayInitializer concept
25template <nda::Array A>
26struct mpi::lazy<mpi::tag::reduce, A> {
27
28 using value_type = typename std::decay_t<A>::value_type;
29 using const_view_type = decltype(std::declval<const A>()());
30
31 const_view_type rhs; // the rhs array
32 mpi::communicator c; // mpi comm
33 const int root; //
34 const bool all;
35 const MPI_Op op;
36
38 [[nodiscard]] auto shape() const { return rhs.shape(); }
39
41 template <nda::Array T>
42 void invoke(T &&target) const {
43 if (not target.is_contiguous()) NDA_RUNTIME_ERROR << "mpi operations require contiguous target.data() to be contiguous";
44
45 static_assert(std::decay_t<A>::layout_t::stride_order_encoded == std::decay_t<T>::layout_t::stride_order_encoded,
46 "Array types for rhs and target have incompatible stride order");
47
48 if (not mpi::has_env) {
49 target = rhs;
50 return;
51 }
52
53 if constexpr (not mpi::has_mpi_type<value_type>) {
54 target = nda::map([this](auto const &x) { return mpi::reduce(x, this->c, this->root, this->all, this->op); })(rhs);
55 } else {
56
57 // some checks.
58 bool in_place = (target.data() == rhs.data());
59 auto sha = shape();
60 if (in_place) {
61 if (rhs.size() != target.size()) NDA_RUNTIME_ERROR << "mpi reduce of array : same pointer to data start, but different number of elements !";
62 } else { // check no overlap
63 if ((c.rank() == root) || all) resize_or_check_if_view(target, sha);
64 if (std::abs(target.data() - rhs.data()) < rhs.size()) NDA_RUNTIME_ERROR << "mpi reduce of array : overlapping arrays !";
65 }
66
67 void *v_p = (void *)target.data();
68 void *rhs_p = (void *)rhs.data();
69 auto rhs_n_elem = rhs.size();
70 auto D = mpi::mpi_type<value_type>::get();
71
72 if (!all) {
73 if (in_place)
74 MPI_Reduce((c.rank() == root ? MPI_IN_PLACE : rhs_p), rhs_p, rhs_n_elem, D, op, root, c.get());
75 else
76 MPI_Reduce(rhs_p, v_p, rhs_n_elem, D, op, root, c.get());
77 } else {
78 if (in_place)
79 MPI_Allreduce(MPI_IN_PLACE, rhs_p, rhs_n_elem, D, op, c.get());
80 else
81 MPI_Allreduce(rhs_p, v_p, rhs_n_elem, D, op, c.get());
82 }
83 }
84 }
85};
