25 // ------- CallableWithLongs<A, R> ----------
26 // A can be called with R longs
27
28#ifndef __clang__
29 // clang does not have an implementation yet of "Lambda in unevaluated context"
30 // we make a workaround below
31 // clang-format off
32 template <typename A, int R>
33 concept CallableWithLongs = requires(A const &a) {
34 {
35 []<auto... Is>(std::index_sequence<Is...>, auto const &aa) -> decltype(aa(long(Is)...)) {return aa(long(Is)...);} // a lambda to call aa(Is...)
36 (std::make_index_sequence<R>{}, a) // apply it
37 };
38 // the expression a(long(Is)...) should compile, with a pack of R longs
39 // the concept says nothing about the return type of this function
40 // NB decltype is crucial for the concept to work, to use SFINAE.
41 // if not present, the concept will fail to compile for an A for which the a(Is...) it not well formed.
42 };
43// clang-format on
44#else
45 namespace details {
46 template <auto... Is, typename A>
47 auto call_on_R_longs(std::index_sequence<Is...>, A const &a) -> decltype(a((long{Is})...)); // no impl needed
48 }
49
50 template <typename A, int R>
51 concept CallableWithLongs = requires(A const &a) {
52 { details::call_on_R_longs(std::make_index_sequence<R>{}, a) };
53 };
54
55#endif

104 template <typename H, typename T = typename std::remove_cvref_t<H>::value_type>
105 concept Handle = requires(H const &h) {
106 requires std::is_same_v<typename std::remove_cvref_t<H>::value_type, T>;
107 { h.is_null() } noexcept -> std::same_as<bool>;
108 { h.data() } noexcept -> std::same_as<T *>;

120 // ------- Array ----------
121
123 template <typename A>
124 concept Array = requires(A const &a) {
125 // A has a shape() which returns an array<long, R>
126 // its length is the rank, as deduced by get_rank
127 { a.shape() } -> StdArrayOfLong;
128
129 // Contract: Expects the size to be equal to the product of the elements in shape()
130 { a.size() } -> std::same_as<long>;
131
132 // a(0,0,0,0... R times) returns something, which is of type value_type by definition
133 requires CallableWithLongs<A, get_rank<A>>;
134 };
135
137 template <typename A, typename A_t = std::remove_cvref_t<A>>
138 concept MemoryArray = Array<A> && requires(A &a) {
139 // Has a storage_t that is a memory handle
140 typename A_t::storage_t;
141 requires mem::Handle<typename A_t::storage_t>;
142
143 // There is a member-type value_type that maybe const
144 typename A_t::value_type;
145
146 // We can acquire the pointer to the underlying data
147 {
148 a.data()
149 } -> std::same_as<std::conditional_t<std::is_const_v<std::remove_reference_t<A>> || std::is_const_v<typename A_t::value_type>,
150 const get_value_t<A>, get_value_t<A>> *>;
151
152 // Exposes the memory stride for each dimension
153 { a.indexmap().strides() } -> StdArrayOfLong;
154 };
