27 template <Chan_t Chan> void chi2_tau<Chan>::accumulate(mc_weight_t sign) {
28 // Accumulate sign
29 Z += sign;
30
31 // Calculate chi2ph
32 for (int bl1 : range(params.n_blocks()))
33 for (int bl2 : range(params.n_blocks())) {
34
35 auto &det1 = qmc_config.dets[bl1];
36 auto &det2 = qmc_config.dets[bl2];
37 auto &chi2_tau = chi2_tau_(bl1, bl2);
38 int bl1_size = chi2_tau.target_shape()[0];
39 int bl2_size = chi2_tau.target_shape()[2];
40
41 for (int a : range(bl1_size))
42 for (int b : range(bl1_size))
43 for (int c : range(bl2_size))
44 for (int d : range(bl2_size))
45 for (auto tau : tau_mesh) {
46
47 cdag_t cdag_a, cdag_c;
48 c_t c_b, c_d;
49
50 auto tau_point = make_tau_t(double(tau));
51 auto taup_point = tau_point;
52
53 // Time-Ordering
54 tau_point.n += 3; // tau -> tau^{+++}
55 taup_point.n += 2; // taup -> tau^{++}
56
57 if (tau.index() == params.n_tau_chi2 - 1) {
58 tau_point = tau_t::get_beta_minus();
59 taup_point = tau_t::get_beta_minus_minus();
60 }
61
62 if constexpr (Chan == Chan_t::PP) { // Particle-particle channel
63
64 cdag_a = cdag_t{tau_point, a};
65 c_b = c_t{tau_t::get_zero_plus(), b};
66 cdag_c = cdag_t{taup_point, c};
67 c_d = c_t{tau_t::get_zero(), d};
68
69 } else if constexpr (Chan == Chan_t::PH) { // Particle-hole channel
70
71 cdag_a = cdag_t{tau_point, a};
72 c_b = c_t{taup_point, b};
73 cdag_c = cdag_t{tau_t::get_zero_plus(), c};
74 c_d = c_t{tau_t::get_zero(), d};
75 }
76
77 if (bl1 == bl2)
78 chi2_tau[tau](a, b, c, d) += sign * det1.try_insert2(0, 1, 0, 1, c_b, c_d, cdag_a, cdag_c);
79 else
80 chi2_tau[tau](a, b, c, d) += sign * det1.try_insert(0, 0, c_b, cdag_a) * det2.try_insert(0, 0, c_d, cdag_c);
81
82 det1.reject_last_try();
83 det2.reject_last_try();
84 }
85 }
86 }
