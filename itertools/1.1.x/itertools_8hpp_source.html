26 template <class Iter, class Value, class Tag = std::forward_iterator_tag, class Reference = Value &, class Difference = std::ptrdiff_t>

29 /*
30 * A helper for the implementation of forward iterators using CRTP
31 *
32 * @tparam Iter
33 * The Iterator Class to be implemented
34 * `Iter` is required to have the following member functions
35 * - value_type [const] [&] dereference()
36 * - void increment()
37 */
38 template <typename Iter, typename Value, typename Reference, typename Difference>
39 struct iterator_facade<Iter, Value, std::forward_iterator_tag, Reference, Difference> {
40
41 private:
42 Iter &self() { return static_cast<Iter &>(*this); }
43 [[nodiscard]] Iter const &self() const { return static_cast<const Iter &>(*this); }
44
45 public:
46 using value_type = Value;
47 using reference = Reference;
48 using pointer = Value *;
49 using difference_type = Difference;
50 using iterator_category = std::forward_iterator_tag;
51
52 Iter &operator++() {
53 self().increment();
54 return self();
55 }
56
57 Iter operator++(int) {
58 Iter c = self();
59 self().increment();
60 return c;
61 }
62
63 decltype(auto) operator*() const { return self().dereference(); }
64 decltype(auto) operator->() const { return operator*(); }
65 };

67 template <typename Iter, typename EndIter>
68 inline typename std::iterator_traits<Iter>::difference_type distance(Iter first, EndIter last) {
69 if constexpr (std::is_same_v<typename std::iterator_traits<Iter>::iterator_category, std::random_access_iterator_tag>) {
70 // Difference should be defined also for the the case that last is a sentinel
71 return last - first;
72 } else {
73 typename std::iterator_traits<Iter>::difference_type r(0);
74 for (; first != last; ++first) ++r;
75 return r;
76 }
77 }

171 /********************* Product Iterator ********************/
172
173 template <typename TupleSentinel, typename... It>
174 struct prod_iter : iterator_facade<prod_iter<TupleSentinel, It...>, std::tuple<typename std::iterator_traits<It>::value_type...>> {
175
176 std::tuple<It...> its_begin;
177 TupleSentinel its_end;
178 std::tuple<It...> its = its_begin;
179
180 prod_iter(std::tuple<It...> its_begin, TupleSentinel its_end) : its_begin(std::move(its_begin)), its_end(std::move(its_end)) {}
181
182 template <int N>
183 void _increment() {
184 ++std::get<N>(its);
185 if constexpr (N < sizeof...(It) - 1) {
186 if (std::get<N>(its) == std::get<N>(its_end)) {
187 std::get<N>(its) = std::get<N>(its_begin);
188 _increment<N + 1>();
189 }
190 }
191 }
192 void increment() { _increment<0>(); }
193
194 bool operator==(prod_iter const &other) const { return its == other.its; }
195
196 template <typename U>
197 bool operator==(sentinel_t<U> const &s) const {
198 return (s.it == std::get<sizeof...(It) - 1>(its));
199 }
200
201 private:
202 template <size_t... Is>
203 [[gnu::always_inline]] [[nodiscard]] auto tuple_map_impl(std::index_sequence<Is...>) const {
204 return std::tuple<decltype(*std::get<Is>(its))...>(*std::get<Is>(its)...);
205 }
206
207 public:
208 [[nodiscard]] decltype(auto) dereference() const { return tuple_map_impl(std::index_sequence_for<It...>{}); }
209 };

528 template <typename R>
529 auto make_vector_from_range(R const &r) {
530 std::vector<std::decay_t<decltype(*(std::begin(r)))>> vec{}; // decltype returns a &
531 if constexpr(std::is_same_v<decltype(std::cbegin(r)), decltype(std::cend(r))>) {

533 vec.reserve(total_size);

698 inline std::pair<std::ptrdiff_t, std::ptrdiff_t> chunk_range(std::ptrdiff_t start, std::ptrdiff_t end, long n_chunks, long rank) {
699 auto total_size = end - start;
700 auto chunk_size = total_size / n_chunks;
701 auto n_large_nodes = total_size - n_chunks * chunk_size;
702 if (rank < n_large_nodes) // larger nodes have size chunk_size + 1
703 return {start + rank * (chunk_size + 1), start + (rank + 1) * (chunk_size + 1)};
704 else // smaller nodes have size chunk_size
705 return {start + n_large_nodes + rank * chunk_size, start + n_large_nodes + (rank + 1) * chunk_size};
706 }
