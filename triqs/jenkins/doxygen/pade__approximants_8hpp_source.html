32 typedef std::complex<double> dcomplex;
33
34 // This implementation is based on a Fortran code written by
35 // A. Poteryaev <Alexander.Poteryaev _at_ cpht.polytechnique.fr>
36 //
37 // The original algorithm is described in
38 // H. J. Vidberg, J. W. Serene. J. Low Temp. Phys. 29, 3-4, 179 (1977)
39
40 struct gmp_complex {
41 mpf_class re, im;
42 gmp_complex operator*(const gmp_complex &rhs) { return {rhs.re * re - rhs.im * im, rhs.re * im + rhs.im * re}; }
43 friend gmp_complex inverse(const gmp_complex &rhs) {
44 mpf_class d = rhs.re * rhs.re + rhs.im * rhs.im;
45 if (d == 0) TRIQS_RUNTIME_ERROR << "pade_approximant: GMP division by zero";
46 return {rhs.re / d, -rhs.im / d};
47 }
48 gmp_complex operator/(const gmp_complex &rhs) { return (*this) * inverse(rhs); }
49 gmp_complex operator+(const gmp_complex &rhs) { return {rhs.re + re, rhs.im + im}; }
50 gmp_complex operator-(const gmp_complex &rhs) { return {re - rhs.re, im - rhs.im}; }
51 friend mpf_class real(const gmp_complex &rhs) { return rhs.re; }
52 friend mpf_class imag(const gmp_complex &rhs) { return rhs.im; }
53 mpf_class norm() const { return real(*this) * real(*this) + imag(*this) * imag(*this); }
54 gmp_complex &operator=(const std::complex<double> &rhs) {
55 re = real(rhs);
56 im = imag(rhs);
57 return *this;
58 }
59 friend std::ostream &operator<<(std::ostream &out, gmp_complex const &r) {
60 return out << " gmp_complex(" << r.re << "," << r.im << ")" << std::endl;
61 }
62 };
63
64 class pade_approximant {
65
66 nda::vector<dcomplex> z_in; // Input complex frequency points
67 nda::vector<dcomplex> a; // Pade coefficients
68
69 public:
70 static const int GMP_default_prec = 256; // Precision of GMP floats to use during a Pade coefficients calculation.
71
72 pade_approximant(const nda::vector<dcomplex> &z_in_, const nda::vector<dcomplex> &u_in) : z_in(z_in_), a(z_in.size()) {
73
74 int N = z_in.size();
75
76 // Change the default precision of GMP floats.
77 unsigned long old_prec = mpf_get_default_prec();
78 mpf_set_default_prec(GMP_default_prec); // How do we determine it?
79
80 nda::array<gmp_complex, 2> g(N, N);
81 gmp_complex MP_0 = {0.0, 0.0};
82 g() = MP_0;
83 for (int f = 0; f < N; ++f) { g(0, f) = u_in(f); };
84
85 gmp_complex MP_1 = {1.0, 0.0};
86
87 for (int p = 1; p < N; ++p) {
88
89 // If |g| is very small, the continued fraction should be truncated.
90 if (g(p - 1, p - 1).norm() < 1.0e-20) break;
91
92 for (int j = p; j < N; ++j) {
93 gmp_complex x = g(p - 1, p - 1) / g(p - 1, j) - MP_1;
94 gmp_complex y;
95 y = z_in(j) - z_in(p - 1);
96 g(p, j) = x / y;
97 }
98 }
99
100 for (int j = 0; j < N; ++j) {
101 gmp_complex gj = g(j, j);
102 a(j) = dcomplex(real(gj).get_d(), imag(gj).get_d());
103 }
104
105 // Restore the precision.
106 mpf_set_default_prec(old_prec);
107 }
108
109 // give the value of the pade continued fraction at complex number e
110 dcomplex operator()(dcomplex e) const {
111
112 dcomplex A1(0);
113 dcomplex A2 = a(0);
114 dcomplex B1(1.0);
115
116 int N = a.size();
117 for (int i = 0; i <= N - 2; ++i) {
118 dcomplex Anew = A2 + (e - z_in(i)) * a(i + 1) * A1;
119 dcomplex Bnew = 1.0 + (e - z_in(i)) * a(i + 1) * B1;
120 A1 = A2 / Bnew;
121 A2 = Anew / Bnew;
122 B1 = 1.0 / Bnew;
123 }
124
125 return A2;
126 }
127 };
