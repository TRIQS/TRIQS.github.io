47 template <MemoryGf G> void mpi_broadcast(G &&g, mpi::communicator c, int root) { // NOLINT (temporary views are allowed)
48 constexpr bool is_view = std::decay_t<G>::is_view;
49
50 // broadcast mesh
51 if constexpr (!is_view) {
52 // for non-view GFs, we broadcast the mesh directly into the GF
53 mpi::broadcast(g._mesh, c, root);
54 } else {
55 // for views, we keep the mesh in the GF but check that it is the same as the mesh on the root process
56 auto m = g.mesh();
57 mpi::broadcast(m, c, root);
58 EXPECTS(m.mesh_hash() == g.mesh().mesh_hash());
59 }
60
61 // broadcast data
62 mpi::broadcast(g.data(), c, root);
63 }

91 template <MemoryGf G1, MemoryGf G2>
92 void mpi_reduce_into(G1 const &g_in, G2 &&g_out, mpi::communicator c, // NOLINT (temporary views are allowed here)
93 int root, bool all, MPI_Op op) {
94 constexpr bool is_view = std::decay_t<G2>::is_view;
95
96 // check the meshes of the input GFs
97 EXPECTS(mpi::all_equal(g_in.mesh().mesh_hash(), c));
98
99 // assign (check) the mesh of the output GF (view) on receiving ranks
100 if (c.rank() == root || all) {
101 if constexpr (is_view) {
102 EXPECTS(g_in.mesh().mesh_hash() == g_out.mesh().mesh_hash());
103 } else {
104 g_out._mesh = g_in.mesh();
105 }
106 }
107
108 // reduce the data
109 mpi::reduce_into(g_in.data(), g_out.data(), c, root, all, op);
110 }
