53 // special case for imtime
54
55 template <typename Tag, nda::AnyOf<mesh::imtime, mesh::dlr_imtime> M> M combine_mesh(M const &l, M const &r) {
56
57 if constexpr (std::is_same<Tag, utility::tags::multiplies>::value or std::is_same<Tag, utility::tags::divides>::value) {
58 bool eq = (std::abs(l.beta() - r.beta()) < 1.e-15) and (l.size() == r.size());
59 if (!eq)
60 TRIQS_RUNTIME_ERROR << "Mesh mismatch: In Green Function Expression, the mesh of the 2 operands should be equal" << l << " vs " << r;
61
62 // compute the stat of the product, divide.
63 int s = (int(l.statistic()) + int(r.statistic())) % 2;
64 statistic_enum stat = (s == 0 ? Boson : Fermion);
65 if constexpr (std::is_same_v<M, mesh::imtime>)
66 return M{l.beta(), stat, l.size()};
67 else // dlr_imtime
68 return M{l.beta(), stat, l.w_max(), l.eps()};
69 } else {
70 if (!(l == r))
71 TRIQS_RUNTIME_ERROR << "Mesh mismatch: In Green Function Expression, the mesh of the 2 operands should be equal" << l << " vs " << r;
72 return l;
73 }
74 }

120 template <typename Tag, typename L, typename R> struct gf_expr : TRIQS_CONCEPT_TAG_NAME(GreenFunction) {
121 using L_t = typename std::remove_reference<L>::type;
122 using R_t = typename std::remove_reference<R>::type;
123 using mesh_t = typename gfs_expr_tools::_or_<typename L_t::mesh_t, typename R_t::mesh_t>::type;
124 using target_t = typename gfs_expr_tools::_or_<typename L_t::target_t, typename R_t::target_t>::type;
125 using regular_t = gf<mesh_t, target_t>;
126 static_assert(!std::is_same<mesh_t, void>::value, "Cannot combine two gf expressions with different variables");
127 static_assert(!std::is_same<target_t, void>::value, "Cannot combine two gf expressions with different target");
128
129 L l;
130 R r;
131 template <typename LL, typename RR> gf_expr(LL &&l_, RR &&r_) : l(std::forward<LL>(l_)), r(std::forward<RR>(r_)) {}
132
133 private:
134 mutable std::optional<mesh_t> _mesh;
135
136 public:
137 auto const &mesh() const {
138 if (not _mesh) _mesh.emplace(gfs_expr_tools::combine_mesh<Tag>(l.mesh(), r.mesh()));
139 return *_mesh;
140 }
141 auto data_shape() const { return gfs_expr_tools::combine_shape()(l, r); }
142
143 template <typename... Keys> decltype(auto) operator[](Keys &&...keys) const {
144 return utility::operation<Tag>()(l.operator[](std::forward<Keys>(keys)...), r.operator[](std::forward<Keys>(keys)...)); // Clang Fix
145 }
146 template <typename... Args> decltype(auto) operator()(Args &&...args) const {
147 return utility::operation<Tag>()(l(std::forward<Args>(args)...), r(std::forward<Args>(args)...));
148 }
149 friend std::ostream &operator<<(std::ostream &sout, gf_expr const &expr) {
150 return sout << "(" << expr.l << " " << utility::operation<Tag>::name << " " << expr.r << ")";
151 }
152 };
