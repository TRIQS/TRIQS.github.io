364 friend void h5_read(h5::group g, std::string const &name, dlr_imtime &m) {
365 h5::group gr = g.open_group(name);
366 h5::assert_hdf5_format(gr, m, true); // NOLINT (downcasting to base class)
367 auto b = h5::read<double>(gr, "beta");
368 auto stat = (h5::read<std::string>(gr, "statistic") == "F" ? Fermion : Boson);
369 auto wmax = h5::read<double>(gr, "w_max");
370 auto epsilon = h5::read<double>(gr, "eps");
371 bool sym = false;
372 h5::try_read(gr, "symmetrize", sym);
373 auto freq = h5::read<nda::vector<double>>(gr, "dlr_freq");
374 auto imt = h5::read<cppdlr::imtime_ops>(gr, "dlr_it");
375 auto imf = h5::read<cppdlr::imfreq_ops>(gr, "dlr_if");
376 m = dlr_imtime{b, stat, wmax, epsilon, sym, {.freq = freq, .imt = imt, .imf = imf}};
377 }
