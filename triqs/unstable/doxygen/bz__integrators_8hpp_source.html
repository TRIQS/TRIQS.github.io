48 // helper to determine the return container dimension
49 int deduce_dim_from_expression(auto const &f_kw) {
50 namespace ph = placeholders;
51 auto f_w_temp = eval(f_kw, ph::kx = 0., ph::ky = 0., ph::kz = 0., ph::w = 0);
52 static_assert(not nda::clef::is_lazy<decltype(f_w_temp)>, "Integration expects a proper expression with placeholders.");
53 // check if this is a matrix or scalar type -- if scalar, return 1 for dim
54 return [&]() {
55 if constexpr (requires { f_w_temp.shape(); }) { // i.e. if this compiles ...
56 return f_w_temp.shape()[0]; // this is a matrix type with block dim
57 } else {
58 return 1; // this is a scalar type, will be handled as a 1D matrix

77 template <typename T>
78 requires(std::convertible_to<T, dcomplex> or std::convertible_to<T, double>) // allow for real or imaginary mesh points or numbers
79 nda::array<dcomplex, 3> integrate_ptr(auto const &f_kw, std::vector<T> const &omega_values, std::array<long, 3> const &k_grid,
80 mpi::communicator comm) {
81
82 for (auto i : {0, 1, 2}) {
83 if (k_grid[i] <= 0) std::runtime_error{"Cannot use PTR integration with kgrid dim <= 0."};
84 }
85
86 namespace ph = placeholders;
87 int block_dim = detail::deduce_dim_from_expression(f_kw);
88 auto result = nda::zeros<dcomplex>(omega_values.size(), block_dim, block_dim); // container to return
89
90 // Determine the longest direction and apply MPI chunk to this dimension, otherwise provide a simple iterator
91 auto mpi_chunk_max = [&k_grid, comm](int kdim) {
92 // chunk the biggest one with MPI
93 if (kdim == std::distance(k_grid.begin(), std::ranges::max_element(k_grid))) { return mpi::chunk(nda::range(k_grid[kdim]), comm); }
94 return itertools::slice(nda::range(k_grid[kdim]), 0,
95 k_grid[kdim]); // otherwise complains about different return types
96 };
97
98// omp reduction operation for nda array
99#pragma omp declare reduction(array_add_c_3 : nda::array<dcomplex, 3> : omp_out += omp_in) \
100 initializer(omp_priv = nda::array<dcomplex, 3>(omp_orig.shape()))
101
102#pragma omp parallel for collapse(3) reduction(array_add_c_3 : result) default(none) \
103 shared(k_grid, omega_values, f_kw, ph::kx, ph::ky, ph::kz, ph::w, mpi_chunk_max)
104 for (auto ikx : mpi_chunk_max(0)) {
105 for (auto iky : mpi_chunk_max(1)) {
106 for (auto ikz : mpi_chunk_max(2)) {
107 double kx = ikx / double(k_grid[0]);
108 double ky = iky / double(k_grid[1]);
109 double kz = ikz / double(k_grid[2]);
110 auto f_w = eval(f_kw, ph::kx = kx, ph::ky = ky, ph::kz = kz);
111 for (auto &&[n, omega] : itertools::enumerate(omega_values)) result(n, nda::range::all, nda::range::all) += eval(f_w, ph::w = omega);
112 }
113 }
114 }
115 result = mpi::all_reduce(result, comm);
116
117 // apply normalization, divide by number of k-points on integrated grid
118 result /= double(k_grid[0] * k_grid[1] * k_grid[2]);
119 return result;
120 }

135 template <typename Mesh> auto integrate_ptr(auto const &f_kw, Mesh const &w_mesh, std::array<long, 3> const &k_grid, mpi::communicator comm = {}) {
136
137 int dim = detail::deduce_dim_from_expression(f_kw);
138 auto g_out = gf{w_mesh, {dim, dim}};
139 std::vector<typename Mesh::mesh_point_t> mesh_points(w_mesh.begin(), w_mesh.end());
140 auto ptr_result = integrate_ptr(f_kw, mesh_points, k_grid, comm);
141 // fill in the GF to return
142 for (auto &&[n, w] : itertools::enumerate(mpi::chunk(w_mesh, comm))) { g_out[w] = calc(w); }

211 template <typename Mesh>
212 gf<Mesh, matrix_valued> integrate_bz(auto const &f_kw, Mesh const &w_mesh, bz_int_options const &opt, mpi::communicator comm = {}) {
213
214 namespace ph = placeholders;
215
216 // set up kgrid
217 auto k_grid = opt.k_grid;
218 auto kgrid_above_max = [&](auto &k_grid) {
219 for (auto ik : {0, 1, 2})
220 if (k_grid[ik] >= opt.k_grid_max[ik]) { return true; }
221 return false;
222 };
223
224 // REFACTOR this seems like it should go into a constructor for a bz_int_opt object
225 // note that the PTR will always run once as long as run_ptr = true
226 if (opt.tolerance <= 0) { throw std::runtime_error("Must provide a positive tolerance."); }
227 if (!opt.run_ptr and !opt.run_adaptive) {
228 throw std::runtime_error("Must choose at least one of run_ptr or run_adaptive for BZ integration to work.");
229 }
230 if (opt.run_ptr) { // check PTR options are sound
231 if (!std::all_of(k_grid.begin(), k_grid.end(), [&](int k) { return k > 0; })) { // check if values are positive
232 throw std::runtime_error("Cannot run integraton with k_grid <= 0. ");
233 }
234 // check that kgrid increment is meaningful
235 if (!std::all_of(opt.delta_k_grid.begin(), opt.delta_k_grid.end(), [&](int k) { return k >= 0; })) {
236 throw std::runtime_error("delta_k_grid cannot be negative.");
237 }
238 // delta_k_grid = 0 is reasonable only if k_grid >= k_grid_max, otherwise this will run doing nothing
239 if (std::all_of(opt.delta_k_grid.begin(), opt.delta_k_grid.end(), [&](int k) { return k == 0; }) and !kgrid_above_max(k_grid)) {
240 throw std::runtime_error("delta_k_grid can only be zero if k_grid >= k_grid_max.");
241 }
242 }
243
244 int dim = detail::deduce_dim_from_expression(f_kw);
245 auto g_out = gf{w_mesh, {dim, dim}};
246
247 // set up containers to check if ptr has converged for different points
248 std::vector<bool> ptr_converged(w_mesh.size(), false);
249 auto all_converged = [&]() { return std::ranges::all_of(ptr_converged, std::identity()); };
250
251 // set up initialize set of omega values to be run (all of them for first loop)
252 std::vector<typename Mesh::mesh_point_t> mesh_points(w_mesh.begin(), w_mesh.end());
253
254 // ------ Do the PTR -------
255 if (opt.run_ptr) {
256 do { // run loop at least once if PTR is chosen
257
258 if (opt.verbose) {
259 int remaining_ptr = ptr_converged.size() - std::reduce(ptr_converged.begin(), ptr_converged.end());
260 std::cout << "Points remaining unconverged: " << remaining_ptr << ", now running with k-grid " << k_grid[0] << " " << k_grid[1] << " "
261 << k_grid[2] << std::endl;
262 }
263
264 // update the list of omega values we need to cover
265 auto ptr_result = integrate_ptr(f_kw, mesh_points, k_grid, comm);
266
267 // check which ones are converged after this run
268 for (auto &&[n, w] : itertools::enumerate(mesh_points)) {
269 ptr_converged[w.data_index()] = (max_element(abs(ptr_result(n, nda::range::all, nda::range::all) - g_out[w])) < opt.tolerance);
270 g_out[w] = ptr_result(n, nda::range::all, nda::range::all);
271 }
272 // update list of unconverged frequencies to work on
273 // REFACTOR it's much nicer to use the below line if we later can
274 // mesh_points = mesh_points | std::views::filter([&](auto om) { return !ptr_converged[om.data_index()]; }) | std::ranges::to<std::vector>();
275 std::vector<typename Mesh::mesh_point_t> unconv_mesh_points;
276 for (auto w : mesh_points) {
277 if (!ptr_converged[w.data_index()]) unconv_mesh_points.emplace_back(w);
278 }
279 mesh_points = unconv_mesh_points;
280
281 // increment the grid for the next iteration
282 for (auto ik : {0, 1, 2}) k_grid[ik] += opt.delta_k_grid[ik];
283
284 } while (!all_converged() and !kgrid_above_max(k_grid));
285 }
286
287 // ------- Execute adaptive algo for the frequencies PTR could not do ----------
288 if (opt.run_adaptive) {
289 adaptive_options adaptive_opt = {.tolerance = opt.tolerance};
290 if (opt.verbose) std::cout << "Running adaptive integration on remaining points." << std::endl;
291 auto calc = integrate_adaptive(f_kw, adaptive_opt);
292 // adaptive evaluation at each frequency is MPI parallel;
293 // possibly could be done better but this is ok for now
294 for (auto &&[n, w] : itertools::enumerate(mpi::chunk(g_out.mesh(), comm))) {
295 if (not ptr_converged[n]) g_out[w] = calc(w);
296 }
297 }
298 return g_out;
299 }
