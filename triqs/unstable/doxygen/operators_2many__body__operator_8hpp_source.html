79 struct canonical_ops_t {
84 bool dagger;
85
87 indices_t indices;
88
109 auto operator<=>(canonical_ops_t const &b) const {
110 if (dagger != b.dagger) return !dagger <=> !b.dagger; // c+ < c
111
112 // Compare indices element-wise, ordering by type index first (long < string < double < array)
113 auto compare_indices = [](indices_t const &a, indices_t const &b) -> std::strong_ordering {
114 for (size_t i = 0; i < std::min(a.size(), b.size()); ++i) {
115 // Compare by type index first
116 if (a[i].index() != b[i].index()) return a[i].index() <=> b[i].index();
117 // Same type: compare values
118 auto cmp = std::visit(
119 [](auto const &x, auto const &y) -> std::strong_ordering {
120 if constexpr (std::is_same_v<std::decay_t<decltype(x)>, std::decay_t<decltype(y)>>) {
121 if (x == y) return std::strong_ordering::equal;
122 return x < y ? std::strong_ordering::less : std::strong_ordering::greater;
123 }
124 return std::strong_ordering::equal; // unreachable due to index check
125 },
126 a[i], b[i]);
127 if (cmp != std::strong_ordering::equal) return cmp;
128 }
129 return a.size() <=> b.size();
130 };
131
132 return dagger ? compare_indices(indices, b.indices) : compare_indices(b.indices, indices);
133 }
134
141 bool operator==(canonical_ops_t const &b) const { return (*this <=> b) == 0; }
142
147 void serialize(auto &ar) const { ar & dagger & indices; }
148
153 void deserialize(auto &ar) { ar & dagger & indices; }
154 };

637 many_body_operator_generic &operator*=(many_body_operator_generic const &op) {
638 monomials_map_t tmp_map; // product will be stored here
639 for (auto const &m : monomials)
640 for (auto const &op_m : op.monomials) {
641 monomial_t product_m;
642 product_m.reserve(m.first.size() + op_m.first.size());
643 for (auto const &tmp_op : m.first) product_m.push_back(tmp_op);
644 for (auto const &tmp_op : op_m.first) product_m.push_back(tmp_op);
645 normalize_and_insert(product_m, m.second * op_m.second, tmp_map);
646 }
647 std::swap(monomials, tmp_map);
648 return *this;
649 }

760 // Normalize a monomial and insert into a map
761 static void normalize_and_insert(monomial_t m, scalar_t coeff, monomials_map_t &target) {
762 // The normalization is done by employing a simple bubble sort algorithms.
763 // Apart from sorting elements this function keeps track of the sign and
764 // recursively calls itself if a permutation of two operators produces a new
765 // monomial
766 if (m.size() >= 2) {
767 bool is_swapped;
768 do {
769 is_swapped = false;
770 for (std::size_t n = 1; n < m.size(); ++n) {
771 canonical_ops_t &prev_index = m[n - 1];
772 canonical_ops_t &cur_index = m[n];
773 if (prev_index == cur_index) return; // The monomial is effectively zero
774 if (prev_index > cur_index) {
775 // Are we swapping C and C^+ with the same indices?
776 // A bit ugly ...
777 canonical_ops_t cur_index_flipped_type(cur_index);
778 cur_index_flipped_type.dagger = !cur_index_flipped_type.dagger;
779 if (prev_index == cur_index_flipped_type) {
780 monomial_t new_m;
781 new_m.reserve(m.size() - 2);
782 std::copy(m.begin(), m.begin() + n - 1, std::back_inserter(new_m));
783 std::copy(m.begin() + n + 1, m.end(), std::back_inserter(new_m));
784 normalize_and_insert(new_m, coeff, target);
785 }
786 coeff = -coeff;
787 std::swap(prev_index, cur_index);
788 is_swapped = true;
789 }
790 }
791 } while (is_swapped);
792 }
793
794 // Insert the result
795 bool is_new_monomial;
796 typename monomials_map_t::iterator it;
797 std::tie(it, is_new_monomial) = target.insert(std::make_pair(std::move(m), coeff));
798 if (!is_new_monomial) {
799 it->second += coeff;
800 erase_zero_monomial(target, it);
801 }
802 }
