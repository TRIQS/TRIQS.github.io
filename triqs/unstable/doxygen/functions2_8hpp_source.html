70 template <int N = 0, typename BG, typename BA = std::vector<typename BG::g_t::data_t::regular_type>>
71 std::pair<std::vector<typename BG::g_t::data_t::regular_type>, double> fit_tail(BG const &bg, BA const &known_moments = {})
72 requires(is_block_gf_v<BG, 1>)
73 {
74 double max_err = 0.0;
75 std::vector<typename BG::g_t::data_t::regular_type> tail_vec;
76 for (auto [i, g_bl] : itertools::enumerate(bg)) {
77 auto [tail, err] = known_moments.empty() ? fit_tail<N, typename BG::g_t>(g_bl) : fit_tail<N, typename BG::g_t>(g_bl, known_moments[i]);
78 tail_vec.emplace_back(std::move(tail));
79 max_err = std::max(err, max_err);
80 }
81 return std::make_pair(tail_vec, max_err);
82 }

96 template <int N = 0, typename G, typename A = typename G::data_t>
97 std::pair<typename A::regular_type, double> fit_hermitian_tail(G const &g, A const &known_moments = {})
98 requires(is_gf_v<G>)
99 {
100 std::optional<long> inner_matrix_dim;
101 constexpr int rank = G::target_t::rank;
102 if (rank == 0)
103 inner_matrix_dim = 1;
104 else if (rank == 2 && g.target_shape()[0] == g.target_shape()[1]) {
105 inner_matrix_dim = g.target_shape()[0];
106 } else
107 TRIQS_RUNTIME_ERROR << "Incompatible target_shape for fit_hermitian_tail\n";
108
109 auto const &m = get_mesh<N>(g);
110 return m.get_tail_fitter().template fit_hermitian<N>(m, make_array_const_view(g.data()), true, make_array_const_view(known_moments),
111 inner_matrix_dim);
112 }

159 template <int N = 0, typename G> auto make_zero_tail(G const &g, int n_moments = 10) {
160 if constexpr (is_gf_v<G>) { // gf[_const][_view]<V, T>
161 auto sh = nda::rotate_index_view<N>(make_const_view(g.data())).shape();
162 sh[0] = n_moments;
163 return nda::zeros<dcomplex>(sh);
164 } else if constexpr (is_block_gf_v<G>) { // block[2]_gf[_const][_view]<V, T>
165 return map_block_gf([&](auto const &g_bl) { return make_zero_tail<N>(g_bl, n_moments); }, g);

201 // auxiliary function : invert the data : one function for all matrix valued gf (save code).
202 template <typename M> void invert_in_place(gf_view<M, matrix_valued> g) {
203 auto &a = g.data();
204 auto mesh_lengths = stdutil::mpop<2>(a.indexmap().lengths());
205 nda::for_each(mesh_lengths, [&a](auto &&...i) { nda::linalg::inv_in_place(make_matrix_view(a(i..., range::all, range::all))); });
206 }

208 template <typename M> gf<M, matrix_valued> inverse(gf<M, matrix_valued> g) {
209 invert_in_place(g());
210 return g;

293 template <typename A3, typename T> void _gf_data_mul_R(A3 &&a, matrix<T> const &r) {
294 for (int i = 0; i < first_dim(a); ++i) { // Rely on the ordering
295 matrix_view<T> v = a(i, nda::range::all, nda::range::all);
296 v = v * r;
297 }
298 }
299
300 template <typename A3, typename T> void _gf_data_mul_L(matrix<T> const &l, A3 &&a) {
301 for (int i = 0; i < first_dim(a); ++i) { // Rely on the ordering
302 matrix_view<T> v = a(i, nda::range::all, nda::range::all);
303 v = l * v;
304 }
305 }

307 template <typename M, typename T> gf<M, matrix_valued> operator*(gf<M, matrix_valued> g, matrix<T> r) {
308 _gf_data_mul_R(g.data(), r);
309 return g;
310 }

312 template <typename M, typename T> gf<M, matrix_valued> operator*(matrix<T> l, gf<M, matrix_valued> g) {
313 _gf_data_mul_L(l, g.data());
314 return g;

317 /*------------------------------------------------------------------------------------------------------
318 * Multiply by matrices left and right, in place.
319 * Optimized for speed.
320 *-----------------------------------------------------------------------------------------------------*/
321
322 template <typename A, typename B, typename M> void set_from_gf_data_mul_LR(A &a, M const &l, B const &b, M const &r) {
323 auto tmp = matrix<typename M::value_type>(second_dim(b), second_dim(r));
324 auto _ = nda::range::all;
325 for (int i = 0; i < first_dim(a); ++i) { // Rely on the ordering
326 auto rhs_v = make_matrix_view(b(i, _, _));
327 auto lhs_v = make_matrix_view(a(i, _, _));
328 nda::blas::gemm(1, rhs_v, r, 0, tmp);
329 nda::blas::gemm(1, l, tmp, 0, lhs_v);
330 }
331 }

333 template <typename G1, typename G2, typename M> void set_from_L_G_R(G1 &g1, M const &l, G2 const &g2, M const &r) {
334 set_from_gf_data_mul_LR(g1.data(), l, g2.data(), r);
