39 template <typename T> struct lin_binning {
40 long max_n_bins = 0; // Maximum number of bins
41 long bin_capacity = 1; // Current capacity of each bin (must be >= 1)
42 long last_bin_count = 0; // Number of data points the currently active bin [bins.back()]
43 std::vector<T> bins; // Bins with accumulated data (stores means)
44 long count = 0; // Total number of elements added to accumulator: Information only
45
46 // static std::string hdf5_format() { return "linear_bins"; }
47
48 lin_binning() = default;
49
50 lin_binning(T const &data_instance, long max_n_bins, long bin_capacity) : max_n_bins(max_n_bins), bin_capacity(bin_capacity) {
51 if (max_n_bins == 0) return;
52 T data_instance_local = data_instance;
53 data_instance_local = 0;
54 if (max_n_bins > 0) { bins.reserve(max_n_bins); }
55 bins.emplace_back(std::move(data_instance_local));
56 }
57
58 [[nodiscard]] long n_bins() const { return bins.size(); }
59
60 template <typename U> lin_binning<T> &operator<<(U &&x) {
61 ++count;
62 if (max_n_bins == 0) return *this;
63 // Check if all bins are full and compress if needed
64 if (max_n_bins > 1 && n_bins() == max_n_bins && last_bin_count == bin_capacity) {
65 compress(2); // Adjusts bin_capacity & last_bin_count
66 }
67 // Check if current bin full: push new bin or add data to current bin
68 if (last_bin_count == bin_capacity && max_n_bins != 1) {
69 bins.emplace_back(std::forward<U>(x));
70 last_bin_count = 1;
71 } else {
72 last_bin_count++;
73 bins.back() += (x - bins.back()) / last_bin_count;
74 }
75 return *this;
76 }
77
78 // Compresses bins by scaling up bin_capacity by compression_factor (>= 2).
79 void compress(int compression_factor) {
80 if (max_n_bins == 0 || compression_factor < 2) return;
81 const int bins_left = n_bins() % compression_factor;
82 int n_bins_new = n_bins() / compression_factor;
83 int n_bins_last_chunk = compression_factor;
84 if (bins_left != 0) {
85 n_bins_new++;
86 n_bins_last_chunk = bins_left;
87 }
88 // Compress data into new bins, except the last new bin
89 for (int i = 0; i < n_bins_new - 1; ++i) {
90 if (i != 0) { bins[i] = std::move(bins[compression_factor * i]); }
91 for (int j = 1; j < compression_factor; j++) { bins[i] += bins[compression_factor * i + j]; }
92 bins[i] /= compression_factor;
93 }
94 // Last new bin is special: last old bin could be filled below capacity
95 int new_last_bin_count = last_bin_count + (n_bins_last_chunk - 1) * bin_capacity;
96 auto &new_last_bin = bins[n_bins_new - 1];
97 new_last_bin = std::move(bins[compression_factor * (n_bins_new - 1)]);
98 // If n_bins_last_chunk == 1, we have already copied its value above
99 if (n_bins_last_chunk > 1) {
100 for (int j = 1; j < n_bins_last_chunk - 1; j++) { new_last_bin += bins[compression_factor * (n_bins_new - 1) + j]; }
101 new_last_bin *= bin_capacity; // full bins in last chunk
102 new_last_bin += bins[compression_factor * (n_bins_new - 1) + (n_bins_last_chunk - 1)] * last_bin_count;
103 new_last_bin /= new_last_bin_count;
104 }
105 // Adjust final parameters
106 bins.resize(n_bins_new);
107 last_bin_count = new_last_bin_count;
108 bin_capacity *= compression_factor;
109 }
110 };

129 // Updating Formulae
130 // Mk \equiv \frac{1}{k} \sum_{i=1}^k x_i
131 // Qk \equiv \sum_{i=1}^k (x_i - M_k)^2
132 // Update formula
133 // M_{k} = M_{k-1} + \frac{x- M_{k-1}}{k}
134 // Q_{k} = Q_{k-1} + \frac{(k-1)(x- M_{k-1})^2}{k}
135 // Ref: see e.g. Chan, Golub, LeVeque. Am. Stat. 37, 242 (1983) and therein
136 template <typename T> struct log_binning {
137
138 std::vector<T> Mk; // Cf comments
139 using Q_t = get_real_t<T>;
140 std::vector<Q_t> Qk; // Cf comments
141 int max_n_bins = 0;
142 std::vector<T> acc; // partial accumulators at size 2^(n+1). WARNING: acc[n] correspond to sum[n+1] to save unecessary temporaries
143 std::vector<int> acc_count; // number of elements for partial accumulators at size 2^(n+1)
144 long count = 0; // Number of elements added to accumulator
145
146 //static std::string hdf5_format() { return "log_binning"; }
147
148 log_binning() = default;
149
150 log_binning(T const &data_instance, int max_n_bins) : max_n_bins(max_n_bins) {
151 if (max_n_bins == 0) return;
152 T data_instance_local = data_instance;
153 data_instance_local = 0;
154 if (max_n_bins > 0) {
155 Qk.reserve(max_n_bins);
156 Mk.reserve(max_n_bins);
157 acc.reserve(max_n_bins - 1);
158 acc_count.reserve(max_n_bins - 1);
159 }
160 // If max_n_bins == 1, we don't need acc / acc_count, otherwise we initialize the first element
161 if (max_n_bins != 1) {
162 acc.emplace_back(data_instance_local);
163 acc_count.push_back(0);
164 }
165 // Multiply to ensure element can be element-wise multiplied [FIXME with concepts]
166 Qk.emplace_back(make_real(data_instance_local * data_instance_local));
167 Mk.emplace_back(std::move(data_instance_local));
168 }
169
170 [[nodiscard]] long n_bins() const { return Qk.size(); }
171
172 template <typename U> log_binning<T> &operator<<(U const &x) {
173 if (max_n_bins == 0) return *this;
174
175 using nda::conj;
176 using nda::real;
177
178 ++count;
179
180 // If max_n_bins == 1, there is only one (Mk, Qk) and we skip direclty to updating that below
181 if (max_n_bins != 1) {
182 // go up in n as long as the acc_count becomes full and add the acc in the
183 // then go down, and store the acc
184
185 int n = 0;
186 for (; n < acc.size(); ++n) {
187 if (n == 0) {
188 acc[n] += x; // n = 0 case is special, as it involves new data input
189 } else {
190 acc[n] += acc[n - 1];
191 }
192 acc_count[n]++;
193 if (acc_count[n] < 2) break;
194 }
195
196 // When reaching power of 2 nr data points, add a new pair of (Mk, Qk) = (acc.back(), 0)
197 // and add an extra space in acc, acc_count
198 if (count == (1 << acc.size()) && (max_n_bins < 0 || n_bins() < max_n_bins)) {
199 T last_acc = acc.back(); // Force copy
200
201 if (max_n_bins < 0 || n_bins() < (max_n_bins - 1)) {
202 acc.emplace_back(last_acc);
203 acc_count.push_back(1);
204 }
205
206 Mk.emplace_back(std::move(last_acc));
207 auto data_instance_q = Qk[0];
208 data_instance_q = 0;
209 Qk.emplace_back(std::move(data_instance_q));
210 }
211
212 n--;
213 for (; n >= 0; n--) {
214 auto bin_capacity = (1ul << (n + 1)); // 2^(n+1)
215 T x_m = (acc[n] / bin_capacity - Mk[n + 1]); // Force T if expression template.
216 auto k = count / bin_capacity;
217 Qk[n + 1] += static_cast<nda::get_value_t<Q_t>>((k - 1) / double(k)) * make_real(conj(x_m) * x_m);
218 Mk[n + 1] += x_m / k;
219 acc_count[n] = 0;
220 acc[n] = 0;
221 }
222 }
223
224 // Update the (Mk, Qk) pair with no binning (bin capacity: 2^0)
225 auto k = count;
226 T x_m = (x - Mk[0]);
227 Qk[0] += static_cast<nda::get_value_t<Q_t>>((k - 1) / double(k)) * make_real(conj(x_m) * x_m);
228 Mk[0] += x_m / k;
229
230 return *this;
231 }
232 };

402 [[nodiscard]] auto log_bin_errors() const {
403 auto res1 = log_bins.Qk;
404 std::vector<long> count_vec{};
405
406 if (res1.size() == 0) return std::make_pair(res1, count_vec);
407 count_vec.reserve(res1.size());
408
409 for (int n = 0; n < res1.size(); ++n) {
410 long count_n = (log_bins.count >> n); // == count / 2^n (rounded down)
411 if (count_n <= 1) {
412 res1[n] = 0;
413 } else {
414 using std::sqrt;
415 res1[n] = sqrt(res1[n] / (count_n * (count_n - 1)));
416 }
417 count_vec.emplace_back(count_n);
418 }
419 return std::make_pair(res1, count_vec);
420 }

427 [[nodiscard]] auto log_bin_errors_all_reduce(mpi::communicator c) const {
428 std::vector<get_real_t<T>> result_vec{};
429 std::vector<long> count_vec{};
430
431 // M_k, Q_k can be different lengths on different mpi threads.
432 long n_log_bins_i = n_log_bins();
433 std::vector<long> n_log_bins_vec = mpi::all_gather(std::vector<long>{n_log_bins_i}, c);
434 auto [min_n_bins_it, max_n_bins_it] = std::minmax_element(n_log_bins_vec.crbegin(), n_log_bins_vec.crend());
435 long max_n_bins = *max_n_bins_it;
436 long min_n_bins = *min_n_bins_it;
437
438 int max_n_bins_rank = c.size() - 1 - std::distance(n_log_bins_vec.crbegin(), max_n_bins_it);
439
440 if (c.rank() == max_n_bins_rank) {
441 result_vec.reserve(max_n_bins);
442 count_vec.reserve(max_n_bins);
443 }
444
445 for (int n = 0; n < min_n_bins; n++) {
446 auto [Mn, Qn, count_n] = details::mpi_reduce_MQ(log_bins.Mk[n], log_bins.Qk[n], (log_bins.count >> n), c, max_n_bins_rank);
447 if (c.rank() == max_n_bins_rank) {
448 result_vec.emplace_back(std::move(Qn));
449 count_vec.emplace_back(std::move(count_n));
450 }
451 }
452
453 for (auto n = min_n_bins; n < max_n_bins; n++) {
454 int split_color = (n < n_log_bins_i) ? 0 : MPI_UNDEFINED;
455 int split_key = (c.rank() == max_n_bins_rank) ? 0 : 1 + c.rank();
456 mpi::communicator c_split = c.split(split_color, split_key);
457
458 if (split_color == 0) {
459 auto [Mn, Qn, count_n] = details::mpi_reduce_MQ(log_bins.Mk[n], log_bins.Qk[n], (log_bins.count >> n), c_split, 0);
460 if (c.rank() == max_n_bins_rank) {
461 result_vec.emplace_back(std::move(Qn));
462 count_vec.emplace_back(std::move(count_n));
463 }
464 }
465 }
466
467 if (c.rank() == max_n_bins_rank) {
468 for (int n = 0; n < max_n_bins; n++) {
469 if (count_vec[n] <= 1) {
470 result_vec[n] = 0;
471 } else {
472 using std::sqrt;
473 result_vec[n] = sqrt(result_vec[n] / (count_vec[n] * (count_vec[n] - 1)));
474 }
475 }
476 }
477
478 // FIXME: Option not to bcast to all
479 mpi::broadcast(result_vec, c, max_n_bins_rank);
480 mpi::broadcast(count_vec, c, max_n_bins_rank);
481
482 return std::make_pair(result_vec, count_vec);
483 }

510 template <typename T> auto tau_estimate_from_errors(T const &error_with_binning, T const &error_no_binning) {
511 // Last part is simply 1.0, but could be an array
512 return 0.5 * ((error_with_binning * error_with_binning) / (error_no_binning * error_no_binning) - error_no_binning / error_no_binning);
