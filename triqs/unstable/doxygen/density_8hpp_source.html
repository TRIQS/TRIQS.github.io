174 template <typename BGf, int R>
175 auto density(BGf const &gin, std::vector<array<dcomplex, R>> const &known_moments)
176 requires(is_block_gf_v<BGf>)
177 {
178
179 using mesh_t = typename BGf::mesh_t;
180 static_assert(std::is_same_v<mesh_t, mesh::imfreq> or std::is_same_v<mesh_t, mesh::refreq>,
181 "Density Function must be called with either an imfreq or a refreq Green Function");
182
183 using r_t = decltype(density(gin[0], known_moments[0]));
184 std::vector<r_t> dens_vec;
185
186 TRIQS_ASSERT2(gin.size() == known_moments.size(), "Density: Require equal number of blocks in block_gf and known_moments vector");
187
188 for (auto [gin_bl, km_bl] : itertools::zip(gin, known_moments)) dens_vec.push_back(density(gin_bl, km_bl));
189 return dens_vec;
190 }
