39 // Check the shape of block GFs across all processes.
40 template <typename G> bool have_mpi_equal_shape(const G &bg, const mpi::communicator &comm) {
41 if constexpr (G::arity == 1) {
42 return mpi::all_equal(bg.size(), comm);
43 } else {
44 return mpi::all_equal(bg.size1(), comm) && mpi::all_equal(bg.size2(), comm);
45 }
46 }

48 // Hash the block names for comparison.
49 template <typename G> std::size_t hash_names(G const &bg) {
50 auto bin_op = [](std::size_t acc, auto const &name) { return acc + std::hash<std::string>{}(name); };
51 if constexpr (G::arity == 1) {
52 return std::accumulate(bg.block_names().begin(), bg.block_names().end(), std::size_t{0}, bin_op);
53 } else {
54 uint64_t hash = 0;
55 for (auto const &vec : bg.block_names()) { hash += std::accumulate(vec.begin(), vec.end(), std::size_t{0}, bin_op); }
56 return hash;
57 }
58 }

74 template <typename G>
75 requires(BlockGreenFunction_v<G>)
76 void mpi_broadcast(G &&bg, mpi::communicator c, int root) { // NOLINT (temporary views are allowed)
77 constexpr bool is_view = std::decay_t<G>::is_view;
78
79 // broadcast block names
80 if constexpr (!is_view) {
81 // for non-view block GFs, we broadcast the block names directly into the block GF
82 mpi::broadcast(bg._block_names, c, root);
83 } else {
84 // for views, we keep the block names in the block GF but check that they are the same as on the root process
85 auto names = bg.block_names();
86 mpi::broadcast(names, c, root);
87 EXPECTS(bg.block_names() == names);
88 }
89
90 // broadcast data
91 mpi::broadcast(bg.data(), c, root);
92 }

120 template <typename G1, typename G2>
121 requires(BlockGreenFunction_v<G1> and BlockGreenFunction_v<G2>)
122 void mpi_reduce_into(G1 const &bg_in, G2 &&bg_out, mpi::communicator c, int root, // NOLINT (temporary views are allowed here)
123 bool all, MPI_Op op) {
124 constexpr bool is_view = std::decay_t<G2>::is_view;
125
126 // check the shape and block names of the input block GFs
127 EXPECTS(mpi::all_equal(detail::hash_names(bg_in), c));
128 if (not detail::have_mpi_equal_shape(bg_in, c))
129 TRIQS_RUNTIME_ERROR << "Error in triqs::gfs::mpi_reduce_into: Shapes of input block GFs must be equal";
130
131 // assign (check) the block names of the output block GF (view) on receiving ranks
132 if ((c.rank() == root || all)) {
133 if constexpr (is_view) {
134 EXPECTS(bg_in.block_names() == bg_out.block_names());
135 } else {
136 bg_out._block_names = bg_in.block_names();
137 }
138 }
139
140 // reduce the vector (of vectors) of GF objects
141 mpi::reduce_into(bg_in.data(), bg_out.data(), c, root, all, op);
142 }

166 auto mpi_reduce(G const &bg, mpi::communicator c = {}, int root = 0, bool all = false, MPI_Op op = MPI_SUM) {
167 auto res = typename G::regular_type{};
