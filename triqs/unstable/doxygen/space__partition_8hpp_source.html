103 space_partition(state_t const &st, operator_t const &H, bool store_matrix_elements = true, operator_t const &Hyb = operator_t())
104 : tmp_state(make_zero_state(st)), subspaces(st.size()) {
105 auto size = tmp_state.size();
106
107 // Iteration over all initial basis states
108 for (idx_t i = 0; i < size; ++i) {
109 tmp_state(i) = amplitude_t(1);
110 state_t final_state = H(tmp_state);
111
112 auto mapping = [&](idx_t f, amplitude_t amplitude) {
113 using triqs::utility::is_zero;
114 if (is_zero(amplitude)) return;
115 auto i_subspace = subspaces.find_set(i);
116 auto f_subspace = subspaces.find_set(f);
117 if (i_subspace != f_subspace) subspaces.link(i_subspace, f_subspace);
118
119 if (store_matrix_elements) matrix_elements[std::make_pair(i, f)] = amplitude;
120 };
121
122 // Iterate over non-zero final amplitudes
123 foreach (final_state, mapping)
124 ;
125
126 // redo for additionnal Hyb
127 if (not Hyb.is_empty()) {
128 final_state = Hyb(tmp_state);
129 foreach (final_state, mapping)
130 ;
131 }
132
133 tmp_state(i) = amplitude_t(0.);
134 }
135
136 _update_index();
137 }

157 std::pair<matrix_element_map_t, matrix_element_map_t> merge_subspaces(operator_t const &Cd, operator_t const &C,
158 bool store_matrix_elements = true) {
159
160 matrix_element_map_t Cd_elements, C_elements;
161 std::multimap<idx_t, idx_t> Cd_connections, C_connections;
162
163 auto size = tmp_state.size();
164
165 // Fill connection multimaps
166 for (idx_t i = 0; i < size; ++i) {
167 tmp_state(i) = amplitude_t(1);
168 auto i_subspace = subspaces.find_set(i);
169
170 auto fill_conn = [this, i, i_subspace, store_matrix_elements](operator_t const &op, std::multimap<idx_t, idx_t> &conn,
171 matrix_element_map_t &elem) {
172 state_t final_state = op(tmp_state);
173 // Iterate over non-zero final amplitudes
174 foreach (final_state, [&](idx_t f, amplitude_t amplitude) {
175 using triqs::utility::is_zero;
176 if (is_zero(amplitude)) return;
177 auto f_subspace = subspaces.find_set(f);
178 conn.insert({i_subspace, f_subspace});
179 if (store_matrix_elements) elem[{i, f}] = amplitude;
180 })
181 ;
182 };
183
184 fill_conn(Cd, Cd_connections, Cd_elements);
185 fill_conn(C, C_connections, C_elements);
186
187 tmp_state(i) = amplitude_t(0.);
188 }
189
190 // 'Zigzag' traversal algorithm
191 while (!Cd_connections.empty()) {
192
193 // Take one C^+ - connection
194 // C^+|lower_subspace> = |upper_subspace>
195 idx_t lower_subspace, upper_subspace;
196 std::tie(lower_subspace, upper_subspace) = *std::begin(Cd_connections);
197
198 // - Reveals all subspaces reachable from lower_subspace by application of
199 // a 'zigzag' product C^+ C C^+ C C^+ ... of any length.
200 // - Removes all visited connections from Cd_connections/C_connections.
201 // - Merges lower_subspace with all subspaces generated from lower_subspace by application of (C C^+)^(2*n).
202 // - Merges upper_subspace with all subspaces generated from upper_subspace by application of (C^+ C)^(2*n).
203 std::function<void(idx_t, bool)> zigzag_traversal = [this, lower_subspace, upper_subspace, &Cd_connections, &C_connections,
204 &zigzag_traversal](idx_t i_subspace, // find all connections starting from i_subspace
205 bool upwards // if true, C^+ connection, otherwise C connection
206 ) {
207 std::multimap<idx_t, idx_t>::iterator it;
208 while ((it = (upwards ? Cd_connections : C_connections).find(i_subspace)) != (upwards ? Cd_connections : C_connections).end()) {
209
210 auto f_subspace = it->second;
211 (upwards ? Cd_connections : C_connections).erase(it);
212
213 if (upwards)
214 subspaces.link(f_subspace, upper_subspace);
215 else
216 subspaces.link(f_subspace, lower_subspace);
217
218 // Recursively apply to all found f_subspace's with a 'flipped' direction
219 zigzag_traversal(f_subspace, !upwards);
220 }
221 };
222
223 // Apply to all C^+ connections starting from lower_subspace
224 zigzag_traversal(lower_subspace, true);
225 }
226
227 _update_index();
228
229 return std::make_pair(Cd_elements, C_elements);
230 }

280 block_mapping_t find_mappings(operator_t const &op, bool diagonal_only = false) {
281
282 block_mapping_t mapping;
283
284 // Iteration over all initial basis states
285 for (idx_t i = 0; i < tmp_state.size(); ++i) {
286 state_t initial_state = tmp_state;
287 initial_state(i) = amplitude_t(1);
288 auto i_subspace = subspaces.find_set(i);
289
290 state_t final_state = op(initial_state);
291
292 // Iterate over non-zero final amplitudes
293 foreach (final_state, [&](idx_t f, amplitude_t amplitude) {
294 using triqs::utility::is_zero;
295 if (is_zero(amplitude)) return;
296 auto f_subspace = subspaces.find_set(f);
297 if ((!diagonal_only) || i_subspace == f_subspace)
298 mapping.insert(std::make_pair(representative_to_index[i_subspace], representative_to_index[f_subspace]));
299 })
300 ;
301 }
302
303 return mapping;
304 }

306 private:
307 void _update_index() {
308 auto p = subspaces.parents();
309 subspaces.compress_sets(p.begin(), p.end()); // parents are representatives
310 subspaces.normalize_sets(p.begin(), p.end()); // the representative has the smallest index in the set
311
312 // Update representative_to_index
313 representative_to_index.clear();
314 for (idx_t n = 0; n < tmp_state.size(); ++n) {
315 representative_to_index.insert(std::make_pair(subspaces.find_set(n), representative_to_index.size()));
316 }
317 }
318
319 // Temporary zero state
320 mutable state_t tmp_state;
321 // Subspaces
322 boost::disjoint_sets_with_storage<> subspaces;
323 // Matrix elements of the Hamiltonian
324 matrix_element_map_t matrix_elements;
325 // Map representative basis state to subspace index
326 std::map<idx_t, idx_t> representative_to_index;
327 };
