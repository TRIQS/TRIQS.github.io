32 template <typename V> auto mean(V const &data) {
33 // ENSURE(data.size() > 0);
34 auto mean_calc = data[0];
35 mean_calc = 0;
36 for (auto const &[n, x] : itertools::enumerate(data)) { mean_calc += (x - mean_calc) / (n + 1); }
37 return mean_calc;
38 }

46 template <typename V> auto mean_mpi(mpi::communicator c, V const &data) {
47 // ENSURE(data.size() > 0);
48 auto M = mean(data);
49 long N = mpi::all_reduce(data.size(), c);
50 M *= double(data.size()) / N;
51 mpi::all_reduce_in_place(M, c); //M = mpi::all_reduce(M, c, 0);
52 return M;
53 }

61 template <typename V> auto mean_and_err(V const &data) {
62 // ENSURE(data.size() > 0);
63 long length = data.size();
64 auto mean_calc = mean(data);
65 auto err_calc = make_real(mean_calc);
66 err_calc = 0;
67 for (auto const &x : data) { err_calc += nda::real(nda::conj(x - mean_calc) * (x - mean_calc) / (length * (length - 1))); }
68 err_calc = std::sqrt(err_calc);
69 return std::make_pair(mean_calc, err_calc);
70 }

78 template <typename V> auto mean_and_err_mpi(mpi::communicator c, V const &data) {
79 // ENSURE(data.size() > 0);
80 long length = mpi::all_reduce(data.size(), c);
81 auto mean_calc = mean_mpi(c, data);
82 auto err_calc = make_real(mean_calc);
83 err_calc = 0;
84 for (auto const &x : data) { err_calc += nda::real(nda::conj(x - mean_calc) * (x - mean_calc) / (length * (length - 1))); }
85 mpi::all_reduce_in_place(err_calc, c);
86 err_calc = std::sqrt(err_calc);
87 return std::make_pair(mean_calc, err_calc);

92 template <typename T> auto mpi_reduce_MQ(const T &Mi, const get_real_t<T> &Qi, const long &count_i, mpi::communicator c, int root = 0) {
93 using nda::conj;
94 using nda::real;
95
96 long count_total = mpi::all_reduce(count_i, c);
97 double count_frac = double(count_i) / count_total;
98
99 T M = Mi * count_frac;
100 mpi::all_reduce_in_place(M, c);
101
102 T diff = Mi - M;
103 get_real_t<T> Q = Qi + count_i * make_real(conj(diff) * diff);
104
105 mpi::reduce_in_place(Q, c, root);
106
107 return std::make_tuple(M, Q, count_total);
108 }
