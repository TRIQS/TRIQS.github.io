78 template <typename scalar_t> dict2_t<scalar_t> extract_h_dict(op_t<scalar_t> const &h, bool ignore_irrelevant = false) {
79
80 auto h_dict = dict2_t<scalar_t>{};
81
82 for (auto const &term : h) {
83 auto const &coef = term.coef;
84 auto const &m = term.monomial;
85
86 if (m.size() == 2) {
87 if (!(m[0].dagger && !m[1].dagger)) {
88 if (!ignore_irrelevant) TRIQS_RUNTIME_ERROR << "extract_h_dict: monomial is not of the form C^+(i) C(j)";
89 } else { // everything ok
90 h_dict.insert({std::make_tuple(m[0].indices, m[1].indices), coef});
91 }
92 } else {
93 if (!ignore_irrelevant) TRIQS_RUNTIME_ERROR << "extract_h_dict: monomial must have 2 operators";
94 }
95 }
96
97 return h_dict;
98 }

121 template <typename scalar_t> dict2_t<scalar_t> extract_U_dict2(op_t<scalar_t> const &h, bool ignore_irrelevant = false) {
122
123 auto U_dict = dict2_t<scalar_t>{};
124
125 for (auto const &term : h) {
126 auto const &coef = term.coef;
127 auto const &m = term.monomial;
128
129 if (m.size() == 4) {
130 if (!(m[0].dagger && m[1].dagger && !m[2].dagger && !m[3].dagger) || (m[0].indices != m[3].indices) || (m[1].indices != m[2].indices)) {
131 if (!ignore_irrelevant) TRIQS_RUNTIME_ERROR << "extract_U_dict2: monomial is not of the form C^+(i) C^+(j) C(j) C(i)";
132 } else { //everything ok
133 U_dict.insert({std::make_tuple(m[0].indices, m[1].indices), coef});
134 U_dict.insert({std::make_tuple(m[1].indices, m[0].indices), coef});
135 }
136 } else {
137 if (!ignore_irrelevant) TRIQS_RUNTIME_ERROR << "extract_U_dict2: monomial must have 4 operators";
138 }
139 }
140
141 return U_dict;
142 }

166 template <typename scalar_t> dict4_t<scalar_t> extract_U_dict4(op_t<scalar_t> const &h, bool ignore_irrelevant = false) {
167
168 auto U_dict = dict4_t<scalar_t>{};
169
170 for (auto const &term : h) {
171 scalar_t const &coef = term.coef;
172 auto const &m = term.monomial;
173
174 if (m.size() == 4) {
175 if (!(m[0].dagger && m[1].dagger && !m[2].dagger && !m[3].dagger)) {
176 if (!ignore_irrelevant) TRIQS_RUNTIME_ERROR << "extract_U_dict4: monomial is not of the form C^+(i) C^+(j) C(l) C(k)";
177 } else { // everything ok
178 U_dict.insert({std::make_tuple(m[0].indices, m[1].indices, m[3].indices, m[2].indices), scalar_t(0.5) * coef});
179 U_dict.insert({std::make_tuple(m[1].indices, m[0].indices, m[2].indices, m[3].indices), scalar_t(0.5) * coef});
180 U_dict.insert({std::make_tuple(m[0].indices, m[1].indices, m[2].indices, m[3].indices), scalar_t(-0.5) * coef});
181 U_dict.insert({std::make_tuple(m[1].indices, m[0].indices, m[3].indices, m[2].indices), scalar_t(-0.5) * coef});
182 }
183 } else {
184 if (!ignore_irrelevant) TRIQS_RUNTIME_ERROR << "extract_U_dict4: monomial must have 4 operators";
185 }
186 }
187
188 return U_dict;
189 }

210 template <typename ValueType = double, typename DictType>
211 array<ValueType, std::tuple_size<typename DictType::key_type>::value> dict_to_matrix(DictType const &dict,
212 hilbert_space::fundamental_operator_set const &fs) {
213
214 using namespace triqs::tuple;
215 using triqs::hilbert_space::format_indices;
216 using matrix_t = array<ValueType, std::tuple_size<typename DictType::key_type>::value>;
217
218 auto indices_to_linear = [&fs](indices_t const &indices) {
219 if (!fs.has_indices(indices))
220 TRIQS_RUNTIME_ERROR << "dict_to_matrix: key [" << format_indices(indices) << "] of dict not in fundamental_operator_set/gf_struct";
221 return fs[indices];
222 };
223
224 auto dims = make_tuple_repeat<std::tuple_size<typename DictType::key_type>::value>(fs.size());
225 auto mat = apply_construct_parenthesis<matrix_t>(dims);
226 mat() = ValueType{};
227
228 for (auto const &kv : dict) triqs::tuple::apply(mat, map(indices_to_linear, kv.first)) = ValueType(kv.second);
229
230 return mat;
231 }

271 template <typename scalar_t> op_t<scalar_t> filter_op(op_t<scalar_t> const &h, long len) {
272
273 auto h_filtered = op_t<scalar_t>{};
274
275 for (auto const &term : h) {
276 if (term.monomial.size() == len) h_filtered += term;
277 }

320 template <typename scalar_t>
321 nda::array<nda::matrix<scalar_t>, 1> block_matrix_from_op(op_t<scalar_t> const &h, hilbert_space::gf_struct_t const &gf_struct,
322 bool ignore_irrelevant = false) {
323
324 int n_bl = gf_struct.size();
325 auto bl_mat = nda::array<nda::matrix<scalar_t>, 1>(n_bl);
326 for (auto bl : range(n_bl)) {
327 auto bl_size = gf_struct[bl].second;
328 bl_mat[bl] = nda::zeros<scalar_t>(bl_size, bl_size);
329 }
330
331 auto name_to_bl = [&](auto &bl_name) {
332 auto it =
333 std::find_if(cbegin(gf_struct), cend(gf_struct), [&bl_name](auto const &blname_and_size) { return bl_name == blname_and_size.first; });
334 return std::distance(cbegin(gf_struct), it);
335 };
336
337 for (auto const &term : h) {
338 auto const &m = term.monomial;
339 scalar_t const &coef = term.coef;
340
341 if (m.size() == 2 and m[0].dagger and not m[1].dagger and m[0].indices[0] == m[1].indices[0] and m[0].indices.size() == 2
342 and m[1].indices.size() == 2) {
343 auto bl_name = std::get<std::string>(m[0].indices[0]);
344 auto op1_idx = std::get<long>(m[0].indices[1]);
345 auto op2_idx = std::get<long>(m[1].indices[1]);
346
347 bl_mat[name_to_bl(bl_name)](op1_idx, op2_idx) = coef;
348 } else {
349 if (!ignore_irrelevant) TRIQS_RUNTIME_ERROR << "block_matrix_from_op: Operator term is not of the form 'coeff * c_dag{bl,i} * c_{bl,j}'";
350 ;
351 }
352 }
353
354 return bl_mat;
355 }

371 template <typename scalar_t>
372 op_t<scalar_t> op_from_block_matrix(nda::array<nda::matrix<scalar_t>, 1> const &bl_mat, hilbert_space::gf_struct_t const &gf_struct) {
373
374 EXPECTS(bl_mat.size() == gf_struct.size());
375
376 auto h = op_t<scalar_t>{};
377
378 int n_bl = gf_struct.size();
379 for (auto bl : range(n_bl)) {
380
381 auto [bl_name, bl_size] = gf_struct[bl];
382
383 EXPECTS(bl_mat[bl].shape() == (std::array{bl_size, bl_size}));
384
385 for (auto [i, j] : itertools::product_range(bl_size, bl_size)) h += bl_mat[bl](i, j) * c_dag(bl_name, i) * c(bl_name, j);
386 }
387
388 return h;
389 }
