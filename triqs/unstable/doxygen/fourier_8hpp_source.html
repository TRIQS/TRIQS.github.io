63 template <int N, typename M1, typename M2, typename T1, typename T2, typename... OptArgs>
64 void _fourier(gf_const_view<M1, T1> gin, gf_view<M2, T2> gout, OptArgs const &...opt_args) {
65
66 static_assert(std::is_same<typename T1::complex_t, T2>::value, "Incompatible target types for fourier transform");
67
68 // pb std::get<0> would not work on a non composite mesh. We use a little lambda to deduce ref and type
69 auto const &out_mesh = [&gout]() -> auto const & { // NB must return a reference
70 using m_t = std::decay_t<decltype(gout.mesh())>;
71 if constexpr (mesh::is_product<m_t>)
72 return std::get<N>(gout.mesh());
73 else
74 return gout.mesh();
75 }();
76
77 // FIXME : Code failed with nda optimisation relaxing assumption on iterator order.
78 // between nda::for_each and the flatten which were not inverse of each other any more
79 // TODO : put back the optimisation in nda (MACRO ??)
80 // some tests will fail and check in a test that flatten_2d and the inverse op are indeed inverse...
81 auto gin_fl = [&gin]() {
82 using gfl_t = typename decltype(flatten_gf_2d<N>(gin))::complex_t;
83 return gfl_t{flatten_gf_2d<N>(gin)};
84 }();
85
86 auto gout_fl = _fourier_impl(out_mesh, gin_fl, flatten_2d(opt_args)...);
87 unflatten_gf_2d<N>(gout, gout_fl);
88 }

121 template <int N = 0, typename M1, typename M2, typename T, typename... OptArgs>
122 auto make_gf_from_fourier(gf_const_view<M1, T> gin, M2 const &mesh, OptArgs const &...opt_args) {
123 static_assert(N >= 0 && N < n_variables<M1>, "Mesh index exceeds Gf Mesh Rank");
124 static_assert(n_variables<M2> == 1, "Cannot fourier transform on cartesian product mesh");
125
126 if constexpr (n_variables<M1> == 1) { // === single mesh
127 static_assert(n_variables<M2> == 1, "Incompatible mesh ranks");
128 static_assert(N == 0, "Fourier transforming gf with mesh of rank 1 but fourier index N > 1");
129 static_assert(std::is_same_v<M2, _mesh_fourier_image<M1>>, "There is no Fourier transform between these two meshes");
130 auto gout = gf<M2, typename T::complex_t>{mesh, gin.target_shape()};
131 _fourier<N>(gin, gout(), opt_args...);
132 return gout;
133 } else { // === prod mesh
134 static_assert(std::is_same_v<M2, _mesh_fourier_image<std::tuple_element_t<N, M1>>>, "There is no Fourier transform between these two meshes");
135 auto mesh_tpl = triqs::tuple::replace<N>(gin.mesh().components(), mesh);
136 auto out_mesh = mesh::prod{mesh_tpl};
137 using mesh_t = typename std::decay_t<decltype(out_mesh)>;
138 auto gout = gf<mesh_t, typename T::complex_t>{out_mesh, gin.target_shape()};
139 _fourier<N>(gin, gout(), opt_args...);
140 return gout;
141 }
142 }

225 template <int N = 0, typename G, typename M, int R>
226 auto make_gf_from_fourier(G const &gin, M const &m, std::vector<array<dcomplex, R>> const &known_moments)
227 requires(is_block_gf_v<G>)
228 {
229
230 using r_t = decltype(make_gf_from_fourier<N>(gin[0], m, known_moments[0]));
231 std::vector<r_t> g_vec;
232
233 TRIQS_ASSERT2(gin.size() == known_moments.size(), "Fourier: Require equal number of blocks in block_gf and known_moments vector");
234
235 for (auto [gin_bl, km_bl] : itertools::zip(gin, known_moments)) g_vec.push_back(make_gf_from_fourier<N>(gin_bl, m, km_bl));
236 return make_block_gf(gin.block_names(), std::move(g_vec));
237 }

239 template <int N = 0, typename G, typename M, int R>
240 auto make_gf_from_fourier(G const &gin, M const &m, std::vector<std::vector<array<dcomplex, R>>> const &known_moments)
241 requires(is_block_gf_v<G>)
242 {
243
244 using r_t = decltype(make_gf_from_fourier<N>(gin(0, 0), m, known_moments[0][0]));
245 std::vector<std::vector<r_t>> g_vecvec;
246
247 TRIQS_ASSERT2(gin.size1() == known_moments.size(), "Fourier: Require matching block structure between gin and known_moments");
248
249 for (int i : range(gin.size1())) {
250 TRIQS_ASSERT2(gin.size2() == known_moments[i].size(), "Fourier: Require matching block structure between gin and known_moments");
251
252 std::vector<r_t> g_vec;
253 for (int j : range(gin.size2())) g_vec.push_back(make_gf_from_fourier<N>(gin(i, j), m, known_moments[i][j]));
254
255 g_vecvec.push_back(std::move(g_vec));
256 }
257 return block2_gf_of<r_t>{gin.block_names(), std::move(g_vecvec)};
258 }

297 // realize the call for gx = fourier(gy);
298 template <int N, typename M1, typename T1, typename M2, typename T2, typename... Args>
299 void triqs_gf_view_assign_delegation(gf_view<M1, T1> lhs_g, _fourier_lazy<N, gf_const_view<M2, T2>, Args...> const &rhs) {
300 static_assert(std::is_same_v<typename T1::real_t, typename T2::real_t>, "Error : in gx = fourier(gy), gx and gy must have the same target");
301
302 if constexpr (n_variables<M1> == 1) // === single mesh
303 static_assert(std::is_same_v<M2, _mesh_fourier_image<M1>>, "There is no Fourier transform between these two meshes");
304 else { // === prod mesh
305 using mesh_res_t = decltype(triqs::tuple::replace<N>(rhs.g.mesh().components(), make_adjoint_mesh(std::get<N>(rhs.g.mesh()))));
306 static_assert(std::is_same_v<typename M1::m_tuple_t, mesh_res_t>, "Meshes in assignment don't match");
307 }
308
309 // check the size of the "inactive" dimensions
310
311 std::apply([&](auto &&...u) { _fourier<N>(rhs.g, lhs_g, u...); }, rhs.args); // calls _fourier( rhs.g, lhs_g, rhs.args...)
312 }
