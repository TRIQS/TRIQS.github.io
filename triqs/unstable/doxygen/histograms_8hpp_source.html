209 inline histogram pdf(histogram const &h) {
210 auto pdf = h;
211 pdf._data /= double(h.n_data_pts());
212 return pdf;
213 }
214
219 inline histogram cdf(histogram const &h) {
220 auto cdf = h;
221 for (int i = 1; i < h.size(); ++i) cdf._data[i] += cdf._data[i - 1];
222 cdf._data /= double(h.n_data_pts());
223 return cdf;
224 }
