50 report_stream(std::ostream &out_, int verbosity_ = 1) : out(&out_), verbosity(verbosity_) {}
51
52 report_stream operator()(int n) const { return {out, verbosity - n + 1}; }
53
54 template <class T> report_stream &operator<<(T const &x) {
55 if (verbosity > 0) (*out) << x;
56 return *this;
57 }
58
59 // operator<< for manipulators like std::endl
60 report_stream &operator<<(std::ostream &(*manip)(std::ostream &)) {
61 if (verbosity > 0) manip(*out);
62 return *this;
63 }
