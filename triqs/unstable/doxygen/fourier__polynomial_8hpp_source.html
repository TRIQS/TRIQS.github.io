18 // ---------------------------------------------------------------------------
19 // Free Fourier-evaluation kernels.
20 //
21 // These take the packed R-matrix of lattice vectors (shape [nR, kdim]) and a
22 // packed coefficient array (shape [nR, coeff_shape...]) and evaluate
23 //
24 // f(k) = sum_R coeff(R) * exp(2 pi i k . R)
25 //
26 // at a single k-point, a matrix of k-points, or any contiguous range of
27 // k-like objects (anything with operator[](int)). Both fourier_polynomial's
28 // member operator() and triqs::experimental::gfs::fourier_eval /
29 // make_gf_from_fourier call them.
30 // ---------------------------------------------------------------------------
31
33 template <int kdim> nda::matrix<double> make_R_mat(auto const &r_list) {
34 nda::matrix<double> rm(r_list.size(), kdim);
35 for (long i = 0; i < static_cast<long>(r_list.size()); ++i)
36 for (int d = 0; d < kdim; ++d) rm(i, d) = static_cast<double>(r_list[i][d]);
37 return rm;
38 }

41 template <int coeff_dim, int kdim, typename CoeffArr>
42 auto fourier_eval(nda::matrix_const_view<double> R_mat, CoeffArr const &coeff_arr, std::array<double, kdim> const &k) {
43 auto c_shape = coeff_arr.shape();
44
45 // coeff_dim == 0 (scalar target): accumulate into a plain dcomplex since
46 // nda::array<T, 0> is not a valid container type. For matrix-valued
47 // targets we preserve the nda::matrix type so matmul/linalg work directly.
48 auto res = [&]() {
49 if constexpr (coeff_dim == 0)
50 return dcomplex{0};
51 else if constexpr (coeff_dim == 2)
52 return nda::matrix<dcomplex>::zeros(nda::stdutil::front_pop(c_shape));
53 else
54 return nda::array<dcomplex, coeff_dim>::zeros(nda::stdutil::front_pop(c_shape));
55 }();
56
57 long nR = R_mat.shape(0);
58 for (long idx = 0; idx < nR; ++idx) {
59 double kR = 0;
60 for (int d = 0; d < kdim; ++d) kR += k[d] * R_mat(idx, d);
61 res += std::exp(2i * M_PI * kR) * coeff_arr(idx, nda::ellipsis{});
62 }
63 return res;
64 }

67 template <int coeff_dim, typename CoeffArr>
68 nda::array<dcomplex, coeff_dim + 1> fourier_eval(nda::matrix_const_view<double> R_mat, CoeffArr const &coeff_arr,
69 nda::array_const_view<double, 2> k_list) {
70 auto ks = nda::matrix_const_view<double>(k_list);
71 long nk = ks.shape(0);
72
73 // set up the phases
74 nda::array<double, 2> kdotR(nk, R_mat.shape(0));
75 nda::blas::gemm(1., ks, transpose(R_mat), 0, kdotR);
76 nda::matrix<dcomplex> phases = nda::exp(2i * M_PI * kdotR);
77
78 auto c_shape = coeff_arr.shape();
79
80 // flat product of the trailing target dims (1 if the target is scalar)
81 long trailing = 1;
82 for (int i = 1; i < coeff_dim + 1; ++i) trailing *= c_shape[i];
83
84 auto coeff_arr_view = nda::reshape(coeff_arr, std::array{c_shape[0], trailing});
85
86 auto result_shape = c_shape;
87 result_shape[0] = nk;
88 auto result = nda::array<dcomplex, coeff_dim + 1>(result_shape);
89 auto result_mat_view = nda::reshape(result, std::array{nk, trailing});
90
91 nda::blas::gemm(1, phases, coeff_arr_view, 0, result_mat_view);
92 return result;
93 }

96 template <int coeff_dim, int kdim, typename CoeffArr, typename V>
97 requires(std::ranges::contiguous_range<V>)
98 nda::array<dcomplex, coeff_dim + 1> fourier_eval(nda::matrix_const_view<double> R_mat, CoeffArr const &coeff_arr, V const &k_iterator) {
99 auto kvecs = nda::matrix<double>(k_iterator.size(), kdim);
100 for (auto [ik, k] : itertools::enumerate(k_iterator)) {
101 for (auto idim : nda::range(kdim)) { kvecs(ik, idim) = k[idim]; }
102 }
103 return fourier_eval<coeff_dim>(R_mat, coeff_arr, nda::array_const_view<double, 2>(kvecs));
104 }

207 template <typename... T>
208 requires((std::is_same_v<T, double> or nda::clef::is_lazy<T>) and ...) // double OR placeholder
209 auto operator()(T... ks) const {
210
211 static_assert(sizeof...(T) == kdim, "Incorrect number of arguments"); // # arguments is same as dimension
212 constexpr auto ph_number = ((nda::clef::is_lazy<T> + ...)); // How many placeholders in ks
213
214 // Called with 0 placeholder
215 if constexpr (ph_number == 0) return operator()(std::array{ks...});
216 // Called with 1 placeholder, not valid for kdim = 1
217 else if constexpr ((ph_number == 1) and (kdim != 1)) {
218 // find the position of the placeholder
219 constexpr auto is_ph = std::array{(nda::clef::is_lazy<T> ? 1 : 0)...}; // 0 or 1 if it is a placeholder
220 constexpr int ph_position = [](auto &&v) {
221 for (int i = 0; i < v.size(); ++i)
222 if (v[i] == 1) return i;
223 return -1; // to silence compiler warning
224 }(is_ph); // immediately calling the lambda. executed at compile time
225
226 return nda::clef::make_expr_call(partial_eval<ph_position>(ks...), // auto{k...[ph_position]}); // C++26
227 auto{std::get<ph_position>(std::tie(ks...))});
228 }
229 // Called with > 1 placeholders in kdim > 1 or 1 placeholder in kdim == 1
230 else
231 return nda::clef::make_expr_call(fourier_polynomial{*this}, auto{ks}...);
232 }

238 template <int placeholder_pos>
239 fourier_polynomial<coeff_dim, 1> partial_eval(auto const &...ks) const
240 requires((kdim == 3) or (kdim == 2))
241 {
242
243 auto [it1, it2] = std::ranges::minmax_element(R_list, {}, [](auto const &r) { return r[placeholder_pos]; });
244
245 auto Rz_min = (*it1)[placeholder_pos];
246 auto Rz_max = (*it2)[placeholder_pos];
247
248 long n_new = Rz_max - Rz_min + 1;
249 auto new_Rs = std::vector<std::array<long, 1>>(n_new);
250 auto coeff_shape = nda::stdutil::front_pop(coeff_arr.shape());
251 auto new_coeff_arr = nda::array<dcomplex, coeff_dim + 1>::zeros(nda::stdutil::front_append(coeff_shape, n_new));
252
253 constexpr int p1 = (placeholder_pos + 1) % kdim;
254 constexpr int p2 = (placeholder_pos + 2) % kdim;
255 auto kst = std::tie(ks...);
256
257 for (long idx = 0; idx < n_R(); ++idx) {
258 auto Rz = R_list[idx][placeholder_pos];
259 long new_idx = Rz - Rz_min;
260 new_Rs[new_idx] = {Rz};
261
262 double kR = std::get<p1>(kst) * R_list[idx][p1];
263 if constexpr (kdim == 3) kR += std::get<p2>(kst) * R_list[idx][p2];
264 new_coeff_arr(new_idx, nda::ellipsis{}) += coeff_arr(idx, nda::ellipsis{}) * std::exp(2i * M_PI * kR);
265 }
266
267 return {std::move(new_Rs), std::move(new_coeff_arr)};
268 }
