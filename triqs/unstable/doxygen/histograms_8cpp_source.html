24 void histogram::_init() {
25 if (a >= b) TRIQS_RUNTIME_ERROR << "histogram construction: one must have a<b";
26 _step = (n_bins - 1) / (b - a);
27 }

29 bool histogram::operator==(histogram const &h) const {
30 return this->a == h.a && //
31 this->b == h.b && //
32 this->n_bins == h.n_bins && //
33 this->_data == h._data && //
34 this->_n_data_pts == h._n_data_pts && //
35 this->_n_lost_pts == h._n_lost_pts && //
36 this->_step == h._step;
37 }

39 histogram &histogram::operator<<(double x) {
40 if ((x < a) || (x > b))
41 ++_n_lost_pts;
42 else {
43 auto n = int(std::floor(((x - a) * _step) + 0.5));
44 ++_data[n];
45 ++_n_data_pts;
46 }
47 return *this;
48 }

50 histogram operator+(histogram h1, histogram const &h2) {
51 auto l1 = h1.limits(), l2 = h2.limits();
52 if (l1 != l2 || h1.size() != h2.size()) {
53 TRIQS_RUNTIME_ERROR << "Histograms with different limits in addition: [" << l1.first << ";" << l1.second << "] vs [" << l2.first << ";"
54 << l2.second << "]";
55 }
56 h1._data += h2._data;
57 h1._n_data_pts += h2._n_data_pts;
58 h1._n_lost_pts += h2._n_lost_pts;
59 return h1;
60 }

84 std::ostream &operator<<(std::ostream &os, histogram const &h) {
85 auto normed = pdf(h);
86 auto integrated = cdf(h);

88 os << h.mesh_point(i) << " " << h.data()[i] << " " << normed.data()[i] << " " << integrated.data()[i] << std::endl;
