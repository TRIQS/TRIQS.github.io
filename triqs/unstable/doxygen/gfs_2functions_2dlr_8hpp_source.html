38 template <int N, int... Ns, typename F, typename G>
39 requires(MemoryGf<G> or is_block_gf_v<G>)
40 auto apply_to_mesh(F const &f, G const &g) {
41 using M = typename G::mesh_t;
42 if constexpr (sizeof...(Ns) > 0) {
43 return apply_to_mesh<Ns...>(f, apply_to_mesh<N>(f, g));
44 } else if constexpr (is_block_gf_v<G>) {
45 return map_block_gf([&](auto const &gbl) { return apply_to_mesh<N>(f, gbl); }, g);
46 } else {
47 static_assert(mesh::is_product<M>, "requires product mesh");
48 auto gfl = f(flatten_gf_2d<N>(g));
49 auto out_mesh = mesh::prod{triqs::tuple::replace<N>(g.mesh(), gfl.mesh())};
50 auto g_out = gf{out_mesh, g.target_shape()};
51 unflatten_gf_2d<N>(g_out, gfl);
52 return g_out;
53 }
54 }

70 template <int N = 0, int... Ns, typename G>
71 requires(MemoryGf<G> or is_block_gf_v<G>)
72 auto make_gf_dlr(G const &g) {
73 using M = typename G::mesh_t;
74 if constexpr (is_block_gf_v<G>) {
75 return map_block_gf([&](auto const &gbl) { return make_gf_dlr<N, Ns...>(gbl); }, g);
76 } else if constexpr (mesh::is_product<M>) {
77 return apply_to_mesh<N, Ns...>([&](auto const &gfl) { return make_gf_dlr(gfl); }, g);
78 } else {
79 static_assert(N == 0, "N must be 0 for non-product meshes");
80 static_assert(nda::AnyOf<M, dlr_imtime, dlr_imfreq>, "Mesh must be dlr_imtime or dlr_imfreq");
81 auto result = gf{dlr{g.mesh()}, g.target_shape()};
82 if constexpr (std::is_same_v<M, dlr_imtime>)
83 result.data() = result.mesh().dlr_it().vals2coefs(g.data());
84 else { // dlr_imfreq
85 auto beta_inv = 1. / result.mesh().beta();
86 result.data() = beta_inv * result.mesh().dlr_if().vals2coefs(g.data());
87 }
88 return result;
89 }

121 auto result = gf{mesh, g.target_shape()};
122 result.data() = result.mesh().dlr_it().fitvals2coefs(make_regular(tvals), g.data());
123 return result;

200 // Factored out so make_gf_dlr_imfreq and find_w_max build the dlr_imfreq mesh
201 // once (the cppdlr SVD is not cheap) and reuse it across blocks / search iterations.
202 template <typename G>
203 requires(MemoryGf<G> or is_block_gf_v<G>)
204 auto sample_on_dlr_imfreq(G const &g, dlr_imfreq const &dlr_mesh) {
205 if constexpr (is_block_gf_v<G>) {
206 return map_block_gf([&](auto const &gbl) { return sample_on_dlr_imfreq(gbl, dlr_mesh); }, g);
207 } else {
208 static_assert(std::is_same_v<typename G::mesh_t, mesh::imfreq>, "Input mesh must be imfreq");
209 auto const &iw_mesh = g.mesh();
210 auto [iw_lo, iw_hi] = dlr_mesh.min_max_frequencies();
211 if (not iw_mesh.is_index_valid(iw_lo.n) or not iw_mesh.is_index_valid(iw_hi.n))
212 TRIQS_RUNTIME_ERROR << "make_gf_dlr_imfreq: input imfreq mesh does not cover the requested DLR frequency range";
213 auto result = gf{dlr_mesh, g.target_shape()};
214 for (auto const &mp : dlr_mesh) result[mp] = g[iw_mesh(mp.index())];
215 return result;
216 }
217 }

289 template <int = 0, typename G>
290 requires(MemoryGf<G> or is_block_gf_v<G>)
291 double find_w_max(G const &g, double eps = 1e-10, bool symmetrize = true, double w_max_init = 1.0, double w_max_max = 200.0) {
292 if (w_max_init > w_max_max)
293 TRIQS_RUNTIME_ERROR << "find_w_max: w_max_init (" << w_max_init << ") > w_max_max (" << w_max_max << ")";
294
295 auto const &m0 = detail::imfreq_mesh_of(g);
296
297 for (double w_max = w_max_init; w_max <= w_max_max; w_max *= 1.5) {
298 auto dlr_mesh = dlr_imfreq{m0.beta(), m0.statistic(), w_max, eps, symmetrize};
299 auto [iw_lo, iw_hi] = dlr_mesh.min_max_frequencies();
300 if (not m0.is_index_valid(iw_lo.n) or not m0.is_index_valid(iw_hi.n)) continue;
301
302 auto g_dlr_iw = detail::sample_on_dlr_imfreq(g, dlr_mesh);
303 auto g_rec = make_gf_imfreq(make_gf_dlr(g_dlr_iw), m0.n_iw());
304 auto data_err = [](auto const &x, auto const &y) { return double(max_element(abs(x.data() - y.data()))); };
305 double max_err = 0.0;
306 if constexpr (is_block_gf_v<G>) {
307 for (long b = 0; b < long(g.size()); ++b) max_err = std::max(max_err, data_err(g[b], g_rec[b]));
308 } else {
309 max_err = data_err(g, g_rec);
310 }
311 if (max_err < eps) return w_max;
312 }
313 TRIQS_RUNTIME_ERROR << "find_w_max: no w_max <= " << w_max_max << " yields round-trip error < eps = " << eps;
314 }

388 template <typename G>
389 requires(MemoryGf<G> or is_block_gf_v<G>)
390 auto tau_L2_norm(G const &g) {
391 using M = typename G::mesh_t;
392 static_assert(nda::AnyOf<M, dlr, dlr_imfreq, dlr_imtime>, "Input mesh must be one of dlr, dlr_imfreq, dlr_imtime");
393 if constexpr (is_block_gf_v<G>) {
394 // Manual loop with a concrete element type. map_block_gf would deduce
395 // std::vector<std::invoke_result_t<lambda, gf>> (the lambda is local),
396 // and a function-scoped `using elem_t = ...` alias would also leak its
397 // spelling -- in both cases clair-c2py fails to render the type.
398 constexpr int rank = std::decay_t<G>::g_t::target_t::rank;
399 if constexpr (rank == 0) {
400 std::vector<double> result;
401 result.reserve(g.data().size());
402 for (auto const &gbl : g.data()) result.push_back(tau_L2_norm(gbl));
403 return result;
404 } else {
405 std::vector<nda::array<double, rank>> result;
406 result.reserve(g.data().size());
407 for (auto const &gbl : g.data()) result.push_back(tau_L2_norm(gbl));
408 return result;
409 }
410 } else if constexpr (nda::AnyOf<M, dlr_imtime, dlr_imfreq>) {
411 return tau_L2_norm(make_gf_dlr(g));
412 } else { // M == dlr
413 if constexpr (G::target_rank == 0) {
414 // use abs here to avoid sqrt of negative numbers, which can happen due to numerical errors
415 return std::sqrt(std::abs(g.mesh().dlr_it().innerprod(g.data(), g.data())));
416 } else {
417 auto result = nda::array<double, G::target_rank>(g.target_shape());
418 nda::for_each(g.target_shape(), [&](auto... is) {
419 auto dat = g.data()(nda::range::all, is...);
420 result(is...) = std::sqrt(std::abs(g.mesh().dlr_it().innerprod(dat, dat)));
421 });
422
423 return result;
424 }
425 }
