38 template <int N, int... Ns, typename F, typename G>
39 requires(MemoryGf<G> or is_block_gf_v<G>)
40 auto apply_to_mesh(F const &f, G const &g) {
41 using M = typename G::mesh_t;
42 if constexpr (sizeof...(Ns) > 0) {
43 return apply_to_mesh<Ns...>(f, apply_to_mesh<N>(f, g));
44 } else if constexpr (is_block_gf_v<G>) {
45 return map_block_gf([&](auto const &gbl) { return apply_to_mesh<N>(f, gbl); }, g);
46 } else {
47 static_assert(mesh::is_product<M>, "requires product mesh");
48 auto gfl = f(flatten_gf_2d<N>(g));
49 auto out_mesh = mesh::prod{triqs::tuple::replace<N>(g.mesh(), gfl.mesh())};
50 auto g_out = gf{out_mesh, g.target_shape()};
51 unflatten_gf_2d<N>(g_out, gfl);
52 return g_out;
53 }
54 }

70 template <int N = 0, int... Ns, typename G>
71 requires(MemoryGf<G> or is_block_gf_v<G>)
72 auto make_gf_dlr(G const &g) {
73 using M = typename G::mesh_t;
74 if constexpr (is_block_gf_v<G>) {
75 return map_block_gf([&](auto const &gbl) { return make_gf_dlr<N, Ns...>(gbl); }, g);
76 } else if constexpr (mesh::is_product<M>) {
77 return apply_to_mesh<N, Ns...>([&](auto const &gfl) { return make_gf_dlr(gfl); }, g);
78 } else {
79 static_assert(N == 0, "N must be 0 for non-product meshes");
80 static_assert(nda::AnyOf<M, dlr_imtime, dlr_imfreq>, "Mesh must be dlr_imtime or dlr_imfreq");
81 auto result = gf{dlr{g.mesh()}, g.target_shape()};
82 if constexpr (std::is_same_v<M, dlr_imtime>)
83 result.data() = result.mesh().dlr_it().vals2coefs(g.data());
84 else { // dlr_imfreq
85 auto beta_inv = 1. / result.mesh().beta();
86 result.data() = beta_inv * result.mesh().dlr_if().vals2coefs(g.data());
87 }
88 return result;
89 }

118 result.data() = result.mesh().dlr_it().fitvals2coefs(make_regular(tvals), g.data());

222 // Factored out so make_gf_dlr_imfreq and find_w_max build the dlr_imfreq mesh
223 // once (the cppdlr SVD is not cheap) and reuse it across blocks / search iterations.
224 template <typename G>
225 requires(MemoryGf<G> or is_block_gf_v<G>)
226 auto sample_on_dlr_imfreq(G const &g, dlr_imfreq const &dlr_mesh) {
227 if constexpr (is_block_gf_v<G>) {
228 return map_block_gf([&](auto const &gbl) { return sample_on_dlr_imfreq(gbl, dlr_mesh); }, g);
229 } else {
230 static_assert(std::is_same_v<typename G::mesh_t, mesh::imfreq>, "Input mesh must be imfreq");
231 auto const &iw_mesh = g.mesh();
232 auto [iw_lo, iw_hi] = dlr_mesh.min_max_frequencies();
233 if (not iw_mesh.is_index_valid(iw_lo.n) or not iw_mesh.is_index_valid(iw_hi.n))
234 TRIQS_RUNTIME_ERROR << "make_gf_dlr_imfreq: input imfreq mesh does not cover the requested DLR frequency range";
235 auto result = gf{dlr_mesh, g.target_shape()};
236 for (auto const &mp : dlr_mesh) result[mp] = g[iw_mesh(mp.index())];
237 return result;
238 }

311 template <int = 0, typename G>
312 requires(MemoryGf<G> or is_block_gf_v<G>)
313 double find_w_max(G const &g, double eps = 1e-10, bool symmetrize = true, double w_max_init = 1.0, double w_max_max = 200.0) {
314 if (w_max_init > w_max_max) TRIQS_RUNTIME_ERROR << "find_w_max: w_max_init (" << w_max_init << ") > w_max_max (" << w_max_max << ")";
315
316 auto const &m0 = detail::imfreq_mesh_of(g);
317
318 for (double w_max = w_max_init; w_max <= w_max_max; w_max *= 1.5) {
319 auto dlr_mesh = dlr_imfreq{m0.beta(), m0.statistic(), w_max, eps, symmetrize};
320 auto [iw_lo, iw_hi] = dlr_mesh.min_max_frequencies();
321 if (not m0.is_index_valid(iw_lo.n) or not m0.is_index_valid(iw_hi.n)) continue;
322
323 auto g_dlr_iw = detail::sample_on_dlr_imfreq(g, dlr_mesh);
324 auto g_rec = make_gf_imfreq(make_gf_dlr(g_dlr_iw), m0.n_iw());
325 auto data_err = [](auto const &x, auto const &y) { return double(max_element(abs(x.data() - y.data()))); };
326 double max_err = 0.0;
327 if constexpr (is_block_gf_v<G>) {
328 for (long b = 0; b < long(g.size()); ++b) max_err = std::max(max_err, data_err(g[b], g_rec[b]));
329 } else {
330 max_err = data_err(g, g_rec);
331 }
332 if (max_err < eps) return w_max;
333 }
334 TRIQS_RUNTIME_ERROR << "find_w_max: no w_max <= " << w_max_max << " yields round-trip error < eps = " << eps;
335 }

409 template <typename G>
410 requires(MemoryGf<G> or is_block_gf_v<G>)
411 auto tau_L2_norm(G const &g) {
412 using M = typename G::mesh_t;
413 static_assert(nda::AnyOf<M, dlr, dlr_imfreq, dlr_imtime>, "Input mesh must be one of dlr, dlr_imfreq, dlr_imtime");
414 if constexpr (is_block_gf_v<G>) {
415 // Manual loop with a concrete element type. map_block_gf would deduce
416 // std::vector<std::invoke_result_t<lambda, gf>> (the lambda is local),
417 // and a function-scoped `using elem_t = ...` alias would also leak its
418 // spelling -- in both cases clair-c2py fails to render the type.
419 constexpr int rank = std::decay_t<G>::g_t::target_t::rank;
420 if constexpr (rank == 0) {
421 std::vector<double> result;
422 result.reserve(g.data().size());
423 for (auto const &gbl : g.data()) result.push_back(tau_L2_norm(gbl));
424 return result;
425 } else {
426 std::vector<nda::array<double, rank>> result;
427 result.reserve(g.data().size());
428 for (auto const &gbl : g.data()) result.push_back(tau_L2_norm(gbl));
429 return result;
430 }
431 } else if constexpr (nda::AnyOf<M, dlr_imtime, dlr_imfreq>) {
432 return tau_L2_norm(make_gf_dlr(g));
433 } else { // M == dlr
434 if constexpr (G::target_rank == 0) {
435 // use abs here to avoid sqrt of negative numbers, which can happen due to numerical errors
436 return std::sqrt(std::abs(g.mesh().dlr_it().innerprod(g.data(), g.data())));
437 } else {
438 auto result = nda::array<double, G::target_rank>(g.target_shape());
439 nda::for_each(g.target_shape(), [&](auto... is) {
440 auto dat = g.data()(nda::range::all, is...);
441 result(is...) = std::sqrt(std::abs(g.mesh().dlr_it().innerprod(dat, dat)));
442 });
443
444 return result;
445 }
446 }
