158 state &operator+=(state const &phi) {
159 for (auto const &[f, b_f] : phi.map_) {
160 auto [it, inserted] = map_.insert({f, b_f});
161 if (!inserted) it->second += b_f;
162 }
163 prune();
164 return *this;
165 }

176 state &operator-=(state const &phi) {
177 for (auto const &[f, b_f] : phi.map_) {
178 auto [it, inserted] = map_.insert({f, -b_f});
179 if (!inserted) it->second -= b_f;
180 }
181 prune();
182 return *this;
183 }
184
194 state &operator*=(value_type x) {
195 for (auto &[f, a_f] : map_) { a_f *= x; }
196 prune();
197 return *this;
198 }

209 state &operator/=(value_type x) { return operator*=(value_type(1.0) / x); }

225 friend value_type dot_product(state const &phi, state const &psi) {
226 value_type res(0.0);
227 for (auto const &[f, b_f] : phi.map_) {
228 using triqs::utility::conj;
229 if (psi.map_.count(f) == 1) res += conj(b_f) * psi.map_.at(f);
230 }
231 return res;
232 }

259 // Discard all amplitudes smaller than a specific threshold.
260 void prune() {
261 for (auto it = map_.begin(); it != map_.end();) {
262 using triqs::utility::is_zero;
263 if (is_zero(it->second)) {
264 it = map_.erase(it);
265 } else {
266 ++it;
267 }
268 }

471 template <typename HilbertSpace, typename ScalarType, bool BasedOnMap>
472 std::ostream &operator<<(std::ostream &sout, state<HilbertSpace, ScalarType, BasedOnMap> const &psi) {
473 bool something_written = false;
474 auto const &hs = psi.get_hilbert();
475
476 foreach (psi, [&sout, hs, &something_written](int i, auto a_f_i) {
477 using triqs::utility::is_zero;
478 if (!is_zero(a_f_i)) {
479 sout << " +(" << a_f_i << ")"
480 << "|" << hs.get_fock_state(i) << ">";
481 something_written = true;
482 }
483 });
484
485 if (!something_written) sout << 0;
486 return sout;
487 }

507 template <typename TargetState, typename OriginalState> TargetState project(OriginalState const &psi, hilbert_space const &proj_hs) {
508 TargetState proj_psi(proj_hs);
509 auto const &hs = psi.get_hilbert();
510 foreach (psi, [&](int i, auto a_f_i) { proj_psi(hs.get_fock_state(i)) = a_f_i; });
511 return proj_psi;
512 }
513
531 template <typename TargetState, typename OriginalState> TargetState project(OriginalState const &psi, sub_hilbert_space const &proj_hs) {
532 TargetState proj_psi(proj_hs);
533 auto const &hs = psi.get_hilbert();
534 foreach (psi, [&](int i, auto a_f_i) {
535 auto f = hs.get_fock_state(i);
536 if (proj_hs.has_state(f)) proj_psi(proj_hs.get_state_index(f)) = a_f_i;
537 });
538 return proj_psi;
539 }
