45 using data_t = std::conditional_t<Arity == 1, std::vector<g_t>, std::vector<std::vector<g_t>>>;

67 block_gf_view(block_names_t b, data_t d) : _block_names(std::move(b)), _glist(std::move(d)) {
68 if constexpr (Arity == 1) {
69 if (_glist.size() != _block_names.size())
70 TRIQS_RUNTIME_ERROR << "block_gf(vector<string>, vector<gf>) : the two vectors do not have the same size !";
71 } else {
72 if (_glist.size() != _block_names[0].size())
73 TRIQS_RUNTIME_ERROR << "block2_gf(vector<vector<string>>, vector<vector<gf>>) : Outer vectors have different sizes !";
74 if (_glist.size() != 0)
75 if (_glist[0].size() != _block_names[1].size())
76 TRIQS_RUNTIME_ERROR << "block2_gf(vector<vector<string>>, vector<vector<gf>>) : Inner vectors have different sizes !";
77 }
78 }

114 template <typename RHS>
115 block_gf_view &operator=(RHS const &rhs)
116 requires(not IsConst)
117 {
118 if constexpr (not nda::is_scalar_v<RHS>) {
119 if (!(size() == rhs.size())) TRIQS_RUNTIME_ERROR << "Gf Assignment in View : incompatible size" << size() << " vs " << rhs.size();
120 _assign_impl(rhs);
121 } else {
122 if constexpr (Arity == 1) {
123 for (auto &y : _glist) y = rhs;
124 } else {
125 for (auto &x : _glist)
126 for (auto &y : x) y = rhs;
127 }
128 }
129 return *this;
130 }
131
137 template <typename L, typename G>
138 block_gf_view &operator=(lazy_transform_t<L, G> const &rhs)
139 requires(not IsConst)
140 {
141 if constexpr (Arity == 1) {
142 for (int i = 0; i < rhs.value.size(); ++i) (*this)[i] = rhs.lambda(rhs.value[i]);
143 } else {
144
145 for (int i = 0; i < rhs.value.size1(); ++i)
146 for (int j = 0; j < rhs.value.size2(); ++j) (*this)(i, j) = rhs.lambda(rhs.value(i, j));
147 }
148 return *this;
149 }

156 name = x.name;
157 }
