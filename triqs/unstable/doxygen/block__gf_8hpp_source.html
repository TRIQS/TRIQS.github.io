241 // Create Block Green function from Mesh and gf_struct
242 block_gf(Mesh const &m, gf_struct_t const &gf_struct)
243 requires(Arity == 1)
244 {
245
246 for (auto const &[bl_name, bl_size] : gf_struct) {
247 _block_names.push_back(bl_name);
248 if constexpr (Target::rank == 0)
249 _glist.emplace_back(m);
250 else
251 _glist.emplace_back(m, make_shape(bl_size, bl_size));
252 }
253 }
254
255 // Create Block Green function from Mesh and gf_struct
256 template <typename Int>
257 block_gf(Mesh const &m, std::vector<Int> const &bl_sizes)
258 requires(Arity == 1 && std::is_integral_v<Int>)
259 {
260
261 for (auto const &[bl, bl_size] : itertools::enumerate(bl_sizes)) {
262 _block_names.push_back(std::to_string(bl));
263 if constexpr (Target::rank == 0)
264 _glist.emplace_back(m);
265 else
266 _glist.emplace_back(m, make_shape(bl_size, bl_size));
267 }
268 }

296 template <typename RHS>
297 requires(BlockGreenFunction_v<RHS>)
298 block_gf &operator=(RHS &&rhs) {
299 if constexpr (Arity == 1) {
300 _glist.resize(rhs.size());
301 _block_names.resize(rhs.size());
302 _assign_impl(rhs);
303 } else {
304 _block_names.resize(2);
305 _glist.resize(rhs.size1());
306 for (auto &g_bl : _glist) g_bl.resize(rhs.size2());
307 _block_names[0].resize(rhs.size1());
308 _block_names[1].resize(rhs.size2());
309 _assign_impl(rhs);
310 }
311 return *this;
312 }
