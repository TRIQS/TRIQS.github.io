23 TRIQS_ASSERT(r_l.size() == dim());
24 auto R1 = nda::vector<double>(r_l);
25 auto R = transfo_SL_L * r_l;

27 if (max_element(abs(R - R_rounded)) < epsilon) {
28 return R_rounded; // Return the rounded vector if it is close enough to an integer vector

46 template <int kdim> fourier_polynomial<2, kdim> fold(superlattice const &sl, fourier_polynomial<2, kdim> const &fp) {
47 using itertools::enumerate;
48 if (sl.dim() != kdim) throw std::runtime_error("Superlattice dimension does not match the Fourier polynomial dimension.");
49
50 auto n_orb_L = fp.get_coeff_arr().shape(1); // number of orbitals in the lattice
51 auto n_orb_SL = n_orb_L * sl.n_cluster_sites(); // number of orbitals in the superlattice
52 auto orb_range = [&](long a) { return nda::range(a * n_orb_L, (a + 1) * n_orb_L); };
53
54 // create containers for the new R vectors and coefficients
55 std::vector<std::array<long, kdim>> SL_R_list;
56 std::map<std::array<long, kdim>, long> SL_R_idx;
57 std::vector<nda::array<dcomplex, 2>> SL_coeff_list;
58
59 auto const &cpts = sl.get_cluster_pts();
60 for (auto i : nda::range(sl.n_cluster_sites()))
61 for (auto j : nda::range(sl.n_cluster_sites())) {
62 auto ai = cpts(i, _all_);
63 auto aj = cpts(j, _all_);
64 for (auto &&[r_idx, r] : enumerate(fp.get_R_list())) {
65 auto r1 = nda::vector<long>(kdim);
66 for (int u = 0; u < kdim; ++u) r1[u] = r[u] + ai[u] - aj[u]; // r1 = r + ai - aj
67 auto R_rounded = sl.L_to_SL(r1); // transform r + ai - aj to superlattice coordinates and round it
68 if (R_rounded) { // if the rounding was successful
69 auto R_rounded_2 = detail::make_std_array_from_vector<kdim>(R_rounded.value()); // convert to std::array<long, kdim>
70 auto [it, is_new] = SL_R_idx.emplace(R_rounded_2, SL_R_list.size()); // store the R_sl if not already seen
71 if (is_new) {
72 SL_R_list.push_back(R_rounded_2);
73 SL_coeff_list.emplace_back(nda::zeros<dcomplex>(n_orb_SL, n_orb_SL));
74 }
75 SL_coeff_list[it->second](orb_range(i), orb_range(j)) += fp[r_idx];
76 }
77 }
78 }
79 return {std::move(SL_R_list), std::move(SL_coeff_list)};
80 }
