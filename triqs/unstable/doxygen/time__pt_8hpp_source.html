69 inline friend time_pt operator+(time_pt const &a, time_pt const &b) {
70 bool wrapped = ((Nmax - std::max(a.n, b.n)) < std::min(a.n, b.n));
71 if (!wrapped)
72 return time_pt((a.n + b.n) % Nmax, a.beta);
73 else
74 return time_pt(((a.n + b.n) + 1) % Nmax, a.beta);
75 }
76 inline friend time_pt operator-(time_pt const &a, time_pt const &b) {
77 uint64_t nres = (a.n >= b.n ? (a.n - b.n) % Nmax : Nmax - (b.n - a.n));
78 return time_pt(nres, a.beta);
79 }
