34 class real_or_complex {
35
36 bool _is_real = true;
37 std::complex<double> _x;
38
39 public:
40 real_or_complex() : _x(0) {}
41 real_or_complex(double x) : _x(x) {}
42 real_or_complex(std::complex<double> x) : _is_real(false), _x(std::move(x)) {}
43
44 bool is_real() const { return _is_real; }
45
46 explicit operator std::complex<double>() const { return _x; }
47
48 explicit operator double() const {
49 if (!_is_real) TRIQS_RUNTIME_ERROR << "Logic error : the number is not real, it is complex";
50 return real(_x);
51 }
52
53 friend real_or_complex operator-(real_or_complex a) {
54 a._x = -a._x;
55 return a;
56 }
57
58 friend double real(real_or_complex const &x) { return real(x._x); }
59 friend double imag(real_or_complex const &x) { return imag(x._x); }
60 friend double abs(real_or_complex const &x) { return std::abs(x._x); }
61 friend real_or_complex conj(real_or_complex x) {
62 x._x = std::conj(x._x);
63 return x;
64 }
65
66 friend bool is_zero(real_or_complex const &x, double tolerance = 100 * std::numeric_limits<double>::epsilon()) {
67 using triqs::utility::is_zero;
68 if (x.is_real()) return is_zero(double(x), tolerance);
69 return is_zero(x._x, tolerance);
70 }
71
72 friend std::ostream &operator<<(std::ostream &out, real_or_complex const &x) {
73 if (x._is_real)
74 out << double(x);
75 else {
76 out << '(' << x._x.real();
77 if (not std::signbit(x._x.imag())) out << '+';
78 out << x._x.imag() << "j)";
79 }
80 return out;
81 }
82
83 bool operator==(real_or_complex const &x) const { return abs(_x - x._x) == 0; }
84 bool operator!=(real_or_complex const &x) const { return !operator==(x); }
85
86#define MAKE_OP(OP) \
87 inline real_or_complex &operator OP(double y) { \
88 _x OP y; \
89 return *this; \
90 } \
91 inline real_or_complex &operator OP(std::complex<double> const &y) { \
92 _x OP y; \
93 _is_real = false; \
94 return *this; \
95 } \
96 inline real_or_complex &operator OP(real_or_complex const &y) { \
97 _x OP y._x; \
98 _is_real &= y._is_real; \
99 return *this; \
100 }
101
102 MAKE_OP(+=);
103 MAKE_OP(-=);
104 MAKE_OP(*=);
105 MAKE_OP(/=);
106#undef MAKE_OP
107 };
108
109#define MAKE_OP(OP, OPC) \
110 inline real_or_complex operator OP(real_or_complex a, real_or_complex const &b) { \
111 a OPC b; \
112 return a; \
113 } \
114 inline real_or_complex operator OP(real_or_complex a, std::complex<double> const &b) { \
115 a OPC b; \
116 return a; \
117 } \
118 inline real_or_complex operator OP(real_or_complex a, double b) { \
119 a OPC b; \
120 return a; \
121 } \
122 inline real_or_complex operator OP(std::complex<double> const &a, real_or_complex b) { return a OP std::complex<double>(b); } \
123 inline real_or_complex operator OP(double a, real_or_complex b) { \
124 if (b.is_real()) \
125 return a OP double(b); \
126 else \
127 return a OP std::complex<double>(b); \
128 }
129
130 MAKE_OP(+, +=);
131 MAKE_OP(-, -=);
132 MAKE_OP(*, *=);
133 MAKE_OP(/, /=);
134#undef MAKE_OP
