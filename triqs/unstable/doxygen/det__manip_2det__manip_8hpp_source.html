40 template <typename x_type, typename y_type, typename value_type> struct work_data_type1 {
41 x_type x;
42 y_type y;

50 MB.resize(N);
51 MC.resize(N);
52 B.resize(N);
53 C.resize(N);
54 }
55 };

57 // For multiple-row/column operations
58 template <typename x_type, typename y_type, typename value_type> struct work_data_typek {
59 std::vector<x_type> x;
60 std::vector<y_type> y;
61 std::vector<long> i, j, ireal, jreal;
62 // MB = A^(-1)*B,
63 // MC = C*A^(-1)
64 nda::matrix<value_type> MB, MC, B, C, ksi;
65 void resize(long N, long k) {
66 if (k < 2) return;
67 x.resize(k);
68 y.resize(k);
69 i.resize(k);
70 j.resize(k);
71 ireal.resize(k);
72 jreal.resize(k);
73 MB.resize(N, k);
74 MC.resize(k, N);
75 B.resize(N, k);
76 C.resize(k, N);
77 ksi.resize(k, k);
78 }
79 value_type det_ksi(long k) const {
80 if (k == 2) {
81 return ksi(0, 0) * ksi(1, 1) - ksi(1, 0) * ksi(0, 1);
82 } else if (k == 3) {
83 return // Rule of Sarrus
84 ksi(0, 0) * ksi(1, 1) * ksi(2, 2) + //
85 ksi(0, 1) * ksi(1, 2) * ksi(2, 0) + //
86 ksi(0, 2) * ksi(1, 0) * ksi(2, 1) - //
87 ksi(2, 0) * ksi(1, 1) * ksi(0, 2) - //
88 ksi(2, 1) * ksi(1, 2) * ksi(0, 0) - //
89 ksi(2, 2) * ksi(1, 0) * ksi(0, 1); //
90 } else {
91 auto Rk = range(k);
92 return nda::linalg::det(ksi(Rk, Rk));
93 };
94 }
95 };

303 template <typename ArgumentContainer1, typename ArgumentContainer2>
304 det_manip(FunctionType F, ArgumentContainer1 const &X, ArgumentContainer2 const &Y) : f(std::move(F)), Nmax(0) {
305 if (X.size() != Y.size()) TRIQS_RUNTIME_ERROR << " X.size != Y.size";
306 N = X.size();
307 if (N == 0) {
308 det = 1;
309 reserve(30);
310 return;
311 }
312 reserve(N);
313 std::copy(X.begin(), X.end(), std::back_inserter(x_values));
314 std::copy(Y.begin(), Y.end(), std::back_inserter(y_values));
315 mat_inv() = 0;
316 for (long i = 0; i < N; ++i) {
317 row_num.push_back(i);
318 col_num.push_back(i);
319 for (long j = 0; j < N; ++j) mat_inv(i, j) = f(x_values[i], y_values[j]);
320 }
321 range RN(N);
322 det = nda::linalg::det(mat_inv(RN, RN));
323 mat_inv(RN, RN) = nda::linalg::inv(mat_inv(RN, RN));
324 }

482 value_type try_insert(long i, long j, x_type const &x, y_type const &y) {
483
484 // check input and store it for complete_operation
485 TRIQS_ASSERT(last_try == NoTry);
486 TRIQS_ASSERT(0 <= i and i <= N);
487 TRIQS_ASSERT(0 <= j and j <= N);
488 reserve(N + 1);
489 last_try = Insert;
490 w1.i = i;
491 w1.j = j;
492 w1.x = x;
493 w1.y = y;
494
495 // treat empty matrix separately
496 if (N == 0) {
497 newdet = f(x, y);
498 newsign = 1;
499 return value_type(newdet);
500 }
501
502 // I add the row and col and the end. If the move is rejected,
503 // no effect since N will not be changed : Minv(i,j) for i,j>=N has no meaning.
504 for (long l = 0; l < N; l++) {
505 w1.B(l) = f(x_values[l], y);
506 w1.C(l) = f(x, y_values[l]);
507 }
508 range RN(N);
509 //w1.MB(R) = mat_inv(R,R) * w1.B(R);// OPTIMIZE BELOW
510 blas::gemv(1.0, mat_inv(RN, RN), w1.B(RN), 0.0, w1.MB(RN));
511 w1.ksi = f(x, y) - nda::blas::dot(w1.C(RN), w1.MB(RN));
512 newdet = det * w1.ksi;
513 newsign = ((i + j) % 2 == 0 ? sign : -sign); // since N-i0 + N-j0 = i0+j0 [2]
514 return w1.ksi * (newsign * sign); // sign is unity, hence 1/sign == sign
515 }

517 //fx gives the new line coefficients, fy gives the new column coefficients and ksi is the last coeff (at the intersection of the line and the column).
518 template <typename Fx, typename Fy> value_type try_insert_from_function(long i, long j, Fx fx, Fy fy, value_type const ksi) {
519
520 // check input and store it for complete_operation
521 TRIQS_ASSERT(last_try == NoTry);
522 TRIQS_ASSERT(0 <= i and i <= N);
523 TRIQS_ASSERT(0 <= j and j <= N);
524 reserve(N + 1);
525 last_try = Insert;
526 w1.i = i;
527 w1.j = j;
528
529 // treat empty matrix separately
530 if (N == 0) {
531 newdet = ksi;
532 newsign = 1;
533 return newdet;
534 }
535
536 // I add the row and col and the end. If the move is rejected,
537 // no effect since N will not be changed : Minv(i,j) for i,j>=N has no meaning.
538 for (long l = 0; l < N; l++) {
539 w1.B(l) = fx(x_values[l]);
540 w1.C(l) = fy(y_values[l]);
541 }
542 range RN(N);
543 //w1.MB(R) = mat_inv(R,R) * w1.B(R);// OPTIMIZE BELOW
544 blas::gemv(1.0, mat_inv(RN, RN), w1.B(RN), 0.0, w1.MB(RN));
545 w1.ksi = ksi - nda::blas::dot(w1.C(RN), w1.MB(RN));
546 newdet = det * w1.ksi;
547 newsign = ((i + j) % 2 == 0 ? sign : -sign); // since N-i0 + N-j0 = i0+j0 [2]
548 return w1.ksi * (newsign * sign); // sign is unity, hence 1/sign == sign
549 }

553 void complete_insert() {
554 // store the new value of x,y. They are seen through the same permutations as rows and cols resp.
555 x_values.push_back(w1.x);
556 y_values.push_back(w1.y);
557 row_num.push_back(0);
558 col_num.push_back(0);
559
560 // special empty case again
561 if (N == 0) {
562 N = 1;
563 mat_inv(0, 0) = 1 / value_type(newdet);
564 return;
565 }
566
567 range RN(N);
568 //w1.MC(R1) = transpose(mat_inv(R1,R1)) * w1.C(R1); //OPTIMIZE BELOW
569 blas::gemv(1.0, transpose(mat_inv(RN, RN)), w1.C(RN), 0.0, w1.MC(RN));
570 w1.MC(N) = -1;
571 w1.MB(N) = -1;
572
573 N++;
574 RN = range(N);
575
576 // keep the real position of the row/col
577 // since we insert a col/row, we have first to push the col at the right
578 // and then say that col w1.i is stored in N, the last col.
579 // same for rows
580 for (long i = N - 2; i >= w1.i; i--) row_num[i + 1] = row_num[i];
581 row_num[w1.i] = N - 1;
582 for (long i = N - 2; i >= w1.j; i--) col_num[i + 1] = col_num[i];
583 col_num[w1.j] = N - 1;
584
585 // Minv is ok, we need to complete
586 w1.ksi = 1 / w1.ksi;
587
588 // compute the change to the inverse
589 // M += w1.ksi w1.MB w1.MC with BLAS. first put the 0
590 mat_inv(RN, N - 1) = 0;
591 mat_inv(N - 1, RN) = 0;
592 //mat_inv(R,R) += w1.ksi* w1.MB(R) * w1.MC(R)// OPTIMIZE BELOW
593 blas::ger(w1.ksi, w1.MB(RN), w1.MC(RN), mat_inv(RN, RN));
594 }

611 value_type try_insert_k(std::vector<long> i, std::vector<long> j, std::vector<x_type> x, std::vector<y_type> y) {
612 TRIQS_ASSERT(last_try == NoTry);
613 TRIQS_ASSERT(i.size() == j.size());
614 TRIQS_ASSERT(j.size() == x.size());
615 TRIQS_ASSERT(x.size() == y.size());
616
617 k_tried = i.size();
618 reserve(N + k_tried, k_tried);
619 last_try = InsertK;
620
621 auto const argsort = [](auto const &vec) {
622 std::vector<long> idx(vec.size());
623 std::iota(idx.begin(), idx.end(), static_cast<long>(0));
624 std::stable_sort(idx.begin(), idx.end(), [&vec](long const lhs, long const rhs) { return vec[lhs] < vec[rhs]; });
625 return idx;
626 };
627 std::vector<long> idx = argsort(i);
628 std::vector<long> idy = argsort(j);
629
630 // store it for complete_operation
631 for (long l = 0; l < k_tried; ++l) {
632 wk.i[l] = i[idx[l]];
633 wk.x[l] = x[idx[l]];
634 wk.j[l] = j[idy[l]];
635 wk.y[l] = y[idy[l]];
636 };
637
638 // check consistency
639 for (int l = 0; l < k_tried - 1; ++l) {
640 TRIQS_ASSERT(wk.i[l] != wk.i[l + 1] and 0 <= wk.i[l] and wk.i[l] < N + k_tried);
641 TRIQS_ASSERT(wk.j[l] != wk.j[l + 1] and 0 <= wk.j[l] and wk.j[l] < N + k_tried);
642 }
643
644 // w1.ksi = Delta(x_values,y_values) - Cw.MB using BLAS
645 for (long m = 0; m < k_tried; ++m) {
646 for (long n = 0; n < k_tried; ++n) { wk.ksi(m, n) = f(wk.x[m], wk.y[n]); }
647 }
648
649 // treat empty matrix separately
650 if (N == 0) {
651 newdet = wk.det_ksi(k_tried);
652 newsign = 1;
653 return value_type(newdet);
654 }
655
656 // I add the rows and cols and the end. If the move is rejected,
657 // no effect since N will not be changed : inv_mat(i,j) for i,j>=N has no meaning.
658 for (long n = 0; n < N; n++) {
659 for (long l = 0; l < k_tried; ++l) {
660 wk.B(n, l) = f(x_values[n], wk.y[l]);
661 wk.C(l, n) = f(wk.x[l], y_values[n]);
662 }
663 }
664 range RN(N), Rk(k_tried);
665 //wk.MB(RN,Rk) = mat_inv(RN,N) * wk.B(RN,Rk); // OPTIMIZE BELOW
666 blas::gemm(1.0, mat_inv(RN, RN), wk.B(RN, Rk), 0.0, wk.MB(RN, Rk));
667 //ksi -= wk.C (Rk, RN) * wk.MB(RN, Rk); // OPTIMIZE BELOW
668 blas::gemm(-1.0, wk.C(Rk, RN), wk.MB(RN, Rk), 1.0, wk.ksi(Rk, Rk));
669 auto ksi = wk.det_ksi(k_tried);
670 newdet = det * ksi;
671 long idx_sum = 0;
672 for (long l = 0; l < k_tried; ++l) { idx_sum += wk.i[l] + wk.j[l]; }
673 newsign = (idx_sum % 2 == 0 ? sign : -sign); // since N-i0 + N-j0 + N + 1 -i1 + N+1 -j1 = i0+j0 [2]
674 return ksi * (newsign * sign); // sign is unity, hence 1/sign == sign

693 template <nda::Array X, nda::Array Y>
694 requires(nda::get_rank<X> == nda::get_rank<Y>)
695 auto insert_ratios(long i, long j, X const &xs, Y const &ys) const -> nda::array<value_type, nda::get_rank<X>> {
696 constexpr int R = nda::get_rank<X>;
697 TRIQS_ASSERT(xs.shape() == ys.shape());
698 TRIQS_ASSERT(0 <= i and i <= N);
699 TRIQS_ASSERT(0 <= j and j <= N);
700
701 long nbatch = xs.size();
702 value_type sign_fac = ((i + j) % 2 == 0 ? 1 : -1);
703 nda::array<value_type, R> result(xs.shape());
704
705 if (nbatch == 0) return result;
706
707 // Flatten inputs for BLAS
708 auto xs_flat = flatten_array(xs);
709 auto ys_flat = flatten_array(ys);
710
711 if (N == 0) {
712 for (long m = 0; m < nbatch; ++m) result.data()[m] = sign_fac * f(xs_flat[m], ys_flat[m]);
713 return result;
714 }
715
716 range RN(N);
717
718 // Build B(N, nbatch) and C(nbatch, N) matrices
719 nda::matrix<value_type> B(N, nbatch), C(nbatch, N), MB(N, nbatch);
720 for (long l = 0; l < N; ++l)
721 for (long m = 0; m < nbatch; ++m) B(l, m) = f(x_values[l], ys_flat[m]);
722 for (long m = 0; m < nbatch; ++m)
723 for (long l = 0; l < N; ++l) C(m, l) = f(xs_flat[m], y_values[l]);
724
725 // MB = mat_inv * B -- single BLAS3 gemm
726 blas::gemm(1.0, mat_inv(RN, RN), B, 0.0, MB);
727
728 // Compute each ratio: ksi_m = f(xs[m], ys[m]) - C[m,:] . MB[:,m]
729 for (long m = 0; m < nbatch; ++m) {
730 value_type dot = 0;
731 for (long l = 0; l < N; ++l) dot += C(m, l) * MB(l, m);
732 result.data()[m] = sign_fac * (f(xs_flat[m], ys_flat[m]) - dot);
733 }
734
735 return result;
736 }
