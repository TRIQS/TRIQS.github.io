373 friend void h5_read(h5::group g, std::string const &name, dlr_imfreq &m) {
374 h5::group gr = g.open_group(name);
375 h5::assert_hdf5_format(gr, m, true); // NOLINT (downcasting to base class)
376 auto b = h5::read<double>(gr, "beta");
377 auto stat = (h5::read<std::string>(gr, "statistic") == "F" ? Fermion : Boson);
378 auto wmax = h5::read<double>(gr, "w_max");
379 auto epsilon = h5::read<double>(gr, "eps");
380 bool sym = false;
381 h5::try_read(gr, "symmetrize", sym);
382 auto freq = h5::read<nda::vector<double>>(gr, "dlr_freq");
383 auto imt = h5::read<cppdlr::imtime_ops>(gr, "dlr_it");
384 auto imf = h5::read<cppdlr::imfreq_ops>(gr, "dlr_if");
385 m = dlr_imfreq(b, stat, wmax, epsilon, sym, {.freq = freq, .imt = imt, .imf = imf});
386 }
