380 friend void h5_read(h5::group g, std::string const &name, dlr &m) {
381 h5::group gr = g.open_group(name);
382 h5::assert_hdf5_format(gr, m, true); // NOLINT (downcasting to base class)
383 auto b = h5::read<double>(gr, "beta");
384 auto stat = (h5::read<std::string>(gr, "statistic") == "F" ? Fermion : Boson);
385 auto wmax = h5::read<double>(gr, "w_max");
386 auto epsilon = h5::read<double>(gr, "eps");
387 bool sym = false;
388 h5::try_read(gr, "symmetrize", sym);
389 auto freq = h5::read<nda::vector<double>>(gr, "dlr_freq");
390 auto imt = h5::read<cppdlr::imtime_ops>(gr, "dlr_it");
391 auto imf = h5::read<cppdlr::imfreq_ops>(gr, "dlr_if");
392 m = dlr(b, stat, wmax, epsilon, sym, {.freq = freq, .imt = imt, .imf = imf});
393 }

428 auto evaluate(dlr const &m, auto const &f, double tau) {
429 EXPECTS(m.size() > 0);
430 EXPECTS(tau >= 0 and tau <= m.beta());
431 return detail::sum_to_regular(nda::range(m.size()), [&](auto l) { return f(l) * cppdlr::k_it(tau / m.beta(), m.dlr_freq()[l]); });
432 }
433
452 auto evaluate(dlr const &m, auto const &f, matsubara_freq const &iw) {
453 EXPECTS(m.size() > 0);
454 return detail::sum_to_regular(nda::range(m.size()),
455 [&](auto l) { return f(l) * cppdlr::k_if(iw.n, m.dlr_freq()[l], (cppdlr::statistic_t)iw.statistic) * m.beta(); });
456 }
