24 std::tuple<std::vector<std::array<long, 3>>, std::vector<nda::array<dcomplex, 2>>, std::vector<nda::array<dcomplex, 3>>, nda::matrix<double>>
25 read_wannier90_tb_data(std::string const &w90_path_and_seedname) {
26
27 // open the file
28 std::string tb_filename = w90_path_and_seedname + "_tb.dat";
29 std::ifstream file(tb_filename);
30 if (!file.is_open()) throw std::runtime_error("Could not open *_tb.dat file with provided name " + tb_filename);
31
32 // skip first line, it's a 5 item line of the date / time
33 for ([[maybe_unused]] auto i : detail::read_N<std::string>(file, 5)) {}
34
35 // read the lattice vectors
36 auto latt_vec = nda::matrix<double>(3, 3);
37 for (auto x : {0, 1, 2})
38 for (auto y : {0, 1, 2}) { latt_vec(x, y) = detail::read<double>(file); }
39
40 // next two lines have number of wannier centers, number of R vectors
41 int n_wannier = detail::read<int>(file);
42 int n_r = detail::read<int>(file);
43
44 // read in the R vector degeneracies
45 std::vector<int> r_degeneracies;
46 for (auto i : detail::read_N<int>(file, n_r)) { r_degeneracies.push_back(i); };
47
48 // set up containers for R and H_r
49 std::vector<std::array<long, 3>> r_vectors;
50 std::vector<nda::array<dcomplex, 2>> H_r;
51 std::vector<nda::array<dcomplex, 3>> position_op_r;
52
53 for ([[maybe_unused]] auto r : nda::range(n_r)) {
54 r_vectors.emplace_back(std::array<long, 3>{0, 0, 0});
55 H_r.emplace_back(n_wannier, n_wannier);
56 position_op_r.emplace_back(n_wannier, n_wannier, 3);
57 }
58
59 // read the R vectors and the Wannier Hamiltonian -------
60 std::array<long, 3> R;
61 for (auto ir : nda::range(n_r)) {
62
63 file >> R[0] >> R[1] >> R[2];
64 for ([[maybe_unused]] auto [x, y] : product(nda::range(n_wannier), nda::range(n_wannier))) {
65
66 double re, im;
67 int ii, jj; // read these from the file rather than as loop indices
68 file >> ii >> jj >> re >> im;
69
70 // check state
71 // NB if we want to use ws_cells vec data, we will likely want to use all wannier indices instead
72 // (wannier90 indexes this from 1)
73 if ((ii == 1) and (jj == 1))
74 for (int i : {0, 1, 2}) r_vectors[ir][i] = R[i];
75
76 // merge degeneracy vector into Wannier Hamiltonian (note, W90 indexes from 1)
77 H_r[ir](ii - 1, jj - 1) = dcomplex{re, im} / r_degeneracies[ir];
78 }
79 }
80
81 // read in the position operator elements in the Wannier basis --------------------
82 // These appear at the end of the tb.dat file in units of Angstrom
83 for (auto ir : nda::range(n_r)) {
84 // R vector is listed before each r_matrix block in the file
85 file >> R[0] >> R[1] >> R[2];
86 for ([[maybe_unused]] auto [x, y] : product(nda::range(n_wannier), nda::range(n_wannier))) {
87
88 std::array<double, 3> re, im;
89 int ii, jj; // read these from the file rather than as loop indices
90 file >> ii >> jj >> re[0] >> im[0] >> re[1] >> im[1] >> re[2] >> im[2];
91 // merge degeneracy information into the position operator
92 for (auto idx_cart : {0, 1, 2})
93 position_op_r[ir](ii - 1, jj - 1, idx_cart) = dcomplex{re[idx_cart], im[idx_cart]} / r_degeneracies[ir]; // in Angstrom
94 }
95 }
96 return {std::move(r_vectors), std::move(H_r), std::move(position_op_r), std::move(latt_vec)};
97 }
