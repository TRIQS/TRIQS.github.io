181 // Apply a function to each element of a range and sum the results into a regular type.
182 [[nodiscard]] auto sum_to_regular(std::ranges::forward_range auto &&rg, auto f) {
183 auto it = std::ranges::begin(rg);
184 auto e = std::ranges::end(rg);
185 auto res = nda::make_regular(f(*it));
186 for (++it; it != e; ++it) res += f(*it);
187 return res;
188 }
