240 template <typename M, typename T, typename L, typename RHS> void triqs_gf_view_assign_delegation(gf_view<M, T, L> g, RHS const &rhs) {
241 if constexpr (nda::is_scalar_v<RHS>) {
242 for (auto w : g.mesh()) g[w] = rhs;
243 } else {
244 if (!(g.mesh() == rhs.mesh())) TRIQS_RUNTIME_ERROR << "Gf Assignment in View : incompatible mesh \n" << g.mesh() << "\n vs \n" << rhs.mesh();
245 for (auto w : g.mesh()) g[w] = rhs[w];
246 }
