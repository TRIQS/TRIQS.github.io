126 // convert from gf to nda symmetry
127 data_sym_func_t to_data_symmetry(F const &f, G const &g) const {
128
129 auto fp = [f, m = g.mesh()](data_index_t const &x) -> std::tuple<data_index_t, nda::operation> {
130 // init new data index and residual operation
131 data_index_t xp;
132
133 if constexpr (target_rank == 0) { // scalar valued gfs
134
135 if constexpr (mesh_rank == 1) {
136 auto [new_mesh_index, op] = f(m.to_index(x[0]));
137 xp[0] = m.to_data_index(new_mesh_index);
138 return {xp, op};
139
140 } else { // product mesh
141 auto [new_mesh_index, op] = f(m.to_index(to_tuple(x)));
142 xp = to_array(m.to_data_index(new_mesh_index));
143 return {xp, op};
144 }
145
146 } else { // tensor valued gfs
147
148 // convert data index to target index
149 target_index_t target_index;
150 for (auto i : range(target_rank)) target_index[i] = x[i + mesh_rank];
151
152 if constexpr (mesh_rank == 1) {
153 // evaluate symmetry
154 auto [new_mesh_index, new_target_index, op] = f(m.to_index(x[0]), target_index);
155
156 // convert mesh index + target index back to data index
157 xp[0] = m.to_data_index(new_mesh_index);
158 for (auto i : range(target_rank)) xp[i + mesh_rank] = new_target_index[i];
159
160 return {xp, op};
161
162 } else { // product mesh
163 // evaluate symmetry
164 auto [new_mesh_index, new_target_index, op] = f(m.to_index(to_tuple<mesh_rank>(x)), target_index);
165
166 // convert mesh index + target index back to data index
167 auto new_mesh_arr = to_array(m.to_data_index(new_mesh_index));
168 for (auto i : range(mesh_rank)) xp[i] = new_mesh_arr[i];
169 for (auto i : range(target_rank)) xp[i + mesh_rank] = new_target_index[i];
170
171 return {xp, op};
172 }
173 }
174 };
175
176 return fp;
177 };

179 // convert from list of gf symmetries to list of nda symmetries
180 std::vector<data_sym_func_t> to_data_symmetry_list(G const &g, std::vector<F> const &sym_list) const {
181 std::vector<data_sym_func_t> data_sym_list;
182 for (auto f : sym_list) data_sym_list.push_back(to_data_symmetry(f, g));
183 return data_sym_list;

186 // convert from gf to nda init function
187 template <typename H> data_init_func_t to_data_init_func(G const &g, H const &h) const {
188
189 auto hp = [h, m = g.mesh()](data_index_t const &x) {
190 if constexpr (target_rank == 0) { // scalar valued gfs
191
192 if constexpr (mesh_rank == 1) {
193 return h(m.to_index(x[0]));
194
195 } else { // product mesh
196 return h(m.to_index(to_tuple(x)));
197 }
198
199 } else { // tensor valued gfs
200
201 target_index_t target_index;
202 for (auto i : range(target_rank)) target_index[i] = x[i + mesh_rank];
203
204 if constexpr (mesh_rank == 1) {
205 return h(m.to_index(x[0]), target_index);
206
207 } else { // product mesh
208 return h(m.to_index(to_tuple<mesh_rank>(x)), target_index);
209 }
210 }
211 };
212
213 return hp;
214 }

278 std::tuple<double, mesh_index_t, target_index_t> symmetrize(G &g) const {
279 auto const &[max_diff, max_index] = data_sym_grp.symmetrize(g.data());
280 auto const m = g.mesh();
281
282 if constexpr (target_rank == 0) { // scalar valued gfs
283
284 if constexpr (mesh_rank == 1) {
285 return {max_diff, m.to_index(max_index[0]), {}};
286
287 } else { // product mesh
288 return {max_diff, m.to_index(to_tuple(max_index)), {}};
289 }
290
291 } else { // tensor valued gfs
292
293 // convert data index to target index
294 target_index_t target_index;
295 for (auto i : range(target_rank)) target_index[i] = max_index[i + mesh_rank];
296
297 if constexpr (mesh_rank == 1) {
298 return {max_diff, m.to_index(max_index[0]), target_index};
299
300 } else { // product mesh
301 return {max_diff, m.to_index(to_tuple<mesh_rank>(max_index)), target_index};
