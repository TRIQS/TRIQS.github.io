71 template <typename HilbertType, typename ScalarType = double, bool UseMap = false> class imperative_operator {

73 // Fock state convention:
74 // |0,...,k> = C^+_0 ... C^+_k |0>
75 // Operator monomial convention:
76 // C^+_0 ... C^+_i ... C_j ... C_0

78 using scalar_t = ScalarType;

80 // Precomputed bitmask representation of a single monomial: coeff is the coefficient,
81 // d_mask / dag_mask mark the annihilation / creation operator bit positions, and
82 // d_count_mask / dag_count_mask are the precomputed parity masks used for the fermionic sign.
83 struct one_term_t {
84 scalar_t coeff;
85 uint64_t d_mask, dag_mask, d_count_mask, dag_count_mask;
86 };
87 std::vector<one_term_t> all_terms;
88
89 std::vector<sub_hilbert_space> const *sub_spaces;
90 using hilbert_map_t = std::vector<int>;
91 hilbert_map_t hilbert_map;

95 imperative_operator() {}

101 bool is_empty() const { return (all_terms.size() == 0); }

123 imperative_operator(triqs::operators::many_body_operator_generic<scalar_t> const &op, fundamental_operator_set const &fops,
124 hilbert_map_t hmap = hilbert_map_t(), std::vector<sub_hilbert_space> const *sub_spaces_set = nullptr) {
125
126 sub_spaces = sub_spaces_set;
127 hilbert_map = hmap;
128 if ((hilbert_map.size() == 0) != !UseMap) TRIQS_RUNTIME_ERROR << "Internal error";
129
130 auto greater = [&fops](triqs::operators::canonical_ops_t const &op1, triqs::operators::canonical_ops_t const &op2) {
131 if (op1.dagger != op2.dagger) return op2.dagger;
132 return op1.dagger ? (fops[op1.indices] > fops[op2.indices]) : (fops[op1.indices] < fops[op2.indices]);
133 };
134
135 // The goal here is to have a transcription of the many_body_operator in terms
136 // of simple vectors (maybe the code below could be more elegant)
137 for (auto const &term : op) {
138 auto monomial = term.monomial;
139 auto coef = term.coef;
140
141 // Sort monomial according to the order established by fops
142 int n = monomial.size();
143 bool swapped;
144 do {
145 swapped = false;
146 for (int i = 1; i < n; ++i) {
147 if (greater(monomial[i - 1], monomial[i])) {
148 using std::swap;
149 swap(monomial[i - 1], monomial[i]);
150 swapped = true;
151 coef *= scalar_t(-1);
152 }
153 }
154 --n;
155 } while (swapped);
156
157 // Given the environment variable CHECK_ISSUE819 was set by the user
158 // throw an exception if the result of this model was effected by issue 819
159 // https://github.com/TRIQS/triqs/issues/819
160 static const bool check_issue819 = std::getenv("CHECK_ISSUE819");
161 if (check_issue819 && term.coef != coef)
162 TRIQS_RUNTIME_ERROR << "ERROR: The Atom-Diag result of this model is affected by issue 819 (https://github.com/TRIQS/triqs/issues/819).\n"
163 "If you have solved the same model with release 2.2.0, 2.2.1 or 3.0.0 of TRIQS the result was incorrect.";
164
165 std::vector<int> dag, ndag;
166 uint64_t d_mask = 0, dag_mask = 0;
167 for (auto const &canonical_op : monomial) {
168 (canonical_op.dagger ? dag : ndag).push_back(fops[canonical_op.indices]);
169 (canonical_op.dagger ? dag_mask : d_mask) |= (uint64_t(1) << fops[canonical_op.indices]);
170 }
171 auto compute_count_mask = [](std::vector<int> const &d) {
172 uint64_t mask = 0;
173 bool is_on = (d.size() % 2 == 1);
174 for (int i = 0; i < 64; ++i) {
175 if (std::find(begin(d), end(d), i) != end(d))
176 is_on = !is_on;
177 else if (is_on)
178 mask |= (uint64_t(1) << i);
179 }
180 return mask;
181 };
182 uint64_t d_count_mask = compute_count_mask(ndag), dag_count_mask = compute_count_mask(dag);
183 all_terms.push_back(one_term_t{scalar_t(coef), d_mask, dag_mask, d_count_mask, dag_count_mask});
184 }
185 }

196 template <typename w_max> void update(w_max L) {
197 for (auto &M : all_terms) L(M.coeff);

201 // Return a zero state in the target Hilbert space of `st` under this operator.
202 // When UseMap is true and the connection map sends `st`'s subspace to `-1`, returns a default-constructed
203 // (empty) sub_hilbert_space state.
204 template <typename StateType> StateType get_target_st(StateType const &st) const {
205 if constexpr (UseMap) {
206 auto n = hilbert_map[st.get_hilbert().get_index()];
207 if (n == -1) return StateType{};
208 return StateType{(*sub_spaces)[n]};
209 } else {
210 return StateType(st.get_hilbert());
211 }
212 }

214 static bool parity_number_of_bits(uint64_t v) {
215 // http://graphics.stanford.edu/~seander/bithacks.html#CountBitsSetNaive
216 // v ^= v >> 16;
217 // only ok until 16 orbitals ! assert this or put the >> 16
218 v ^= v >> 8;
219 v ^= v >> 4;
220 v ^= v >> 2;
221 v ^= v >> 1;
222 return v & 0x01;
223 }
224
225 // Return x(args...) when scalar_t is callable, x otherwise. Supports parameterized operators
226 // whose coefficients are themselves callable objects.
227 template <typename... Args> static auto apply_if_possible(scalar_t const &x, Args &&...args) -> std::invoke_result_t<scalar_t, Args...> {
228 return x(std::forward<Args>(args)...);
229 }
230 static auto apply_if_possible(scalar_t const &x) -> scalar_t { return x; }

232 public:
251 template <typename StateType, typename... Args> StateType operator()(StateType const &st, Args &&...args) const {
252
253 StateType target_st = get_target_st(st);
254 auto const &hs = st.get_hilbert();
255
256 using amplitude_t = typename StateType::value_type;
257
258 for (int i = 0; i < all_terms.size(); ++i) { // loop over monomials
259 auto M = all_terms[i];
260 foreach (st, [M, &target_st, hs, args...](int j, typename StateType::value_type amplitude) {
261 fock_state_t f2 = hs.get_fock_state(j);
262 if ((f2 & M.d_mask) != M.d_mask) return;
263 f2 &= ~M.d_mask;
264 if (((f2 ^ M.dag_mask) & M.dag_mask) != M.dag_mask) return;
265 fock_state_t f3 = ~(~f2 & ~M.dag_mask);
266 auto sign_is_minus = parity_number_of_bits((f2 & M.d_count_mask) ^ (f3 & M.dag_count_mask));
267 // update state vector in target Hilbert space
268 auto ind = target_st.get_hilbert().get_state_index(f3);
269 target_st(ind) += amplitude * apply_if_possible(M.coeff, args...) * (sign_is_minus ? -amplitude_t(1) : amplitude_t(1));
270 })
271 ; // foreach
272 }
273 return target_st;
274 }
