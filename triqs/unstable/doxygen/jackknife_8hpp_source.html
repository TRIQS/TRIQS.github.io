32 // Represent the jacknifed series.
33 // @tparam V A vector like object, with size() and []
34 //
35 template <typename V> struct jackknifed_t {
36 V const &original_series;
37 using T = std::decay_t<decltype(original_series[0])>;
38 T sum;
39 long n_tot;
40
41 // On each node independently
42 jackknifed_t(V const &bins) : original_series(bins) {
43 // ENSURE(bins.size()>0)
44 n_tot = original_series.size();
45 // compute the sum of B into a new T.
46 sum = original_series[0];
47 for (long i = 1; i < original_series.size(); ++i) sum += original_series[i];
48 }
49
50 // Averages are taken over the whole communicator
51 jackknifed_t(V const &bins, mpi::communicator c) : jackknifed_t(bins) {
52 // first sum on each node (delegated constructor) and then reduce
53 n_tot = mpi::all_reduce(n_tot, c);
54 sum = mpi::all_reduce(sum, c);
55 }
56
57 // average of the series (over the communicator given at construction)
58 auto average() const { return sum / n_tot; }
59
60 // Call. NB : if T is an array, returns an expression template to retard the evaluation
61 auto operator[](long i) const { return (sum - original_series[i]) / (n_tot - 1); }
62 };

66 // implementation of jacknife.
67 // mpi iif the c pointer is not null. If null the computation is on this node only.
68 template <typename F, typename... Jacknifed> auto jackknife_impl(mpi::communicator *c, F &&f, Jacknifed const &...ja) {
69
70 // N is the size of the series, it should be all equal and !=0
71 std::array<long, sizeof...(Jacknifed)> dims{long(ja.original_series.size())...};
72 long N = dims[0];
73 if (N == 0) TRIQS_RUNTIME_ERROR << "No data !";
74 if (not((ja.original_series.size() == N) and ...)) TRIQS_RUNTIME_ERROR << " Jackknife : size mismatch";
75
76 // WARNING NB : In this function, we must be very careful never to store an expression template using f.
77 // f is provided by the user. It will be provided with ja[i] which
78 // might be an expression template itself (hence a temporary).
79 // The user is likely be careless with rvalue references
80 // e.g. f = [](auto const & x, auto const y){ return x/y;};
81 // in such cases, the ja[0] will be dangling unless we use the expression template
82 // containing f immediately.
83 // Cf below, M and Q computation.
84 // Check by ASAN sanitizer on tests
85 auto M = make_regular(f(ja[0]...));
86
87 for (long i = 1; i < N; ++i) {
88 M += (f(ja[i]...) - M) / (i + 1); // cf NB
89 }
90
91 // Now the reduction over the nodes.
92 long Ntot = N;
93 if (c) { // MPI version
94 Ntot = mpi::reduce(N, *c);
95
96 M *= double(N) / Ntot; // remove the 1/N
97 M = mpi::all_reduce(M, *c);
98 }
99
100 // Compute the variance
101 auto Q = make_regular(nda::real(M));
102 Q = 0;
103 for (long i = 0; i < N; ++i) {
104 Q += nda::abs2(f(ja[i]...) - M); // Cf NB
105 }
106
107 // reduce Q
108 if (c) Q = mpi::all_reduce(Q, *c);
109
110 auto f_on_averages = make_regular(f(ja.average()...));
111 auto std_err = make_regular(std::sqrt((Ntot - 1) / double(Ntot)) * sqrt(Q));
112 auto average = make_regular(N * (f_on_averages - M) + M);
113 return std::make_tuple(average, std_err, M, f_on_averages);
114 }

141 template <typename F, typename... A> auto jackknife(F &&f, A const &...a) {
142 static_assert(not std::is_same_v<std::decay_t<F>, mpi::communicator>,
143 "I see that you pass a mpi:::communicator, you probably want to use jackknife_mpi");
144 return details::jackknife_impl(nullptr, std::forward<F>(f), details::jackknifed_t{a}...);
145 }

149 template <typename F, typename... T> auto jackknife(F &&f, accumulator<T> const &...a) {
150 static_assert(not std::is_same_v<std::decay_t<F>, mpi::communicator>,
151 "I see that you pass a mpi:::communicator, you probably want to use jackknife_mpi");
152 return jackknife(std::forward<F>(f), a.linear_bins()...);
153 }
