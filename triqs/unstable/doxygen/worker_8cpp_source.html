42 ATOM_DIAG_WORKER_METHOD(bool, fock_state_filter(fock_state_t s)) {
43 auto c = std::bitset<64>(s).count();
44 return ((c >= n_min) && (c <= n_max));

49 ATOM_DIAG_WORKER_METHOD(void, autopartition(many_body_op_t const &hyb)) {
50 //ATOM_DIAG_WORKER_METHOD(void, autopartition()) {
51
52 fundamental_operator_set const &fops = hdiag->get_fops();
53 many_body_op_t const &h = hdiag->get_h_atomic();
54
55 using imperative_operator_t = imperative_operator<class hilbert_space, scalar_t, false>;
56 imperative_operator_t hamiltonian(h, fops);
57 imperative_operator_t hybridization(hyb, fops);
58 state<class hilbert_space, scalar_t, true> st(hdiag->full_hs);
59
60 using space_partition_t = space_partition<state<class hilbert_space, scalar_t, true>, imperative_operator_t>;
61 // Split the Hilbert space
62 space_partition_t SP(st, hamiltonian, false, hybridization);
63
64 std::vector<typename space_partition_t::matrix_element_map_t> creation_melem(fops.size());
65 std::vector<typename space_partition_t::matrix_element_map_t> annihilation_melem(fops.size());
66 // Merge subspaces
67 for (auto const &o : fops) {
68 auto create = many_body_op_t::make_canonical(true, o.index);
69 auto destroy = many_body_op_t::make_canonical(false, o.index);
70
71 imperative_operator_t op_c_dag(create, fops), op_c(destroy, fops);
72
73 int n = o.linear_index;
74 std::tie(creation_melem[n], annihilation_melem[n]) = SP.merge_subspaces(op_c_dag, op_c, true);
75 }
76
77 // Fill subspaces
78 auto &h_spaces = hdiag->sub_hilbert_spaces;
79
80 h_spaces.reserve(SP.n_subspaces());
81 for (int sp = 0; sp < SP.n_subspaces(); ++sp) h_spaces.emplace_back(sp);
82
83 foreach (SP, [&](fock_state_t s, int spn) {
84 if (fock_state_filter(s)) h_spaces[spn].add_fock_state(s);
85 })
86 ;
87
88 // Discard empty subspaces
89 h_spaces.erase(std::remove_if(h_spaces.begin(), h_spaces.end(), [](sub_hilbert_space const &sp) { return sp.size() == 0; }), h_spaces.end());
90
91 // Correspondence between subspace indices before and after filtering
92 std::map<int, int> remap;
93 for (auto const &sp : h_spaces) remap.emplace(sp.get_index(), remap.size());
94
95 // Fill connections
96 hdiag->creation_connection.resize(fops.size(), h_spaces.size());
97 hdiag->annihilation_connection.resize(fops.size(), h_spaces.size());
98 hdiag->creation_connection.as_array_view() = -1;
99 hdiag->annihilation_connection.as_array_view() = -1;
100
101 for (auto const &o : fops) {
102 int n = o.linear_index;
103 for (auto const &[fs_pair, val] : creation_melem[n]) {
104 auto [i, f] = fs_pair;
105 auto i_remap_it = remap.find(SP.lookup_basis_state(i));
106 if (i_remap_it == remap.end()) continue;
107 auto f_remap_it = remap.find(SP.lookup_basis_state(f));
108 if (f_remap_it == remap.end()) continue;
109 hdiag->creation_connection(n, i_remap_it->second) = f_remap_it->second;
110 }
111 for (auto const &[fs_pair, val] : annihilation_melem[n]) {
112 auto [i, f] = fs_pair;
113 auto i_remap_it = remap.find(SP.lookup_basis_state(i));
114 if (i_remap_it == remap.end()) continue;
115 auto f_remap_it = remap.find(SP.lookup_basis_state(f));
116 if (f_remap_it == remap.end()) continue;
117 hdiag->annihilation_connection(n, i_remap_it->second) = f_remap_it->second;
118 }
119 }
120
121 // Reindex subspaces
122 for (int i = 0; i < h_spaces.size(); ++i) h_spaces[i].set_index(i);
123
124 complete();
125 }

129 ATOM_DIAG_WORKER_METHOD(void, partition_with_qn(std::vector<many_body_op_t> const &qn_vector)) {
130
131 fundamental_operator_set const &fops = hdiag->get_fops();
132 class hilbert_space const &full_hs = hdiag->full_hs;
133
134 // Define a more tolerant comparison between vectors for the quantum numbers
135 auto lt_dbl = [](std::vector<double> const &v1, std::vector<double> const &v2) {
136 for (int i = 0; i < v1.size(); ++i) {
137 if (v1[i] < (v2[i] - 1e-8))
138 return true;
139 else if (v2[i] < (v1[i] - 1e-8))
140 return false;
141 }
142 return false;
143 };
144
145 // Quantum numbers -> Hilbert subspace mapping
146 std::map<std::vector<double>, int, decltype(lt_dbl)> map_qn_n(lt_dbl);
147
148 // The QN as operators: a vector of imperative operators for the quantum numbers
149 std::vector<imperative_operator<class hilbert_space, scalar_t>> qsize;
150 for (auto &qn : qn_vector) qsize.emplace_back(qn, fops);
151
152 // Helper function to get quantum numbers
153 auto get_quantum_numbers = [&qsize](state<class hilbert_space, scalar_t, false> const &s) {
154 std::vector<quantum_number_t> qn;
155 for (auto const &op : qsize) {
156 auto y = dot_product(s, op(s));
157 if (std::abs(std::imag(y)) > 1.e-10) TRIQS_RUNTIME_ERROR << "Quantum number is complex !";
158 qn.push_back(std::real(y));
159 }
160 return qn;
161 };
162
163 // The first part consists in dividing the full Hilbert space
164 // into smaller subspaces using the quantum numbers
165 for (int r = 0; r < full_hs.size(); ++r) {
166
167 // fock_state corresponding to r
168 fock_state_t fs = full_hs.get_fock_state(r);
169
170 // The state we'll act on
171 state<class hilbert_space, scalar_t, false> s(full_hs);
172 s(r) = 1.0;
173
174 // Create the vector with the quantum numbers
175 std::vector<quantum_number_t> qn = get_quantum_numbers(s);
176
177 // If first time we meet these quantum numbers create partial Hilbert space
178 if (map_qn_n.count(qn) == 0) {
179 auto n_blocks = hdiag->sub_hilbert_spaces.size();
180 hdiag->sub_hilbert_spaces.emplace_back(n_blocks); // a new sub_hilbert_space
181 hdiag->quantum_numbers.push_back(qn);
182 map_qn_n[qn] = n_blocks;
183 }
184
185 // Add fock state to partial Hilbert space
186 hdiag->sub_hilbert_spaces[map_qn_n[qn]].add_fock_state(fs);
187 }
188
189 // ---- Now make the creation/annihilation maps -----
190
191 auto creation_map = std::vector<std::vector<int>>(fops.size(), std::vector<int>(hdiag->sub_hilbert_spaces.size(), -1));
192 auto annihilation_map = creation_map;
193
194 // init the mapping tables
195 hdiag->creation_connection.resize(fops.size(), hdiag->sub_hilbert_spaces.size());
196 hdiag->annihilation_connection.resize(fops.size(), hdiag->sub_hilbert_spaces.size());
197 hdiag->creation_connection.as_array_view() = -1;
198 hdiag->annihilation_connection.as_array_view() = -1;
199
200 for (auto const &x : fops) {
201
202 // Get the operators and their index
203 int n = x.linear_index;
204 auto create = many_body_op_t::make_canonical(true, x.index);
205 auto destroy = many_body_op_t::make_canonical(false, x.index);
206
207 // Construct their imperative counterpart
208 imperative_operator<class hilbert_space, scalar_t> op_c_dag(create, fops), op_c(destroy, fops);
209
210 // to avoid declaring every time in the loop below
211 std::vector<quantum_number_t> qn_before, qn_after;
212
213 // Now act on the state with the c, c_dag to see how quantum numbers change
214 for (int r = 0; r < full_hs.size(); ++r) {
215
216 // The state we'll act on and its quantum numbers
217 state<class hilbert_space, scalar_t, false> s(full_hs);
218 s(r) = 1.0;
219 qn_before = get_quantum_numbers(s);
220
221 // Apply creation on state to figure quantum numbers
222 qn_after = get_quantum_numbers(op_c_dag(s));
223
224 // Insert in creation map checking whether it was already there
225 if (!triqs::utility::is_zero(dot_product(op_c_dag(s), op_c_dag(s)))) {
226 auto origin = hdiag->sub_hilbert_spaces[map_qn_n[qn_before]].get_index();
227 auto target = hdiag->sub_hilbert_spaces[map_qn_n[qn_after]].get_index();
228 if (creation_map[n][origin] == -1)
229 creation_map[n][origin] = target;
230 else if (creation_map[n][origin] != target)
231 TRIQS_RUNTIME_ERROR << "partition_with_qn(): internal error while filling creation_connection";
232 hdiag->creation_connection(n, map_qn_n[qn_before]) = map_qn_n[qn_after];
233 }
234
235 // apply annihilation on state to figure quantum numbers
236 qn_after = get_quantum_numbers(op_c(s));
237
238 // insert in annihilation map checking whether it was already there
239 if (!triqs::utility::is_zero(dot_product(op_c(s), op_c(s)))) {
240 auto origin = hdiag->sub_hilbert_spaces[map_qn_n[qn_before]].get_index();
241 auto target = hdiag->sub_hilbert_spaces[map_qn_n[qn_after]].get_index();
242 if (annihilation_map[n][origin] == -1)
243 annihilation_map[n][origin] = target;
244 else if (annihilation_map[n][origin] != target)
245 TRIQS_RUNTIME_ERROR << "partition_with_qn(): internal error while filling annihilation_connection";
246 hdiag->annihilation_connection(n, map_qn_n[qn_before]) = map_qn_n[qn_after];
247 }
248 }
249 }
250 complete();
251 }

255 ATOM_DIAG_WORKER_METHOD(matrix_t, make_op_matrix(many_body_op_t const &op, int from_spn, int to_spn) const) {
256
257 fundamental_operator_set const &fops = hdiag->get_fops();
258 class hilbert_space const &full_hs = hdiag->full_hs;
259 auto const &from_sp = hdiag->sub_hilbert_spaces[from_spn];
260 auto const &to_sp = hdiag->sub_hilbert_spaces[to_spn];
261
262 imperative_operator<class hilbert_space, scalar_t> imp_op(op, fops);
263
264 auto M = matrix_t::zeros({to_sp.size(), from_sp.size()});
265
266 for (int i = 0; i < from_sp.size(); ++i) { // loop on all fock states of the blocks
267 state<class hilbert_space, scalar_t, true> from_s(full_hs);
268 from_s(from_sp.get_fock_state(i)) = 1.0;
269 auto to_s = imp_op(from_s);
270 auto proj_s = project<state<sub_hilbert_space, scalar_t, true>>(to_s, to_sp);
271 foreach (proj_s, [&](int j, scalar_t ampl) { M(j, i) = ampl; })
272 ;
273 }
274
275 return dagger(hdiag->eigensystems[to_spn].unitary_matrix) * M * hdiag->eigensystems[from_spn].unitary_matrix;
276 }

280 ATOM_DIAG_WORKER_METHOD(void, complete()) {
281
282 fundamental_operator_set const &fops = hdiag->get_fops();
283 many_body_op_t const &h = hdiag->get_h_atomic();
284
285 imperative_operator<class hilbert_space, scalar_t, false> hamiltonian(h, fops);
286
287 // Compute energy levels and eigenvectors of the local Hamiltonian
288 int n_subspaces = hdiag->sub_hilbert_spaces.size();
289 hdiag->eigensystems.resize(n_subspaces);
290 hdiag->gs_energy = std::numeric_limits<double>::infinity();
291
292 // Prepare the eigensystem in a temporary map to sort them by energy !
293 std::map<std::pair<double, int>, typename atom_diag<Complex>::eigensystem_t> eign_map;
294 double energy_split = 1.e-10; // to split the eigenvalues, which are numerically very close
295 for (int spn = 0; spn < n_subspaces; ++spn) {
296 auto const &sp = hdiag->sub_hilbert_spaces[spn];
297 typename atom_diag<Complex>::eigensystem_t eigensystem;
298
299 state<sub_hilbert_space, scalar_t, false> i_state(sp);
300 auto h_matrix = matrix_t(sp.size(), sp.size());
301
302 for (int i = 0; i < sp.size(); ++i) {
303 i_state.amplitudes()() = 0;
304 i_state(i) = 1;
305 auto f_state = hamiltonian(i_state);
306 h_matrix(range::all, i) = f_state.amplitudes();
307 }
308
309 auto eig = linalg::eigh(h_matrix);
310 eigensystem.eigenvalues = eig.first;
311 eigensystem.unitary_matrix = eig.second;
312 hdiag->gs_energy = std::min(hdiag->gs_energy, eigensystem.eigenvalues[0]);
313
314 eign_map.insert({{eigensystem.eigenvalues(0) + energy_split * spn, spn}, eigensystem});
315 }
316
317 // Reorder the block along their minimal energy
318 {
319 auto tmp = hdiag->sub_hilbert_spaces;
320 std::map<int, int> remap;
321 int i = 0;
322 for (auto const &x : eign_map) { // in order of min energy !
323 hdiag->eigensystems[i] = x.second;
324 tmp[i] = hdiag->sub_hilbert_spaces[x.first.second];
325 tmp[i].set_index(i);
326 remap[x.first.second] = i;
327 ++i;
328 }
329 std::swap(tmp, hdiag->sub_hilbert_spaces);
330 auto remap_connection = [&](matrix<long> &connection) {
331 auto c2 = connection;
332 for (int n = 0; n < first_dim(connection); ++n)
333 for (int j = 0; j < second_dim(connection); ++j) connection(n, remap[j]) = (c2(n, j) == -1 ? -1 : remap[c2(n, j)]);
334 };
335 remap_connection(hdiag->creation_connection);
336 remap_connection(hdiag->annihilation_connection);
337 } // end reordering
338
339 // Shift the ground state energy of the local Hamiltonian to zero.
340 for (auto &eigensystem : hdiag->eigensystems) eigensystem.eigenvalues() -= hdiag->get_gs_energy();
341
342 for (auto const &x : fops) {
343 // get the operators and their index
344 int n = x.linear_index;
345 // n is guaranteed to be 0, 1, 2, 3, ... by the fundamental_operator_set class ... but it is very weird...
346 // otherwise the push_back below is false.
347 auto create = many_body_op_t::make_canonical(true, x.index);
348 auto destroy = many_body_op_t::make_canonical(false, x.index);
349
350 // Compute the matrices of c, c dagger in the diagonalization base of H_loc
351 // first a lambda, since it is almost the same code for c and cdag
352 auto make_c_mat = [&](int m, matrix<long> const &connection, many_body_op_t const &op) {
353 std::vector<matrix_t> cmat(second_dim(connection));
354 for (int B = 0; B < second_dim(connection); ++B) {
355 auto Bp = connection(m, B);
356 if (Bp == -1) continue;
357 cmat[B] = make_op_matrix(op, B, Bp);
358 }
359 return cmat;
360 };
361
362 // now execute code...
363 hdiag->c_matrices.push_back(make_c_mat(n, hdiag->annihilation_connection, destroy));
364 hdiag->cdag_matrices.push_back(make_c_mat(n, hdiag->creation_connection, create));
365
366 } // end of loop on operators
367 }
