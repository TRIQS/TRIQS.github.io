33 template <int N = 0, nda::MemoryArray A> auto flatten_2d(A const &v) {
34 long nrows = v.extent(N); // # rows of the result, i.e. n-th dim, which is now at 0.
35 long ncols = v.size() / nrows; // # columns of the result. Everything but n-th dim.
36 return reshape(typename A::regular_type{nda::transposed_view<0, N>(v)}, nrows, ncols);
37 }

42 template <int N = 0> void unflatten_2d(nda::MemoryArray auto &&out, nda::MemoryArray auto const &afl) {
43 auto sh_fl = out.shape();
44 std::swap(sh_fl[0], sh_fl[N]);
45 auto data_fl = reshape(afl, sh_fl);
46 out() = nda::transposed_view<0, N>(data_fl);
47 }
