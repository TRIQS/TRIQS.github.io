32 template <typename T, typename Layout> gf<mesh::imfreq, T> make_gf_from_real_gf(gf_const_view<mesh::imfreq, T, Layout> g) {
33 if (!g.mesh().positive_only()) TRIQS_RUNTIME_ERROR << "gf imfreq is not for omega_n >0, real_to_complex does not apply";
34 auto const &dat = g.data();
35 auto sh = dat.shape();
36 int is_boson = (g.mesh().statistic() == Boson);
37 long L = sh[0];
38 sh[0] = 2 * sh[0] - is_boson;
39 array<dcomplex, std::decay_t<decltype(dat)>::rank> new_data(sh);
40 auto ___ = nda::ellipsis{};
41 if (is_boson) new_data(L - 1, ___) = dat(0, ___);
42 int L1 = (is_boson ? L - 1 : L);
43 for (int u = is_boson; u < L; ++u) {
44 new_data(L1 + u, ___) = dat(u, ___);
45 new_data(L - 1 - u, ___) = conj(dat(u, ___));
46 }
47 return {mesh::imfreq{g.mesh().beta(), g.mesh().statistic(), L}, std::move(new_data)};
48 }

51 template <typename G>
52 view_or_type_t<std::decay_t<G>> positive_freq_view(G &&g)
53 requires(is_gf_v<G>)
54 {
55 static_assert(std::is_same<typename std::decay_t<G>::mesh_t, mesh::imfreq>::value, "positive_freq_view only makes senses for imfreq gf");
56 static_assert(std::decay_t<G>::is_view or std::is_lvalue_reference_v<G>, "Cannot construct a positive_freq_view from a temporary gf");
57 if (g.mesh().positive_only()) return g;
58 long L = g.mesh().size();
59 long L1 = (L + 1) / 2; // fermion : L is even. boson, L = 2p+1 --> p+1
60 int is_boson = (g.mesh().statistic() == Boson);
61 return {g.mesh().get_positive_freq(), g.data()(range(L1 - is_boson, L), nda::ellipsis())};
62 }

83 template <typename G>
84 bool is_gf_hermitian(G const &g, double tolerance = 1.e-12)
85 requires(is_gf_v<G> or is_block_gf_v<G>)
86 {
87 if constexpr (is_gf_v<G>) {
88 using target_t = typename G::target_t;
89 using mesh_t = typename std::decay_t<G>::mesh_t;
90 static_assert(std::is_same_v<mesh_t, mesh::imfreq> or std::is_same_v<mesh_t, mesh::imtime> or std::is_same_v<mesh_t, mesh::dlr_imfreq>
91 or std::is_same_v<mesh_t, mesh::dlr_imtime>,
92 "is_gf_hermitian requires an imfreq, imtime, dlr_imfreq or dlr_imtime Green function");
93 static_assert(target_t::rank == 0 or target_t::rank == 2 or target_t::rank == 4,
94 "is_gf_hermitian requires a Green function with a target rank of 0, 2 or 4.");
95
96 if constexpr (std::is_same_v<mesh_t, mesh::imfreq>) { // === gf<imfreq>
97 if (g.mesh().positive_only()) return true;
98 //for (auto w : g.mesh().get_positive_freq()) {
99 for (auto w : g.mesh()) {
100 if constexpr (target_t::rank == 0) { // ------ scalar_valued
101 if (abs(conj(g[-w]) - g[w]) > tolerance) return false;
102 } else if constexpr (target_t::rank == 2) { // matrix_valued
103 if (max_element(abs(dagger(g[-w]) - g[w])) > tolerance) return false;
104 } else { // ---------------------------------- tensor_valued<4>
105 for (auto [i, j, k, l] : g.target_indices())
106 if (abs(conj(g[-w](k, l, i, j)) - g[w](i, j, k, l)) > tolerance) return false;
107 }
108 }
109 return true;
110 } else if constexpr (std::is_same_v<mesh_t, mesh::dlr_imfreq>) { // === gf<dlr_imfreq>
111 // Requires symmetrize=true so that DLR nodes come in (iw, -iw) pairs
112 TRIQS_ASSERT(g.mesh().symmetrize());
113 long N = g.mesh().size();
114 for (long d = 0; d < N / 2; ++d) {
115 long d_neg = N - 1 - d;
116 if constexpr (target_t::rank == 0) { // ------ scalar_valued
117 if (abs(conj(g.data()(d_neg)) - g.data()(d)) > tolerance) return false;
118 } else if constexpr (target_t::rank == 2) { // matrix_valued
119 if (max_element(abs(dagger(g.data()(d_neg, ellipsis())) - g.data()(d, ellipsis()))) > tolerance) return false;
120 } else { // ---------------------------------- tensor_valued<4>
121 for (auto [i, j, k, l] : g.target_indices())
122 if (abs(conj(g.data()(d_neg, k, l, i, j)) - g.data()(d, i, j, k, l)) > tolerance) return false;
123 }
124 }
125 return true;
126 } else if constexpr (std::is_same_v<mesh_t, mesh::dlr_imtime>) { // === gf<dlr_imtime>
127 for (long d = 0; d < g.mesh().size(); ++d) {
128 if constexpr (target_t::rank == 0) { // ------ scalar_valued
129 if (abs(conj(g.data()(d)) - g.data()(d)) > tolerance) return false;
130 } else if constexpr (target_t::rank == 2) { // matrix_valued
131 for (auto [i, j] : g.target_indices())
132 if (abs(conj(g.data()(d, j, i)) - g.data()(d, i, j)) > tolerance) return false;
133 } else { // ---------------------------------- tensor_valued<4>
134 for (auto [i, j, k, l] : g.target_indices())
135 if (abs(conj(g.data()(d, k, l, i, j)) - g.data()(d, i, j, k, l)) > tolerance) return false;
136 }
137 }
138 return true;
139 } else { // === gf<imtime>
140 for (auto t : g.mesh()) {
141 if constexpr (target_t::rank == 0) { // ------ scalar_valued
142 if (abs(conj(g[t]) - g[t]) > tolerance) return false;
143 } else if constexpr (target_t::rank == 2) { // matrix_valued
144 for (auto [i, j] : g.target_indices())
145 if (abs(conj(g[t](j, i)) - g[t](i, j)) > tolerance) return false;
146 } else { // ---------------------------------- tensor_valued<4>
147 for (auto [i, j, k, l] : g.target_indices())
148 if (abs(conj(g[t](k, l, i, j)) - g[t](i, j, k, l)) > tolerance) return false;
149 }
150 }
151 return true;
152 }
153 } else { // Block Green function
154 return std::all_of(g.begin(), g.end(), [&](auto &g_bl) { return is_gf_hermitian<typename G::g_t>(g_bl, tolerance); });
155 }
156 }

171 template <typename G> bool is_gf_real_in_tau(G const &g, double tolerance = 1.e-12) {
172 if constexpr (is_gf_v<G>) {
173 using target_t = typename G::target_t;
174 using mesh_t = typename std::decay_t<G>::mesh_t;
175 static_assert(std::is_same_v<mesh_t, mesh::imfreq>, "is_gf_hermitian requires an imfreq Green function");
176
177 if (g.mesh().positive_only()) return true;
178 for (auto w : g.mesh()) {
179 if constexpr (target_t::rank == 0) { // ---- scalar_valued
180 if (abs(conj(g[-w]) - g[w]) > tolerance) return false;
181 } else {
182 if (max_element(abs(conj(g[-w]) - g[w])) > tolerance) return false;
183 }
184 }
185 return true;
186 } else { // Block Green function
187 return std::all_of(g.begin(), g.end(), [&](auto &g_bl) { return is_gf_real_in_tau<typename G::g_t>(g_bl, tolerance); });
188 }

209 template <typename G>
210 typename G::regular_type make_hermitian(G const &g)
211 requires(is_gf_v<G> or is_block_gf_v<G>)
212 {
213 if constexpr (is_gf_v<G>) {
214 using target_t = typename G::target_t;
215 using mesh_t = typename std::decay_t<G>::mesh_t;
216 static_assert(std::is_same_v<mesh_t, mesh::imfreq> or std::is_same_v<mesh_t, mesh::imtime> or std::is_same_v<mesh_t, mesh::dlr_imfreq>
217 or std::is_same_v<mesh_t, mesh::dlr_imtime>,
218 "make_hermitian requires an imfreq, imtime, dlr_imfreq or dlr_imtime Green function");
219 static_assert(target_t::rank == 0 or target_t::rank == 2 or target_t::rank == 4,
220 "make_hermitian requires a Green function with a target rank of 0, 2 or 4.");
221
222 if constexpr (std::is_same_v<mesh_t, mesh::imfreq>) { // === gf<imfreq>
223 if (g.mesh().positive_only()) return typename G::regular_type{g};
224 auto g_sym = typename G::regular_type{g};
225 for (auto w : g.mesh()) {
226 if constexpr (target_t::rank == 0) // ---- scalar_valued
227 g_sym[w] = 0.5 * (g[w] + conj(g[-w]));
228 else if constexpr (target_t::rank == 2) // matrix_valued
229 g_sym[w] = 0.5 * (g[w] + dagger(g[-w]));
230 else // ---------------------------------- tensor_valued<4>
231 for (auto [i, j, k, l] : g.target_indices()) g_sym[w](i, j, k, l) = 0.5 * (g[w](i, j, k, l) + conj(g[-w](k, l, i, j)));
232 }
233 return g_sym;
234
235 } else if constexpr (std::is_same_v<mesh_t, mesh::dlr_imfreq>) { // === gf<dlr_imfreq>
236 TRIQS_ASSERT(g.mesh().symmetrize());
237 auto g_sym = typename G::regular_type{g};
238 long N = g.mesh().size();
239 for (long d = 0; d < N; ++d) {
240 long d_neg = N - 1 - d;
241 if constexpr (target_t::rank == 0)
242 g_sym.data()(d) = 0.5 * (g.data()(d) + conj(g.data()(d_neg)));
243 else if constexpr (target_t::rank == 2)
244 g_sym.data()(d, ellipsis()) = 0.5 * (g.data()(d, ellipsis()) + dagger(g.data()(d_neg, ellipsis())));
245 else
246 for (auto [i, j, k, l] : g.target_indices())
247 g_sym.data()(d, i, j, k, l) = 0.5 * (g.data()(d, i, j, k, l) + conj(g.data()(d_neg, k, l, i, j)));
248 }
249 return g_sym;
250 } else { // === gf<imtime> or gf<dlr_imtime>
251 auto g_sym = typename G::regular_type{g};
252 for (long d = 0; d < g.mesh().size(); ++d) {
253 if constexpr (target_t::rank == 0) // ---- scalar_valued
254 g_sym.data()(d) = 0.5 * (g.data()(d) + conj(g.data()(d)));
255 else if constexpr (target_t::rank == 2) // matrix_valued
256 for (auto [i, j] : g.target_indices()) g_sym.data()(d, i, j) = 0.5 * (g.data()(d, i, j) + conj(g.data()(d, j, i)));
257 else // ---------------------------------- tensor_valued<4>
258 for (auto [i, j, k, l] : g.target_indices())
259 g_sym.data()(d, i, j, k, l) = 0.5 * (g.data()(d, i, j, k, l) + conj(g.data()(d, k, l, i, j)));
260 }
261 return g_sym;
262 }
263 } else if (is_block_gf_v<G>) { // Block Green function
264 return map_block_gf(make_hermitian<typename G::g_t>, g);
265 }
266 }

299 template <template <typename, typename, typename...> typename G, typename T> auto restricted_view(G<mesh::imfreq, T> const &g, int n_max) {
300 auto iw_mesh = mesh::imfreq{g.mesh().beta(), Fermion, n_max};
301
302 auto const &old_mesh = g.mesh();
303 int idx_min = old_mesh.to_data_index(iw_mesh.first_index());
304 int idx_max = old_mesh.to_data_index(iw_mesh.last_index());
305 auto data_view = g.data()(range(idx_min, idx_max + 1), ellipsis());
306
307 return typename G<mesh::imfreq, T>::const_view_type{iw_mesh, data_view};
308 }
