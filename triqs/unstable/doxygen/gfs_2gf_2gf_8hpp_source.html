315 template <typename RHS>
316 gf &operator=(RHS &&rhs)
317 requires(GreenFunction<RHS>::value and not std::is_same_v<std::decay_t<RHS>, gf>)
318 {
319 _mesh = rhs.mesh();
320 _data.resize(rhs.data_shape());
321 for (auto w : _mesh) (*this)[w] = rhs[w];
322 return *this;
