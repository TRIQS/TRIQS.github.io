99 log_binning(T const &sample, int max_n_bins) : max_n_bins_(max_n_bins) {
100 if (max_n_bins != -1 && max_n_bins < 1) throw std::runtime_error("log_binning: max_n_bins must be -1 (unbounded) or >= 1.");
101
102 if (max_n_bins > 0) {
103 mean_bins_.reserve(max_n_bins);
104 var_bins_.reserve(max_n_bins);
105 // bare bins start at block size 2; the size-1 bin is folded into mean_bins_/var_bins_ directly
106 bare_bins_.reserve(max_n_bins - 1);
107 bare_counts_.reserve(max_n_bins - 1);
108 }
109
110 mean_bins_.emplace_back(zeroed_sample(sample));
111 var_bins_.emplace_back(make_real(zeroed_sample(sample)));
112 if (max_n_bins != 1) {
113 bare_bins_.emplace_back(zeroed_sample(sample));
114 bare_counts_.push_back(0);
115 }
116 }

194 template <typename U> log_binning<T> &operator<<(U const &x) {
195 ++count_;
196
197 // seed shape from the first sample for default-constructed accumulators
198 if (mean_bins_.empty()) {
199 T s = x;
200 mean_bins_.emplace_back(zeroed_sample(s));
201 var_bins_.emplace_back(make_real(zeroed_sample(s)));
202 if (max_n_bins_ != 1) {
203 bare_bins_.emplace_back(zeroed_sample(s));
204 bare_counts_.push_back(0);
205 }
206 }
207
208 T x_m = (x - mean_bins_[0]);
209 var_bins_[0] += abs_square(x_m) * (static_cast<double>(count_ - 1) / count_);
210 mean_bins_[0] += x_m / count_;
211
212 if (max_n_bins_ == 1) return *this;
213
214 if (count_ == (1L << bare_bins_.size()) && (is_unbounded() || n_bins() < max_n_bins_)) {
215 mean_bins_.emplace_back(zeroed_sample(mean_bins_[0]));
216 var_bins_.emplace_back(zeroed_sample(var_bins_[0]));
217
218 // n_bins() has just been incremented; re-check whether another mean bin could still follow
219 if (is_unbounded() || n_bins() < max_n_bins_) {
220 bare_bins_.emplace_back(zeroed_sample(bare_bins_[0]));
221 bare_counts_.push_back(0);
222 }
223 }
224
225 bare_bins_[0] += x;
226 ++bare_counts_[0];
227
228 for (int i = 0; i < bare_bins_.size() && bare_counts_[i] == 2; ++i) {
229 if (i + 1 < bare_bins_.size()) {
230 bare_bins_[i + 1] += bare_bins_[i];
231 ++bare_counts_[i + 1];
232 }
233
234 auto const bc = static_cast<double>(1ul << (i + 1));
235 auto const k = count_ / bc;
236 x_m = (bare_bins_[i] / bc - mean_bins_[i + 1]);
237 var_bins_[i + 1] += abs_square(x_m) * ((k - 1) / k);
238 mean_bins_[i + 1] += x_m / k;
239
240 bare_counts_[i] = 0;
241 bare_bins_[i] = 0;
242 }
243
244 return *this;
245 }

313 [[nodiscard]] auto mpi_all_reduce(mpi::communicator c) const {
314 auto mean_red = mean_bins_;
315 auto var_red = var_bins_;
316 auto nsamples_red = effective_counts();
317
318 // make sure all ranks have the same number of bins
319 auto const nbins = mpi::all_reduce(n_bins(), c, MPI_MAX);
320 mean_red.resize(nbins, zeroed_sample(mean_bins_[0]));
321 var_red.resize(nbins, zeroed_sample(var_bins_[0]));
322 nsamples_red.resize(nbins, 0);
323
324 for (int i = 0; i < nbins; ++i) {
325 auto const ns = nsamples_red[i];
326 mpi::all_reduce_in_place(nsamples_red[i], c);
327
328 // skip the per-rank reweighting (and its 0/0) if no rank has samples at this bin level
329 if (nsamples_red[i] > 0) mean_red[i] *= static_cast<double>(ns) / static_cast<double>(nsamples_red[i]);
330 mpi::all_reduce_in_place(mean_red[i], c);
331
332 if (i < mean_bins_.size()) var_red[i] += ns * abs_square(mean_bins_[i] - mean_red[i]);
333 mpi::all_reduce_in_place(var_red[i], c);
334 }
335
336 return std::make_tuple(mean_red, var_red, nsamples_red);
337 }

382 // Get the overall mean as well as the standard error, integrated autocorrelation time and effective number of
383 // samples for each bin with at least a given minimum number of samples.
384 [[nodiscard]] auto calculate_mean_errors_and_taus(std::vector<value_t> const &mk, std::vector<real_t> const &qk,
385 std::vector<long> const &nsamples, int min_samples) const {
386 // early return for empty vectors
387 if (mk.empty()) return std::make_tuple(value_t{}, std::vector<real_t>{}, std::vector<real_t>{}, std::vector<long>{});
388
389 // only consider bins with at least min_samples effective samples
390 auto const size = std::distance(nsamples.begin(), std::ranges::upper_bound(nsamples, min_samples, std::greater{}));
391 auto effs = std::vector(nsamples.begin(), nsamples.begin() + size);
392 auto errs = std::vector(qk.begin(), qk.begin() + size);
393 auto taus = std::vector<real_t>(size);
394
395 // calculate errors and taus
396 real_t var0 = qk[0] / (static_cast<double>(nsamples[0]) * static_cast<double>(nsamples[0] - 1));
397 for (int i = 0; i < size; ++i) {
398 real_t var = errs[i] / (static_cast<double>(effs[i]) * static_cast<double>(effs[i] - 1));
399 errs[i] = nda::sqrt(var);
400 taus[i] = 0.5 * (var / var0 - 1.0);
401 }
402 return std::make_tuple(mk[0], errs, taus, effs);
403 }
