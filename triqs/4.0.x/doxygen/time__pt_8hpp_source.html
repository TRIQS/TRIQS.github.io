103 inline friend time_pt operator+(time_pt const &tp1, time_pt const &tp2) {
104 bool wrapped = ((Nmax - std::max(tp1.n, tp2.n)) < std::min(tp1.n, tp2.n));
105 if (!wrapped)
106 return {(tp1.n + tp2.n) % Nmax, tp1.beta};
