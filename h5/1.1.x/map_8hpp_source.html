35 template <typename keyT, typename valueT>
36 void h5_write(group f, std::string const &name, std::map<keyT, valueT> const &M) {
37 auto gr = f.create_group(name);
38 write_hdf5_format(gr, M);

40 if constexpr (std::is_same_v<keyT, std::string>) {

53 template <typename keyT, typename valueT>
54 void h5_read(group f, std::string const &name, std::map<keyT, valueT> &M) {
55 auto gr = f.open_group(name);
56 M.clear();
57
58 for (auto const &x : gr.get_all_subgroup_dataset_names()) {
59 valueT val;
60 if constexpr (std::is_same_v<keyT, std::string>) {
61 h5_read(gr, x, val);
62 M.emplace(x, std::move(val));
63 } else {
64 auto element_gr = gr.open_group(x);
65 keyT key;
66 h5_read(element_gr, "key", key);
67 h5_read(element_gr, "val", val);
68 M.emplace(std::move(key), std::move(val));
69 }
70 }
71 }
