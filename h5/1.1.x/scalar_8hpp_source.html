36 template <typename T>
37 void h5_read(group g, std::string const &name, T &x) H5_REQUIRES(std::is_arithmetic_v<T> or is_complex_v<T> or std::is_same_v<T, dcplx_t>) {
38
39 if constexpr (is_complex_v<T>) {
40 // Backward compatibility to read complex stored the old way
41 if (g.has_subgroup(name)) {
42 group gr = g.open_group(name);
43 H5_ASSERT(gr.has_key("r") and gr.has_key("i"));
44 double r = NAN, i = NAN;
45 h5_read(gr, "r", r);
46 h5_read(gr, "i", i);
47 x = std::complex<double>{r, i};
48 return;
49 }
50 }
51
52 auto lt = array_interface::get_h5_lengths_type(g, name);
53
54 if constexpr (is_complex_v<T>) {
55 // Allow reading compound hdf5 dataype into complex
56 if (hdf5_type_equal(lt.ty, hdf5_type<dcplx_t>())) {
57 h5_read(g, name, reinterpret_cast<dcplx_t &>(x));
58 return;
59 }
60 }
61
62 array_interface::read(g, name, array_interface::h5_array_view_from_scalar(x), lt);
63 }
