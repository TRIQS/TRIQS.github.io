57 /*
58 * A handle to the a general HDF5 object
59 *
60 * HDF5 uses a reference counting system, similar to python.
61 * h5::object handles the proper reference couting (similar to pybind11::object e.g.)
62 * using a RAII pattern (hence exception safety).
63 */
64 class object {
65
66 protected:
67 hid_t id = 0; //NOLINT Ok, I want a protected variable ...

74 object(hid_t id = 0) : id(id) {}
75
77 object(object const &x);

80 object(object &&x) noexcept : id(x.id) { x.id = 0; }
81
83 object &operator=(object const &x) {
84 operator=(object(x));
85 return *this;

89 object &operator=(object &&x) noexcept;
90
92 ~object() { close(); }

95 void close();
96
98 operator hid_t() const { return id; }

101 [[nodiscard]] int get_ref_count() const;
102
104 [[nodiscard]] bool is_valid() const;
105 };
