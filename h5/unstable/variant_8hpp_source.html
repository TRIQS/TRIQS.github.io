38 template <typename VT, typename U, typename... T>
39 void h5_read_variant_helper(VT &v, datatype dt, group gr, std::string const &name) {
40 if (hdf5_type_equal(hdf5_type<U>(), dt)) {
41 v = VT{h5_read<U>(gr, name)};
42 return;
43 }
44 if constexpr (sizeof...(T) > 0)
45 h5_read_variant_helper<VT, T...>(v, dt, gr, name);

48 }
