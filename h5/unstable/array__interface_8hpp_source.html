37 // Store HDF5 hyperslab info, as in HDF5 manual
38 // http://davis.lbl.gov/Manuals/HDF5-1.8.7/UG/12_Dataspaces.html
39 struct hyperslab {
40 v_t offset; // offset of the pointer from the start of data
41 v_t stride; // stride in each dimension (in the HDF5 sense). 1 if contiguous. Always >0.
42 v_t count; // length in each dimension
43 v_t block; // block in each dimension
44
45 // Constructor : unique to enforce the proper sizes of array
46 // rank : the array will have rank + is_complex
47 hyperslab(int rank, bool is_complex)
48 : offset(rank + is_complex, 0), stride(rank + is_complex, 1), count(rank + is_complex, 0), block() { // block is often unused
49 if (is_complex) {
50 stride[rank] = 1;
51 count[rank] = 2;
52 }
53 }
54
55 //
56 [[nodiscard]] int rank() const { return count.size(); }
57 };

59 // Stores a view of an array.
60 // scalar are array of rank 0, lengths, strides are empty, rank is 0, start is the scalar
61 struct h5_array_view {
62 datatype ty; // HDF5 type
63 void *start; // start of data. It MUST be a pointer of T* with ty = hdf5_type<T>()
64 v_t L_tot; // lengths of the parent contiguous array
65 hyperslab slab; // hyperslab
66 bool is_complex;
67
68 // Constructor : unique to enforce the proper sizes of array
69 h5_array_view(datatype ty, void *start, int rank, bool is_complex)
70 : ty(std::move(ty)), start(start), L_tot(rank + is_complex), slab(rank, is_complex), is_complex(is_complex) {
71 if (is_complex) L_tot[rank] = 2;
72 }
73
74 [[nodiscard]] int rank() const { return slab.rank(); }
75 };
