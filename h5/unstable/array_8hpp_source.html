35 template <typename T, size_t N>
36 void h5_write(group g, std::string const &name, std::array<T, N> const &a) {
37
38 if constexpr (std::is_same_v<T, std::string>) {
39 auto char_arr = std::array<const char *, N>{};
40 std::transform(cbegin(a), cend(a), begin(char_arr), [](std::string const &s) { return s.c_str(); });
41 h5_write(g, name, char_arr);
42
43 } else if constexpr (std::is_arithmetic_v<
44 T> or is_complex_v<T> or std::is_same_v<T, dcplx_t> or std::is_same_v<T, char *> or std::is_same_v<T, const char *>) {
45
46 h5::array_interface::h5_array_view v{hdf5_type<T>(), (void *)a.data(), 1, is_complex_v<T>};
47 v.slab.count[0] = N;
48 v.slab.stride[0] = 1;
49 v.L_tot[0] = N;
50
51 h5::array_interface::write(g, name, v, true);
52
53 } else { // generic unknown type to hdf5
54 auto g2 = g.create_group(name);
55 h5_write(g2, "shape", std::array<long, 1>{N});
56 for (int i = 0; i < N; ++i) h5_write(g2, std::to_string(i), a[i]);
57 }
58 }

70 template <typename T, size_t N>
71 void h5_read(group g, std::string name, std::array<T, N> &a) {
72
73 if constexpr (std::is_same_v<T, std::string>) {
74 auto char_arr = std::array<char *, N>{};
75 h5_read(g, name, char_arr);
76 std::copy(cbegin(char_arr), cend(char_arr), begin(a));
77 std::for_each(begin(char_arr), end(char_arr), [](char *cb) { free(cb); });
78
79 } else if constexpr (std::is_arithmetic_v<
80 T> or is_complex_v<T> or std::is_same_v<T, dcplx_t> or std::is_same_v<T, char *> or std::is_same_v<T, const char *>) {
81
82 auto lt = array_interface::get_h5_lengths_type(g, name);
83
84 H5_EXPECTS(lt.rank() == 1 + lt.has_complex_attribute);
85 H5_EXPECTS(N == lt.lengths[0]);
86
87 if constexpr (is_complex_v<T>) {
88 // Allow reading compound hdf5 dataype into array<complex>
89 if (hdf5_type_equal(lt.ty, hdf5_type<dcplx_t>())) {
90 h5_read(g, name, reinterpret_cast<std::array<dcplx_t, N> &>(a));
91 return;
92 }
93
94 // Allow to read non-complex data into array<complex>
95 if (!lt.has_complex_attribute) {
96 std::cerr << "WARNING: Mismatching types in h5_read. Expecting a complex " + get_name_of_h5_type(hdf5_type<T>())
97 + " while the array stored in the hdf5 file has type " + get_name_of_h5_type(lt.ty) + "\n";
98 std::array<double, N> tmp{};
99 h5_read(g, name, tmp);
100 std::copy(begin(tmp), end(tmp), begin(a));
101 return;
102 }
103 }
104
105 array_interface::h5_array_view v{hdf5_type<T>(), (void *)(a.data()), 1 /*rank*/, is_complex_v<T>};
106 v.slab.count[0] = N;
107 v.slab.stride[0] = 1;
108 v.L_tot[0] = N;
109
110 array_interface::read(g, name, v, lt);
111
112 } else { // generic unknown type to hdf5
113 auto g2 = g.open_group(name);
114
115 // Check that shapes are compatible
116 auto h5_shape = std::array<long, 1>{};
117 h5_read(g2, "shape", h5_shape);
118 H5_EXPECTS(N == h5_shape[0]);
119
120 // Read using appropriate h5_read implementation
121 for (int i = 0; i < N; ++i) h5_read(g2, std::to_string(i), a[i]);
122 }
123 }
