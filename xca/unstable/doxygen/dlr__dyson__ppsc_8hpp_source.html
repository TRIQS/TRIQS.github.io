57 dyson_it_ppsc(double beta, imtime_ops itops, Ht const &h)
58 : beta(beta),
59 itops_ptr(std::make_shared<imtime_ops>(itops)),
60 r(itops.rank()),
61 norb(std::floating_point<Ht> ? 1 : h.shape(0)),
62 time_order(true),
63 g0(r, norb, norb),
64 g0c(r, norb, norb),
65 sysmat(r * norb, r * norb) {
66
67 // dyson_it_ppsc object contains a shared pointer to the imtime_ops object
68 // itops. This is done to avoid making a copy of itops, which is meant to
69 // handle all imaginary time operations on the given DLR imaginary time
70 // grid.
71
72 g0() = free_gf_ppsc(beta, itops, h); // Free Green's function (right hand side of Dyson equation
73 g0c() = itops_ptr->vals2coefs(g0); // DLR coefficients of free Green's function
74
75 // Get right hand side of Dyson equation
76 if constexpr (std::floating_point<Ht>) { // If h is real scalar, rhs is a vector
77 rhs.resize(r);
78 rhs() = g0;
79 } else { // Otherwise, rhs is given by g0 w/ some indices transposed (for compatibility w/ LAPACK)
80 rhs.resize(norb, r, norb);
81 rhs() = permuted_indices_view<nda::encode<3>({1, 2, 0})>(g0);
82 }

97 // Tsig is type of sig. Tg is type of Green's
98 // function, which is of type Tsig, with scalar type replaced by the common
99 // type of the Hamiltonian's scalar type, and the self-energy's scalar type
100 // (real if both are real, complex otherwise).
101 template <nda::MemoryArray Tsig, nda::MemoryArray Tg = make_common_t<Tsig, Sh, nda::get_value_t<Tsig>>> Tg solve(Tsig const &sig, double eta) {
102
103 int r = itops_ptr->rank(); // DLR rank
104 auto sigc = itops_ptr->vals2coefs(sig); // DLR coefficients of self-energy
105
106 // Obtain Dyson equation system matrix I - G0 * eta - G0 * Sig by only
107 // building a single matrix of convolution for (G0 * eta - G0 * Sig)
108 // the product G0 * Sig is computed using matrix free convolution
109 // avoiding matrix matrix operations (2x speedup and 1/2 memory foot print)
110
111 auto ggs = itops_ptr->convolve(beta, g0c, sigc, time_order);
112 ggs += eta * g0;
113 ggs *= -1.0;
114
115 itops_ptr->convmat_inplace(nda::matrix_view<Sh, nda::C_layout>(sysmat), beta, Fermion, itops_ptr->vals2coefs(ggs), time_order);
116
117 sysmat += nda::eye<double>(r * norb);
118
119 // Factorize system matrix
120 auto ipiv = nda::vector<int>(r * norb);
121 nda::lapack::getrf(sysmat, ipiv);
122
123 // Solve Dyson equation
124 auto g = Tg(sig.shape()); // Declare Green's function
125 g = rhs; // Get right hand side of Dyson equation
126 auto g_rs = nda::matrix_view<nda::get_value_t<Tg>>(nda::reshape(g, norb, r * norb)); // Reshape g to be compatible w/ LAPACK
127 nda::lapack::getrs(sysmat, g_rs, ipiv); // Back solve
128
129 if constexpr (std::floating_point<Ht>) { // If h is scalar, g is scalar-valued
130 return g;
131 } else { // Otherwise, g is matrix-valued, and need to transpose some indices after solve to undo LAPACK formatting
132 return permuted_indices_view<nda::encode<3>({2, 0, 1})>(g);
133 }
134 }
135
149 // Tsig is type of sig. Tg is type of Green's
150 // function, which is of type Tsig, with scalar type replaced by the common
151 // type of the Hamiltonian's scalar type, and the self-energy's scalar type
152 // (real if both are real, complex otherwise).
153 template <nda::MemoryArray Tsig, nda::MemoryArray Tg = make_common_t<Tsig, Sh, nda::get_value_t<Tsig>>>
154 Tg solve_with_op(Tsig const &sig, double eta, nda::matrix_view<dcomplex> op) {
155
156 int r = itops_ptr->rank(); // DLR rank
157 auto sigc = itops_ptr->vals2coefs(sig); // DLR coefficients of self-energy
158
159 auto ggs = itops_ptr->convolve(beta, g0c, sigc, time_order);
160 ggs += eta * g0;
161 for (int k = 0; k < r; ++k) ggs(k, _, _) += nda::matmul(g0(k, _, _), op);
162 ggs *= -1.0;
163
164 itops_ptr->convmat_inplace(nda::matrix_view<Sh, nda::C_layout>(sysmat), beta, Fermion, itops_ptr->vals2coefs(ggs), time_order);
165
166 sysmat += nda::eye<double>(r * norb);
167
168 // Factorize system matrix
169 auto ipiv = nda::vector<int>(r * norb);
170 nda::lapack::getrf(sysmat, ipiv);
171
172 // Solve Dyson equation
173 auto g = Tg(sig.shape()); // Declare Green's function
174 g = rhs; // Get right hand side of Dyson equation
175 auto g_rs = nda::matrix_view<nda::get_value_t<Tg>>(nda::reshape(g, norb, r * norb)); // Reshape g to be compatible w/ LAPACK
176 nda::lapack::getrs(sysmat, g_rs, ipiv); // Back solve
177
178 if constexpr (std::floating_point<Ht>) { // If h is scalar, g is scalar-valued
179 return g;
180 } else { // Otherwise, g is matrix-valued, and need to transpose some indices after solve to undo LAPACK formatting
181 return permuted_indices_view<nda::encode<3>({2, 0, 1})>(g);
182 }
183 }

215 template <typename Ht>
216 requires(std::floating_point<Ht> || nda::MemoryMatrix<Ht>)
217 // If h is scalar, return scalar-valued Green's function; if h is matrix,
218 // return matrix-valued Green's function
219 auto free_gf_ppsc(double beta, imtime_ops const &itops, Ht const &h) {
220
221 int r = itops.rank();
222
223 if constexpr (std::floating_point<Ht>) { // If h is scalar, return scalar-valued Green's function
224 auto g = nda::array<Ht, 1>(r);
225 double mu = h;
226 // Free Green's function for time-ordered Dyson equation
227 for (int i = 0; i < r; i++) { g(i) = -exp(-beta * itops.get_itnodes(i) * (h - mu)); }
228 return g;
229 } else { // Otherwise, return matrix-valued Green's function
230
231 int norb = h.shape(0);
232
233 // Diagonalize Hamiltonian
234 auto [eval, evec] = nda::linalg::eigenelements(h);
235
236 auto E0 = nda::min_element(eval);
237 eval -= E0;
238 auto Z = nda::sum(nda::exp(-beta * eval));
239 auto eta = log(Z) / beta;
240 eval += eta;
241
242 // Get free Green's function
243 auto g = nda::array<nda::get_value_t<Ht>, 3>(r, norb, norb);
244 g = 0;
245 for (int i = 0; i < r; i++) {
246 for (int j = 0; j < norb; j++) { g(i, j, j) = -exp(-beta * rel2abs(itops.get_itnodes(i)) * eval(j)); }
247 g(i, _, _) = matmul(evec, matmul(g(i, _, _), transpose(conj(evec))));
248 }
249
250 return g;
251 }
252 }
