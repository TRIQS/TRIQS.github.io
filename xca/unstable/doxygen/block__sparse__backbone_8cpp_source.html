169void DiagramBlockSparseEvaluator::eval_diagram_block_sparse(Backbone &backbone) {
170
171 int m = backbone.m;
172 nda::vector<int> ind_path(2 * m - 1); // tracks block indices of factors for computing a particular block of the self-energy
173 // nda::vector<int> ind_path(backbone.get_topology(0, 1) - 1);
174 // ^ tracks block indices of factors for computing a particular block of the self-energy before the vertex connected to zero
175 // nda::array<int, 2> ind_path_forks(2 * m - backbone.get_topology(0, 1), q);
176 nda::vector<int> block_dims(2 * m + 1); // tracks the dimensions of the blocks in these factors
177 // nda::vector<int> block_dims(backbone.get_topology(0, 1) + 1);
178 // ^ tracks the dimensions of the blocks in these factors before the vertex connected to zero
179 // nda::array<int, 2> block_dims_forks(2 * m - backbone.get_topology(0, 1), q);
180
181 // loop over all flat indices
182 int f_ix_max = static_cast<int>(backbone.fb_ix_max * backbone.o_ix_max * pow(hyb_poles.size(), m - 1));
183 for (int f_ix = 0; f_ix < f_ix_max; f_ix++) {
184 backbone.set_flat_index(f_ix, hyb_poles); // set directions, pole indices, and orbital indices from a single integer index
185 /* Example of block_dims for m = 2 (OCA): each number is an index of block_dims, and each square represents a block of a matrix
186 3 3 2 2 1 1 0
187 -------- -------- -------- -------- -------- -------- --------
188 4 | F | 3 | G | 3 | F | 2 | G | 2 | F | 1 | G | 1 | F |
189 | | | | | | | | | | | | | |
190 -------- -------- -------- -------- -------- -------- --------
191 */
192 // ind_path can diverge at the vertex connected to vertex 0
193 for (int b_ix = 0; b_ix < Gt.get_num_block_cols(); b_ix++) { // loop over blocks of self-energy
194 for (int p_kap = 0; p_kap < q; p_kap++) { // loop over symmetry sets on the zero vertex
195 bool path_all_nonzero = true;
196 int w = 0, ip = 0; // w loops over the vertices and edges, ip is the current block index
197
198 if (backbone.has_vertex_dag(0)) { // if line is connected to zero is backward
199 ip = Fq.F_dags[p_kap].get_block_index(b_ix);
200 if (ip != -1) {
201 block_dims(0) = Fq.F_dags[p_kap].get_block_size(b_ix, 1);
202 block_dims(1) = Fq.F_dags[p_kap].get_block_size(b_ix, 0);
203 } else {
204 path_all_nonzero = false;
205 }
206 } else {
207 ip = Fq.Fs[p_kap].get_block_index(b_ix);
208 if (ip != -1) {
209 block_dims(0) = Fq.Fs[p_kap].get_block_size(b_ix, 1);
210 block_dims(1) = Fq.Fs[p_kap].get_block_size(b_ix, 0);
211 } else {
212 path_all_nonzero = false;
213 }
214 }
215 // traverse factors in two halves
216 // first half: before vertex connected to zero
217 while (w < backbone.get_topology(0, 1) && path_all_nonzero) { // only continue if we have not hit a zero block
218 if (w != 0) {
219 ip = (backbone.has_vertex_dag(w)) ? Fq.F_dags[Fq.sym_set_labels(backbone.get_orb_ind(w))].get_block_index(ip) :
220 Fq.Fs[Fq.sym_set_labels(backbone.get_orb_ind(w))].get_block_index(ip); // update block index
221 }
222 if (ip == -1 || (w < 2 * m - 1 && Gt.get_zero_block_index(ip) == -1)) { // check if we hit a zero block in F or Gt
223 path_all_nonzero = false;
224 } else { // inner 'if' block unnecessary in first half
225 ind_path(w) = ip; // store the block index for the current vertex/edge, unless we are at the last vertex
226 if (w != backbone.get_topology(0, 1) - 1) { // if so, then orb_ind = -1, because w is the vertex connected to zero
227 block_dims(w + 2) = (backbone.has_vertex_dag(w + 1)) ? Fq.F_dags[Fq.sym_set_labels(backbone.get_orb_ind(w + 1))].get_block_size(ip, 0) :
228 Fq.Fs[Fq.sym_set_labels(backbone.get_orb_ind(w + 1))].get_block_size(ip, 0);
229 }
230 }
231 w += 1;
232 }
233
234 int ip1 = ip;
235 // second half
236 if (path_all_nonzero) {
237 for (int p_mu = 0; p_mu < q; p_mu++) {
238 bool fork_all_nonzero = true;
239 w = backbone.get_topology(0, 1); // reset w to the vertex connected to vertex 0
240 // save block_dims(backbone.get_topology(0, 1) + 1)
241 block_dims(w + 1) = (backbone.has_vertex_dag(w)) ? Fq.F_dags[p_mu].get_block_size(ip1, 0) : Fq.Fs[p_mu].get_block_size(ip1, 0);
242 ip = (backbone.has_vertex_dag(w)) ? Fq.F_dags[p_mu].get_block_index(ip1) :
243 Fq.Fs[p_mu].get_block_index(ip1); // update block index for the vertex connected to zero
244 while (w < 2 * m && fork_all_nonzero) {
245 if (w != backbone.get_topology(0, 1)) {
246 ip = (backbone.has_vertex_dag(w)) ? Fq.F_dags[Fq.sym_set_labels(backbone.get_orb_ind(w))].get_block_index(ip) :
247 Fq.Fs[Fq.sym_set_labels(backbone.get_orb_ind(w))].get_block_index(ip); // update block index
248 }
249 if (ip == -1 || (w < 2 * m - 1 && Gt.get_zero_block_index(ip) == -1)) { // check if we hit a zero block in F or Gt
250 fork_all_nonzero = false;
251 } else {
252 if (w < 2 * m - 1) { // only store the block index if we are not at the last vertex
253 ind_path(w) = ip; // store the block index for the current vertex/edge
254 block_dims(w + 2) = (backbone.has_vertex_dag(w + 1)) ?
255 Fq.F_dags[Fq.sym_set_labels(backbone.get_orb_ind(w + 1))].get_block_size(ip, 0) :
256 Fq.Fs[Fq.sym_set_labels(backbone.get_orb_ind(w + 1))].get_block_size(ip, 0);
257 }
258 }
259 w += 1;
260 }
261 if (fork_all_nonzero) {
262 // evaluate the diagram with these directions, poles, and orbital indices
263 // b_ix is the block index for the first edge
264 eval_backbone_fixed_indices_block_sparse(backbone, b_ix, p_kap, p_mu, ind_path, block_dims);
265 }
266 }
267 }
268 }
269 }
270 backbone.reset_all_inds(); // reset directions, pole indices, and orbital indices for the next iteration
271 }
272 Sigma.set_zero_block_indices(); // set zero_block_indices according to current blocks
273}

275void DiagramBlockSparseEvaluator::eval_backbone_fixed_indices_block_sparse(Backbone &backbone, int b_ix, int p_kap, int p_mu,
276 nda::vector_const_view<int> ind_path,
277 nda::vector_const_view<int> block_dims) {
278
279 int m = backbone.m;
280
281 T(_, range(0, block_dims(1)), range(0, block_dims(1))) = Gt.get_block(ind_path(0));
282 for (int v = 1; v < backbone.get_topology(0, 1); v++) {
283 multiply_vertex_block(backbone, v, ind_path, block_dims);
284 compose_with_edge_block(backbone, v, ind_path, block_dims);
285 }
286
287 multiply_zero_vertex_block(backbone, (not backbone.has_vertex_dag(0)), b_ix, p_kap, p_mu, ind_path, block_dims);
288
289 for (int v = backbone.get_topology(0, 1) + 1; v < 2 * m; v++) {
290 compose_with_edge_block(backbone, v - 1, ind_path, block_dims);
291 multiply_vertex_block(backbone, v, ind_path, block_dims);
292 }
293
294 for (int m_ix = 0; m_ix < m - 1; m_ix++) {
295 int exp = backbone.get_prefactor_Kexp(m_ix);
296 if (exp != 0) {
297 int Ksign = backbone.get_prefactor_Ksign(m_ix);
298 double om = hyb_poles(backbone.get_pole_ind(m_ix));
299 double k = k_it(0, Ksign * om);
300 for (int x = 0; x < exp; x++) T(_, range(0, block_dims(2 * m)), range(0, block_dims(0))) /= k;
301 }
302 }
303 int diag_order_sign = (m % 2 == 0) ? -1 : 1;
304 if (backbone.get_fb(0) == 0) diag_order_sign *= -1;
305 T(_, range(0, block_dims(2 * m)), range(0, block_dims(0))) *= diag_order_sign * backbone.prefactor_sign;
306 // TODO: temporary fix: have backward pass consider sparsity of hybridization function (hyb, hyb_refl) during zero vertex
307 if (nda::max_element(nda::abs(T(_, range(0, block_dims(2 * m)), range(0, block_dims(0))))) > 1e-16)
308 Sigma.add_block(b_ix, T(_, range(0, block_dims(2 * m)), range(0, block_dims(0))));
309}
