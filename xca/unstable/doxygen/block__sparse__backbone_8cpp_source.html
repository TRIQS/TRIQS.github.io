39void DiagramBlockSparseEvaluator::multiply_vertex_block(Backbone &backbone, int v_ix, nda::vector_const_view<int> ind_path,
40 nda::vector_const_view<int> block_dims) {
41 int o_ix = backbone.get_vertex_orb(v_ix); // orbital_index
42 // split backbone orbital index into symmetry set index and orbital index within the symmetry set
43 // i.e. have mapping between backbone orbital index and symmetry set index
44 int q_ix = static_cast<int>(Fq.sym_set_labels(o_ix)); // symmetry set index
45 int qo_ix = static_cast<int>(Fq.sym_set_inds(o_ix)); // index within the symmetry set
46 int l_ix = backbone.get_pole_ind(backbone.get_vertex_hyb_ind(v_ix));
47 int n_col_r = v_ix < backbone.get_topology(0, 1) ? block_dims(1) : block_dims(0);
48 int b_ix = ind_path(v_ix - 1); // block index for the vertex v_ix
49
50 if (backbone.has_vertex_bar(v_ix)) { // F has bar
51 if (backbone.has_vertex_dag(v_ix)) { // F has dagger
52 for (int t = 0; t < r; t++) {
53 T(t, range(0, block_dims(v_ix + 1)), range(0, n_col_r)) =
54 matmul(Fq.F_dag_bars[q_ix].get_block(b_ix)(qo_ix, l_ix, _, _), T(t, range(0, block_dims(v_ix)), range(0, n_col_r)));
55 }
56 } else {
57 for (int t = 0; t < r; t++) {
58 T(t, range(0, block_dims(v_ix + 1)), range(0, n_col_r)) =
59 -matmul(Fq.F_bars_refl[q_ix].get_block(b_ix)(qo_ix, l_ix, _, _), T(t, range(0, block_dims(v_ix)), range(0, n_col_r)));
60 }
61 }
62 } else {
63 if (backbone.has_vertex_dag(v_ix)) { // F has dagger
64 for (int t = 0; t < r; t++) {
65 T(t, range(0, block_dims(v_ix + 1)), range(0, n_col_r)) =
66 matmul(Fq.F_dags[q_ix].get_block(b_ix)(qo_ix, _, _), T(t, range(0, block_dims(v_ix)), range(0, n_col_r)));

68 } else {
69 for (int t = 0; t < r; t++) {
70 T(t, range(0, block_dims(v_ix + 1)), range(0, n_col_r)) =

73 }
74 }

78 double pole = hyb_poles(l_ix);

160 // Multiply by prefactor

162 int exp = backbone.get_prefactor_Kexp(m_ix);

182void DiagramBlockSparseEvaluator::eval_diagram_block_sparse(Backbone &backbone) {
183
184 int m = backbone.m;
185 nda::vector<int> ind_path(2 * m - 1); // tracks block indices of factors for computing a particular block of the self-energy
186 nda::vector<int> block_dims(2 * m + 1); // tracks the dimensions of the blocks in these factors
187
188 // loop over all flat indices
189 int f_ix_max = static_cast<int>(backbone.fb_ix_max * backbone.o_ix_max * pow(hyb_poles.size(), m - 1));
190 for (int f_ix = 0; f_ix < f_ix_max; f_ix++) {
191 backbone.set_flat_index(f_ix, hyb_poles); // set directions, pole indices, and orbital indices from a single integer index
192 /* Example of block_dims for m = 2 (OCA): each number is an index of block_dims, and each square represents a block of a matrix
193 3 3 2 2 1 1 0
194 -------- -------- -------- -------- -------- -------- --------
195 4 | F | 3 | G | 3 | F | 2 | G | 2 | F | 1 | G | 1 | F |
196 | | | | | | | | | | | | | |
197 -------- -------- -------- -------- -------- -------- --------
198 */
199 // ind_path can diverge at the vertex connected to vertex 0
200 for (int b_ix = 0; b_ix < Gt.get_num_block_cols(); b_ix++) { // loop over blocks of self-energy
201 for (int p_kap = 0; p_kap < q; p_kap++) { // loop over symmetry sets on the zero vertex
202 bool path_all_nonzero = true;
203 int w = 0, ip = 0; // w loops over the vertices and edges, ip is the current block index
204
205 if (backbone.has_vertex_dag(0)) { // if line is connected to zero is backward
206 ip = Fq.F_dags[p_kap].get_block_index(b_ix);
207 if (ip != -1) {
208 block_dims(0) = Fq.F_dags[p_kap].get_block_size(b_ix, 1);
209 block_dims(1) = Fq.F_dags[p_kap].get_block_size(b_ix, 0);
210 } else {
211 path_all_nonzero = false;
212 }
213 } else {
214 ip = Fq.Fs[p_kap].get_block_index(b_ix);
215 if (ip != -1) {
216 block_dims(0) = Fq.Fs[p_kap].get_block_size(b_ix, 1);
217 block_dims(1) = Fq.Fs[p_kap].get_block_size(b_ix, 0);
218 } else {
219 path_all_nonzero = false;
220 }
221 }
222 // traverse factors in two halves
223 // first half: before vertex connected to zero
224 while (w < backbone.get_topology(0, 1) && path_all_nonzero) { // only continue if we have not hit a zero block
225 if (w != 0) {
226 ip = (backbone.has_vertex_dag(w)) ? Fq.F_dags[Fq.sym_set_labels(backbone.get_orb_ind(w))].get_block_index(ip) :
227 Fq.Fs[Fq.sym_set_labels(backbone.get_orb_ind(w))].get_block_index(ip); // update block index
228 }
229 if (ip == -1 || (w < 2 * m - 1 && Gt.get_zero_block_index(ip) == -1)) { // check if we hit a zero block in F or Gt
230 path_all_nonzero = false;
231 } else { // inner 'if' block unnecessary in first half
232 ind_path(w) = ip; // store the block index for the current vertex/edge, unless we are at the last vertex
233 if (w != backbone.get_topology(0, 1) - 1) { // if so, then orb_ind = -1, because w is the vertex connected to zero
234 block_dims(w + 2) = (backbone.has_vertex_dag(w + 1)) ? Fq.F_dags[Fq.sym_set_labels(backbone.get_orb_ind(w + 1))].get_block_size(ip, 0) :
235 Fq.Fs[Fq.sym_set_labels(backbone.get_orb_ind(w + 1))].get_block_size(ip, 0);
236 }
237 }
238 w += 1;
239 }
240
241 int ip1 = ip;
242 // second half
243 if (path_all_nonzero) {
244 for (int p_mu = 0; p_mu < q; p_mu++) {
245 bool fork_all_nonzero = true;
246 w = backbone.get_topology(0, 1); // reset w to the vertex connected to vertex 0
247 // save block_dims(backbone.get_topology(0, 1) + 1)
248 block_dims(w + 1) = (backbone.has_vertex_dag(w)) ? Fq.F_dags[p_mu].get_block_size(ip1, 0) : Fq.Fs[p_mu].get_block_size(ip1, 0);
249 ip = (backbone.has_vertex_dag(w)) ? Fq.F_dags[p_mu].get_block_index(ip1) :
250 Fq.Fs[p_mu].get_block_index(ip1); // update block index for the vertex connected to zero
251 while (w < 2 * m && fork_all_nonzero) {
252 if (w != backbone.get_topology(0, 1)) {
253 ip = (backbone.has_vertex_dag(w)) ? Fq.F_dags[Fq.sym_set_labels(backbone.get_orb_ind(w))].get_block_index(ip) :
254 Fq.Fs[Fq.sym_set_labels(backbone.get_orb_ind(w))].get_block_index(ip); // update block index
255 }
256 if (ip == -1 || (w < 2 * m - 1 && Gt.get_zero_block_index(ip) == -1)) { // check if we hit a zero block in F or Gt
257 fork_all_nonzero = false;
258 } else {
259 if (w < 2 * m - 1) { // only store the block index if we are not at the last vertex
260 ind_path(w) = ip; // store the block index for the current vertex/edge
261 block_dims(w + 2) = (backbone.has_vertex_dag(w + 1)) ?
262 Fq.F_dags[Fq.sym_set_labels(backbone.get_orb_ind(w + 1))].get_block_size(ip, 0) :
263 Fq.Fs[Fq.sym_set_labels(backbone.get_orb_ind(w + 1))].get_block_size(ip, 0);
264 }
265 }
266 w += 1;
267 }
268 if (fork_all_nonzero) {
269 // evaluate the diagram with these directions, poles, and orbital indices
270 // b_ix is the block index for the first edge
271 eval_backbone_fixed_indices_block_sparse(backbone, b_ix, p_kap, p_mu, ind_path, block_dims);
272 }
273 }
274 }
275 }
276 }
277 backbone.reset_all_inds(); // reset directions, pole indices, and orbital indices for the next iteration
278 }
279 Sigma.set_zero_block_indices(); // set zero_block_indices according to current blocks
280}

282void DiagramBlockSparseEvaluator::eval_backbone_fixed_indices_block_sparse(Backbone &backbone, int b_ix, int p_kap, int p_mu,
283 nda::vector_const_view<int> ind_path,
284 nda::vector_const_view<int> block_dims) {
285
286 int m = backbone.m;
287
288 T(_, range(0, block_dims(1)), range(0, block_dims(1))) = Gt.get_block(ind_path(0));
289 for (int v = 1; v < backbone.get_topology(0, 1); v++) {
290 multiply_vertex_block(backbone, v, ind_path, block_dims);
291 compose_with_edge_block(backbone, v, ind_path, block_dims);
292 }
293
294 multiply_zero_vertex_block(backbone, (not backbone.has_vertex_dag(0)), b_ix, p_kap, p_mu, ind_path, block_dims);
295
296 for (int v = backbone.get_topology(0, 1) + 1; v < 2 * m; v++) {
297 compose_with_edge_block(backbone, v - 1, ind_path, block_dims);
298 multiply_vertex_block(backbone, v, ind_path, block_dims);
299 }
300
301 multiply_prefactor(backbone, block_dims);
302 int diag_order_sign = (m % 2 == 0) ? -1 : 1;
303 if (backbone.get_fb(0) == 0) diag_order_sign *= -1;
304 T(_, range(0, block_dims(2 * m)), range(0, block_dims(0))) *= diag_order_sign * backbone.prefactor_sign;
305 // TODO: temporary fix: have backward pass consider sparsity of hybridization function (hyb, hyb_refl) during zero vertex
306 if (nda::max_element(nda::abs(T(_, range(0, block_dims(2 * m)), range(0, block_dims(0))))) > 1e-16)
307 Sigma.add_block(b_ix, T(_, range(0, block_dims(2 * m)), range(0, block_dims(0))));
308}

410void CorrelatorDiagramBlockSparseEvaluator::eval_backbone_fixed_indices_block_sparse(Backbone &backbone, int b_ix, int p_kap, int p_mu,
411 nda::vector_const_view<int> ind_path,
412 nda::vector_const_view<int> block_dims) {
413
414 int m = backbone.m;
415
416 // evaluate the first sequence of backbone products and convolutions from tau_1 to tau, proceeding right to left, not inculding the creation and
417 // annihilation matrices at the end points. the result is an N x N matrix-valued function of tau.
418 T(_, range(0, block_dims(1)), range(0, block_dims(1))) = Gt.get_block(ind_path(0));
419 for (int v = 1; v < backbone.get_topology(0, 1); ++v) {
420 multiply_vertex_block(backbone, v, ind_path, block_dims);
421 compose_with_edge_block(backbone, v, ind_path, block_dims);
422 }
423
424 // evaluate the second sequence of backbone products and convolutions from tau to beta, using a change of variables to perform convolutions. the
425 // result is another N x N matrix-valued function of tau.
426 U(_, range(0, block_dims(backbone.get_topology(0, 1) + 1)), range(0, block_dims(backbone.get_topology(0, 1) + 1))) =
427 Gt.get_block(ind_path(backbone.get_topology(0, 1)));
428 int be = 0;
429 for (int i = 0; i < m - 1; ++i) {
430 be = backbone.get_edge(backbone.get_topology(0, 1), i);
431 if (be != 0) {
432 for (int t = 0; t < r; t++) {
433 U(t, range(0, block_dims(backbone.get_topology(0, 1) + 1)), range(0, block_dims(backbone.get_topology(0, 1) + 1))) =
434 k_it(dlr_it(t), be * hyb_poles(backbone.get_pole_ind(i)))
435 * U(t, range(0, block_dims(backbone.get_topology(0, 1) + 1)), range(0, block_dims(backbone.get_topology(0, 1) + 1)));
436 }
437 }
438 }
439 for (int v = backbone.get_topology(0, 1) + 1; v < 2 * m - 1; ++v) {
440 multiply_vertex_corr_left_block(backbone, v, ind_path, block_dims);
441 compose_with_edge_corr_left_block(backbone, v, ind_path, block_dims);
442 }
443 // multiply by the last vertex
444 multiply_vertex_corr_left_block(backbone, 2 * m - 1, ind_path, block_dims);
445 // convolve with last edge
446 GKt(_, range(0, block_dims(2 * m)), range(0, block_dims(2 * m))) = Gt.get_block(ind_path(2 * m - 1));
447 int bv = backbone.get_vertex_Ksign(2 * m - 1); // sign on K
448 int l_ix = backbone.get_pole_ind(backbone.get_vertex_hyb_ind(2 * m - 1));
449 double pole = hyb_poles(l_ix);
450 if (bv != 0) {
451 for (int t = 0; t < r; t++) {
452 GKt(t, range(0, block_dims(2 * m)), range(0, block_dims(2 * m))) =
453 k_it(dlr_it(t), bv * pole) * GKt(t, range(0, block_dims(2 * m)), range(0, block_dims(2 * m)));
454 }
455 }
456 U(_, range(0, block_dims(2 * m)), range(0, block_dims(backbone.get_topology(0, 1) + 1))) =
457 itops.convolve(beta, itops.vals2coefs(GKt(_, range(0, block_dims(2 * m)), range(0, block_dims(2 * m)))),
458 itops.vals2coefs(U(_, range(0, block_dims(2 * m)), range(0, block_dims(backbone.get_topology(0, 1) + 1)))), TIME_ORDERED);
459 U = itops.reflect(U);
460
461 multiply_prefactor(backbone, block_dims);
462 int diag_order_sign = (m % 2 == 0) ? -1 : 1;
463 if (backbone.get_fb(0) == 0) diag_order_sign *= -1;
464 T(_, range(0, block_dims(backbone.get_topology(0, 1))), range(0, block_dims(1))) *= diag_order_sign * backbone.prefactor_sign;
465}
