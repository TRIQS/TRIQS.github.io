172 void DenseDiagramEvaluator::multiply_left_vertex_and_right_zero_vertex(nda::array_view<dcomplex, 3> T_buf, Backbone &backbone, int vct0) {
173
174 bool is_forward = backbone.has_vertex_dag(vct0);
175 nda::array_const_view<dcomplex, 3> hyb_too = is_forward ? hyb.values : hyb.values_reflect;
176
177 // Save compute by precomputing Tkaps = T_buf * F_kap for all kappa, since this is needed for each mu
178 // at the cost of storing an n x r x N x N array (Tkaps) instead of an r x N x N array
179
180 for (int kap = 0; kap < n; kap++) {
181 nda::array_const_view<dcomplex, 2> F_kap = Fset.get_operator(backbone, 0, kap);
182 for (int t = 0; t < r; t++) Tkaps(kap, t, _, _) = matmul(T_buf(t, _, _), F_kap);
183 }
184
185 T_buf = 0;
186
187 for (int mu = 0; mu < n; mu++) {
188 Tmu = 0;
189 for (int kap = 0; kap < n; kap++) {
190 nda::array_const_view<dcomplex, 1> hyb_t = hyb_too(_, mu, kap);
191
192 // set the orbital indices of the vertex connected to zero
193 // and compute the fermionic permutation parity of the resulting diagram.
194
195 backbone.set_orb_inds_of_0_and_vct0(kap, mu);
196 int fermionic_parity = backbone.get_parity();
197
198 for (int t = 0; t < r; t++) Tmu(t, _, _) += fermionic_parity * hyb_t(t) * Tkaps(kap, t, _, _);
199 }
200 nda::array_const_view<dcomplex, 2> F_mu = Fset.get_operator(backbone, vct0, mu);
201 for (int t = 0; t < r; t++) T_buf(t, _, _) += matmul(F_mu, Tmu(t, _, _));
202 }
203 }

223 void DenseDiagramEvaluator::eval_self_energy_fixed_indices(nda::array_const_view<dcomplex, 3> Gt, Backbone &backbone, int f_ix) {
224 int m = backbone.m;
225 int vct0 = backbone.get_topology(0, 1); // Vertex Connected To zero
226
227 backbone.set_flat_index(f_ix, hyb.poles); // set directions, pole indices, and orbital indices from a single integer index
228
229 // 1. Starting from tau_1, proceed right to left, performing multiplications at vertices and convolutions at edges, until reaching the vertex
230 // containing the undecomposed hybridization line Delta_{mu kappa}.
231 T = Gt;
232 // T is initialized to Gt, which is always the function at the rightmost edge
233 for (int v = 1; v < vct0; v++) { // loop from the first vertex to before the special vertex
234 multiply_left_vertex(T, backbone, v);
235 integrate_left_edge(T, Gt, backbone, v);
236 }
237
238 // 2. For each kappa, multiply by F_kappa(^dag). Then for each mu, kappa, multiply by Delta_{mu kappa}, and sum over kappa. Finally for each mu,
239 // multiply F_mu[^dag] and sum over mu.
240 multiply_left_vertex_and_right_zero_vertex(T, backbone, vct0);
241
242 // 3. Continue right to left until the final vertex multiplication is complete.
243 for (int v = vct0 + 1; v < 2 * m; v++) { // loop from the special vertex to the last vertex
244 integrate_left_edge(T, Gt, backbone, v - 1);
245 multiply_left_vertex(T, backbone, v);
246 }
247
248 multiply_prefactor(T, backbone);
249 int diag_order_sign = (m % 2 == 0) ? -1 : 1;
250 if (backbone.get_fb(0) == 0) diag_order_sign *= -1; // if the first hybridization line is backward, there is an additional sign change
251 T *= diag_order_sign * backbone.prefactor_sign;
252 Sigma += T;
253
254 backbone.reset_all_inds(); // reset directions, pole indices, and orbital indices for the next iteration
255 }

289 nda::array<dcomplex, 3> DenseDiagramEvaluator::eval_correlator(nda::array_const_view<dcomplex, 3> Gt, CorrelatorBackbone &backbone,
290 nda::array<dcomplex, 3> mu_ops, nda::array<dcomplex, 3> kap_ops, int f_ix, bool is_fermionic) {
291
292 int m = backbone.m;
293
294 backbone.set_flat_index(f_ix, hyb.poles); // set directions, pole indices, and orbital indices from a single integer index
295
296 nda::array<dcomplex, 3> correlator = nda::zeros<dcomplex>(r, mu_ops.extent(0), kap_ops.extent(0));
297
298 // evaluate the first sequence of backbone products and convolutions from tau_1 to tau, proceeding right to left, not inculding the creation and
299 // annihilation matrices at the end points. the result is an N x N matrix-valued function of tau.
300 T = Gt;
301 // T is initialized to Gt, which is always the function at the rightmost edge
302 for (int v = 1; v < backbone.get_topology(0, 1); ++v) { // loop from the first vertex to before the special vertex
303 multiply_left_vertex(T, backbone, v);
304 integrate_left_edge(T, Gt, backbone, v);
305 }
306
307 // evaluate the second sequence of backbone products and convolutions from tau to beta, using a change of variables to perform convolutions. the
308 // result is another N x N matrix-valued function of tau.
309 U = Gt;
310
311 for (int v = 2 * m - 1; v > backbone.get_topology(0, 1); v--) {
312 multiply_right_vertex(U, backbone, v);
313 integrate_right_edge(U, Gt, backbone, v - 1);
314 }
315
316 U = itops.reflect(U);
317
318 multiply_prefactor(T, backbone);
319 int diag_order_sign = 1; // (m % 2 == 1) ? -1 : 1;
320 T *= diag_order_sign * backbone.prefactor_sign;
321
322 nda::array<dcomplex, 3> Tmuop = nda::zeros<dcomplex>(r, Gt.extent(1), Gt.extent(1));
323
324 for (int mu = 0; mu < mu_ops.extent(0); ++mu) {
325 for (int t = 0; t < r; ++t) Tmuop(t, _, _) = matmul(U(t, _, _), matmul(mu_ops(mu, _, _), T(t, _, _)));
326 for (int kap = 0; kap < kap_ops.extent(0); ++kap) {
327 for (int t = 0; t < r; ++t) correlator(t, mu, kap) += trace(matmul(Tmuop(t, _, _), kap_ops(kap, _, _)));
328 }
329 }
330
331 // use special orbital index values to indicate whether the vertex connected to zero
332 // is fermionic or bosonic, since this affects the fermionic parity calculation.
333 // -2 for fermionic, -3 for bosonic
334
335 int orb_idx_flag = is_fermionic ? -2 : -3;
336 backbone.set_orb_inds_of_0_and_vct0(orb_idx_flag, orb_idx_flag);
337 int fermionic_sign = backbone.get_parity();
338 correlator *= fermionic_sign;
339
340 backbone.reset_all_inds(); // reset directions, pole indices, and orbital indices for the next iteration
341
342 return correlator;
343 }

400 nda::array<dcomplex, 3> DenseDiagramEvaluator::compute_one_time_correlator(gf_vt G_ppsc,
401 std::vector<triqs::operators::many_body_operator_real> const & ops_tau,
402 std::vector<triqs::operators::many_body_operator_real> const & ops_0,
403 triqs::atom_diag::atom_diag<isComplex> const &ad,
404 nda::array_const_view<int, 2> topology, nda::array_const_view<int, 1> f_ix_vec){
405
406 CorrelatorBackbone backbone(topology, n);
407
408 // Check Hilbert space dimension.
409 assert( N == ad.get_full_hilbert_space_dim() );
410 assert( ad.n_subspaces() == 1 );
411
412 auto U = ad.get_unitary_matrix(0);
413
414 nda::array<dcomplex, 3> mu_ops = nda::zeros<dcomplex>(ops_tau.size(), N, N);
415 nda::array<dcomplex, 3> kap_ops = nda::zeros<dcomplex>(ops_0.size(), N, N);
416
417 for (auto [i, op] : itertools::enumerate(ops_tau)) {
418 mu_ops(i, _, _) = U * ad.get_op_mat(op).block_mat[0] * nda::conj(nda::transpose(U));
419 }
420
421 for (auto [i, op] : itertools::enumerate(ops_0)) {
422 kap_ops(i, _, _) = U * ad.get_op_mat(op).block_mat[0] * nda::conj(nda::transpose(U));
423 }
424
425 nda::array<dcomplex, 3> correlator = nda::zeros<dcomplex>(r, mu_ops.extent(0), kap_ops.extent(0));
426
427 // Figure out whether ops_tau and ops_0 are fermionic or bosonic.
428 // Require that all operators have the same statistics
429
430 nda::array<bool, 2> is_fermionic_ops(ops_tau.size(), ops_0.size());
431
432 for (auto [i, op_tau] : itertools::enumerate(ops_tau)) {
433 for (auto [j, op_0] : itertools::enumerate(ops_0)) {
434
435 // test bosonic property: [A, B] = 0
436 is_fermionic_ops(i, j) = !(op_tau * op_0 - op_0 * op_tau).is_zero();
437
438 /*
439 std::cout << "i, j = " << i << ", " << j
440 << ", op_tau = " << op_tau << ", op_0 = " << op_0
441 << ", is_fermionic_ops(i, j) = " << is_fermionic_ops(i, j) << "\n";
442 */
443 }
