34void DenseDiagramEvaluator::multiply_vertex(Backbone &backbone, int v_ix) {
35 int o_ix = backbone.get_vertex_orb(v_ix); // orbital index
36 int l_ix = backbone.get_pole_ind(backbone.get_vertex_hyb_ind(v_ix));
37 // backbone.get_vertex_hyb_ind(v_ix) = i, where i is the # of primes on l
38 // l_ix = value of l with i primes
39
40 if (backbone.has_vertex_bar(v_ix)) { // F has bar
41 if (backbone.has_vertex_dag(v_ix)) { // F has dagger
42 for (int t = 0; t < r; t++) T(t, _, _) = matmul(Fset.F_dag_bars(o_ix, l_ix, _, _), T(t, _, _));
43 } else {
44 for (int t = 0; t < r; t++) T(t, _, _) = -matmul(Fset.F_bars_refl(o_ix, l_ix, _, _), T(t, _, _));
45 }
46 } else {
47 if (backbone.has_vertex_dag(v_ix)) { // F has dagger
48 for (int t = 0; t < r; t++) T(t, _, _) = matmul(Fset.F_dags(o_ix, _, _), T(t, _, _));
49 } else {
50 for (int t = 0; t < r; t++) T(t, _, _) = matmul(Fset.Fs(o_ix, _, _), T(t, _, _));
51 }
52 }
53
54 // K factor
55 int bv = backbone.get_vertex_Ksign(v_ix); // sign on K
56 double pole = hyb_poles(l_ix);
57 if (backbone.get_vertex_direction(v_ix) == 0) pole = -pole; // MODIFIED LOGIC -- HYB_POLES->(-HYB_POLES) FOR BACKWARD LINES
58 if (bv != 0) {
59 for (int t = 0; t < r; t++) { T(t, _, _) = k_it(dlr_it(t), bv * pole) * T(t, _, _); }
60 }
61}

63void DenseDiagramEvaluator::compose_with_edge(Backbone &backbone, int e_ix) {
64 GKt = Gt;
65 int m = backbone.m;
66 for (int x = 0; x < m - 1; x++) {
67 int be = backbone.get_edge(e_ix, x); // sign on K
68 double pole = hyb_poles(backbone.get_pole_ind(x));
69 if (backbone.get_fb(x + 1) == 0) pole = -pole; // MODIFIED LOGIC -- HYB_POLES->(-HYB_POLES) FOR BACKWARD LINES
70 if (be != 0) {
71 for (int t = 0; t < r; t++) { GKt(t, _, _) = k_it(dlr_it(t), be * pole) * GKt(t, _, _); }
72 }
73 }
74 T = itops.convolve(beta, itops.vals2coefs(GKt), itops.vals2coefs(T), TIME_ORDERED);

77void DenseDiagramEvaluator::multiply_prefactor(Backbone &backbone) {
78 int m = backbone.m;
79 // Multiply by prefactor
80 for (int m_ix = 0; m_ix < m - 1; m_ix++) { // loop over hybridization indices
81 int exp = backbone.get_prefactor_Kexp(m_ix); // exponent on K for this hybridization index
82 if (exp != 0) {
83 int Ksign = backbone.get_prefactor_Ksign(m_ix); // sign on K for this hybridization index
84 double om = hyb_poles(backbone.get_pole_ind(m_ix)); // DLR frequency for this value of this hybridization index
85 if (backbone.get_fb(m_ix + 1) == 0) om = -om; // MODIFIED LOGIC -- HYB_POLES->(-HYB_POLES) FOR BACKWARD LINES
86 for (int q = 0; q < exp; q++) T /= k_it(0, Ksign * om);
87 }

131void DenseDiagramEvaluator::eval_self_energy_fixed_indices(Backbone &backbone) {
132 int m = backbone.m;
133
134 // 1. Starting from tau_1, proceed right to left, performing multiplications at vertices and convolutions at edges, until reaching the vertex
135 // containing the undecomposed hybridization line Delta_{mu kappa}.
136 T = Gt; // T stores the result moving left to right
137 // T is initialized to Gt, which is always the function at the rightmost edge
138 for (int v = 1; v < backbone.get_topology(0, 1); v++) { // loop from the first vertex to before the special vertex
139 multiply_vertex(backbone, v);
140 compose_with_edge(backbone, v);
141 }
142
143 // 2. For each kappa, multiply by F_kappa(^dag). Then for each mu, kappa, multiply by Delta_{mu kappa}, and sum over kappa. Finally for each mu,
144 // multiply F_mu[^dag] and sum over mu.
145 multiply_zero_vertex(backbone, (not backbone.has_vertex_dag(0)));
146
147 // 3. Continue right to left until the final vertex multiplication is complete.
148 for (int v = backbone.get_topology(0, 1) + 1; v < 2 * m; v++) { // loop from the special vertex to the last vertex
149 compose_with_edge(backbone, v - 1);
150 multiply_vertex(backbone, v);
151 }
152
153 multiply_prefactor(backbone);
154 int diag_order_sign = (m % 2 == 0) ? -1 : 1;
155 if (backbone.get_fb(0) == 0) diag_order_sign *= -1; // if the first hybridization line is backward, there is an additional sign change
156 T *= diag_order_sign * backbone.prefactor_sign;
157 Sigma += T;
158}

206nda::array<dcomplex, 3> DenseDiagramEvaluator::eval_correlator(CorrelatorBackbone &backbone, nda::array<dcomplex, 3> mu_ops,
207 nda::array<dcomplex, 3> kap_ops) {
208 int m = backbone.m;
209 // loop over all flat indices
210 int f_ix_max = static_cast<int>(backbone.fb_ix_max * backbone.o_ix_max * pow(hyb_poles.size(), m - 1));
211 nda::array<dcomplex, 3> correlator = nda::zeros<dcomplex>(r, mu_ops.extent(0), kap_ops.extent(0));
212 nda::array<dcomplex, 3> Tmuop = nda::zeros<dcomplex>(r, Gt.extent(1), Gt.extent(1));
213 for (int f_ix = 0; f_ix < f_ix_max; ++f_ix) {
214 backbone.set_flat_index(f_ix, hyb_poles); // set directions, pole indices, and orbital indices from a single integer index
215 eval_correlator_fixed_indices(backbone); // evaluate the diagram with these directions, poles, and orbital indices
216 for (int mu = 0; mu < mu_ops.extent(0); ++mu) {
217 for (int t = 0; t < r; ++t) { Tmuop(t, _, _) = matmul(U(t, _, _), matmul(mu_ops(mu, _, _), T(t, _, _))); }
218 for (int kap = 0; kap < kap_ops.extent(0); ++kap) {
219 for (int t = 0; t < r; ++t) { correlator(t, mu, kap) += trace(matmul(Tmuop(t, _, _), kap_ops(kap, _, _))); }
220 }
221 }
222 backbone.reset_all_inds(); // reset directions, pole indices, and orbital indices for the next iteration
223 }
224 return correlator;

227void DenseDiagramEvaluator::eval_correlator_fixed_indices(CorrelatorBackbone &backbone) {
228 int m = backbone.m;
229
230 // evaluate the first sequence of backbone products and convolutions from tau_1 to tau, proceeding right to left, not inculding the creation and
231 // annihilation matrices at the end points. the result is an N x N matrix-valued function of tau.
232 T = Gt; // T stores the result moving right to left
233 // T is initialized to Gt, which is always the function at the rightmost edge
234 for (int v = 1; v < backbone.get_topology(0, 1); ++v) { // loop from the first vertex to before the special vertex
235 multiply_vertex(backbone, v);
236 compose_with_edge(backbone, v);
237 }
238
239 // evaluate the second sequence of backbone products and convolutions from tau to beta, using a change of variables to perform convolutions. the
240 // result is another N x N matrix-valued function of tau.
241 U = Gt; // U stores the result moving right to left
242 if (m > 1) { // only do this for second-order and higher
243 // compute U, which is the edge immediately to the left of the vertex connected to 0
244 int be = 0;
245 for (int i = 0; i < m - 1; ++i) {
246 be = backbone.get_edge(backbone.get_topology(0, 1), i);
247 double pole = hyb_poles(backbone.get_pole_ind(i));
248 if (backbone.get_fb(i + 1) == 0) pole = -pole; // MODIFIED LOGIC -- HYB_POLES->(-HYB_POLES) FOR BACKWARD LINES
249 if (be != 0) {
250 for (int t = 0; t < r; t++) { U(t, _, _) = k_it(dlr_it(t), be * pole) * U(t, _, _); }
251 }
252 }
253 for (int v = backbone.get_topology(0, 1) + 1; v < 2 * m - 1; ++v) { // loop from the special vertex to before the last vertex
254 multiply_vertex_corr(backbone, v);
255 compose_with_edge_corr(backbone, v);
256 }
257 // multiply by the last vertex
258 multiply_vertex_corr(backbone, 2 * m - 1);
259 // convolve with last edge
260 GKt = Gt;
261 int bv = backbone.get_vertex_Ksign(2 * m - 1); // sign on K
262 int l_ix = backbone.get_pole_ind(backbone.get_vertex_hyb_ind(2 * m - 1));
263 double pole = hyb_poles(l_ix);
264 if (backbone.get_vertex_direction(2 * m - 1) == 0) pole = -pole; // MODIFIED LOGIC -- HYB_POLES->(-HYB_POLES) FOR BACKWARD LINES
265 if (bv != 0) {
266 for (int t = 0; t < r; t++) { GKt(t, _, _) = k_it(dlr_it(t), -bv * pole) * GKt(t, _, _); }
267 }
268 U = itops.convolve(beta, itops.vals2coefs(GKt), itops.vals2coefs(U), TIME_ORDERED);
269 }
270 U = itops.reflect(U);
271
272 multiply_prefactor(backbone);
273 int diag_order_sign = 1; // (m % 2 == 1) ? -1 : 1;
274 T *= diag_order_sign * backbone.prefactor_sign;
275}
