128void DiagramEvaluator::eval_backbone_fixed_indices_dense(Backbone &backbone) {
129 int m = backbone.m;
130
131 // 1. Starting from tau_1, proceed right to left, performing multiplications at vertices and convolutions at edges, until reaching the vertex
132 // containing the undecomposed hybridization line Delta_{mu kappa}.
133 T = Gt; // T stores the result moving left to right
134 // T is initialized to Gt, which is always the function at the rightmost edge
135 for (int v = 1; v < backbone.get_topology(0, 1); v++) { // loop from the first vertex to before the special vertex
136 multiply_vertex_dense(backbone, v);
137 compose_with_edge_dense(backbone, v);
138 }
139
140 // 2. For each kappa, multiply by F_kappa(^dag). Then for each mu, kappa, multiply by Delta_{mu kappa}, and sum over kappa. Finally for each mu,
141 // multiply F_mu[^dag] and sum over mu.
142 multiply_zero_vertex(backbone, (not backbone.has_vertex_dag(0)));
143
144 // 3. Continue right to left until the final vertex multiplication is complete.
145 for (int v = backbone.get_topology(0, 1) + 1; v < 2 * m; v++) { // loop from the special vertex to the last vertex
146 compose_with_edge_dense(backbone, v - 1);
147 multiply_vertex_dense(backbone, v);
148 }
149
150 multiply_prefactor(backbone);
151 int diag_order_sign = (m % 2 == 0) ? -1 : 1;
152 if (backbone.get_fb(0) == 0) diag_order_sign *= -1; // if the first hybridization line is backward, there is an additional sign change
153 T *= diag_order_sign * backbone.prefactor_sign;
154 Sigma += T;
155}

230void CorrelatorDiagramEvaluator::eval_backbone_fixed_indices_dense(CorrelatorBackbone &backbone) {
231 int m = backbone.m;
232
233 // evaluate the first sequence of backbone products and convolutions from tau_1 to tau, proceeding right to left, not inculding the creation and
234 // annihilation matrices at the end points. the result is an N x N matrix-valued function of tau.
235 T = Gt; // T stores the result moving right to left
236 // T is initialized to Gt, which is always the function at the rightmost edge
237 for (int v = 1; v < backbone.get_topology(0, 1); ++v) { // loop from the first vertex to before the special vertex
238 multiply_vertex_dense(backbone, v);
239 compose_with_edge_dense(backbone, v);
240 }
241
242 // evaluate the second sequence of backbone products and convolutions from tau to beta, using a change of variables to perform convolutions. the
243 // result is another N x N matrix-valued function of tau.
244 U = Gt; // U stores the result moving right to left
245 // compute U, which is the edge immediately to the left of the vertex connected to 0
246 int be = 0;
247 for (int i = 0; i < m - 1; ++i) {
248 be = backbone.get_edge(backbone.get_topology(0, 1), i);
249 if (be != 0) {
250 for (int t = 0; t < r; t++) { U(t, _, _) = k_it(dlr_it(t), be * hyb_poles(backbone.get_pole_ind(i))) * U(t, _, _); }
251 }
252 }
253 for (int v = backbone.get_topology(0, 1) + 1; v < 2 * m - 1; ++v) { // loop from the special vertex to before the last vertex
254 multiply_vertex_corr_left_dense(backbone, v);
255 compose_with_edge_corr_left_dense(backbone, v);
256 }
257 // multiply by the last vertex
258 multiply_vertex_corr_left_dense(backbone, 2 * m - 1);
259 // convolve with last edge
260 GKt = Gt;
261 int bv = backbone.get_vertex_Ksign(2 * m - 1); // sign on K
262 int l_ix = backbone.get_pole_ind(backbone.get_vertex_hyb_ind(2 * m - 1));
263 double pole = hyb_poles(l_ix);
264 if (bv != 0) {
265 for (int t = 0; t < r; t++) { GKt(t, _, _) = k_it(dlr_it(t), -bv * pole) * GKt(t, _, _); }
266 }
267 U = itops.convolve(beta, itops.vals2coefs(GKt), itops.vals2coefs(U), TIME_ORDERED);
268 U = itops.reflect(U);
269
270 multiply_prefactor(backbone);
271 int diag_order_sign = (m % 2 == 1) ? -1 : 1;
272 T *= diag_order_sign * backbone.prefactor_sign;
