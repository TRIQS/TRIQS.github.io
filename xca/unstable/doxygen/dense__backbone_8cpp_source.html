116void DiagramEvaluator::eval_backbone_fixed_indices_dense(Backbone &backbone) {
117 int m = backbone.m;
118
119 // 1. Starting from tau_1, proceed right to left, performing multiplications at vertices and convolutions at edges, until reaching the vertex
120 // containing the undecomposed hybridization line Delta_{mu kappa}.
121 T = Gt; // T stores the result moving left to right
122 // T is initialized to Gt, which is always the function at the rightmost edge
123 for (int v = 1; v < backbone.get_topology(0, 1); v++) { // loop from the first vertex to before the special vertex
124 multiply_vertex_dense(backbone, v);
125 compose_with_edge_dense(backbone, v);
126 }
127
128 // 2. For each kappa, multiply by F_kappa(^dag). Then for each mu, kappa, multiply by Delta_{mu kappa}, and sum over kappa. Finally for each mu,
129 // multiply F_mu[^dag] and sum over mu.
130 multiply_zero_vertex(backbone, (not backbone.has_vertex_dag(0)));
131
132 // 3. Continue right to left until the final vertex multiplication is complete.
133 for (int v = backbone.get_topology(0, 1) + 1; v < 2 * m; v++) { // loop from the special vertex to the last vertex
134 compose_with_edge_dense(backbone, v - 1);
135 multiply_vertex_dense(backbone, v);
136 }
137
138 // Multiply by prefactor
139 for (int m_ix = 0; m_ix < m - 1; m_ix++) { // loop over hybridization indices
140 int exp = backbone.get_prefactor_Kexp(m_ix); // exponent on K for this hybridization index
141 if (exp != 0) {
142 int Ksign = backbone.get_prefactor_Ksign(m_ix); // sign on K for this hybridization index
143 double om = hyb_poles(backbone.get_pole_ind(m_ix)); // DLR frequency for this value of this hybridization index
144 double k = k_it(0, Ksign * om);
145 for (int q = 0; q < exp; q++) T /= k;
146 }
147 }
148 int diag_order_sign = (m % 2 == 0) ? -1 : 1;
149 if (backbone.get_fb(0) == 0) diag_order_sign *= -1; // if the first hybridization line is backward, there is an additional sign change
150 T *= diag_order_sign * backbone.prefactor_sign;
151 Sigma += T;
152}
