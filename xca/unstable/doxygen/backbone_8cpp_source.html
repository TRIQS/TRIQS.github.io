40 // @param[in] fb forward/backward line information

42 this->fb = fb;

91void Backbone::set_pole_inds(nda::vector_const_view<int> pole_inds, nda::vector_const_view<double> hyb_poles) {
92
93 this->pole_inds = pole_inds; // values of l, l`, etc.
94 for (int i = 1; i < m; i++) {
95 if (fb(i) == 1) { // line i is forward
96 if (hyb_poles(pole_inds(i - 1)) <= 0) {
97 // step 4(a)
98 // place K^-_l F_nu at the right vertex
99 vertices[topology(i, 0)].set_hyb_ind(i - 1);
100 vertices[topology(i, 0)].set_Ksign(-1);
101 // place K^+_l F^bar^dag_nu_l at the left vertex
102 vertices[topology(i, 1)].set_hyb_ind(i - 1);
103 vertices[topology(i, 1)].set_Ksign(1);
104 // divide by K^-_l(0)
105 prefactor_Ksigns(i - 1) = -1;
106 prefactor_Kexps(i - 1) = 1;
107 } else {
108 // step 4(b)
109 // no K's on vertices, but F bar on left vertex
110 vertices[topology(i, 1)].set_hyb_ind(i - 1);
111 // place K^+_l on each edge between the two vertices
112 for (int j = topology(i, 0); j < topology(i, 1); j++) edges(j, i - 1) = 1;
113 // divide by (K^+_l(0))^(# edges between vertices - 1)
114 prefactor_Ksigns(i - 1) = 1;
115 prefactor_Kexps(i - 1) = topology(i, 1) - topology(i, 0) - 1;
116 }
117 } else { // line i is backward
118 if (hyb_poles(pole_inds(i - 1)) >= 0) {
119 // step 4(a)
120 // place K^+_l F^dag_pi at the right vertex
121 vertices[topology(i, 0)].set_hyb_ind(i - 1);
122 vertices[topology(i, 0)].set_Ksign(1);
123 // place K^-_l F^bar_pi_l at the left vertex
124 vertices[topology(i, 1)].set_hyb_ind(i - 1);
125 vertices[topology(i, 1)].set_Ksign(-1);
126 // divide by -K^-+l(0)
127 prefactor_sign *= -1;
128 prefactor_Ksigns(i - 1) = 1;
129 prefactor_Kexps(i - 1) = 1;
130 } else {
131 // step 4(b)
132 // no K's on vertices, but F bar on left vertex
133 vertices[topology(i, 1)].set_hyb_ind(i - 1);
134 // place K^-_l on each edge between the two vertices
135 for (int j = topology(i, 0); j < topology(i, 1); j++) edges(j, i - 1) = -1;
136 // divide by (-K^-_l(0))^(# edges between vertices - 1)
137 prefactor_sign *= -1;
138 // if ((topology(i,1) - topology(i,0) - 1) % 2 == 1) prefactor_sign *= -1;
139 prefactor_Ksigns(i - 1) = -1;
140 prefactor_Kexps(i - 1) = topology(i, 1) - topology(i, 0) - 1;
141 }
142 }
143 }

179void Backbone::set_orb_inds(int o_ix) {
180 // set orbital indices from a single integer index
181 auto orb_inds = nda::vector<int>(2 * m);
182 orb_inds(0) = -1;
183 orb_inds(topology(0, 1)) = -1; // special vertex 0 and the one connected to it have no orbital indices explicitly summed over
184 for (int i = 1; i < m; i++) { // loop over lines, skipping the one connected to vertex 0
185 orb_inds(topology(i, 0)) = o_ix % n;
186 orb_inds(topology(i, 1)) = o_ix % n;
187 // orbital indices on vertices connected by a line are the same
188 o_ix /= n;
189 }

197void Backbone::set_flat_index(int f_ix, nda::vector_const_view<double> hyb_poles) {
198 // set directions, pole indices, and orbital indices from a single integer index.
199 // In terms of fb_ix, p_ix, and o_ix,
200 // f_ix = o_ix + n^(m-1) * p_ix + (n * r)^(m-1) * fb_ix, where r is the number of hybridization indices.
201
202 this->f_ix = f_ix;
203 int p = hyb_poles.size();
204 int o_ix = f_ix % o_ix_max; // orbital indices
205 f_ix /= o_ix_max;
206 int p_ix_max = static_cast<int>(pow(p, m - 1));
207 int p_ix = f_ix % p_ix_max; // pole indices
208 int fb_ix = f_ix / p_ix_max; // directions
209
210 set_directions(fb_ix);
211 set_pole_inds(p_ix, hyb_poles);
212 set_orb_inds(o_ix);
213}
