75void fastdiagram::hyb_init(nda::array<dcomplex, 3> Deltat0, bool poledlrflag) {
76
77 Deltat = Deltat0;
78 auto Deltat_reflect_intermediate = itops.reflect(Deltat); // obtain Delta(-t) from Delta(t)
79 Deltat_reflect = Deltat_reflect_intermediate;
80
81 for (int i = 0; i < r; ++i) Deltat_reflect(i, _, _) = transpose(Deltat_reflect_intermediate(i, _, _));
82
83 if (poledlrflag == false) {
84 auto ifops = imfreq_ops(lambda, dlr_rf, Fermion);
85 int Nwmax = int(max_element(abs((2.0 * ifops.get_ifnodes()) + 1.0)));
86 if (Nwmax % 2 == 0) Nwmax += 1;
87 dlr_if_dense = nda::arange(-Nwmax, Nwmax + 1, 2) * std::atan2(0, -1) / beta;
88 }
89}

101void fastdiagram::hyb_decomposition(bool poledlrflag, double eps) {
102 // this->hyb_init(Deltat0,poledlrflag);
103
104 if (poledlrflag == false) {
105 auto Delta_decomp = hyb_decomp(weights, pol, eps); // decomposition of Delta(t) using DLR coefficient
106 auto Delta_decomp_reflect = hyb_decomp(weights_reflect, pol_reflect, eps); // decomposition of Delta(-t) using DLR coefficient
107
108 Delta_F.update_inplace(Delta_decomp, dlr_it, F, F_dag); // Compression of Delta(t) and F, F_dag matrices
109 Delta_F_reflect.update_inplace(Delta_decomp_reflect, dlr_it, F_dag, F); // Compression of Delta(-t) and F, F_dag matrices
110
111 } else {
112
113 auto Deltadlr = itops.vals2coefs(Deltat); //obtain dlr coefficient of Delta(t)
114
115 nda::vector<double> dlr_rf_reflect = -dlr_rf;
116 nda::array<dcomplex, 3> Deltadlr_reflect = Deltadlr * 1.0;
117
118 for (int i = 0; i < Deltadlr.shape(0); ++i) { Deltadlr_reflect(i, _, _) = transpose(Deltadlr(i, _, _)); }
119
120 auto Delta_decomp = hyb_decomp(Deltadlr, dlr_rf, eps); // decomposition of Delta(t) using DLR coefficient
121 auto Delta_decomp_reflect = hyb_decomp(Deltadlr_reflect, dlr_rf_reflect, eps); // decomposition of Delta(-t) using DLR coefficient
122
123 Delta_F.update_inplace(Delta_decomp, dlr_it, F, F_dag); // Compression of Delta(t) and F, F_dag matrices
124 Delta_F_reflect.update_inplace(Delta_decomp_reflect, dlr_it, F_dag, F); // Compression of Delta(-t) and F, F_dag matrices
125 }
126
127 P = Delta_F.P;
128}

132nda::array<dcomplex, 3> fastdiagram::Sigma_calc_group(nda::array<dcomplex, 3> Gt, nda::array<int, 2> D, nda::array<int, 1> diagramindex) {
133
134 auto N = Gt.shape(1);
135 auto Nd = diagramindex.shape(0);
136 auto m = D.shape(0);
137 auto num_diagram_per_fb = pown(P, m - 1);
138 auto Diagram = nda::array<dcomplex, 3>::zeros({r, N, N});
139
140 for (int id = 0; id < Nd; ++id) {
141
142 auto fb = nda::vector<int64_t>(m); //utility for iteration
143 auto num = diagramindex(id);
144 auto num0 = num / num_diagram_per_fb;
145 auto num2 = num % num_diagram_per_fb;
146
147 for (int v = 1; v < m; ++v) {
148 fb[v] = num0 % 2;
149 num0 = int64_t(num0 / 2);
150 }
151
152 Diagram =
153 Diagram + evaluate_one_diagram(Delta_F, Delta_F_reflect, D, Deltat, Deltat_reflect, Gt, itops, beta, F, F_dag, fb, true, num2, m, n, r, N, P);
154 }
155
156 return Diagram;
157}

159nda::array<dcomplex, 3> fastdiagram::Sigma_calc(nda::array<dcomplex, 3> Gt, std::string order) {
160
161 if (order.compare("NCA") != 0 && order.compare("OCA") != 0 && order.compare("TCA") != 0)
162 throw std::runtime_error("order needs to be NCA, OCA or TCA\n");
163
164 // First do NCA calculation
165 std::cout << "S-NCA: start\n";
166 nda::array<dcomplex, 3> Sigma_NCA = -Sigma_Diagram_calc_sum_all(Delta_F, Delta_F_reflect, D_NCA, Deltat, Deltat_reflect, Gt, itops, beta, F, F_dag);
167 std::cout << "S-NCA: done\n";
168
169 if (order.compare("NCA") == 0) return Sigma_NCA;
170
171 // Do OCA calculation
172 std::cout << "S-OCA: start\n";
173 auto m = D_OCA.shape(0);
174 auto Num_diagram_oca = this->number_of_diagrams(m);
175
176 std::cout << "number of OCA diagrams = " << Num_diagram_oca << "\n";
177

179 std::cout << "S-OCA: done\n";
180
181 if (order.compare("OCA") == 0) return make_regular(Sigma_NCA + Sigma_OCA);
182
183 // Do TCA calculation
184 std::cout << "S-TCA: start\n";
185 nda::array<dcomplex, 3> Sigma_TCA =
186 -Sigma_Diagram_calc_sum_all(Delta_F, Delta_F_reflect, D_TCA_1, Deltat, Deltat_reflect, Gt, itops, beta, F, F_dag)
187 - Sigma_Diagram_calc_sum_all(Delta_F, Delta_F_reflect, D_TCA_2, Deltat, Deltat_reflect, Gt, itops, beta, F, F_dag)
188 - Sigma_Diagram_calc_sum_all(Delta_F, Delta_F_reflect, D_TCA_3, Deltat, Deltat_reflect, Gt, itops, beta, F, F_dag)
189 + Sigma_Diagram_calc_sum_all(Delta_F, Delta_F_reflect, D_TCA_4, Deltat, Deltat_reflect, Gt, itops, beta, F, F_dag);
190 std::cout << "S-TCA: done\n";
191
192 return make_regular(Sigma_NCA + Sigma_OCA + Sigma_TCA);
193}
