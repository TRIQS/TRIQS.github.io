75 void Hybridization::multiply_kernels_prefactor(nda::array_view<dcomplex, 3> T_buf, Backbone &backbone) {
76 // Multiply the kernels associated with the prefactor into T_buf
77
78 int m = backbone.m;
79 double prefactor = 1.;
80
81 for (int m_ix = 0; m_ix < m - 1; m_ix++) { // loop over hybridization indices

84 int Ksign = backbone.get_prefactor_Ksign(m_ix); // sign on K for this hybridization index
