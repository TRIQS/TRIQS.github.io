426BlockOpSymQuartet::BlockOpSymQuartet(std::vector<BlockOpSymSet> Fs, std::vector<BlockOpSymSet> F_dags, nda::array_const_view<dcomplex, 3> hyb_coeffs,
427 nda::array_const_view<dcomplex, 3> hyb_refl_coeffs, nda::vector_const_view<long> sym_set_labels)
428 : Fs(Fs), F_dags(F_dags), sym_set_labels(sym_set_labels) {
429
430 // Fs and F_dags are vectors of BOSS
431 // Each entry corresponds to a set of operators with the same block-sparse structure
432 // Fs and F_dags have the same number of entries
433 // if k = # of entries of Fs, and each entry f_i has q_i operators, then n = sum(q_i) = number of orbital indices
434 int k = Fs.size();
435 if (k != F_dags.size()) { throw std::invalid_argument("Fs and F_dags must have the same number of entries"); }
436
437 // initialize F_dag_bars and F_bars_refl
438 int p = hyb_coeffs.extent(0);
439 std::vector<BlockOpSymSetBar> F_dag_bars, F_bars_refl;
440 nda::vector<int> block_indices_dag = F_dags[0].get_block_indices();
441 nda::vector<int> block_indices_f = Fs[0].get_block_indices();
442 for (int i = 0; i < k; i++) {
443 block_indices_dag = F_dags[i].get_block_indices();
444 block_indices_f = Fs[i].get_block_indices();
445 std::cout << "Block indices dag: " << block_indices_dag << "\n";
446 F_dag_bars.emplace_back(F_dags[i].get_size_sym_set(), p, block_indices_dag, F_dags[i].get_block_sizes());
447 std::cout << "Block indices f: " << block_indices_f << "\n";
448 F_bars_refl.emplace_back(Fs[i].get_size_sym_set(), p, block_indices_f, Fs[i].get_block_sizes());
449 }
450
451 // calculate symmetry set indices
452 long n = sym_set_labels.size(); // number of orbital indices
453 long q = nda::max_element(sym_set_labels) + 1; // number of symmetry sets
454 sym_set_inds = nda::zeros<long>(n); // indices of orbital indices in symmetry sets
455 sym_set_sizes = nda::zeros<long>(q); // sizes of symmetry sets
456 sym_set_to_orb = nda::ones<long>(q, n); // map label and index to orbital index
457 sym_set_to_orb *= -1; // initialize with -1
458 for (int i = 0; i < n; i++) {
459 sym_set_inds(i) = sym_set_sizes(sym_set_labels(i));
460 sym_set_sizes(sym_set_labels(i))++;
461 sym_set_to_orb(sym_set_labels(i), sym_set_inds(i)) = i; // map symmetry set index to backbone orbital index
462 }
463
464 // compute F_dag_bars and F_bars_refl
465 for (int l = 0; l < p; l++) {
466 for (int p_lam = 0; p_lam < q; p_lam++) {
467 for (int p_nu = 0; p_nu < q; p_nu++) {
468 for (int lam = 0; lam < sym_set_sizes(p_lam); lam++) {
469 for (int nu = 0; nu < sym_set_sizes(p_nu); nu++) {
470 long lam_orb = sym_set_to_orb(p_lam, lam);
471 long nu_orb = sym_set_to_orb(p_nu, nu);
472 for (int b = 0; b < F_dags[p_lam].get_num_block_cols(); b++) {
473 if (F_dags[p_lam].get_block_index(b) != -1) {
474 F_dag_bars[p_lam].add_block(b, lam, l, nda::make_regular(hyb_coeffs(l, nu_orb, lam_orb) * F_dags[p_lam].get_block(b)(nu, _, _)));
475 }
476 }
477 for (int b = 0; b < Fs[p_nu].get_num_block_cols(); b++) {
478 if (Fs[p_nu].get_block_index(b) != -1) {
479 F_bars_refl[p_nu].add_block(b, nu, l, nda::make_regular(hyb_refl_coeffs(l, nu_orb, lam_orb) * Fs[p_nu].get_block(b)(lam, _, _)));
480 }
481 }
482 }
483 }
484 }
485 }
486 }
487 this->F_dag_bars = F_dag_bars;
488 this->F_bars_refl = F_bars_refl;
489}

555 // Evaluate F^dagger in block-sparse storage
556 // @param[in] F F operator
557 // @return F^dagger operator
558
559 int num_block_cols = F.get_num_block_cols();
560
561 // find block indices for F^dagger

563 // initialize indices with -1
564 block_indices_dag = -1;
565 std::vector<nda::array<dcomplex, 2>> blocks_dag(num_block_cols);
566 for (int i = 0; i < num_block_cols; ++i) {
567 int j = F.get_block_indices()[i];
568 if (j != -1) {
569 block_indices_dag[j] = i;
570 blocks_dag[j] = nda::transpose(F.get_blocks()[i]);
571 }
572 }
573 BlockOp F_dag(block_indices_dag, blocks_dag);
574 return F_dag;

622BlockDiagOpFun BOFtoBDOF(BlockOpFun const &A) {
623 // Convert a BlockOpFun with diagonal structure to a BlockDiagOpFun
624 // @param[in] A BlockOpFun
625 // @return BlockDiagOpFun
626
627 int num_block_cols = A.get_num_block_cols();
628 auto diag_blocks = A.get_blocks();
629 auto zero_block_indices = nda::zeros<int>(num_block_cols);
630 for (int i = 0; i < num_block_cols; i++) {
631 int block_index = A.get_block_index(i);
632 if (block_index == -1) {
633 diag_blocks[i] = nda::zeros<dcomplex>(1, 1, 1);
634 zero_block_indices(i) = -1;
635 } else if (block_index != i) {
636 throw std::invalid_argument("BOF is not diagonal");
637 }
638 }
639
640 return BlockDiagOpFun(diag_blocks, zero_block_indices);
641}

643BlockDiagOpFun nonint_gf_BDOF(std::vector<nda::array<double, 2>> H_blocks, nda::vector<int> H_block_inds, double beta,
644 nda::vector_const_view<double> dlr_it_abs) {
645
646 int num_block_cols = H_block_inds.size();
647 nda::vector<int> H_block_sizes(num_block_cols);
648 for (int i = 0; i < num_block_cols; i++) { H_block_sizes(i) = H_blocks[i].extent(0); }
649
650 int r = dlr_it_abs.size();
651
652 double tr_exp_minusbetaH = 0;
653 std::vector<nda::array<double, 1>> H_evals(num_block_cols);
654 std::vector<nda::array<double, 2>> H_evecs(num_block_cols);
655 for (int i = 0; i < num_block_cols; i++) {
656 if (H_block_inds(i) != -1) {
657 if (H_block_sizes(i) == 1) {
658 H_evals[i] = nda::array<double, 1>{H_blocks[i](0, 0)};
659 H_evecs[i] = nda::array<double, 2>{{1}};
660 } else {
661 auto H_block_eig = nda::linalg::eigenelements(H_blocks[i]);
662 H_evals[i] = std::get<0>(H_block_eig);
663 H_evecs[i] = std::get<1>(H_block_eig);
664 }
665 tr_exp_minusbetaH += nda::sum(exp(-beta * H_evals[i]));
666 } else {
667 H_evals[i] = nda::zeros<double>(H_block_sizes(i));
668 H_evecs[i] = nda::eye<double>(H_block_sizes(i));
669 tr_exp_minusbetaH += 1.0 * H_block_sizes(i); // 0 entry in the diagonal
670 }
671 }
672
673 auto eta_0 = nda::log(tr_exp_minusbetaH) / beta;
674 auto Gt = BlockDiagOpFun(r, H_block_sizes);
675 for (int i = 0; i < num_block_cols; i++) {
676 auto Gt_block = nda::array<dcomplex, 3>(r, H_block_sizes(i), H_block_sizes(i));
677 auto Gt_temp = nda::make_regular(0 * H_blocks[i]);
678 for (int t = 0; t < r; t++) {
679 for (int j = 0; j < H_block_sizes(i); j++) { Gt_temp(j, j) = -exp(-beta * dlr_it_abs(t) * (H_evals[i](j) + eta_0)); }
680 Gt_block(t, _, _) = nda::matmul(H_evecs[i], nda::matmul(Gt_temp, nda::transpose(H_evecs[i])));
681 }
682 Gt.set_block(i, Gt_block);
683 }
684
685 return Gt;
686}
