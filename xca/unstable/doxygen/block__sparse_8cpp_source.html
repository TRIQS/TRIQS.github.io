436BlockOpSymQuartet::BlockOpSymQuartet(std::vector<BlockOpSymSet> Fs, std::vector<BlockOpSymSet> F_dags, nda::array_const_view<dcomplex, 3> hyb_coeffs,
437 nda::array_const_view<dcomplex, 3> hyb_refl_coeffs, nda::vector_const_view<long> sym_set_labels)
438 : Fs(Fs), F_dags(F_dags), sym_set_labels(sym_set_labels) {
439
440 // Fs and F_dags are vectors of BOSS
441 // Each entry corresponds to a set of operators with the same block-sparse structure
442 // Fs and F_dags have the same number of entries
443 // if k = # of entries of Fs, and each entry f_i has q_i operators, then n = sum(q_i) = number of orbital indices
444 int k = Fs.size();
445 if (k != F_dags.size()) { throw std::invalid_argument("Fs and F_dags must have the same number of entries"); }
446
447 // initialize F_dag_bars and F_bars_refl
448 int p = hyb_coeffs.extent(0);
449 std::vector<BlockOpSymSetBar> F_dag_bars, F_bars_refl;
450 nda::vector<int> block_indices_dag = F_dags[0].get_block_indices();
451 nda::vector<int> block_indices_f = Fs[0].get_block_indices();
452 for (int i = 0; i < k; i++) {
453 block_indices_dag = F_dags[i].get_block_indices();
454 block_indices_f = Fs[i].get_block_indices();
455 std::cout << "Block indices dag: " << block_indices_dag << "\n";
456 F_dag_bars.emplace_back(F_dags[i].get_size_sym_set(), p, block_indices_dag, F_dags[i].get_block_sizes());
457 std::cout << "Block indices f: " << block_indices_f << "\n";
458 F_bars_refl.emplace_back(Fs[i].get_size_sym_set(), p, block_indices_f, Fs[i].get_block_sizes());
459 }
460
461 // calculate symmetry set indices
462 long n = sym_set_labels.size(); // number of orbital indices
463 long q = nda::max_element(sym_set_labels) + 1; // number of symmetry sets
464 sym_set_inds = nda::zeros<long>(n); // indices of orbital indices in symmetry sets
465 sym_set_sizes = nda::zeros<long>(q); // sizes of symmetry sets
466 sym_set_to_orb = nda::ones<long>(q, n); // map label and index to orbital index
467 sym_set_to_orb *= -1; // initialize with -1
468 for (int i = 0; i < n; i++) {
469 sym_set_inds(i) = sym_set_sizes(sym_set_labels(i));
470 sym_set_sizes(sym_set_labels(i))++;
471 sym_set_to_orb(sym_set_labels(i), sym_set_inds(i)) = i; // map symmetry set index to backbone orbital index
472 }
473
474 // compute F_dag_bars and F_bars_refl
475 for (int l = 0; l < p; l++) {
476 for (int p_lam = 0; p_lam < q; p_lam++) {
477 for (int p_nu = 0; p_nu < q; p_nu++) {
478 for (int lam = 0; lam < sym_set_sizes(p_lam); lam++) {
479 for (int nu = 0; nu < sym_set_sizes(p_nu); nu++) {
480 long lam_orb = sym_set_to_orb(p_lam, lam);
481 long nu_orb = sym_set_to_orb(p_nu, nu);
482 for (int b = 0; b < F_dags[p_lam].get_num_block_cols(); b++) {
483 if (F_dags[p_lam].get_block_index(b) != -1) {
484 F_dag_bars[p_lam].add_block(b, lam, l, nda::make_regular(hyb_coeffs(l, nu_orb, lam_orb) * F_dags[p_lam].get_block(b)(nu, _, _)));
485 }
486 }
487 for (int b = 0; b < Fs[p_nu].get_num_block_cols(); b++) {
488 if (Fs[p_nu].get_block_index(b) != -1) {
489 F_bars_refl[p_nu].add_block(b, nu, l, nda::make_regular(hyb_refl_coeffs(l, nu_orb, lam_orb) * Fs[p_nu].get_block(b)(lam, _, _)));
490 }
491 }
492 }
493 }
494 }
495 }
496 }
497 this->F_dag_bars = F_dag_bars;
498 this->F_bars_refl = F_bars_refl;
499}

564BlockOp dagger_bs(BlockOp const &F) {
565 // Evaluate F^dagger in block-sparse storage
566 // @param[in] F F operator
567 // @return F^dagger operator
568
569 int num_block_cols = F.get_num_block_cols();
570
571 // find block indices for F^dagger
572 nda::vector<int> block_indices_dag(num_block_cols);
573 // initialize indices with -1
574 block_indices_dag = -1;
575 std::vector<nda::array<dcomplex, 2>> blocks_dag(num_block_cols);
576 for (int i = 0; i < num_block_cols; ++i) {
577 int j = F.get_block_indices()[i];
578 if (j != -1) {
579 block_indices_dag[j] = i;
580 blocks_dag[j] = nda::transpose(F.get_blocks()[i]);
581 }
582 }
583 BlockOp F_dag(block_indices_dag, blocks_dag);
584 return F_dag;
585}

632BlockDiagOpFun BOFtoBDOF(BlockOpFun const &A) {
633 // Convert a BlockOpFun with diagonal structure to a BlockDiagOpFun
634 // @param[in] A BlockOpFun
635 // @return BlockDiagOpFun
636
637 int num_block_cols = A.get_num_block_cols();
638 auto block_views = A.get_blocks();
639 std::vector<nda::array<dcomplex, 3>> diag_blocks(block_views.begin(), block_views.end());
640 auto zero_block_indices = nda::zeros<int>(num_block_cols);
641 for (int i = 0; i < num_block_cols; i++) {
642 int block_index = A.get_block_index(i);
643 if (block_index == -1) {
644 diag_blocks[i] = nda::zeros<dcomplex>(1, 1, 1);
645 zero_block_indices(i) = -1;
646 } else if (block_index != i) {
647 throw std::invalid_argument("BOF is not diagonal");
648 }
649 }
650
651 return BlockDiagOpFun(std::move(diag_blocks), zero_block_indices);
652}

654BlockDiagOpFun nonint_gf_BDOF(std::vector<nda::array<double, 2>> H_blocks, nda::vector<int> H_block_inds, double beta,
655 nda::vector_const_view<double> dlr_it_abs) {
656
657 int num_block_cols = H_block_inds.size();
658 nda::vector<int> H_block_sizes(num_block_cols);
659 for (int i = 0; i < num_block_cols; i++) { H_block_sizes(i) = H_blocks[i].extent(0); }
660
661 int r = dlr_it_abs.size();
662
663 double tr_exp_minusbetaH = 0;
664 std::vector<nda::array<double, 1>> H_evals(num_block_cols);
665 std::vector<nda::array<double, 2>> H_evecs(num_block_cols);
666 for (int i = 0; i < num_block_cols; i++) {
667 if (H_block_inds(i) != -1) {
668 if (H_block_sizes(i) == 1) {
669 H_evals[i] = nda::array<double, 1>{H_blocks[i](0, 0)};
670 H_evecs[i] = nda::array<double, 2>{{1}};
671 } else {
672 auto H_block_eig = nda::linalg::eigenelements(H_blocks[i]);
673 H_evals[i] = std::get<0>(H_block_eig);
674 H_evecs[i] = std::get<1>(H_block_eig);
675 }
676 tr_exp_minusbetaH += nda::sum(exp(-beta * H_evals[i]));
677 } else {
678 H_evals[i] = nda::zeros<double>(H_block_sizes(i));
679 H_evecs[i] = nda::eye<double>(H_block_sizes(i));
680 tr_exp_minusbetaH += 1.0 * H_block_sizes(i); // 0 entry in the diagonal
681 }
682 }
683
684 auto eta_0 = nda::log(tr_exp_minusbetaH) / beta;
685 auto Gt = BlockDiagOpFun(r, H_block_sizes);
686 for (int i = 0; i < num_block_cols; i++) {
687 auto Gt_block = nda::array<dcomplex, 3>(r, H_block_sizes(i), H_block_sizes(i));
688 auto Gt_temp = nda::make_regular(0 * H_blocks[i]);
689 for (int t = 0; t < r; t++) {
690 for (int j = 0; j < H_block_sizes(i); j++) { Gt_temp(j, j) = -exp(-beta * dlr_it_abs(t) * (H_evals[i](j) + eta_0)); }
691 Gt_block(t, _, _) = nda::matmul(H_evecs[i], nda::matmul(Gt_temp, nda::transpose(H_evecs[i])));
692 }
693 Gt.set_block(i, Gt_block);
694 }
695
696 return Gt;
697}
