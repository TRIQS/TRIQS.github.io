452BlockOpSymQuartet::BlockOpSymQuartet(std::vector<BlockOpSymSet> Fs, std::vector<BlockOpSymSet> F_dags, nda::array_const_view<dcomplex, 3> hyb_coeffs,
453 nda::vector_const_view<long> sym_set_labels)
454 : Fs(Fs), F_dags(F_dags), sym_set_labels(sym_set_labels), p(hyb_coeffs.extent(0)) {
455
456 // Fs and F_dags are vectors of BOSS
457 // Each entry corresponds to a set of operators with the same block-sparse structure
458 // Fs and F_dags have the same number of entries
459 // if k = # of entries of Fs, and each entry f_i has q_i operators, then n = sum(q_i) = number of orbital indices
460 int k = Fs.size();
461 if (k != F_dags.size()) { throw std::invalid_argument("Fs and F_dags must have the same number of entries"); }
462
463 // initialize F_dag_bars and F_bars_refl
464 int p = hyb_coeffs.extent(0);
465 std::vector<BlockOpSymSetBar> F_dag_bars, F_bars_refl;
466 nda::vector<int> block_indices_dag = F_dags[0].get_block_indices();
467 nda::vector<int> block_indices_f = Fs[0].get_block_indices();
468 for (int i = 0; i < k; i++) {
469 block_indices_dag = F_dags[i].get_block_indices();
470 block_indices_f = Fs[i].get_block_indices();
471 F_dag_bars.emplace_back(F_dags[i].get_size_sym_set(), p, block_indices_dag, F_dags[i].get_block_sizes());
472 F_bars_refl.emplace_back(Fs[i].get_size_sym_set(), p, block_indices_f, Fs[i].get_block_sizes());
473 }
474
475 // calculate symmetry set indices
476 long n = sym_set_labels.size(); // number of orbital indices
477 long q = nda::max_element(sym_set_labels) + 1; // number of symmetry sets
478 sym_set_inds = nda::zeros<long>(n); // indices of orbital indices in symmetry sets
479 sym_set_sizes = nda::zeros<long>(q); // sizes of symmetry sets
480 sym_set_to_orb = nda::ones<long>(q, n); // map label and index to orbital index
481 sym_set_to_orb *= -1; // initialize with -1
482 for (int i = 0; i < n; i++) {
483 sym_set_inds(i) = sym_set_sizes(sym_set_labels(i));
484 sym_set_sizes(sym_set_labels(i))++;
485 sym_set_to_orb(sym_set_labels(i), sym_set_inds(i)) = i; // map symmetry set index to backbone orbital index
486 }
487
488 // compute F_dag_bars and F_bars_refl
489 for (int l = 0; l < p; l++) {
490 for (int p_lam = 0; p_lam < q; p_lam++) {
491 for (int p_nu = 0; p_nu < q; p_nu++) {
492 for (int lam = 0; lam < sym_set_sizes(p_lam); lam++) {
493 for (int nu = 0; nu < sym_set_sizes(p_nu); nu++) {
494 long lam_orb = sym_set_to_orb(p_lam, lam);
495 long nu_orb = sym_set_to_orb(p_nu, nu);
496 for (int b = 0; b < F_dags[p_lam].get_num_block_cols(); b++) {
497 if (F_dags[p_lam].get_block_index(b) != -1) {
498 F_dag_bars[p_lam].add_block(b, lam, l, nda::make_regular(hyb_coeffs(l, nu_orb, lam_orb) * F_dags[p_lam].get_block(b)(nu, _, _)));
499 }
500 }
501 for (int b = 0; b < Fs[p_nu].get_num_block_cols(); b++) {
502 if (Fs[p_nu].get_block_index(b) != -1) {
503 F_bars_refl[p_nu].add_block(b, nu, l, nda::make_regular(hyb_coeffs(l, nu_orb, lam_orb) * Fs[p_nu].get_block(b)(lam, _, _)));
504 }
505 }
506 }
507 }
508 }
509 }
510 }
511 this->F_dag_bars = F_dag_bars;
512 this->F_bars_refl = F_bars_refl;
513}

578BlockOp dagger_bs(BlockOp const &F) {
579 // Evaluate F^dagger in block-sparse storage
580 // @param[in] F F operator
581 // @return F^dagger operator
582
583 int num_block_cols = F.get_num_block_cols();
584
585 // find block indices for F^dagger

587 // initialize indices with -1
588 block_indices_dag = -1;

591 int j = F.get_block_indices()[i];

593 block_indices_dag[j] = i;
594 blocks_dag[j] = nda::transpose(F.get_blocks()[i]);
595 }
596 }
597 BlockOp F_dag(block_indices_dag, blocks_dag);
598 return F_dag;

646BlockDiagOpFun BOFtoBDOF(BlockOpFun const &A) {
647 // Convert a BlockOpFun with diagonal structure to a BlockDiagOpFun
648 // @param[in] A BlockOpFun
649 // @return BlockDiagOpFun
650
651 int num_block_cols = A.get_num_block_cols();
652 auto diag_blocks = A.get_blocks();
653 auto zero_block_indices = nda::zeros<int>(num_block_cols);
654 for (int i = 0; i < num_block_cols; i++) {
655 int block_index = A.get_block_index(i);
656 if (block_index == -1) {
657 diag_blocks[i] = nda::zeros<dcomplex>(1, 1, 1);
658 zero_block_indices(i) = -1;
659 } else if (block_index != i) {
660 throw std::invalid_argument("BOF is not diagonal");
661 }
662 }
663
664 return BlockDiagOpFun(diag_blocks, zero_block_indices);
665}

667BlockDiagOpFun nonint_gf_BDOF(std::vector<nda::array<double, 2>> H_blocks, nda::vector<int> H_block_inds, double beta,
668 nda::vector_const_view<double> dlr_it_abs) {
669
670 int num_block_cols = H_block_inds.size();
671 nda::vector<int> H_block_sizes(num_block_cols);
672 for (int i = 0; i < num_block_cols; i++) { H_block_sizes(i) = H_blocks[i].extent(0); }
673
674 int r = dlr_it_abs.size();
675
676 double tr_exp_minusbetaH = 0;
677 std::vector<nda::array<double, 1>> H_evals(num_block_cols);
678 std::vector<nda::array<double, 2>> H_evecs(num_block_cols);
679 for (int i = 0; i < num_block_cols; i++) {
680 if (H_block_inds(i) != -1) {
681 if (H_block_sizes(i) == 1) {
682 H_evals[i] = nda::array<double, 1>{H_blocks[i](0, 0)};
683 H_evecs[i] = nda::array<double, 2>{{1}};
684 } else {
685 auto H_block_eig = nda::linalg::eigh(H_blocks[i]);
686 H_evals[i] = std::get<0>(H_block_eig);
687 H_evecs[i] = std::get<1>(H_block_eig);
688 }
689 tr_exp_minusbetaH += nda::sum(exp(-beta * H_evals[i]));
690 } else {
691 H_evals[i] = nda::zeros<double>(H_block_sizes(i));
692 H_evecs[i] = nda::eye<double>(H_block_sizes(i));
693 tr_exp_minusbetaH += 1.0 * H_block_sizes(i); // 0 entry in the diagonal
694 }
695 }
696
697 auto eta_0 = nda::log(tr_exp_minusbetaH) / beta;
698 auto Gt = BlockDiagOpFun(r, H_block_sizes);
699 for (int i = 0; i < num_block_cols; i++) {
700 auto Gt_block = nda::array<dcomplex, 3>(r, H_block_sizes(i), H_block_sizes(i));
701 auto Gt_temp = nda::make_regular(0 * H_blocks[i]);
702 for (int t = 0; t < r; t++) {
703 for (int j = 0; j < H_block_sizes(i); j++) { Gt_temp(j, j) = -exp(-beta * dlr_it_abs(t) * (H_evals[i](j) + eta_0)); }
704 Gt_block(t, _, _) = matmul(H_evecs[i], matmul(Gt_temp, nda::transpose(H_evecs[i])));
705 }
706 Gt.set_block(i, Gt_block);
707 }
708
709 return Gt;
710}
