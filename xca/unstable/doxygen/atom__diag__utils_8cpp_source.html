9matrix<double> get_full_h_atomic(const atom_diag::atom_diag<false> &ad) {
10 int dim = ad.get_full_hilbert_space_dim();
11 matrix<double> H_mat = zeros<double>(dim, dim);
12
13 for (int sidx = 0; sidx < ad.n_subspaces(); ++sidx) {
14 auto U = ad.get_unitary_matrix(sidx);
15 auto energies = ad.get_energies()[sidx];
16
17 // Create diagonal energy matrix
18 matrix<double> H_block_diag = zeros<double>(energies.size(), energies.size());
19 for (int i = 0; i < energies.size(); ++i) { H_block_diag(i, i) = energies[i] + ad.get_gs_energy(); }
20
21 // Transform to Fock basis: H_block = U @ H_block_diag @ U^T
22 matrix<double> H_block = U * H_block_diag * transpose(U);
23
24 // Get Fock states for this subspace
25 auto fock_states = ad.get_fock_states(sidx);
26
27 // Copy block into full matrix
28 for (int i = 0; i < fock_states.size(); ++i) {
29 for (int j = 0; j < fock_states.size(); ++j) { H_mat(fock_states[i], fock_states[j]) = H_block(i, j); }
30 }
31 }
32
33 return H_mat;
34}

56matrix<double> get_full_operator_matrix(const atom_diag::atom_diag<false> &ad, int oidx, bool is_creation) {
57 int dim = ad.get_full_hilbert_space_dim();
58 matrix<double> op_mat = zeros<double>(dim, dim);
59
60 for (int s1 = 0; s1 < ad.n_subspaces(); ++s1) {
61 // Get connection to target subspace
62 long s2 = is_creation ? ad.cdag_connection(oidx, s1) : ad.c_connection(oidx, s1);
63 if (s2 < 0) continue;
64
65 // Get matrix block in eigenbasis
66 auto block_mat = is_creation ? ad.cdag_matrix(oidx, s1) : ad.c_matrix(oidx, s1);
67
68 // Transform to Fock basis
69 auto U1 = ad.get_unitary_matrix(s1);
70 auto U2 = ad.get_unitary_matrix(s2);
71 matrix<double> block_mat_fock = U2 * block_mat * transpose(U1);
72
73 // Get Fock states
74 auto f1 = ad.get_fock_states(s1);
75 auto f2 = ad.get_fock_states(s2);
76
77 // Copy block into full matrix
78 for (int i = 0; i < f2.size(); ++i) {
79 for (int j = 0; j < f1.size(); ++j) { op_mat(f2[i], f1[j]) = block_mat_fock(i, j); }
80 }
81 }
82
83 return op_mat;
