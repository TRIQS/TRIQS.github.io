79nda::array<dcomplex, 3> OCA_gf_tpz(nda::array_const_view<dcomplex, 3> hyb_coeffs, nda::array_const_view<dcomplex, 3> hyb_refl_coeffs,

88 auto it_eq = cppdlr::eqptsrel(n_quad + 1);

90 gf_eq = 0;
91 for (int i = 0; i <= n_quad; i++) {
92 hyb_eq(i, _, _) = itops.coefs2eval(hyb_coeffs, it_eq(i));
93 hyb_refl_eq(i, _, _) = itops.coefs2eval(hyb_refl_coeffs, it_eq(i));
94 Gt_eq(i, _, _) = itops.coefs2eval(Gt_coeffs, it_eq(i));
95 }
96
97 double dt = beta / n_quad;

99 nda::array<dcomplex, 2> GFGFGFGF(N, N);
100 for (int fb = 0; fb <= 1; fb++) {
101 auto const &Flams = (fb) ? Fs(_, _, _) : F_dags(_, _, _);
102 auto const &Fnus = (fb) ? F_dags(_, _, _) : Fs(_, _, _);
103 auto const &hyb = (fb) ? hyb_eq : hyb_refl_eq;
104 int sfM = 1; // (fb) ? 1 : -1;
105
106 for (int kap = 0; kap < n; kap++) {
107 for (int mu = 0; mu < n; mu++) {
108 for (int lam = 0; lam < n; lam++) {
109 for (int nu = 0; nu < n; nu++) {
110 for (int i = 1; i <= n_quad - 1; i++) {
111 for (int i1 = 0; i1 <= i; i1++) {
112 for (int i2 = i; i2 <= n_quad; i2++) {
113 double w = 1.0;
114 if (i1 == 0 || i1 == i) w = w / 2;
115 if (i2 == i || i2 == n_quad) w = w / 2;
116 GFGFGFGF = nda::matmul(
117 Gt_eq(n_quad - i2, _, _),
118 nda::matmul(
119 Fnus(nu, _, _),
120 nda::matmul(Gt_eq(i2 - i, _, _),
121 nda::matmul(Fs(mu, _, _),
122 nda::matmul(Gt_eq(i - i1, _, _),
123 nda::matmul(Flams(lam, _, _), nda::matmul(Gt_eq(i1, _, _), F_dags(kap, _, _))))))));
124
125 gf_eq(i, mu, kap) += sfM * w * hyb(i2 - i1, nu, lam) * nda::trace(GFGFGFGF);
126 }
127 }
128 }
129 // std::cout << "gf_eq(_, " << mu << ", " << kap << ") = " << gf_eq(_, mu, kap) << std::endl;
130 }
131 }
132 }
133 }
134 }
135
136 gf_eq = dt * dt * gf_eq;
137
138 return gf_eq;

208nda::array<dcomplex, 3> OCA_gf_dense(nda::array_const_view<dcomplex, 3> hyb_coeffs, nda::array_const_view<dcomplex, 3> hyb_refl_coeffs,
209 nda::vector_const_view<double> hyb_poles, imtime_ops &itops, double beta, nda::array_const_view<dcomplex, 3> Gt,

211
212 nda::vector_const_view<double> dlr_it = itops.get_itnodes();

218 // compute Fbars and Fdagbars
219 auto Fdagbars = nda::array<dcomplex, 4>(n, p, N, N);
220 auto Fbarsrefl = nda::array<dcomplex, 4>(n, p, N, N);
221 for (long lam = 0; lam < n; lam++) {
222 for (long l = 0; l < p; l++) {
223 for (long nu = 0; nu < n; nu++) {
224 Fdagbars(lam, l, _, _) += hyb_coeffs(l, nu, lam) * F_dags(nu, _, _);
225 Fbarsrefl(nu, l, _, _) += hyb_refl_coeffs(l, nu, lam) * Fs(lam, _, _);
226 }
227 }
228 }
229
230 // initialize Green's function
231 nda::array<dcomplex, 3> gf(r, n, n);
232 gf = 0;
233
234 // initialize temporary arrays
235 nda::array<dcomplex, 3> T(r, N, N), U(r, N, N), Tnu(r, N, N);
236
237 // loop over hybridization line directions
238 for (int fb = 0; fb <= 1; fb++) {
239 // fb = 1 for forward line, else = 0
240 auto const &F1 = (fb) ? Fs : F_dags;
241 auto const &Fbar = (fb) ? Fdagbars : Fbarsrefl;
242
243 for (int nu = 0; nu < n; nu++) {
244 Tnu = 0;
245 for (int lam = 0; lam < n; lam++) {
246 for (int l = 0; l < p; l++) {
247 T = 0;
248 U = 0;
249 OCA_gf_dense_right(beta, itops, dlr_it, hyb_poles(l), fb, Gt, F1(lam, _, _), T);
250 OCA_gf_dense_left(beta, itops, dlr_it, hyb_poles(l), fb, Gt, Fbar(lam, l, _, _), U);
251 // std::cout << "dense T(0, :, :) = " << nda::make_regular(nda::real(T(0, _, _))) << std::endl;
252 if (hyb_poles(l) <= 0) {
253 for (int t = 0; t < r; t++) { Tnu(t, _, _) += nda::matmul(U(t, _, _), nda::matmul(Fs(nu, _, _), T(t, _, _))) / k_it(0, -hyb_poles(l)); }
254 } else {
255 for (int t = 0; t < r; t++) { Tnu(t, _, _) += nda::matmul(U(t, _, _), nda::matmul(Fs(nu, _, _), T(t, _, _))) / k_it(0, hyb_poles(l)); }
256 }
257 }
258 }
259 // if (fb == 0) Tnu = -1 * Tnu; // minus sign for backward line
260 for (int kap = 0; kap < n; kap++) {
261 for (int t = 0; t < r; t++) { gf(t, nu, kap) += nda::trace(nda::matmul(Tnu(t, _, _), F_dags(kap, _, _))); }
262 // std::cout << "dense gf(" << nu << ", " << kap << ") = " << gf(_, nu, kap) << std::endl;
263 }
264 }
265 }
266 return gf;
267}

395 // fb = 1 for forward line, else = 0
396 auto const &F1 = (fb) ? Fq.Fs : Fq.F_dags;
397 auto const &Fbar = (fb) ? Fq.F_dag_bars : Fq.F_bars_refl;

400 for (int p_mu = 0; p_mu < q; p_mu++) {
401 for (int b = 0; b < Gt.get_num_block_cols(); b++) {
402 // std::cout << "fb = " << fb << ", b = " << b << std::endl;
403 // backward pass
404 bool path_all_nonzero = true;

406 ind_path(0) = ip;
407 if (ip == -1) {
408 path_all_nonzero = false;
409 } else {

414 } else {

417 ind_path(1) = ip;
418 if (ip == -1 || Gt.get_zero_block_index(ip) == -1) {
419 path_all_nonzero = false;
420 } else {
421 block_dims(3) = Fq.Fs[p_mu].get_block_size(ip, 0);

426 } else {
427 block_dims(4) = Fbar[p_lam].get_block_size(ip, 0);

430 if (ip == -1 || Gt.get_zero_block_index(ip) == -1) { path_all_nonzero = false; }
431 }
432 }
433 }
434 }
435
436 // matmuls
437 if (path_all_nonzero) {
438 for (int mu = 0; mu < Fq.sym_set_sizes(p_mu); mu++) {
439 Tmu = 0;
440 long mu_orb = Fq.sym_set_to_orb(p_mu, mu);
441 for (int lam = 0; lam < Fq.sym_set_sizes(p_lam); lam++) {
442 for (int l = 0; l < p; l++) {
443 T = 0;
444 U = 0;
445 OCA_gf_bs_right(beta, itops, dlr_it, hyb_poles(l), fb, Gt, F1[p_lam], lam, T, ind_path, block_dims);
446 OCA_gf_bs_left(beta, itops, dlr_it, hyb_poles(l), fb, Gt, Fbar[p_lam], lam, l, U, ind_path, block_dims);
447 // std::cout << "bs T(0, :, :) = " << nda::make_regular(nda::real(T(0, range(0, block_dims(2)), range(0, block_dims(1)))))
448 // << std::endl;
449 if (hyb_poles(l) <= 0) {
450 for (int t = 0; t < r; t++) {
451 Tmu(t, range(0, block_dims(4)), range(0, block_dims(1))) +=
452 nda::matmul(
453 U(t, range(0, block_dims(4)), range(0, block_dims(3))),
454 nda::matmul(Fq.Fs[p_mu].get_block(ind_path(1))(mu, _, _), T(t, range(0, block_dims(2)), range(0, block_dims(1)))))
455 / k_it(0, -hyb_poles(l));
456 }
457 } else {
458 for (int t = 0; t < r; t++) {
459 Tmu(t, range(0, block_dims(4)), range(0, block_dims(1))) +=
460 nda::matmul(
461 U(t, range(0, block_dims(4)), range(0, block_dims(3))),
462 nda::matmul(Fq.Fs[p_mu].get_block(ind_path(1))(mu, _, _), T(t, range(0, block_dims(2)), range(0, block_dims(1)))))
463 / k_it(0, hyb_poles(l));
464 }
465 }
466 }
467 }
468 for (int kap = 0; kap < Fq.sym_set_sizes(p_kap); kap++) {
469 long kap_orb = Fq.sym_set_to_orb(p_kap, kap);
470 for (int t = 0; t < r; t++) {
471 gf(t, mu_orb, kap_orb) +=
472 nda::trace(nda::matmul(Tmu(t, range(0, block_dims(4)), range(0, block_dims(1))), Fq.F_dags[p_kap].get_block(b)(kap, _, _)));
473 }
474 }
475 }
476 }
477 }
478 }
479 }
480 }
481 }

483 return gf;
