35void Backbone::set_directions(nda::vector_const_view<int> fb_vec) {
36
37 this->fb = fb_vec;
38 if (m != fb_vec.size()) { throw std::invalid_argument("topology and fb must have same # of vertices"); }
39 for (int i = 0; i < m; i++) {
40 if (topology(i, 0) >= topology(i, 1)) { throw std::invalid_argument("first row of topology must contain smaller-numbered vertices"); }
41 }
42 if (topology(0, 0) != 0) throw std::invalid_argument("topology(0,0) must be 0");
43
44 // set operator flags for each vertex, depending on fb
45 vertices[0].set_bar(false); // operator on vertex 0 has no bar
46 vertices[topology(0, 1)].set_bar(false); // operator on vertex connected to 0 has no bar
47 if (fb_vec(0) == 1) {
48 vertices[0].set_dag(false); // annihilation operator on vertex 0
49 vertices[topology(0, 1)].set_dag(true); // creation operator on vertex connected to 0
50 } else {
51 vertices[0].set_dag(true); // creation operator on vertex 0
52 vertices[topology(0, 1)].set_dag(false); // annihilation operator on vertex connected to 0
53 }
54
55 for (int i = 1; i < m; i++) {
56 vertices[topology(i, 0)].set_bar(false); // operator on vertex i has no bar
57 vertices[topology(i, 1)].set_bar(true); // operator on vertex connected to i has a bar
58 if (fb_vec(i) == 1) {
59 vertices[topology(i, 0)].set_dag(false); // annihilation operator on vertex i
60 vertices[topology(i, 1)].set_dag(true); // creation operator on vertex i
61 } else {
62 vertices[topology(i, 0)].set_dag(true); // creation operator on vertex i
63 vertices[topology(i, 1)].set_dag(false); // annihilation operator on vertex i
64 }
65 }

86void Backbone::set_pole_inds(nda::vector_const_view<int> pole_inds_vec, nda::vector_const_view<double> hyb_poles) {
87
88 this->pole_inds = pole_inds_vec; // values of l, l`, etc.
89 for (int i = 1; i < m; i++) {
90 if (fb(i) == 1) { // line i is forward
91 if (hyb_poles(pole_inds(i - 1)) <= 0) {
92 // step 4(a)
93 // place K^-_l F_nu at the right vertex
94 vertices[topology(i, 0)].set_hyb_ind(i - 1);
95 vertices[topology(i, 0)].set_Ksign(-1);
96 // place K^+_l F^bar^dag_nu_l at the left vertex
97 vertices[topology(i, 1)].set_hyb_ind(i - 1);
98 vertices[topology(i, 1)].set_Ksign(1);
99 // divide by K^-_l(0)
100 prefactor_Ksigns(i - 1) = -1;
101 prefactor_Kexps(i - 1) = 1;
102 } else {
103 // step 4(b)
104 // no K's on vertices, but F bar on left vertex
105 vertices[topology(i, 1)].set_hyb_ind(i - 1);
106 // place K^+_l on each edge between the two vertices
107 for (int j = topology(i, 0); j < topology(i, 1); j++) edges(j, i - 1) = 1;
108 // divide by (K^+_l(0))^(# edges between vertices - 1)
109 prefactor_Ksigns(i - 1) = 1;
110 prefactor_Kexps(i - 1) = topology(i, 1) - topology(i, 0) - 1;
111 }
112 } else { // line i is backward
113 if (hyb_poles(pole_inds(i - 1)) >= 0) {
114 // step 4(a)
115 // place K^+_l F^dag_pi at the right vertex
116 vertices[topology(i, 0)].set_hyb_ind(i - 1);
117 vertices[topology(i, 0)].set_Ksign(1);
118 // place K^-_l F^bar_pi_l at the left vertex
119 vertices[topology(i, 1)].set_hyb_ind(i - 1);
120 vertices[topology(i, 1)].set_Ksign(-1);
121 // divide by -K^-+l(0)
122 prefactor_sign *= -1;
123 prefactor_Ksigns(i - 1) = 1;
124 prefactor_Kexps(i - 1) = 1;
125 } else {
126 // step 4(b)
127 // no K's on vertices, but F bar on left vertex
128 vertices[topology(i, 1)].set_hyb_ind(i - 1);
129 // place K^-_l on each edge between the two vertices
130 for (int j = topology(i, 0); j < topology(i, 1); j++) edges(j, i - 1) = -1;
131 // divide by (-K^-_l(0))^(# edges between vertices - 1)
132 prefactor_sign *= -1;
133 // if ((topology(i,1) - topology(i,0) - 1) % 2 == 1) prefactor_sign *= -1;
134 prefactor_Ksigns(i - 1) = -1;
135 prefactor_Kexps(i - 1) = topology(i, 1) - topology(i, 0) - 1;
136 }
137 }
138 }

174void Backbone::set_orb_inds(int o_ix) {
175 // set orbital indices from a single integer index
176 auto orb_inds_vec = nda::vector<int>(2 * m);
177 orb_inds_vec(0) = -1;
178 orb_inds_vec(topology(0, 1)) = -1; // special vertex 0 and the one connected to it have no orbital indices explicitly summed over
179 for (int i = 1; i < m; i++) { // loop over lines, skipping the one connected to vertex 0
180 orb_inds_vec(topology(i, 0)) = o_ix % n;
181 orb_inds_vec(topology(i, 1)) = o_ix % n;
182 // orbital indices on vertices connected by a line are the same

192void Backbone::set_flat_index(int flat_ix, nda::vector_const_view<double> hyb_poles) {
193 // set directions, pole indices, and orbital indices from a single integer index.
194 // In terms of fb_ix, p_ix, and o_ix,
195 // f_ix = o_ix + n^(m-1) * p_ix + (n * r)^(m-1) * fb_ix, where r is the number of hybridization indices.
196
197 this->f_ix = flat_ix;
198 int p = hyb_poles.size();
199 int o_ix = f_ix % o_ix_max; // orbital indices
200 f_ix /= o_ix_max;
201 int p_ix_max = static_cast<int>(pow(p, m - 1));
202 int p_ix = f_ix % p_ix_max; // pole indices
203 int fb_ix = f_ix / p_ix_max; // directions
204
205 set_directions(fb_ix);
206 set_pole_inds(p_ix, hyb_poles);
207 set_orb_inds(o_ix);
208}
