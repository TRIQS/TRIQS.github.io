87void DiagramEvaluator::multiply_left_vertex(nda::array_view<dcomplex, 3> T_buf, Backbone &backbone, int v_ix, nda::vector_const_view<int> ind_path,
88 nda::vector_const_view<int> block_dims) {
89 int vct0 = backbone.get_topology(0, 1); // vertex connnected to time zero
90 int o_ix = backbone.get_vertex_orb(v_ix); // orbital_index
91 // split backbone orbital index into symmetry set index and orbital index within the symmetry set
92 // i.e. have mapping between backbone orbital index and symmetry set index
93 int q_ix = static_cast<int>(Fq.sym_set_labels(o_ix)); // symmetry set index
94 int qo_ix = static_cast<int>(Fq.sym_set_inds(o_ix)); // index within the symmetry set
95 int l_ix = backbone.get_pole_ind(backbone.get_vertex_hyb_ind(v_ix));
96 int n_col_r = v_ix < vct0 ? block_dims(1) : block_dims(0); // number of columns in T: depends on whether v_ix is before or after vct0
97 int b_ix = ind_path(v_ix - 1); // block index for the vertex v_ix
98
99 // Get the current operator matrix F using the flags and indices of the vertex with index v_ix
100
101 bool has_bar = backbone.has_vertex_bar(v_ix);
102 bool has_dag = backbone.has_vertex_dag(v_ix);
103
104 auto F_selector = [&]() {
105 if (has_bar)
106 return has_dag ? Fq.F_dag_bars[q_ix].get_block(b_ix)(qo_ix, l_ix, _, _) : Fq.F_bars_refl[q_ix].get_block(b_ix)(qo_ix, l_ix, _, _);
107 else
108 return has_dag ? Fq.F_dags[q_ix].get_block(b_ix)(qo_ix, _, _) : Fq.Fs[q_ix].get_block(b_ix)(qo_ix, _, _);
109 };
110
111 nda::array_const_view<dcomplex, 2> F{F_selector()};
112
113 // Get views on temporary storage for input and output
114
115 nda::array_view<dcomplex, 3> T_v = T_buf(_, range(0, block_dims(v_ix)), range(0, n_col_r));
116 nda::array_view<dcomplex, 3> T_vp = T_buf(_, range(0, block_dims(v_ix + 1)), range(0, n_col_r));
117
118 // Multiply with the operator matrix F from the left
119 for (int t = 0; t < r; t++) T_vp(t, _, _) = matmul(F, T_v(t, _, _));
120
121 // Multiply with scalar K(\tau) factor of the pole with index l_ix
122 hyb.multiply_kernel_on_vertex(T_vp, backbone, v_ix, l_ix);
123}

126 nda::vector_const_view<int> block_dims) {
127 int b_ix = ind_path(e_ix); // block index for the edge e_ix
128 int vct0 = backbone.get_topology(0, 1);
129 int n_col_r = e_ix < vct0 ? block_dims(1) : block_dims(0);
130
131 nda::array_view<dcomplex, 3> GKt_ep = GKt(_, range(0, block_dims(e_ix + 1)), range(0, block_dims(e_ix + 1)));
132 GKt_ep = Gt.get_block(b_ix);
133
134 hyb.multiply_kernels_on_edge(GKt_ep, backbone, e_ix);

147void DiagramEvaluator::multiply_left_vertex_and_right_zero_vertex(nda::array_view<dcomplex, 3> T_buf, Backbone &backbone, bool is_forward, int b_ix_0,
148 int p_kap, int p_mu, nda::vector_const_view<int> ind_path,
149 nda::vector_const_view<int> block_dims) {
150 int v_ix = backbone.get_topology(0, 1);
151 int b_ix_mu = ind_path(v_ix - 1); // block index for F_mu
152
153 nda::array_view<dcomplex, 3> T_v = T_buf(_, range(0, block_dims(v_ix)), range(0, block_dims(1)));
154 nda::array_view<dcomplex, 3> T_vp = T_buf(_, range(0, block_dims(v_ix + 1)), range(0, block_dims(0)));
155
156 nda::array_view<dcomplex, 4> Tkaps_v = Tkaps(_, _, range(0, block_dims(v_ix)), range(0, block_dims(0)));
157 nda::array_view<dcomplex, 3> Tmu_v = Tmu(_, range(0, block_dims(v_ix)), range(0, block_dims(0)));
158
159 auto F_selector = [&](bool is_forward, int b_ix_ix, int p_ix, int ix) {
160 return is_forward ? Fq.Fs[p_ix].get_block(b_ix_ix)(ix, _, _) : Fq.F_dags[p_ix].get_block(b_ix_ix)(ix, _, _);
161 };
162
163 // Apply F_kap operator at tau = 0 from the right
164 for (int kap = 0; kap < Fq.sym_set_sizes(p_kap); kap++) {
165 auto F_kap = F_selector(is_forward, b_ix_0, p_kap, kap);
166 for (int t = 0; t < r; t++) Tkaps_v(kap, t, _, _) = matmul(T_v(t, _, _), F_kap);
167 }
168
169 T_vp = 0; // With intermediate result now in Tkaps, reuse T for final result
170 int hyb_sign = is_forward ? +1. : -1.;
171
172 for (int mu = 0; mu < Fq.sym_set_sizes(p_mu); mu++) {
173 Tmu_v = 0;
174 // Multiply with hybridization function
175 for (int kap = 0; kap < Fq.sym_set_sizes(p_kap); kap++) {
176 nda::array_const_view<dcomplex, 1> hyb_oo = is_forward ?
177 hyb.values(_, Fq.sym_set_to_orb(p_mu, mu), Fq.sym_set_to_orb(p_kap, kap)) :
178 hyb.values_reflect(_, Fq.sym_set_to_orb(p_mu, mu), Fq.sym_set_to_orb(p_kap, kap));
179 for (int t = 0; t < r; t++) Tmu_v(t, _, _) += hyb_sign * hyb_oo(t) * Tkaps_v(kap, t, _, _);
180 }
181 // Apply F_mu operator from the left at vertex connected to tau = 0
182 auto F_mu = F_selector(!is_forward, b_ix_mu, p_mu, mu);
183 for (int t = 0; t < r; t++) T_vp(t, _, _) += matmul(F_mu, Tmu_v(t, _, _));
184 }

187void DiagramEvaluator::find_path_self_energy(BlockDiagOpFun &Gt, Backbone &backbone, int f_ix, nda::vector_view<int> ind_path, nda::vector_view<int> block_dims) {
188
189 int m = backbone.m; // Diagram order
190 int vct0 = backbone.get_topology(0, 1); // vertex connected to zero
191
192 backbone.set_flat_index(f_ix, hyb.poles); // set directions, pole indices, and orbital indices from a single integer index
193
194 /* Example of block_dims for m = 2 (OCA): each number is an index of block_dims, and each square represents a block of a matrix
195 3 3 2 2 1 1 0
196 -------- -------- -------- -------- -------- -------- --------
197 4 | F | 3 | G | 3 | F | 2 | G | 2 | F | 1 | G | 1 | F |
198 | | | | | | | | | | | | | |
199 -------- -------- -------- -------- -------- -------- --------
200 */
201
202 auto F_selector_p_ix = [&](int w, int p_ix) { return backbone.has_vertex_dag(w) ? Fq.F_dags[p_ix] : Fq.Fs[p_ix]; };
203 auto F_selector = [&](int w) { return F_selector_p_ix(w, Fq.sym_set_labels(backbone.get_orb_ind(w))); };
204
205 bool incomplete_path = false;
206 auto is_path_incomplete = [&](int w, int ip) { return (ip == -1 || (w < 2 * m - 1 && Gt.get_zero_block_index(ip) == -1)); };
207
208 for (int b_ix = 0; b_ix < Gt.get_num_block_cols(); b_ix++) { // loop over blocks of self-energy
209
210 for (int p_kap = 0; p_kap < q; p_kap++) { // loop over symmetry sets on the zero vertex
211
212 // traverse factors in two halves
213 // -----------------------------------------------------------------------------------
214 // first half: all vertices before vertex connected to zero
215
216 // ind_path setup 1
217
218 int ip = -1; // running block index
219
220 for (int w = 0; w < vct0; w++) {
221 ip = (w == 0) ? F_selector_p_ix(w, p_kap).get_block_index(b_ix) : F_selector(w).get_block_index(ip);
222 incomplete_path = is_path_incomplete(w, ip);
223 if (incomplete_path) break;
224 ind_path(w) = ip;
225 }
226
227 if (incomplete_path) continue;
228
229 // block_dims setup 1
230
231 block_dims(0) = F_selector_p_ix(0, p_kap).get_block_size(b_ix, 1);
232 block_dims(1) = F_selector_p_ix(0, p_kap).get_block_size(b_ix, 0);
233
234 for (int w = 0; w < vct0 - 1; w++) { block_dims(w + 2) = F_selector(w + 1).get_block_size(ind_path(w), 0); }
235
236 // -----------------------------------------------------------------------------------
237 // second half: vertex connected to zero and above
238
239 for (int p_mu = 0; p_mu < q; p_mu++) { // loop over symmetry sets on the vertex connected to vertex 0?
240
241 // ind_path setup 2
242
243 ip = ind_path(vct0 - 1);
244
245 for (int w = vct0; w < 2 * m; w++) {
246 ip = (w == vct0) ? F_selector_p_ix(w, p_mu).get_block_index(ip) : F_selector(w).get_block_index(ip);
247 incomplete_path = is_path_incomplete(w, ip);
248 if (incomplete_path) break;
249 if (w < 2 * m - 1) ind_path(w) = ip;
250 }
251
252 if (incomplete_path) continue;
253
254 // block_dims setup 2
255
256 block_dims(vct0 + 1) = F_selector_p_ix(vct0, p_mu).get_block_size(ind_path(vct0 - 1), 0);
257
258 for (int w = vct0; w < 2 * m - 1; w++) { block_dims(w + 2) = F_selector(w + 1).get_block_size(ind_path(w), 0); }
259
260 // evaluate the diagram with these directions, poles, and orbital indices
261 // b_ix is the block index for the first edge
262 eval_self_energy_fixed_indices(Gt, backbone, b_ix, p_kap, p_mu, ind_path, block_dims);
263 }
264 }
265 }
266 backbone.reset_all_inds(); // reset directions, pole indices, and orbital indices for the next iteration
267}

269void DiagramEvaluator::eval_self_energy(BlockDiagOpFun &Gt, Backbone &backbone, int f_ix) {
270 int m = backbone.m;
271 int f_ix_max = static_cast<int>(backbone.fb_ix_max * backbone.o_ix_max * pow(hyb.poles.size(), m - 1));
272 if (f_ix < 0 || f_ix >= f_ix_max) { throw std::runtime_error("DiagramEvaluator::eval_self_energy: f_ix out of range"); }
273
274 nda::vector<int> ind_path(2 * m - 1); // tracks block indices of factors for computing a particular block of the self-energy
275 nda::vector<int> block_dims(2 * m + 1); // tracks the dimensions of the blocks in these factors
276
277 find_path_self_energy(Gt, backbone, f_ix, ind_path, block_dims);
278 Sigma.set_zero_block_indices(); // set zero_block_indices according to current blocks
279}

281void DiagramEvaluator::eval_self_energy(BlockDiagOpFun &Gt, Backbone &backbone) {
282 int m = backbone.m;
283 nda::vector<int> ind_path(2 * m - 1); // tracks block indices of factors for computing a particular block of the self-energy
284 nda::vector<int> block_dims(2 * m + 1); // tracks the dimensions of the blocks in these factors
285
286 // loop over all flat indices
287 int f_ix_max = static_cast<int>(backbone.fb_ix_max * backbone.o_ix_max * pow(hyb.poles.size(), m - 1));
288 for (int f_ix = 0; f_ix < f_ix_max; f_ix++) { find_path_self_energy(Gt, backbone, f_ix, ind_path, block_dims); }
289 Sigma.set_zero_block_indices(); // set zero_block_indices according to current blocks
290}

292void DiagramEvaluator::eval_self_energy_fixed_indices(BlockDiagOpFun &Gt, Backbone &backbone, int b_ix, int p_kap, int p_mu, nda::vector_const_view<int> ind_path,
293 nda::vector_const_view<int> block_dims) {
294
295 int m = backbone.m, vct0 = backbone.get_topology(0, 1); // vertex connnected to time zero
296
297 T(_, range(0, block_dims(1)), range(0, block_dims(1))) = Gt.get_block(ind_path(0));
298 for (int v = 1; v < vct0; v++) {
299 multiply_left_vertex(T, backbone, v, ind_path, block_dims);
300 integrate_left_edge(T, Gt, backbone, v, ind_path, block_dims);
301 }
302
303 multiply_left_vertex_and_right_zero_vertex(T, backbone, (not backbone.has_vertex_dag(0)), b_ix, p_kap, p_mu, ind_path, block_dims);
304
305 for (int v = vct0 + 1; v < 2 * m; v++) {
306 integrate_left_edge(T, Gt, backbone, v - 1, ind_path, block_dims);
307 multiply_left_vertex(T, backbone, v, ind_path, block_dims);
308 }
309
310 multiply_prefactor(T, backbone);
311
312 int diag_order_sign = (m % 2 == 0) ? -1 : 1;
313 if (backbone.get_fb(0) == 0) diag_order_sign *= -1;
314
315 nda::array_view<dcomplex, 3> T_out = T(_, range(0, block_dims(2 * m)), range(0, block_dims(0)));
316 T_out *= diag_order_sign * backbone.prefactor_sign;
317
318 // TODO: temporary fix: have backward pass consider sparsity of hybridization function (hyb) during zero vertex
319 if (nda::max_element(nda::abs(T_out)) > 1e-16) Sigma.add_block(b_ix, T_out);
320}

423void DiagramEvaluator::integrate_right_edge(nda::array_view<dcomplex, 3> U_buf, BlockDiagOpFun &Gt, CorrelatorBackbone &backbone, int e_ix,
424 nda::vector_const_view<int> ind_path, nda::vector_const_view<int> block_dims) {
425
426 int b_ix = ind_path(e_ix); // block index for the edge e_ix
427 int n_row_l = block_dims(2 * backbone.m); // number of rows for the left-hand side of the diagram
428
429 nda::array_view<dcomplex, 3> GKt_ep = GKt(_, range(0, block_dims(e_ix + 1)), range(0, block_dims(e_ix + 1)));
430
431 GKt_ep = Gt.get_block(b_ix);
432
433 hyb.multiply_kernels_on_edge(GKt_ep, backbone, e_ix);
434
435 nda::array_view<dcomplex, 3> U_e = U_buf(_, range(0, n_row_l), range(0, block_dims(e_ix + 1)));
436
437 U_e = itops.convolve(beta, itops.vals2coefs(U_e), itops.vals2coefs(GKt_ep), cppdlr::TIME_ORDERED);
438}

451nda::array<dcomplex, 3> DiagramEvaluator::eval_correlator(BlockDiagOpFun &Gt, CorrelatorBackbone &backbone, std::vector<BlockOp> mu_ops, std::vector<BlockOp> kap_ops,
452 int f_ix) {
453 int m = backbone.m;
454 int vct0 = backbone.get_topology(0, 1);
455
456 nda::vector<int> ind_path(2 * m); // tracks block indices of factors for computing a particular block's contribution to the correlator
457 nda::vector<int> block_dims(2 * m + 1); // tracks the dimensions of the blocks in these factors
458
459 nda::array<dcomplex, 3> correlator = nda::zeros<dcomplex>(r, mu_ops.size(), kap_ops.size());
460 nda::array<dcomplex, 3> Tmuop = nda::zeros<dcomplex>(r, Nmax, Nmax);
461
462 // blocks and block indices for intermediate storage
463 std::vector<nda::array<dcomplex, 3>> right(Gt.get_num_block_cols(), nda::array<dcomplex, 3>(r, Nmax, Nmax));
464 std::vector<nda::array<dcomplex, 3>> left(Gt.get_num_block_cols(), nda::array<dcomplex, 3>(r, Nmax, Nmax));
465
466 auto right_inds = nda::vector<int>(Gt.get_num_block_cols());
467 auto left_inds = nda::vector<int>(Gt.get_num_block_cols());
468
469 nda::vector<int> ind_path_end(3), block_dims_end(4);
470
471 backbone.set_flat_index(f_ix, hyb.poles); // set directions, pole indices, and orbital indices from a single integer index
472
473 /* Example of block_dims for m = 2 (OCA): each number is an index of block_dims, and each square represents a block of a matrix
474 4 3 3 2 2 1 1 0
475 -------- -------- -------- -------- -------- -------- -------- --------
476 4 | G | 4 | F | 3 | G | 3 | F | 2 | G | 2 | F | 1 | G | 1 | F |
477 | | | | | | | | | | | | | | | |
478 -------- -------- -------- -------- -------- -------- -------- --------
479 */
480
481 auto F_selector_p_ix = [&](int w, int p_ix) { return backbone.has_vertex_dag(w) ? Fq.F_dags[p_ix] : Fq.Fs[p_ix]; };
482 auto F_selector = [&](int w) { return F_selector_p_ix(w, Fq.sym_set_labels(backbone.get_orb_ind(w))); };
483
484 auto is_path_incomplete = [&](int ip) { return (ip == -1 || Gt.get_zero_block_index(ip) == -1); };
485
486 bool incomplete_path = false;
487
488 right_inds = -1;
489 left_inds = -1;
490
491 // -- Right-hand side of diagram, with vertices on [tau, 0]
492
493 for (int b_ix = 0; b_ix < Gt.get_num_block_cols(); ++b_ix) { // loop over blocks of right-hand side of diagram
494
495 // ind_path setup (right)
496
497 int ip = b_ix;
498 ind_path(0) = ip;
499
500 for(int w = 1; w < vct0; w++) {
501 ip = F_selector(w).get_block_index(ip);
502 incomplete_path = is_path_incomplete(ip);
503 if (incomplete_path) break;
504 ind_path(w) = ip;
505 }
506
507 if (incomplete_path) continue;
508
509 // block_dims setup (right)
510
511 // block_dims(0) = ?? // Not initalized, is block_dims(0) unused ??
512 block_dims(1) = Gt.get_block_size(b_ix);
513 for(int w = 1; w < vct0; w++) block_dims(w + 1) = F_selector(w).get_block_size(ind_path(w - 1), 0);
514
515 // Evaluate: right-hand side edges and vertices (to buffer T)
516
517 T(_, range(0, block_dims(1)), range(0, block_dims(1))) = Gt.get_block(ind_path(0)); // first edge at 0
518
519 for (int v = 1; v < vct0; v++) {
520 multiply_left_vertex(T, backbone, v, ind_path, block_dims);
521 integrate_left_edge(T, Gt, backbone, v, ind_path, block_dims);
522 }
523 multiply_prefactor(T, backbone);
524
525 nda::array_view<dcomplex, 3> T_out = T(_, range(0, block_dims(vct0)), range(0, block_dims(1)));
526 T_out *= backbone.prefactor_sign; // include sign from diag order and prefactor
527
528 // Store result in T to the "right" buffer
529 right[b_ix] = T_out;
530 right_inds(b_ix) = ind_path(vct0 - 1);
531 }
532
533 // -- Left-hand side of diagram, with vertices on [beta, tau]
534
535 for (int b_ix = 0; b_ix < Gt.get_num_block_cols(); ++b_ix) {
536
537 // ind_path setup (left)
538
539 int ip = b_ix;
540 ind_path(vct0) = ip; // store block index for vertex connected to vertex 0
541
542 for(int w = vct0 + 1; w < 2 * m; w++) {
543 ip = F_selector(w).get_block_index(ip);
544 incomplete_path = is_path_incomplete(ip);
545 if (incomplete_path) break;
546 ind_path(w) = ip;
547 }
548
549 if (incomplete_path) continue;
550
551 // block_dims setup (left)
552
553 block_dims(vct0 + 1) = Gt.get_block_size(b_ix);
554 for(int w = vct0 + 1; w < 2 * m; w++) block_dims(w + 1) = F_selector(w).get_block_size(ind_path(w - 1), 0);
555
556 // Evaluate: left-hand side edges and vertices (to buffer U) going from beta to tau
557
558 nda::array_view<dcomplex, 3> U_beta = U(_, range(0, block_dims(2 * m)), range(0, block_dims(2 * m)));
559
560 U_beta = Gt.get_block(ind_path(2 * m - 1));
561
562 for (int v = 2 * m - 1; v > vct0; v--) {
563 multiply_right_vertex(U, backbone, v, ind_path, block_dims);
564 integrate_right_edge(U, Gt, backbone, v - 1, ind_path, block_dims);
565 }
566
567 U = itops.reflect(U); // Reflect result to account for outer [beta, tau] integral order
568
569 // Store result in U to the "left" buffer
570 left[ind_path(vct0)] = U(_, range(0, block_dims(2 * m)), range(0, block_dims(vct0 + 1)));
571 left_inds(b_ix) = ind_path(2 * m - 1);
572 }
573
574 for (int mu = 0; mu < mu_ops.size(); ++mu) {
575 for (int b_ix = 0; b_ix < Gt.get_num_block_cols(); ++b_ix) { // loop over blocks of right product
576
577 // setup ind_path_end
578
579 int ip = right_inds(b_ix);
580 if (is_path_incomplete(ip)) continue;
581 ind_path_end(0) = ip;
582
583 ip = mu_ops[mu].get_block_index(ip);
584 if (is_path_incomplete(ip)) continue;
585 ind_path_end(1) = ip;
586
587 ip = left_inds(ip);
588 if (is_path_incomplete(ip)) continue;
589 ind_path_end(2) = ip;
590
591 // setup block_dims_end
592
593 block_dims_end(0) = right[b_ix].shape(2); // Is this ever different from Nmax?
594 block_dims_end(1) = right[b_ix].shape(1); // Is this ever different from Nmax?
595 block_dims_end(2) = mu_ops[mu].get_block_size(ind_path_end(0), 0);
596 block_dims_end(3) = left[ind_path_end(1)].shape(1);
597
598 // Evaluate: product trace, correlator = Tr[left O_mu right O_kap]
599
600 nda::array_const_view<dcomplex, 3> left_b = left[ind_path_end(1)](_, range(0, block_dims_end(3)), range(0, block_dims_end(2)));
601 nda::array_const_view<dcomplex, 2> O_mu_b = mu_ops[mu].get_block(ind_path_end(0));
602 nda::array_const_view<dcomplex, 3> right_b = right[b_ix](_, range(0, block_dims_end(1)), range(0, block_dims_end(0)));
603
604 nda::array_view<dcomplex, 3> Tmuop_b = Tmuop(_, range(0, block_dims_end(3)), range(0, block_dims_end(0)));
605
606 for (int t = 0; t < r; ++t) Tmuop_b(t, _, _) = matmul(left_b(t, _, _), matmul(O_mu_b, right_b(t, _, _)));
607
608 for (int kap = 0; kap < kap_ops.size(); ++kap) {
609 for (int c_ix = 0; c_ix < Gt.get_num_block_cols(); ++c_ix) { // Can this loop be removed using c_ix = ind_path_end(2) ?
610 if (c_ix == ind_path_end(2) && kap_ops[kap].get_block_index(c_ix) == b_ix) {
611
612 nda::array_const_view<dcomplex, 2> O_kap_b = kap_ops[kap].get_block(c_ix);
613
614 for (int t = 0; t < r; ++t) correlator(t, mu, kap) += trace(matmul(Tmuop_b(t, _, _), O_kap_b));
615
616 }
617 }
618 }
619
620 }
621 }
622
623 backbone.reset_all_inds(); // reset directions, pole indices, and orbital indices for the next iteration
624
625 return correlator;
626}
