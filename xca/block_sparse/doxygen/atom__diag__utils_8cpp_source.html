7 template <bool IsComplex>
8 nda::matrix<dcomplex> get_full_operator_matrix_impl(const triqs_atom_diag_t<IsComplex> &ad, int oidx, bool is_creation) {
9 int dim = ad.get_full_hilbert_space_dim();
10 nda::matrix<dcomplex> op_mat = nda::zeros<dcomplex>(dim, dim);
11
12 for (int s1 = 0; s1 < ad.n_subspaces(); ++s1) {
13 // Get connection to target subspace
14 long s2 = is_creation ? ad.cdag_connection(oidx, s1) : ad.c_connection(oidx, s1);
15 if (s2 < 0) continue;
16
17 // Get matrix block in eigenbasis
18 auto block_mat = is_creation ? ad.cdag_matrix(oidx, s1) : ad.c_matrix(oidx, s1);
19
20 // Transform to Fock basis
21 auto U1 = ad.get_unitary_matrix(s1);
22 auto U2 = ad.get_unitary_matrix(s2);
23 nda::matrix<dcomplex> block_mat_fock = U2 * block_mat * nda::conj(nda::transpose(U1));
24
25 // Get Fock states
26 auto f1 = ad.get_fock_states(s1);
27 auto f2 = ad.get_fock_states(s2);
28
29 // Copy block into full matrix
30 for (int i = 0; i < f2.size(); ++i) {
31 for (int j = 0; j < f1.size(); ++j) { op_mat(f2[i], f1[j]) = block_mat_fock(i, j); }
32 }
33 }
34
35 return op_mat;
36 }

39 std::tuple<BlockOpSymQuartet, nda::vector<int>> get_operators_impl(const triqs_atom_diag_t<IsComplex> &ad,
40 nda::array_const_view<dcomplex, 3> hyb_coeffs) {
41
42 // Find like rows of c_connection (resp. cdag_connection), which correspond with annihilation (resp. creation) operators that have the same
43 // sparsity pattern
44 int n = hyb_coeffs.extent(1);
45 nda::vector<long> sym_set_labels(n);
46 sym_set_labels = 0;
47 int counter = 0;
48 for (int oidx = 0; oidx < n; ++oidx) { // fill each entry of sym_set_labels
49 bool found_match = false;
50 for (int oidx2 = 0; oidx2 < oidx; ++oidx2) { // compare full c_connection row against all previous operators
51 bool same = true;
52 for (int s = 0; s < ad.n_subspaces(); ++s) {
53 if (ad.c_connection(oidx, s) != ad.c_connection(oidx2, s)) {
54 same = false;
55 break;
56 }
57 }
58 if (same) {
59 sym_set_labels(oidx) = sym_set_labels(oidx2);
60 found_match = true;
61 break;
62 }
63 }
64 if (not found_match) { // no matching operator found, so create new group
65 sym_set_labels(oidx) = counter;
66 counter = counter + 1;
67 }
68 }
69 std::set<int> unique_groups(sym_set_labels.begin(), sym_set_labels.end());
70 std::size_t num_sym_sets = unique_groups.size();
71
72 // First pass: count how many operators belong to each symmetry group
73 std::vector<int> ops_per_group(num_sym_sets, 0);
74 for (int oidx = 0; oidx < n; ++oidx) { ops_per_group[sym_set_labels[oidx]]++; }
75
76 // Initialize operator blocks grouped by symmetry with proper dimensions
77 std::vector<std::vector<nda::array<dcomplex, 3>>> c_blocks(num_sym_sets);
78 std::vector<std::vector<nda::array<dcomplex, 3>>> cdag_blocks(num_sym_sets);
79
80 // Initialize arrays for each symmetry group and subspace
81 for (int gidx = 0; gidx < num_sym_sets; ++gidx) {
82 c_blocks[gidx].resize(ad.n_subspaces());
83 cdag_blocks[gidx].resize(ad.n_subspaces());
84
85 for (int sidx = 0; sidx < ad.n_subspaces(); ++sidx) {
86 // Determine dimensions for this subspace
87 long cidx = -1, didx = -1;
88 int dim_c_final = 0, dim_c_initial = 0;
89 int dim_cdag_final = 0, dim_cdag_initial = 0;
90
91 // Find a representative operator from this symmetry group to get dimensions
92 for (int oidx = 0; oidx < n; ++oidx) {
93 if (sym_set_labels[oidx] == gidx) {
94 if (cidx == -1) {
95 cidx = ad.c_connection(oidx, sidx);
96 if (cidx >= 0) {
97 dim_c_final = ad.get_fock_states(cidx).size();
98 dim_c_initial = ad.get_fock_states(sidx).size();
99 }
100 }
101 if (didx == -1) {
102 didx = ad.cdag_connection(oidx, sidx);
103 if (didx >= 0) {
104 dim_cdag_final = ad.get_fock_states(didx).size();
105 dim_cdag_initial = ad.get_fock_states(sidx).size();
106 }
107 }
108 if (cidx >= 0 && didx >= 0) break;
109 }
110 }
111
112 // Initialize arrays with proper dimensions
113 if (cidx >= 0) { c_blocks[gidx][sidx] = nda::zeros<dcomplex>(ops_per_group[gidx], dim_c_final, dim_c_initial); }
114 if (didx >= 0) { cdag_blocks[gidx][sidx] = nda::zeros<dcomplex>(ops_per_group[gidx], dim_cdag_final, dim_cdag_initial); }
115 }
116 }
117
118 // Second pass: fill the arrays
119 std::vector<int> op_count_per_group(num_sym_sets, 0);
120
121 for (int oidx = 0; oidx < n; ++oidx) {
122 int gidx = sym_set_labels[oidx];
123 int op_idx_in_group = op_count_per_group[gidx];
124
125 for (int sidx = 0; sidx < ad.n_subspaces(); ++sidx) {
126 // Handle annihilation operator
127 long cidx = ad.c_connection(oidx, sidx);
128 if (cidx >= 0) {
129 auto fock_final = ad.get_fock_states(cidx);
130 auto fock_initial = ad.get_fock_states(sidx);
131
132 // Get full operator matrix and extract block
133 auto c_full = get_full_operator_matrix_impl(ad, oidx, false);
134 for (int i = 0; i < fock_final.size(); ++i) {
135 for (int j = 0; j < fock_initial.size(); ++j) { c_blocks[gidx][sidx](op_idx_in_group, i, j) = c_full(fock_final[i], fock_initial[j]); }
136 }
137 }
138
139 // Handle creation operator
140 long didx = ad.cdag_connection(oidx, sidx);
141 if (didx >= 0) {
142 auto fock_final = ad.get_fock_states(didx);
143 auto fock_initial = ad.get_fock_states(sidx);
144
145 // Get full operator matrix and extract block
146 auto cdag_full = get_full_operator_matrix_impl(ad, oidx, true);
147 for (int i = 0; i < fock_final.size(); ++i) {
148 for (int j = 0; j < fock_initial.size(); ++j) {
149 cdag_blocks[gidx][sidx](op_idx_in_group, i, j) = cdag_full(fock_final[i], fock_initial[j]);
150 }
151 }
152 }
153 }
154
155 op_count_per_group[gidx]++;
156 }
157
158 // Fill in BlockOpSymSet objects
159 nda::array<int, 2> F_block_inds = nda::zeros<int>(num_sym_sets, ad.n_subspaces()),
160 F_dag_block_inds = nda::zeros<int>(num_sym_sets, ad.n_subspaces());
161 auto filled_F_block_inds = nda::zeros<int>(n);
162 for (int i = 0; i < n; i++) {
163 long label = sym_set_labels(i);
164 if (filled_F_block_inds(label) == 0) {
165 for (int j = 0; j < ad.n_subspaces(); ++j) { F_block_inds(label, j) = ad.c_connection(i, j); }
166 filled_F_block_inds(label) = 1;
167 }
168 }
169 auto filled_F_dag_block_inds = nda::zeros<int>(n);
170 for (int i = 0; i < n; i++) {
171 long label = sym_set_labels(i);
172 if (filled_F_dag_block_inds(label) == 0) {
173 for (int j = 0; j < ad.n_subspaces(); ++j) { F_dag_block_inds(label, j) = ad.cdag_connection(i, j); }
174 filled_F_dag_block_inds(label) = 1;
175 }
176 }
177
178 std::vector<BlockOpSymSet> F_sym_vec, F_dag_sym_vec;
179 for (int i = 0; i < num_sym_sets; i++) {
180 F_sym_vec.emplace_back(F_block_inds(i, cppdlr::_), c_blocks[i]);
181 F_dag_sym_vec.emplace_back(F_dag_block_inds(i, cppdlr::_), cdag_blocks[i]);
182 }
183 BlockOpSymQuartet Fq(F_sym_vec, F_dag_sym_vec, hyb_coeffs, sym_set_labels);
184 return std::make_tuple(Fq, sym_set_labels);
185 }

211 nda::matrix<dcomplex> get_full_h_atomic(const triqs_atom_diag &ad) {
212 int dim = ad.get_full_hilbert_space_dim();
213 auto H_mat = nda::zeros<dcomplex>(dim, dim);
214
215 for (int sidx = 0; sidx < ad.n_subspaces(); ++sidx) {
216 auto U = ad.get_unitary_matrix(sidx);
217 auto energies = ad.get_energies()[sidx];
218
219 // Create diagonal energy matrix
220 nda::matrix<dcomplex> H_block_diag = nda::zeros<dcomplex>(energies.size(), energies.size());
221 for (int i = 0; i < energies.size(); ++i) { H_block_diag(i, i) = energies[i] + ad.get_gs_energy(); }
222
223 // Transform to Fock basis: H_block = U @ H_block_diag @ U^T
224 nda::matrix<dcomplex> H_block = U * H_block_diag * nda::transpose(U);
225
226 // Get Fock states for this subspace
227 auto fock_states = ad.get_fock_states(sidx);
228
229 // Copy block into full matrix
230 for (int i = 0; i < fock_states.size(); ++i) {
231 for (int j = 0; j < fock_states.size(); ++j) { H_mat(fock_states[i], fock_states[j]) = H_block(i, j); }
232 }
233 }
234
235 return H_mat;
236 }

246 std::tuple<std::vector<nda::array<dcomplex, 2>>, nda::vector<long>> get_hamiltonian_blocks(const triqs_atom_diag &ad) {
247 // Get full Hamiltonian matrix
248 auto H_mat = get_full_h_atomic(ad);
249
250 // Create permutation based on Fock state ordering
251 std::vector<unsigned long> H_perm;
252 std::vector<nda::array<dcomplex, 2>> H_blocks;
253 nda::vector<long> H_block_inds(ad.n_subspaces());
254
255 for (int s = 0; s < ad.n_subspaces(); ++s) {
256 auto fock_states = ad.get_fock_states(s);
257 for (auto state : fock_states) { H_perm.push_back(state); }
258
259 // Extract block from full matrix
260 nda::array<dcomplex, 2> H_block = nda::zeros<dcomplex>(fock_states.size(), fock_states.size());
261 for (int i = 0; i < fock_states.size(); ++i) {
262 for (int j = 0; j < fock_states.size(); ++j) { H_block(i, j) = H_mat(fock_states[i], fock_states[j]); }
263 }
264 H_blocks.push_back(H_block);
265
266 // Check if block is zero
267 double max_elem = 0.0;
268 for (int i = 0; i < H_block.extent(0); ++i) {
269 for (int j = 0; j < H_block.extent(1); ++j) { max_elem = std::max(max_elem, std::abs(H_block(i, j))); }
270 }
271 H_block_inds(s) = (max_elem < 1e-16) ? -1 : 0;
272 }
273 return std::make_tuple(H_blocks, H_block_inds);
274 }

276 template <typename T>
277 std::vector<nda::array<T, 3>> H_to_atom_prop_blocks(std::vector<nda::array<dcomplex, 2>> &H_blocks, nda::vector_const_view<long> H_block_inds,
278 double beta, imtime_ops &itops) {
279 int r = itops.rank();
280 dcomplex tr_exp_minusbetaH = 0;
281 std::vector<nda::array<dcomplex, 1>> H_evals(H_blocks.size());
282 std::vector<nda::array<dcomplex, 2>> H_evecs(H_blocks.size());
283 for (int i = 0; i < H_block_inds.size(); ++i) {
284 if (H_block_inds(i) != -1) {
285 if (H_blocks[i].extent(0) == 1) {
286 H_evals[i] = nda::array<dcomplex, 1>{H_blocks[i](0, 0)};
287 H_evecs[i] = nda::array<dcomplex, 2>{{1}};
288 } else {
289 auto H_block_eig = nda::linalg::eigh(H_blocks[i]);
290 H_evals[i] = std::get<0>(H_block_eig);
291 H_evecs[i] = std::get<1>(H_block_eig);
292 }
293 tr_exp_minusbetaH += nda::sum(exp(-beta * H_evals[i]));
294 } else {
295 H_evals[i] = nda::zeros<dcomplex>(H_blocks[i].extent(0));
296 H_evecs[i] = nda::eye<dcomplex>(H_blocks[i].extent(0));
297 tr_exp_minusbetaH += 1.0 * H_blocks[i].extent(0); // 0 entry in the diagonal
298 }
299 }
300
301 auto eta_0 = nda::log(tr_exp_minusbetaH) / beta;
302 auto dlr_it = itops.get_itnodes();
303 auto dlr_it_abs = cppdlr::rel2abs(dlr_it);
304 std::vector<nda::array<T, 3>> ap_blocks(H_block_inds.size());
305 for (int i = 0; i < H_block_inds.size(); ++i) {
306 ap_blocks[i] = nda::array<T, 3>(r, H_blocks[i].extent(0), H_blocks[i].extent(1));
307 auto Gt_temp = nda::make_regular(0 * H_blocks[i]);
308 for (int t = 0; t < r; t++) {
309 for (int j = 0; j < H_blocks[i].extent(0); j++) { Gt_temp(j, j) = -exp(-beta * dlr_it_abs(t) * (H_evals[i](j) + eta_0)); }
310 ap_blocks[i](t, _, _) = matmul(H_evecs[i], matmul(Gt_temp, nda::transpose(H_evecs[i])));
311 }
312 }
313
314 return ap_blocks;
315 }

373 nda::array<dcomplex, 3> get_tensor_in_atom_diag_subspace(nda::array_const_view<dcomplex, 3> tensor_full, int subspace_index,
374 triqs_atom_diag const &ad) {
375 // Permute a tensor from the full Hilbert space to the Fock state ordered basis
376
377 int r = tensor_full.extent(0);
378
379 std::vector<unsigned long> H_perm;
380 auto fock_states = ad.get_fock_states(subspace_index);
381 for (auto state : fock_states) H_perm.push_back(state);
382
383 int N_sub = fock_states.size();
384 nda::array<dcomplex, 3> tensor_subspace = nda::zeros<dcomplex>(r, N_sub, N_sub);
385
386 for (int t = 0; t < r; ++t) {
387 for (int i = 0; i < N_sub; ++i) {
388 for (int j = 0; j < N_sub; ++j) { tensor_subspace(t, i, j) = tensor_full(t, H_perm[i], H_perm[j]); }
389 }
390 }
391
392 return tensor_subspace;
393 }
