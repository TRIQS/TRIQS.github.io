9 nda::matrix<double> get_full_h_atomic(const triqs_atom_diag &ad) {
10 int dim = ad.get_full_hilbert_space_dim();
11 auto H_mat = nda::zeros<double>(dim, dim);
12
13 for (int sidx = 0; sidx < ad.n_subspaces(); ++sidx) {
14 auto U = ad.get_unitary_matrix(sidx);
15 auto energies = ad.get_energies()[sidx];
16
17 // Create diagonal energy matrix
18 nda::matrix<double> H_block_diag = nda::zeros<double>(energies.size(), energies.size());
19 for (int i = 0; i < energies.size(); ++i) { H_block_diag(i, i) = energies[i] + ad.get_gs_energy(); }
20
21 // Transform to Fock basis: H_block = U @ H_block_diag @ U^T
22 nda::matrix<double> H_block = U * H_block_diag * nda::transpose(U);
23
24 // Get Fock states for this subspace
25 auto fock_states = ad.get_fock_states(sidx);
26
27 // Copy block into full matrix
28 for (int i = 0; i < fock_states.size(); ++i) {
29 for (int j = 0; j < fock_states.size(); ++j) { H_mat(fock_states[i], fock_states[j]) = H_block(i, j); }
30 }
31 }
32
33 return H_mat;
34 }

36 nda::matrix<double> get_full_operator_matrix(const triqs_atom_diag &ad, int oidx, bool is_creation) {
37 int dim = ad.get_full_hilbert_space_dim();
38 nda::matrix<double> op_mat = nda::zeros<double>(dim, dim);
39
40 for (int s1 = 0; s1 < ad.n_subspaces(); ++s1) {
41 // Get connection to target subspace
42 long s2 = is_creation ? ad.cdag_connection(oidx, s1) : ad.c_connection(oidx, s1);
43 if (s2 < 0) continue;
44
45 // Get matrix block in eigenbasis
46 auto block_mat = is_creation ? ad.cdag_matrix(oidx, s1) : ad.c_matrix(oidx, s1);
47
48 // Transform to Fock basis
49 auto U1 = ad.get_unitary_matrix(s1);
50 auto U2 = ad.get_unitary_matrix(s2);
51 nda::matrix<double> block_mat_fock = U2 * block_mat * nda::transpose(U1);
52
53 // Get Fock states
54 auto f1 = ad.get_fock_states(s1);
55 auto f2 = ad.get_fock_states(s2);
56
57 // Copy block into full matrix
58 for (int i = 0; i < f2.size(); ++i) {
59 for (int j = 0; j < f1.size(); ++j) { op_mat(f2[i], f1[j]) = block_mat_fock(i, j); }
60 }
61 }
62
63 return op_mat;
64 }

66 std::tuple<std::vector<nda::array<double, 2>>, nda::vector<long>> get_hamiltonian_blocks(const triqs_atom_diag &ad) {
67 // Get full Hamiltonian matrix
68 auto H_mat = get_full_h_atomic(ad);
69
70 // Create permutation based on Fock state ordering
71 std::vector<unsigned long> H_perm;
72 std::vector<nda::array<double, 2>> H_blocks;
73 nda::vector<long> H_block_inds(ad.n_subspaces());
74
75 for (int s = 0; s < ad.n_subspaces(); ++s) {
76 auto fock_states = ad.get_fock_states(s);
77 for (auto state : fock_states) { H_perm.push_back(state); }
78
79 // Extract block from full matrix
80 nda::array<double, 2> H_block = nda::zeros<double>(fock_states.size(), fock_states.size());
81 for (int i = 0; i < fock_states.size(); ++i) {
82 for (int j = 0; j < fock_states.size(); ++j) { H_block(i, j) = H_mat(fock_states[i], fock_states[j]); }
83 }
84 H_blocks.push_back(H_block);
85
86 // Check if block is zero
87 double max_elem = 0.0;
88 for (int i = 0; i < H_block.extent(0); ++i) {
89 for (int j = 0; j < H_block.extent(1); ++j) { max_elem = std::max(max_elem, std::abs(H_block(i, j))); }
90 }
91 H_block_inds(s) = (max_elem < 1e-16) ? -1 : 0;
92 }
93 return std::make_tuple(H_blocks, H_block_inds);
94 }

97 std::vector<nda::array<T, 3>> H_to_atom_prop_blocks(std::vector<nda::array<double, 2>> &H_blocks, nda::vector_const_view<long> H_block_inds,
98 double beta, imtime_ops &itops) {
99 int r = itops.rank();
100 double tr_exp_minusbetaH = 0;
101 std::vector<nda::array<double, 1>> H_evals(H_blocks.size());
102 std::vector<nda::array<double, 2>> H_evecs(H_blocks.size());
103 for (int i = 0; i < H_block_inds.size(); ++i) {
104 if (H_block_inds(i) != -1) {
105 if (H_blocks[i].extent(0) == 1) {
106 H_evals[i] = nda::array<double, 1>{H_blocks[i](0, 0)};
107 H_evecs[i] = nda::array<double, 2>{{1}};
108 } else {
109 auto H_block_eig = nda::linalg::eigh(H_blocks[i]);
110 H_evals[i] = std::get<0>(H_block_eig);
111 H_evecs[i] = std::get<1>(H_block_eig);
112 }
113 tr_exp_minusbetaH += nda::sum(exp(-beta * H_evals[i]));
114 } else {
115 H_evals[i] = nda::zeros<double>(H_blocks[i].extent(0));
116 H_evecs[i] = nda::eye<double>(H_blocks[i].extent(0));
117 tr_exp_minusbetaH += 1.0 * H_blocks[i].extent(0); // 0 entry in the diagonal
118 }
119 }
120
121 auto eta_0 = nda::log(tr_exp_minusbetaH) / beta;
122 auto dlr_it = itops.get_itnodes();
123 auto dlr_it_abs = cppdlr::rel2abs(dlr_it);
124 std::vector<nda::array<T, 3>> ap_blocks(H_block_inds.size());
125 for (int i = 0; i < H_block_inds.size(); ++i) {
126 ap_blocks[i] = nda::array<T, 3>(r, H_blocks[i].extent(0), H_blocks[i].extent(1));
127 auto Gt_temp = nda::make_regular(0 * H_blocks[i]);
128 for (int t = 0; t < r; t++) {
129 for (int j = 0; j < H_blocks[i].extent(0); j++) { Gt_temp(j, j) = -exp(-beta * dlr_it_abs(t) * (H_evals[i](j) + eta_0)); }
130 ap_blocks[i](t, _, _) = matmul(H_evecs[i], matmul(Gt_temp, nda::transpose(H_evecs[i])));
131 }
132 }
133
134 return ap_blocks;
135 }

165 std::tuple<BlockOpSymQuartet, nda::vector<int>> get_operators(const triqs_atom_diag &ad, nda::array_const_view<dcomplex, 3> hyb_coeffs) {
166
167 // Find like rows of c_connection (resp. cdag_connection), which correspond with annihilation (resp. creation) operators that have the same
168 // sparsity pattern
169 int n = hyb_coeffs.extent(1);
170 nda::vector<long> sym_set_labels(n);
171 sym_set_labels = 0;
172 int counter = 0;
173 for (int oidx = 0; oidx < n; ++oidx) { // fill each entry of sym_set_labels
174 bool found_match = false;
175 for (int oidx2 = 0; oidx2 < oidx; ++oidx2) { // compare full c_connection row against all previous operators
176 bool same = true;
177 for (int s = 0; s < ad.n_subspaces(); ++s) {
178 if (ad.c_connection(oidx, s) != ad.c_connection(oidx2, s)) {
179 same = false;
180 break;
181 }
182 }
183 if (same) {
184 sym_set_labels(oidx) = sym_set_labels(oidx2);
185 found_match = true;
186 break;
187 }
188 }
189 if (not found_match) { // no matching operator found, so create new group
190 sym_set_labels(oidx) = counter;
191 counter = counter + 1;
192 }
193 }
194 std::set<int> unique_groups(sym_set_labels.begin(), sym_set_labels.end());
195 std::size_t num_sym_sets = unique_groups.size();
196
197 // First pass: count how many operators belong to each symmetry group
198 std::vector<int> ops_per_group(num_sym_sets, 0);
199 for (int oidx = 0; oidx < n; ++oidx) { ops_per_group[sym_set_labels[oidx]]++; }
200
201 // Initialize operator blocks grouped by symmetry with proper dimensions
202 std::vector<std::vector<nda::array<dcomplex, 3>>> c_blocks(num_sym_sets);
203 std::vector<std::vector<nda::array<dcomplex, 3>>> cdag_blocks(num_sym_sets);
204
205 // Initialize arrays for each symmetry group and subspace
206 for (int gidx = 0; gidx < num_sym_sets; ++gidx) {
207 c_blocks[gidx].resize(ad.n_subspaces());
208 cdag_blocks[gidx].resize(ad.n_subspaces());
209
210 for (int sidx = 0; sidx < ad.n_subspaces(); ++sidx) {
211 // Determine dimensions for this subspace
212 long cidx = -1, didx = -1;
213 int dim_c_final = 0, dim_c_initial = 0;
214 int dim_cdag_final = 0, dim_cdag_initial = 0;
215
216 // Find a representative operator from this symmetry group to get dimensions
217 for (int oidx = 0; oidx < n; ++oidx) {
218 if (sym_set_labels[oidx] == gidx) {
219 if (cidx == -1) {
220 cidx = ad.c_connection(oidx, sidx);
221 if (cidx >= 0) {
222 dim_c_final = ad.get_fock_states(cidx).size();
223 dim_c_initial = ad.get_fock_states(sidx).size();
224 }
225 }
226 if (didx == -1) {
227 didx = ad.cdag_connection(oidx, sidx);
228 if (didx >= 0) {
229 dim_cdag_final = ad.get_fock_states(didx).size();
230 dim_cdag_initial = ad.get_fock_states(sidx).size();
231 }
232 }
233 if (cidx >= 0 && didx >= 0) break;
234 }
235 }
236
237 // Initialize arrays with proper dimensions
238 if (cidx >= 0) { c_blocks[gidx][sidx] = nda::zeros<dcomplex>(ops_per_group[gidx], dim_c_final, dim_c_initial); }
239 if (didx >= 0) { cdag_blocks[gidx][sidx] = nda::zeros<dcomplex>(ops_per_group[gidx], dim_cdag_final, dim_cdag_initial); }
240 }
241 }
242
243 // Second pass: fill the arrays
244 std::vector<int> op_count_per_group(num_sym_sets, 0);
245
246 for (int oidx = 0; oidx < n; ++oidx) {
247 int gidx = sym_set_labels[oidx];
248 int op_idx_in_group = op_count_per_group[gidx];
249
250 for (int sidx = 0; sidx < ad.n_subspaces(); ++sidx) {
251 // Handle annihilation operator
252 long cidx = ad.c_connection(oidx, sidx);
253 if (cidx >= 0) {
254 auto fock_final = ad.get_fock_states(cidx);
255 auto fock_initial = ad.get_fock_states(sidx);
256
257 // Get full operator matrix and extract block
258 auto c_full = get_full_operator_matrix(ad, oidx, false);
259 for (int i = 0; i < fock_final.size(); ++i) {
260 for (int j = 0; j < fock_initial.size(); ++j) { c_blocks[gidx][sidx](op_idx_in_group, i, j) = c_full(fock_final[i], fock_initial[j]); }
261 }
262 }
263
264 // Handle creation operator
265 long didx = ad.cdag_connection(oidx, sidx);
266 if (didx >= 0) {
267 auto fock_final = ad.get_fock_states(didx);
268 auto fock_initial = ad.get_fock_states(sidx);
269
270 // Get full operator matrix and extract block
271 auto cdag_full = get_full_operator_matrix(ad, oidx, true);
272 for (int i = 0; i < fock_final.size(); ++i) {
273 for (int j = 0; j < fock_initial.size(); ++j) {
274 cdag_blocks[gidx][sidx](op_idx_in_group, i, j) = cdag_full(fock_final[i], fock_initial[j]);
275 }
276 }
277 }
278 }
279
280 op_count_per_group[gidx]++;
281 }
282
283 // Fill in BlockOpSymSet objects
284 nda::array<int, 2> F_block_inds = nda::zeros<int>(num_sym_sets, ad.n_subspaces()),
285 F_dag_block_inds = nda::zeros<int>(num_sym_sets, ad.n_subspaces());
286 auto filled_F_block_inds = nda::zeros<int>(n);
287 for (int i = 0; i < n; i++) {
288 long label = sym_set_labels(i);
289 if (filled_F_block_inds(label) == 0) {
290 for (int j = 0; j < ad.n_subspaces(); ++j) { F_block_inds(label, j) = ad.c_connection(i, j); }
291 filled_F_block_inds(label) = 1;
292 }
293 }
294 auto filled_F_dag_block_inds = nda::zeros<int>(n);
295 for (int i = 0; i < n; i++) {
296 long label = sym_set_labels(i);
297 if (filled_F_dag_block_inds(label) == 0) {
298 for (int j = 0; j < ad.n_subspaces(); ++j) { F_dag_block_inds(label, j) = ad.cdag_connection(i, j); }
299 filled_F_dag_block_inds(label) = 1;
300 }
301 }
302
303 std::vector<BlockOpSymSet> F_sym_vec, F_dag_sym_vec;
304 for (int i = 0; i < num_sym_sets; i++) {
305 F_sym_vec.emplace_back(F_block_inds(i, _), c_blocks[i]);
306 F_dag_sym_vec.emplace_back(F_dag_block_inds(i, _), cdag_blocks[i]);
307 }
308 BlockOpSymQuartet Fq(F_sym_vec, F_dag_sym_vec, hyb_coeffs, sym_set_labels);
309 return std::make_tuple(Fq, sym_set_labels);
310 }

333 nda::array<dcomplex, 3> get_tensor_in_atom_diag_subspace(nda::array_const_view<dcomplex, 3> tensor_full, int subspace_index,
334 triqs_atom_diag const &ad) {
335 // Permute a tensor from the full Hilbert space to the Fock state ordered basis
336
337 int r = tensor_full.extent(0);
338
339 std::vector<unsigned long> H_perm;
340 auto fock_states = ad.get_fock_states(subspace_index);
341 for (auto state : fock_states) H_perm.push_back(state);
342
343 int N_sub = fock_states.size();
344 nda::array<dcomplex, 3> tensor_subspace = nda::zeros<dcomplex>(r, N_sub, N_sub);
345
346 for (int t = 0; t < r; ++t) {
347 for (int i = 0; i < N_sub; ++i) {
348 for (int j = 0; j < N_sub; ++j) { tensor_subspace(t, i, j) = tensor_full(t, H_perm[i], H_perm[j]); }
349 }
350 }
351
352 return tensor_subspace;
