127 void DenseDiagramEvaluator::multiply_left_vertex_and_right_zero_vertex(nda::array_view<dcomplex, 3> T_buf, Backbone &backbone, int vct0) {
128
129 bool is_forward = backbone.has_vertex_dag(vct0);
130 nda::array_const_view<dcomplex, 3> hyb_too = is_forward ? hyb.values : hyb.values_reflect;
131
132 // Save compute by precomputing Tkaps = T_buf * F_kap for all kappa, since this is needed for each mu
133 // at the cost of storing an n x r x N x N array (Tkaps) instead of an r x N x N array
134
135 for (int kap = 0; kap < n; kap++) {
136 nda::array_const_view<dcomplex, 2> F_kap = Fset.get_operator(backbone, 0, kap);
137 for (int t = 0; t < r; t++) Tkaps(kap, t, _, _) = matmul(T_buf(t, _, _), F_kap);
138 }
139
140 T_buf = 0;
141
142 for (int mu = 0; mu < n; mu++) {
143 Tmu = 0;
144 for (int kap = 0; kap < n; kap++) {
145 nda::array_const_view<dcomplex, 1> hyb_t = hyb_too(_, mu, kap);
146 for (int t = 0; t < r; t++) Tmu(t, _, _) += hyb_t(t) * Tkaps(kap, t, _, _);
147 }
148 nda::array_const_view<dcomplex, 2> F_mu = Fset.get_operator(backbone, vct0, mu);
149 for (int t = 0; t < r; t++) T_buf(t, _, _) += matmul(F_mu, Tmu(t, _, _));
150 }
151 }

171 void DenseDiagramEvaluator::eval_self_energy_fixed_indices(nda::array_const_view<dcomplex, 3> Gt, Backbone &backbone, int f_ix) {
172 int m = backbone.m;
173 int vct0 = backbone.get_topology(0, 1); // Vertex Connected To zero
174
175 backbone.set_flat_index(f_ix, hyb.poles); // set directions, pole indices, and orbital indices from a single integer index
176
177 // 1. Starting from tau_1, proceed right to left, performing multiplications at vertices and convolutions at edges, until reaching the vertex
178 // containing the undecomposed hybridization line Delta_{mu kappa}.
179 T = Gt;
180 // T is initialized to Gt, which is always the function at the rightmost edge
181 for (int v = 1; v < vct0; v++) { // loop from the first vertex to before the special vertex
182 multiply_left_vertex(T, backbone, v);
183 integrate_left_edge(T, Gt, backbone, v);
184 }
185
186 // 2. For each kappa, multiply by F_kappa(^dag). Then for each mu, kappa, multiply by Delta_{mu kappa}, and sum over kappa. Finally for each mu,
187 // multiply F_mu[^dag] and sum over mu.
188 multiply_left_vertex_and_right_zero_vertex(T, backbone, vct0);
189
190 // 3. Continue right to left until the final vertex multiplication is complete.
191 for (int v = vct0 + 1; v < 2 * m; v++) { // loop from the special vertex to the last vertex
192 integrate_left_edge(T, Gt, backbone, v - 1);
193 multiply_left_vertex(T, backbone, v);
194 }
195
196 multiply_prefactor(T, backbone);
197 int diag_order_sign = (m % 2 == 0) ? -1 : 1;
198 if (backbone.get_fb(0) == 0) diag_order_sign *= -1; // if the first hybridization line is backward, there is an additional sign change
199 T *= diag_order_sign * backbone.prefactor_sign;
200 Sigma += T;
201
202 backbone.reset_all_inds(); // reset directions, pole indices, and orbital indices for the next iteration
203 }

237 nda::array<dcomplex, 3> DenseDiagramEvaluator::eval_correlator(nda::array_const_view<dcomplex, 3> Gt, CorrelatorBackbone &backbone,
238 nda::array<dcomplex, 3> mu_ops, nda::array<dcomplex, 3> kap_ops, int f_ix) {
239
240 int m = backbone.m;
241
242 backbone.set_flat_index(f_ix, hyb.poles); // set directions, pole indices, and orbital indices from a single integer index
243
244 nda::array<dcomplex, 3> correlator = nda::zeros<dcomplex>(r, mu_ops.extent(0), kap_ops.extent(0));
245
246 // evaluate the first sequence of backbone products and convolutions from tau_1 to tau, proceeding right to left, not inculding the creation and
247 // annihilation matrices at the end points. the result is an N x N matrix-valued function of tau.
248 T = Gt;
249 // T is initialized to Gt, which is always the function at the rightmost edge
250 for (int v = 1; v < backbone.get_topology(0, 1); ++v) { // loop from the first vertex to before the special vertex
251 multiply_left_vertex(T, backbone, v);
252 integrate_left_edge(T, Gt, backbone, v);
253 }
254
255 // evaluate the second sequence of backbone products and convolutions from tau to beta, using a change of variables to perform convolutions. the
256 // result is another N x N matrix-valued function of tau.
257 U = Gt;
258
259 for (int v = 2 * m - 1; v > backbone.get_topology(0, 1); v--) {
260 multiply_right_vertex(U, backbone, v);
261 integrate_right_edge(U, Gt, backbone, v - 1);
262 }
263
264 U = itops.reflect(U);
265
266 multiply_prefactor(T, backbone);
267 int diag_order_sign = 1; // (m % 2 == 1) ? -1 : 1;
268 T *= diag_order_sign * backbone.prefactor_sign;
269
270 nda::array<dcomplex, 3> Tmuop = nda::zeros<dcomplex>(r, Gt.extent(1), Gt.extent(1));
271
272 for (int mu = 0; mu < mu_ops.extent(0); ++mu) {
273 for (int t = 0; t < r; ++t) Tmuop(t, _, _) = matmul(U(t, _, _), matmul(mu_ops(mu, _, _), T(t, _, _)));
274
275 for (int kap = 0; kap < kap_ops.extent(0); ++kap) {
276 for (int t = 0; t < r; ++t) correlator(t, mu, kap) += trace(matmul(Tmuop(t, _, _), kap_ops(kap, _, _)));
277 }
278 }
279
280 backbone.reset_all_inds(); // reset directions, pole indices, and orbital indices for the next iteration
281
282 return correlator;
283 }
