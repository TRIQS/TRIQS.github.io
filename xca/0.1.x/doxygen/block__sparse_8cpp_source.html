27 BlockDiagOpFun::BlockDiagOpFun(const triqs::gfs::block_gf<triqs::mesh::dlr_imtime> &bgf) : num_block_cols(bgf.size()) {
28 // TODO set block to just a single zero if the block gf is numerically zero
29 blocks.resize(num_block_cols);
30 zero_block_indices = nda::zeros<int>(num_block_cols);
31 int r = bgf[0].mesh().size();
32
33 for (int i = 0; i < num_block_cols; i++) {
34 auto const &gf = bgf[i];
35 int block_size = gf.target_shape()[0];
36
37 blocks[i] = nda::array<dcomplex, 3>(r, block_size, block_size);
38
39 for (int t = 0; t < r; t++) { blocks[i](t, _, _) = gf(t); }
40
41 if (nda::max_element(nda::abs(blocks[i])) < 1e-16) {
42 zero_block_indices(i) = -1;
43 } else {
44 zero_block_indices(i) = 0;
45 }

433 BlockOpSymQuartet::BlockOpSymQuartet(std::vector<BlockOpSymSet> Fs, std::vector<BlockOpSymSet> F_dags,
434 nda::array_const_view<dcomplex, 3> hyb_coeffs, nda::vector_const_view<long> sym_set_labels)
435 : Fs(Fs), F_dags(F_dags), sym_set_labels(sym_set_labels), p(hyb_coeffs.extent(0)) {
436
437 // Fs and F_dags are vectors of BOSS
438 // Each entry corresponds to a set of operators with the same block-sparse structure
439 // Fs and F_dags have the same number of entries
440 // if k = # of entries of Fs, and each entry f_i has q_i operators, then n = sum(q_i) = number of orbital indices
441 int k = Fs.size();
442 if (k != F_dags.size()) { throw std::invalid_argument("Fs and F_dags must have the same number of entries"); }
443
444 // initialize F_dag_bars and F_bars_refl
445 p = hyb_coeffs.extent(0);
446 nda::vector<int> block_indices_dag = F_dags[0].get_block_indices();
447 nda::vector<int> block_indices_f = Fs[0].get_block_indices();
448 for (int i = 0; i < k; i++) {
449 block_indices_dag = F_dags[i].get_block_indices();
450 block_indices_f = Fs[i].get_block_indices();
451 this->F_dag_bars.emplace_back(F_dags[i].get_size_sym_set(), p, block_indices_dag, F_dags[i].get_block_sizes());
452 this->F_bars_refl.emplace_back(Fs[i].get_size_sym_set(), p, block_indices_f, Fs[i].get_block_sizes());
453 }
454
455 // calculate symmetry set indices
456 long n = sym_set_labels.size(); // number of orbital indices
457 long q = nda::max_element(sym_set_labels) + 1; // number of symmetry sets
458 sym_set_inds = nda::zeros<long>(n); // indices of orbital indices in symmetry sets
459 sym_set_sizes = nda::zeros<long>(q); // sizes of symmetry sets
460 sym_set_to_orb = nda::ones<long>(q, n); // map label and index to orbital index
461 sym_set_to_orb *= -1; // initialize with -1
462 for (int i = 0; i < n; i++) {
463 sym_set_inds(i) = sym_set_sizes(sym_set_labels(i));
464 sym_set_sizes(sym_set_labels(i))++;
465 sym_set_to_orb(sym_set_labels(i), sym_set_inds(i)) = i; // map symmetry set index to backbone orbital index
466 }
467
468 // compute F_dag_bars and F_bars_refl
469 for (int l = 0; l < p; l++) {
470 for (int p_lam = 0; p_lam < q; p_lam++) {
471 for (int p_nu = 0; p_nu < q; p_nu++) {
472 for (int lam = 0; lam < sym_set_sizes(p_lam); lam++) {
473 for (int nu = 0; nu < sym_set_sizes(p_nu); nu++) {
474 long lam_orb = sym_set_to_orb(p_lam, lam);
475 long nu_orb = sym_set_to_orb(p_nu, nu);
476 for (int b = 0; b < F_dags[p_lam].get_num_block_cols(); b++) {
477 if (F_dags[p_lam].get_block_index(b) != -1) {
478 F_dag_bars[p_lam].add_block(b, lam, l, nda::make_regular(hyb_coeffs(l, nu_orb, lam_orb) * F_dags[p_lam].get_block(b)(nu, _, _)));
479 }
480 }
481 for (int b = 0; b < Fs[p_nu].get_num_block_cols(); b++) {
482 if (Fs[p_nu].get_block_index(b) != -1) {
483 F_bars_refl[p_nu].add_block(
484 b, nu, l, nda::make_regular(-hyb_coeffs(l, nu_orb, lam_orb) * Fs[p_nu].get_block(b)(lam, _, _))); // Add -1 sign for reflected F
485 }
486 }
487 }
488 }
489 }
490 }

558 // Evaluate F^dagger in block-sparse storage
559 // @param[in] F F operator
560 // @return F^dagger operator
561

625 BlockDiagOpFun BOFtoBDOF(BlockOpFun const &A) {
626 // Convert a BlockOpFun with diagonal structure to a BlockDiagOpFun
627 // @param[in] A BlockOpFun
628 // @return BlockDiagOpFun
629
630 int num_block_cols = A.get_num_block_cols();
631 auto diag_blocks = A.get_blocks();
632 auto zero_block_indices = nda::zeros<int>(num_block_cols);
633 for (int i = 0; i < num_block_cols; i++) {
634 int block_index = A.get_block_index(i);
635 if (block_index == -1) {
636 diag_blocks[i] = nda::zeros<dcomplex>(1, 1, 1);
637 zero_block_indices(i) = -1;
638 } else if (block_index != i) {
639 throw std::invalid_argument("BOF is not diagonal");
640 }
641 }
642
643 return BlockDiagOpFun(diag_blocks, zero_block_indices);
644 }

646 BlockDiagOpFun nonint_gf_BDOF(std::vector<nda::array<dcomplex, 2>> H_blocks, nda::vector<int> H_block_inds, double beta,
647 nda::vector_const_view<double> dlr_it_abs) {
648
649 int num_block_cols = H_block_inds.size();
650 nda::vector<int> H_block_sizes(num_block_cols);
651 for (int i = 0; i < num_block_cols; i++) { H_block_sizes(i) = H_blocks[i].extent(0); }
652
653 int r = dlr_it_abs.size();
654
655 dcomplex tr_exp_minusbetaH = 0;
656 std::vector<nda::array<dcomplex, 1>> H_evals(num_block_cols);
657 std::vector<nda::array<dcomplex, 2>> H_evecs(num_block_cols);
658 for (int i = 0; i < num_block_cols; i++) {
659 if (H_block_inds(i) != -1) {
660 if (H_block_sizes(i) == 1) {
661 H_evals[i] = nda::array<dcomplex, 1>{H_blocks[i](0, 0)};
662 H_evecs[i] = nda::array<dcomplex, 2>{{1}};
663 } else {
664 auto H_block_eig = nda::linalg::eigh(H_blocks[i]);
665 H_evals[i] = std::get<0>(H_block_eig);
666 H_evecs[i] = std::get<1>(H_block_eig);
667 }
668 tr_exp_minusbetaH += nda::sum(exp(-beta * H_evals[i]));
669 } else {
670 H_evals[i] = nda::zeros<dcomplex>(H_block_sizes(i));
671 H_evecs[i] = nda::eye<dcomplex>(H_block_sizes(i));
672 tr_exp_minusbetaH += 1.0 * H_block_sizes(i); // 0 entry in the diagonal
673 }
674 }
675
676 auto eta_0 = nda::log(tr_exp_minusbetaH) / beta;
677 auto Gt = BlockDiagOpFun(r, H_block_sizes);
678 for (int i = 0; i < num_block_cols; i++) {
679 auto Gt_block = nda::array<dcomplex, 3>(r, H_block_sizes(i), H_block_sizes(i));
680 auto Gt_temp = nda::make_regular(0 * H_blocks[i]);
681 for (int t = 0; t < r; t++) {
682 for (int j = 0; j < H_block_sizes(i); j++) { Gt_temp(j, j) = -exp(-beta * dlr_it_abs(t) * (H_evals[i](j) + eta_0)); }
683 Gt_block(t, _, _) = matmul(H_evecs[i], matmul(Gt_temp, nda::transpose(H_evecs[i])));
684 }
685 Gt.set_block(i, Gt_block);
686 }
687
688 return Gt;
689 }

692 auto dlr_rf = cppdlr::build_dlr_rf(Lambda, eps);
693 auto itops = cppdlr::imtime_ops(Lambda, dlr_rf);

702 gf_vec[i] = triqs::gfs::gf<triqs::mesh::dlr_imtime>{t_mesh, {BDOF.get_block(i).extent(1), BDOF.get_block(i).extent(2)}};

704 } else {

706 gf_vec[i] = triqs::gfs::gf<triqs::mesh::dlr_imtime>{t_mesh, {0, 0}};
707 }
708 }

712 template<bool isComplex>
713 dcomplex expectation_value(
714 triqs::operators::many_body_operator_real const &op,
715 triqs::atom_diag::atom_diag<isComplex> const &ad,
716 triqs::gfs::block_gf_view<triqs::mesh::dlr_imtime> G_ppsc) {
717
718 auto op_blocks = ad.get_op_mat(op);
719 auto beta = G_ppsc[0].mesh().beta();
720
721 dcomplex sum = 0;
722
723 for( auto bidx : range(op_blocks.block_mat.size()) ) {
724 assert( op_blocks.connection[bidx] == bidx );
725 if( op_blocks.block_mat[bidx].shape(0) == 0 ) continue; // skip empty blocks
726 auto g_dlr = make_gf_dlr(G_ppsc[bidx]);
727 auto U = ad.get_unitary_matrix(bidx);
728 auto op_mat_transf = U * op_blocks.block_mat[bidx] * nda::conj(nda::transpose(U));
729 sum += -trace(matmul(op_mat_transf, g_dlr(beta)));
730 }
731
732 return sum;
733 }

760 auto gg = itops.convolve(beta, itops.vals2coefs(g1.data()), itops.vals2coefs(g2.data()), cppdlr::TIME_ORDERED);

762 }
