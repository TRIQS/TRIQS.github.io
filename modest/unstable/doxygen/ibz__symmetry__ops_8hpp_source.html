35 template <typename T> nda::array<T, 2> symmetrize(nda::array<T, 2> const &X) const {
36 auto [n_alpha, n_sigma] = X.shape();
37 nda::array<T, 2> result(n_alpha, n_sigma);
38 for (auto [a, s] : result.indices()) { result(a, s) = nda::zeros<dcomplex>(X(a, s).shape()); }
39 auto n_symm = double(this->ops.size());
40 for (auto const &sym_op : this->ops) {
41 for (auto &&[ialpha, jalpha, mat] : zip(range(n_alpha), sym_op.permutation, sym_op.mats)) {
42 for (auto sigma : range(n_sigma)) {
43 result(jalpha, sigma) +=
44 mat * (sym_op.time_inv ? transpose(X(ialpha, sigma)) : X(ialpha, sigma)) * dagger(mat) / n_symm;
45 } // end loop on sigma
46 } // end loop on alpha
47 } // end loop on ops
48 return result;
49 }
