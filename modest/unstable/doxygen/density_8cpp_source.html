10 double density_nk_matrix_valued_impl(one_body_elements_on_grid const &obe, double mu, double beta) {
11
12 auto Fermi = [](double x) { return (x > 0 ? exp(-x) / (1 + exp(-x)) : 1 / (1 + exp(x))); };
13 double KS_term = 0;
14
15 for (auto sigma : nda::range(obe.C_space.n_sigma()))
16 for (auto k_idx : nda::range(obe.H.n_k())) {
17 double KS_term_acc = 0;
18 auto eps = nda::linalg::eigenvalues(nda::matrix<dcomplex>{obe.H.H(sigma, k_idx)});
19 for (auto nu : range(obe.H.N_nu(sigma, k_idx))) { KS_term_acc += Fermi(beta * (eps(nu) - mu)); }
20 KS_term += obe.H.k_weights(k_idx) * KS_term_acc;
21 }
22 return KS_term;
23 }

25 double density_nk(one_body_elements_on_grid const &obe, double mu, double beta) {
26 // if band dispersion is matrix valued need to diagonalize H(k) at every k.
27 if (obe.H.matrix_valued) return density_nk_matrix_valued_impl(obe, mu, beta);
28
29 auto Fermi = [](double x) { return (x > 0 ? exp(-x) / (1 + exp(-x)) : 1 / (1 + exp(x))); };
30 double KS_term = 0;
31
32 for (auto sigma : nda::range(obe.C_space.n_sigma()))
33 for (auto k_idx : nda::range(obe.H.n_k())) {
34 double KS_term_acc = 0;
35 auto eps = obe.H.H(sigma, k_idx);
36 for (auto nu : range(obe.H.N_nu(sigma, k_idx))) { KS_term_acc += Fermi(beta * real(eps(nu, nu) - mu)); }
37 KS_term += obe.H.k_weights(k_idx) * KS_term_acc;
38 }
39 return KS_term;
40 }
