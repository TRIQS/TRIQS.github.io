31 template <typename Mesh>
32 block2_gf<Mesh, matrix_valued> gloc_for_matrix_valued_dispersion_impl(one_body_elements_on_grid const &obe, double mu,
33 block2_gf<Mesh, matrix_valued> const &Sigma_dynamic,
34 nda::array<nda::matrix<dcomplex>, 2> const &Sigma_static) {
35
36 auto n_sigma = obe.C_space.n_sigma();
37 auto n_M = obe.C_space.dim();
38 auto n_kpts = long(obe.H.n_k());
39 auto const &mesh = Sigma_dynamic(0, 0).mesh();
40
41 auto embedding_decomp = get_struct(Sigma_dynamic).dims(r_all, 0) | tl::to<std::vector>();
42
43 auto PSP = [&](auto &w, auto &k_idx, auto &sigma) {
44 auto N_nu = obe.H.N_nu(sigma, k_idx);
45 auto out = nda::zeros<dcomplex>(N_nu, N_nu);
46 for (auto &&[alpha, R] : enumerated_sub_slices(embedding_decomp)) {
47 auto P = obe.P.P(sigma, k_idx)(R, r_all);
48 out(r_all, r_all) += dagger(P) * nda::matrix<dcomplex>{Sigma_dynamic(alpha, sigma).data()(w, r_all, r_all) + Sigma_static(alpha, sigma)} * P;
49 }
50 return out;
51 };
52
53 auto gloc_k = [&](auto &k_idx, auto &sigma) {
54 auto out = gf{mesh, {n_M, n_M}};
55 auto P = obe.P.P(sigma, k_idx);
56 for (auto &&[n, w] : enumerate(mesh)) {
57 out.data()(n, r_all, r_all) = P * inverse(w + mu - obe.H.H(sigma, k_idx) - PSP(n, k_idx, sigma)) * dagger(P);
58 }
59 return out;
60 };
61
62 auto gloc_result = make_block2_gf(mesh, obe.C_space.Gc_block_shape());
63 mpi::communicator comm = {};
64#pragma omp parallel for collapse(2) reduction(block2_gf_sum : gloc_result) default(none) shared(comm, gloc_k, n_kpts, n_sigma, obe, r_all)
65 for (auto k_idx : mpi::chunk(range(n_kpts), comm)) {
66 for (auto sigma : range(n_sigma)) { gloc_result(0, sigma).data()(r_all, r_all, r_all) += obe.H.k_weights(k_idx) * gloc_k(k_idx, sigma).data(); }
67 }
68 gloc_result = mpi::all_reduce(gloc_result);
69
70 if (auto const &S = obe.ibz_symm_ops; S) { gloc_result = S->symmetrize(gloc_result, obe.C_space.atomic_decomposition()); }
71 return gloc_result;
72 }

104 template <typename Mesh>
105 block2_gf<Mesh, matrix_valued> gloc(one_body_elements_on_grid const &obe, double mu, block2_gf<Mesh, matrix_valued> const &Sigma_dynamic,
106 nda::array<nda::matrix<dcomplex>, 2> const &Sigma_static) {
107
108 // intercept if the dispersion in obe is matrix valued. The Woodbury offers no performance gain for this case.
109 if (obe.H.matrix_valued) return gloc_for_matrix_valued_dispersion_impl(obe, mu, Sigma_dynamic, Sigma_static);
110
111 auto n_sigma = Sigma_dynamic.size2();
112 auto M = obe.C_space.dim();
113 auto n_kpts = long(obe.H.n_k());
114 auto const &mesh = Sigma_dynamic(0, 0).mesh();
115 auto gloc_result = make_block2_gf(mesh, obe.C_space.Gc_block_shape());
116 auto omegas = mesh | tl::to<std::vector<dcomplex>>();
117
118 // Embedding decomposition from structure of Sigma
119 auto embedding_decomp = get_struct(Sigma_dynamic).dims(r_all, 0) | tl::to<std::vector>();
120
121 auto timer = scoped_timer{};
122 // ---------
123 // NOTE: Is there any reason why sigma loop should be the external one?
124 // Internal is favorable for maximum parallelization.
125 mpi::communicator comm = {};
126#pragma omp parallel for collapse(2) reduction(block2_gf_sum : gloc_result) default(none) \
127 shared(comm, r_all, n_kpts, n_sigma, obe, mu, omegas, mesh, M, embedding_decomp, Sigma_dynamic, Sigma_static)
128 for (auto k_idx : mpi::chunk(range(n_kpts), comm)) {
129 for (auto sigma : range(n_sigma)) {
130 auto Y = detail::G0_C_k_sigma(obe, mu, k_idx, sigma, omegas, false);
131 for (auto &&[n, om] : itertools::enumerate(mesh)) {
132 auto Y1 = Y(n, r_all, r_all);
133 auto B = detail::calc_inv_G_G0(M, embedding_decomp, Sigma_dynamic, Sigma_static, om, sigma, Y1, Y1);
134 gloc_result(0, sigma).data()(n, r_all, r_all) += obe.H.k_weights(k_idx) * B;
135 }
136 }
137 // No normalization: the Pk are in obe ALREADY normalized.
138 }
139 gloc_result = mpi::all_reduce(gloc_result);
140
141 // FIXME :: the IBZ should work on a proper gf_view with atomic decomposition
142 // CHANGE IBZ accordingly ...
143 if (auto const &S = obe.ibz_symm_ops; S) gloc_result = S->symmetrize(gloc_result, obe.C_space.atomic_decomposition());
144 return gloc_result;
145 }

159 template <typename Mesh> block2_gf<Mesh, matrix_valued> gloc(Mesh const &mesh, one_body_elements_on_grid const &obe, double mu) {
160 auto Sigma_dynamic = make_block2_gf(mesh, obe.C_space.Gc_block_shape());
161 auto Sigma_static = nda::array<nda::matrix<dcomplex>, 2>(1, obe.C_space.n_sigma());
162 for (auto [i, j] : Sigma_static.indices()) { Sigma_static(i, j) = nda::zeros<dcomplex>(obe.C_space.dim(), obe.C_space.dim()); }
163 return gloc(obe, mu, Sigma_dynamic, Sigma_static);

178 template <typename Mesh>
179 block_gf<Mesh, matrix_valued> hybridization(std::vector<nda::matrix<dcomplex>> const &epsilon_levels, block_gf<Mesh, matrix_valued> const &Gloc,
180 block_gf<Mesh, matrix_valued> const &Sigma_dynamic,
181 std::vector<nda::matrix<dcomplex>> const &Sigma_static) {
182 auto gf_struct = get_struct(Gloc);
183 auto mesh = Gloc[0].mesh();
184 auto Delta = block_gf{mesh, gf_struct};
185 auto n_blocks = gf_struct.size();
186 for (auto bl : range(n_blocks)) {
187 for (auto &&[n, w] : enumerate(mesh))
188 Delta[bl].data()(n, r_all, r_all) =
189 (w - epsilon_levels[bl] - inverse(Gloc[bl]).data()(n, r_all, r_all) - (Sigma_dynamic[bl].data()(n, r_all, r_all) + Sigma_static[bl]));
190 }
191 return Delta;
192 }
