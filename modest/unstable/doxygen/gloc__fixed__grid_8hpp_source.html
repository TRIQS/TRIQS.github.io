22 template <typename Mesh>
23 constexpr auto upfold_self_energy_at_freq(one_body_elements_on_grid const &obe, downfolding_projector const &Proj,
24 block2_gf<Mesh, matrix_valued> const &Sigma_dynamic,
25 nda::array<nda::matrix<dcomplex>, 2> const &Sigma_static, long w_idx, long k_idx, long sigma_idx) {
26 auto N_nu = obe.H.N_nu(sigma_idx, k_idx);
27 auto out = nda::zeros<dcomplex>(N_nu, N_nu);
28 for (auto &&[alpha, R] : enumerated_sub_slices(get_struct(Sigma_dynamic).dims(r_all, 0) | tl::to<std::vector>())) {
29 auto P = Proj.P(sigma_idx, k_idx)(R, r_all);
30 out(r_all, r_all) += dagger(P) * nda::matrix<dcomplex>{Sigma_dynamic(alpha, sigma_idx).data()(w_idx, r_all, r_all) + Sigma_static(alpha, sigma_idx)} * P;
31 }
32 return out;
33 }

36 template <typename Mesh>
37 constexpr auto local_gf_at_k(one_body_elements_on_grid const &obe, double const &mu, downfolding_projector const &Proj, block2_gf<Mesh, matrix_valued> const &Sigma_dynamic, nda::array<nda::matrix<dcomplex>,2> const &Sigma_static){
38 return [&](auto const &k_idx, auto const &sigma_idx){
39 auto const n_M = obe.C_space.dim();
40 auto const &mesh = Sigma_dynamic(0, 0).mesh();
41 auto out = gf{mesh, {n_M, n_M}};
42 auto P = obe.P.P(sigma_idx, k_idx);
43 for (auto &&[n, w] : enumerate(mesh)){
44 auto PSP = upfold_self_energy_at_freq(obe, Proj, Sigma_dynamic, Sigma_static, n, k_idx, sigma_idx);
45 out.data()(n, r_all, r_all) = P * inverse(w + mu - obe.H.H(sigma_idx, k_idx) - PSP) * dagger(P);
46 }
47 return out;
48 };

78 template <typename Mesh>
79 block2_gf<Mesh, matrix_valued> gloc_for_matrix_valued_dispersion_impl(one_body_elements_on_grid const &obe, double mu,
80 block2_gf<Mesh, matrix_valued> const &Sigma_dynamic,
81 nda::array<nda::matrix<dcomplex>, 2> const &Sigma_static) {
82
83 auto n_sigma = obe.C_space.n_sigma();
84 auto n_kpts = long(obe.H.n_k());
85
86 auto gloc_k = detail::local_gf_at_k(obe, mu, obe.P, Sigma_dynamic, Sigma_static);
87 auto gloc_result = make_block2_gf(Sigma_dynamic(0,0).mesh(), obe.C_space.Gc_block_shape());
88
89 mpi::communicator comm = {};
90#pragma omp parallel for collapse(2) reduction(block2_gf_sum : gloc_result) default(none) shared(comm, gloc_k, n_kpts, n_sigma, obe, r_all)
91 for (auto k_idx : mpi::chunk(range(n_kpts), comm)) {
92 for (auto sigma : range(n_sigma)) {
93 auto P = obe.P.P(sigma, k_idx);
94 gloc_result(0, sigma).data()(r_all, r_all, r_all) += obe.H.k_weights(k_idx) * gloc_k(k_idx, sigma).data();
95 }
96 }
97
98 gloc_result = mpi::all_reduce(gloc_result);
99
100 if (auto const &S = obe.ibz_symm_ops; S) { gloc_result = S->symmetrize(gloc_result, obe.C_space.atomic_decomposition()); }
101 return gloc_result;
102 }

136 template <typename Mesh>
137 block2_gf<Mesh, matrix_valued> gloc(one_body_elements_on_grid const &obe, double mu, block2_gf<Mesh, matrix_valued> const &Sigma_dynamic,
138 nda::array<nda::matrix<dcomplex>, 2> const &Sigma_static) {
139
140 // intercept if the dispersion in obe is matrix valued. The Woodbury offers no performance gain for this case.
141 if (obe.H.matrix_valued) return gloc_for_matrix_valued_dispersion_impl(obe, mu, Sigma_dynamic, Sigma_static);
142
143 auto n_sigma = Sigma_dynamic.size2();
144 auto M = obe.C_space.dim();
145 auto n_kpts = long(obe.H.n_k());
146 auto const &mesh = Sigma_dynamic(0, 0).mesh();
147 auto gloc_result = make_block2_gf(mesh, obe.C_space.Gc_block_shape());
148 auto omegas = mesh | tl::to<std::vector<dcomplex>>();
149
150 // Embedding decomposition from structure of Sigma
151 auto embedding_decomp = get_struct(Sigma_dynamic).dims(r_all, 0) | tl::to<std::vector>();
152
153 auto timer = scoped_timer{};
154 // ---------
155 // NOTE: Is there any reason why sigma loop should be the external one?
156 // Internal is favorable for maximum parallelization.
157 mpi::communicator comm = {};
158#pragma omp parallel for collapse(2) reduction(block2_gf_sum : gloc_result) default(none) \
159 shared(comm, r_all, n_kpts, n_sigma, obe, mu, omegas, mesh, M, embedding_decomp, Sigma_dynamic, Sigma_static)
160 for (auto k_idx : mpi::chunk(range(n_kpts), comm)) {
161 for (auto sigma : range(n_sigma)) {
162 auto Y = detail::G0_C_k_sigma(obe, mu, k_idx, sigma, omegas, false);
163 for (auto &&[n, om] : itertools::enumerate(mesh)) {
164 auto Y1 = Y(n, r_all, r_all);
165 auto B = detail::calc_inv_G_G0(M, embedding_decomp, Sigma_dynamic, Sigma_static, om, sigma, Y1, Y1);
166 gloc_result(0, sigma).data()(n, r_all, r_all) += obe.H.k_weights(k_idx) * B;
167 }
168 }
169 // No normalization: the Pk are in obe ALREADY normalized.
170 }
171 gloc_result = mpi::all_reduce(gloc_result);
172
173 // FIXME :: the IBZ should work on a proper gf_view with atomic decomposition
174 // CHANGE IBZ accordingly ...
175 if (auto const &S = obe.ibz_symm_ops; S) gloc_result = S->symmetrize(gloc_result, obe.C_space.atomic_decomposition());
176 return gloc_result;
177 }

210 template <typename Mesh>
211 block_gf<Mesh, matrix_valued> hybridization(std::vector<nda::matrix<dcomplex>> const &epsilon_levels, block_gf<Mesh, matrix_valued> const &Gloc,
212 block_gf<Mesh, matrix_valued> const &Sigma_dynamic,
213 std::vector<nda::matrix<dcomplex>> const &Sigma_static) {
214 auto gf_struct = get_struct(Gloc);
215 auto mesh = Gloc[0].mesh();
216 auto Delta = block_gf{mesh, gf_struct};
217 auto n_blocks = gf_struct.size();
218 for (auto bl : range(n_blocks)) {
219 for (auto &&[n, w] : enumerate(mesh))
220 Delta[bl].data()(n, r_all, r_all) =
221 (w - epsilon_levels[bl] - inverse(Gloc[bl]).data()(n, r_all, r_all) - (Sigma_dynamic[bl].data()(n, r_all, r_all) + Sigma_static[bl]));
222 }
223 return Delta;
224 }
