84 template <typename Mesh>
85 block2_gf<Mesh, matrix_valued> gloc(one_body_elements_tb const &obe, double mu, block2_gf<Mesh, matrix_valued> const &Sigma_dynamic,
86 nda::array<nda::matrix<dcomplex>, 2> const &Sigma_static, triqs::lattice::bz_int_options const &opt) {
87
88 // Need a sigma Hartree
89 auto &mesh = Sigma_dynamic(0, 0).mesh();
90 auto n_sigma = obe.C_space.n_sigma();
91 auto gloc_result = make_block2_gf(mesh, obe.C_space.Gc_block_shape());
92 // TODO also check safety of orbital space sizes...
93 if (n_sigma != Sigma_static.shape(1)) { throw std::runtime_error("Mismatch between the spin channels in Sigma_Static and Sigma_Dynamic"); }
94 if (n_sigma != obe.H.size()) { throw std::runtime_error("Mismatch between the spin channels in Sigma and spin channels in Hamiltonian."); }
95
96 // Embedding decomposition from structure of Sigma -- provides a list of block names
97 auto embedding_decomp = get_struct(Sigma_dynamic).dims(r_all, 0) | tl::to<std::vector>();
98
99 for (auto sigma : range(n_sigma)) {
100
101 // spin index
102 auto Sigma_full_space = gfs::gf(mesh, {obe.H[sigma].n_orbitals(), obe.H[sigma].n_orbitals()}); //
103 for (auto &&[block, R] : enumerated_sub_slices(embedding_decomp)) {
104 for (auto [n, w] : enumerate(mesh)) {
105 Sigma_full_space.data()(n, R, R) = Sigma_dynamic(block, sigma).data()(n, r_all, r_all) + Sigma_static(block, sigma);
106 }
107 }
108 // Call the TRIQS version of this function
109 gloc_result(0, sigma) = gloc(obe.H[sigma], mu, Sigma_full_space, opt);
110 }
111 return gloc_result;
112 }

126 template <typename Mesh>
127 block2_gf<Mesh, matrix_valued> gloc(Mesh const &mesh, one_body_elements_tb const &obe, double mu, triqs::lattice::bz_int_options const &opt) {
128 auto Sigma_dynamic = make_block2_gf(mesh, obe.C_space.Gc_block_shape());
129 auto Sigma_static = nda::array<nda::matrix<dcomplex>, 2>(1, obe.C_space.n_sigma());
130 for (auto [i, j] : Sigma_static.indices()) { Sigma_static(i, j) = nda::zeros<dcomplex>(obe.C_space.dim(), obe.C_space.dim()); }
131 return gloc(obe, mu, Sigma_dynamic, Sigma_static, opt);
132 }

137 template <typename Mesh>
138 double density(one_body_elements_tb const &obe, double mu, block2_gf<Mesh, matrix_valued> const &Sigma_dynamic,
139 nda::array<nda::matrix<dcomplex>, 2> const &Sigma_static, triqs::lattice::bz_int_options const &opt) {
140
141 //auto n_blocks = Sigma_dynamic.size1();
142 auto n_sigma = obe.C_space.n_sigma();
143
144 double n = 0;
145 auto Gloc = gloc(obe, mu, Sigma_dynamic, Sigma_static, opt); // returns block2gf (1, nsigma, {norb, norb})
146 // return type of Gloc is a B2GF with dimensions (1, nspin, {norb, norb})
147 for (auto sigma : range(n_sigma)) { // spin index
148 n += real(nda::trace(density(Gloc(0, sigma))));
149 }
150 return n;
151 }
