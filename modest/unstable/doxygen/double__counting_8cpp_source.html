11 std::pair<double, double> dc_formulas(std::string const method, double const N_tot, double const N_sigma,
12 long const n_orb, double const U, double const J) {
13 double Mag = N_sigma - (N_tot - N_sigma);
14 double L_orbit = 0.5 * (double(n_orb) - 1.0);
15 if (method == "cFLL") {
16 return {(U * (N_tot - 0.5) - 0.5 * J * (N_tot - 1.0)),
17 0.5 * U * N_tot * (N_tot - 1.0) - 0.5 * J * N_tot * (0.5 * N_tot - 1.0)};
18 } else if (method == "sFLL") {
19 return {(U * (N_tot - 0.5) - J * (N_sigma - 0.5)),
20 0.5 * U * N_tot * (N_tot - 1.0) - 0.5 * J * N_tot * (N_tot * 0.5 - 1.0) - 0.25 * J * Mag * Mag};
21 } else if (method == "cAMF") {
22 double C = (U + 2.0 * J * L_orbit) / (2 * L_orbit + 1.0);
23 return {
24 (U * N_tot - 0.5 * N_tot * C), // DC
25 (0.5 * U - 0.25 * C) * N_tot * N_tot //Energy
26 };
27 } else if (method == "sAMF") {
28 return {(U * N_tot - ((U + 2.0 * L_orbit * J) / (2.0 * L_orbit + 1.0)) * N_sigma), // DC
29 0.5 * U * N_tot * N_tot
30 - 0.25 * ((U + 2.0 * L_orbit * J) / (2.0 * L_orbit + 1.0)) * N_tot * N_tot // Energy
31 - 0.25 * ((U + 2.0 * L_orbit * J) / (2.0 * L_orbit + 1.0)) * Mag * Mag};
32 } else if (method == "cHeld") {
33 // double C = double(n_orb) - 1.0;
34 // double Umean = (U + ((U - 2.0 * J) + (U - 3.0 * J)) * C) / (2.0 * double(n_orb) - 1);
35 double U_mean = (U + (n_orb - 1) * (U - 2 * J) + (n_orb - 1) * (U - 3 * J)) / (2 * n_orb - 1);
36 return {
37 (U_mean * (N_tot - 0.5)), //DC
38 0.5 * U_mean * N_tot * (N_tot - 1.0) //Energy
39 };
40 } else {
41 throw std::runtime_error("Not implemented! Complain to the developers");
42 }
43 }
