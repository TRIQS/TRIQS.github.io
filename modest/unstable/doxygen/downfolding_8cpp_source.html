23 downfolding_projector downfolding_projector::rotate_local_basis(nda::array<nda::matrix<dcomplex>, 2> const &U) const {
24 auto res = *this;
25 auto n_k = res.P_k.extent(0);
26 auto n_sigma = res.P_k.extent(1);
27 for (auto isig : range(n_sigma)) {
28 auto decomp = range(U.extent(0)) | stdv::transform([&](auto &&m) { return U(m, isig).extent(0); }) | tl::to<std::vector>();
29 for (auto ik : range(n_k)) {
30 for (auto const &[a, sli] : enumerated_sub_slices(decomp)) {
31 // P <- dagger(U) * P
32 res.P_k(ik, isig, sli, r_all) = dagger(U(a, isig)) * matrix_const_view<dcomplex>{this->P_k(ik, isig, sli, r_all)};
33 // I can not use x.P(isig,k)(sli, r_all) because of the slice on the n_bands_per_k in this P }
34 }
35 }
36 }
37 return res;
38 }

50 one_body_elements_on_grid permute_local_space(std::vector<std::vector<long>> const &atom_partition, one_body_elements_on_grid const &obe) {
51 auto atom_indices = nda::flatten(atom_partition);
52
53 auto spin_kind = obe.C_space.spin_kind();
54 auto atomic_shells = atom_indices | stdv::transform([&](auto i) { return obe.C_space.atomic_shells()[i]; }) | tl::to<std::vector>();
55
56 auto new_rot_from_dft_to_local_basis = obe.C_space.rotation_from_dft_to_local_basis();
57 auto new_rot_from_sph_to_dft_basis = obe.C_space.rotation_from_spherical_to_dft_basis();
58 for (auto &&[iatom, atom] : enumerate(atom_indices)) {
59 new_rot_from_dft_to_local_basis(iatom, r_all) = obe.C_space.rotation_from_dft_to_local_basis()(atom, r_all);
60 new_rot_from_sph_to_dft_basis(iatom) = obe.C_space.rotation_from_spherical_to_dft_basis()(atom);
61 }
62
63 auto n_atoms = atom_partition.size();
64 auto n_sigma = obe.C_space.n_sigma();
65
66 auto irreps_per_atom = nda::array<std::vector<long>, 2>(n_atoms, n_sigma);
67 for (auto &&[ipart, partition] : enumerate(atom_partition)) {
68 if (partition.size() == 1) {
69 irreps_per_atom(ipart, r_all) = obe.C_space.atoms_block_decomposition()(partition[0], r_all);
70 } else {
71 irreps_per_atom(ipart, r_all) = std::vector{
72 stdr::fold_left(partition | stdv::transform([&](auto idx) { return obe.C_space.atomic_shells()[idx].dim; }), 0, std::plus<>())};
73 }
74 }
75
76 auto new_C_space = local_space{spin_kind, atomic_shells, irreps_per_atom, new_rot_from_dft_to_local_basis, new_rot_from_sph_to_dft_basis};
77
78 auto atom_decomp = obe.C_space.atomic_decomposition() | tl::to<std::vector>();
79
80 auto [n_k, n_sig, n_M, n_nu] = obe.P.P_k.shape();
81
82 auto P_k_list = atom_decomp | stdv::transform([&](auto dim) { return nda::zeros<dcomplex>(n_k, n_sig, dim, n_nu); }) | tl::to<std::vector>();
83 for (auto &&[atom, sli] : enumerated_sub_slices(atom_decomp)) {
84 P_k_list[atom](r_all, r_all, r_all, r_all) = obe.P.P_k(r_all, r_all, sli, r_all);
85 }
86
87 auto new_P_k_list = atom_indices | stdv::transform([&](auto i) { return P_k_list[i]; }) | tl::to<std::vector>();
88
89 auto new_atom_decomp = new_C_space.atomic_decomposition() | tl::to<std::vector>();
90
91 auto P_k = nda::zeros<dcomplex>(n_k, n_sig, n_M, n_nu);
92 for (auto const &[atom, sli] : enumerated_sub_slices(new_atom_decomp)) {
93 P_k(r_all, r_all, sli, r_all) = new_P_k_list[atom](r_all, r_all, nda::range(0, new_atom_decomp[atom]), r_all);
94 }
95
96 downfolding_projector new_P{.spin_kind = spin_kind, .P_k = P_k, .n_bands_per_k = obe.P.n_bands_per_k};
97
98 if (!obe.ibz_symm_ops) return {.H = obe.H, .C_space = new_C_space, .P = new_P, .ibz_symm_ops = {}};
99
100 std::vector<long> inv_atom_indices(atom_indices.size());
101 for (long i = 0; i < atom_indices.size(); ++i) { inv_atom_indices[atom_indices[i]] = i; }
102
103 auto new_ibz_symm_ops = obe.ibz_symm_ops.value();
104 for (auto &op : new_ibz_symm_ops.ops) {
105 auto old_perm = op.permutation;
106 auto old_mats = op.mats;
107 for (auto i : range(atom_indices.size())) {
108 op.permutation[i] = inv_atom_indices[old_perm[atom_indices[i]]];
109 op.mats[i] = old_mats[atom_indices[i]];
110 }
111 }
112 return {.H = obe.H, .C_space = new_C_space, .P = new_P, .ibz_symm_ops = new_ibz_symm_ops};
113 }

117 nda::array<nda::matrix<dcomplex>, 2> impurity_levels(one_body_elements_on_grid const &obe) {
118 auto n_atoms = obe.C_space.n_atoms();
119 auto n_sigma = obe.C_space.n_sigma();
120 auto results_per_atom = nda::array<nda::matrix<dcomplex>, 2>{n_atoms, n_sigma};
121 // Hloc^{σ} = \sum_{k} P^{σ}_{mν}(k) H^{σ}_{νν'} P^{σ}_{m'ν'}(k)
122 for (auto const &[atom_idx, R_atom] : enumerated_sub_slices(obe.C_space.atomic_decomposition())) {
123 auto sh = obe.C_space.atomic_shells()[atom_idx];
124 auto M_a = sh.dim;
125 for (auto sigma : range(n_sigma)) {
126 auto hloc0 = nda::zeros<dcomplex>(M_a, M_a);
127 for (auto ik : range(obe.H.n_k())) { // loop over k-points
128 auto P = obe.P.P(sigma, ik)(R_atom, r_all);
129 hloc0 += obe.H.k_weights(ik) * P * obe.H.H(sigma, ik) * dagger(P);
130 }
131 results_per_atom(atom_idx, sigma) = hloc0;
132 }
133 }
134
135 if (auto const &S = obe.ibz_symm_ops; S) results_per_atom = S->symmetrize(results_per_atom);
136 // FIXME : why are we doing this ? If so, why did we compute all of them before ?
137
138 for (auto atom_idx : range(n_atoms)) {
139 if (auto inequiv = obe.C_space.first_shell_of_its_equiv_cls(atom_idx); atom_idx != inequiv) {
140 results_per_atom(atom_idx, r_all) = results_per_atom(inequiv, r_all);
141 }
142 }
143
144 auto result = nda::array<nda::matrix<dcomplex>, 2>{1, n_sigma};
145 for (auto sigma : range(n_sigma)) {
146 result(0, sigma) = nda::zeros<dcomplex>(obe.C_space.dim(), obe.C_space.dim());
147 for (auto const &[atom, r_atom] : enumerated_sub_slices(obe.C_space.atomic_decomposition())) {
148 result(0, sigma)(r_atom, r_atom) = results_per_atom(atom, sigma);
149 }
150 }
151 return result;
152 }

162 nda::array<dcomplex, 3> detail::G0_C_k_sigma(one_body_elements_on_grid const &obe, double mu, long k_idx, long sigma,
163 std::vector<dcomplex> const &omegas, bool mu_derivative) {
164 // Using Woodbury formula
165
166 auto n_nu = obe.H.N_nu(sigma, k_idx);
167 auto R_nu = nda::range(n_nu);
168 auto M = obe.C_space.dim();
169
170 // Compute Dinv(omega, nu) = 1/(om + mu - Hk(nu))
171 auto n_omega = omegas.size();
172 auto Dinv = nda::matrix<dcomplex>(n_omega, n_nu);
173 auto Hk = obe.H.H(sigma, k_idx); // Hk[nu,nu']
174 constexpr auto sqr = [](auto x) { return x * x; };
175
176 // avoid the if in the inner loop
177 if (not mu_derivative) {
178 for (auto [n, om] : enumerate(omegas))
179 for (auto nu : R_nu) Dinv(n, nu) = 1 / (om + mu - Hk(nu, nu));
180 } else {
181 for (auto [n, om] : enumerate(omegas))
182 for (auto nu : R_nu) Dinv(n, nu) = -1 / sqr(om + mu - Hk(nu, nu));
183 }
184 // Compute M[nu, (alpha, m), (alpha', m')] = P[(alpha, m), nu ] conj(P)[(alpha', m'), nu ]
185 // Compute M[nu, m, m'] = P[m, nu ] conj(P)[m', nu ]
186 auto PP = nda::zeros<dcomplex>(n_nu, M, M);
187 auto P = obe.P.P(sigma, k_idx);
188 for (auto m : range(M))
189 for (auto mp : range(M))
190 for (auto nu : R_nu) PP(nu, m, mp) = P(m, nu) * conj(P(mp, nu));
191
192 // Compute Y [ w, m, m'] = D[w, nu] * PP[nu, m, m']
193 // Sum over nu = matmul.
194 auto Y = nda::zeros<dcomplex>(n_omega, M, M);
195 auto M_asmat = cmat_vt{nda::group_indices_view(PP, nda::idx_group<0>, nda::idx_group<1, 2>)};
196 auto Y_asmat = cmat_vt{nda::group_indices_view(Y, nda::idx_group<0>, nda::idx_group<1, 2>)};
197 nda::blas::gemm(1, Dinv, M_asmat, 0, Y_asmat);
198 return Y;
199 }
