207 template <typename Mesh> std::vector<std::pair<block_gf<Mesh, matrix_valued>, block_matrix_t>> make_zero_imp_self_energies(Mesh const &mesh) {
208
209 auto make_block_matrix = [](auto const &gf_struct) {
210 return gf_struct | stdv::transform([](auto &x) {
211 auto bl_size = x.second;
212 return nda::zeros<dcomplex>(bl_size, bl_size);
213 })
214 | tl::to<std::vector<nda::matrix<dcomplex>>>();
215 };
216
217 auto make_self_energy = [&](auto const &gf_struct) {
218 auto Sigma_static = make_block_matrix(gf_struct);
219 auto Sigma_dynamic = block_gf<Mesh, matrix_valued>{mesh, gf_struct};
220 return std::make_pair(Sigma_dynamic, Sigma_static);
221 };
222
223 return imp_block_shape() | stdv::transform(make_self_energy) | tl::to<std::vector>();
224 }

367 template <typename Mesh> block2_gf<Mesh, matrix_valued> embedding::embed(std::vector<block_gf<Mesh, matrix_valued>> const &Sigma_imp_vec) const {
368 // Check that all meshes are the same for Sigma_imp_vec
369 if (not all_equal(Sigma_imp_vec | stdv::transform([](auto &&x) -> decltype(auto) { return x[0].mesh(); })))
370 throw std::runtime_error{"[embedding_desc::embed]: meshes of solvers are not all equal"};
371
372 auto const &mesh = Sigma_imp_vec[0][0].mesh();
373
374 // build the result
375 auto Sigma_embed = make_block2_gf(mesh, this->sigma_embed_block_shape());
376 for (auto &&[S, m] : zip(Sigma_embed, psi)) {
377 if (m.imp_idx == -1) continue;
378 S() = Sigma_imp_vec[m.imp_idx][m.gamma + n_gamma(m.imp_idx) * m.tau];
379 }
380 return Sigma_embed;
381 }

409 template <typename Mesh> std::vector<block_gf<Mesh, matrix_valued>> embedding::extract(block2_gf<Mesh, matrix_valued> const &g) const {
410
411 if (auto decomp = get_struct(g).dims(r_all, 0) | tl::to<std::vector>(); decomp != this->sigma_embed_decomp) {
412 if (decomp.size() != 1) throw std::runtime_error{"extract: g should have decomp = sigma_embedding_decomp or [1]"};
413 return extract(decomposition_view(g, this->sigma_embed_block_shape()));
414 }
415
416 // FIXME : check all meshes are the same
417 auto imp_gf_stru_list = imp_block_shape();
418 auto extract_one_imp = [&](long n_imp) {
419 auto gimp = block_gf{g(0, 0).mesh(), imp_gf_stru_list[n_imp]};
420 auto const &rpsi = reverse_psi[n_imp];
421 for (auto [gamma, tau] : rpsi.indices()) {
422 auto [alpha, sigma] = rpsi(gamma, tau)[0];
423 gimp[gamma + n_gamma(n_imp) * tau].data() = g(alpha, sigma).data();
424 }
425 return gimp;
426 };
427 return range(n_impurities()) | stdv::transform(extract_one_imp) | tl::to<std::vector>();
428 }
