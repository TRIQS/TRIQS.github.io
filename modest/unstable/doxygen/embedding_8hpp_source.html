270 template <typename Mesh> block2_gf<Mesh, matrix_valued> embedding::embed(std::vector<block_gf<Mesh, matrix_valued>> const &Sigma_imp_vec) const {
271 // Check that all meshes are the same for Sigma_imp_vec
272 if (not all_equal(Sigma_imp_vec | stdv::transform([](auto &&x) -> decltype(auto) { return x[0].mesh(); })))
273 throw std::runtime_error{"[embedding_desc::embed]: meshes of solvers are not all equal"};
274 //
275 auto const &mesh = Sigma_imp_vec[0][0].mesh();
276 // build the result
277 auto Sigma_embed = make_block2_gf(mesh, this->sigma_embed_block_shape());
278 // auto Sigma_embed= nda::array<gf<Mesh, matrix_valued>,2>(this->psi.shape());
279 for (auto &&[S, m] : zip(Sigma_embed, psi)) {
280 if (m.imp_idx == -1) continue;
281 S() = Sigma_imp_vec[m.imp_idx][m.gamma + n_gamma(m.imp_idx) * m.tau];
282 }
283 return Sigma_embed;
284 }
