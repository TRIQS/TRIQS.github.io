32 template <typename T> bool is_diagonal(nda::matrix<T> const &M) {
33 auto dim = M.shape()[0];
34 for (size_t i = 0; i < dim; i++)
35 for (size_t j = i + 1; j < dim; j++)
36 if (abs(M(i, j)) > 1.e-14) return false;
37 return true;
38 }

40 template <typename T> nda::array<T, 2> block_diag(std::vector<nda::matrix<T>> const &matrices) {
41 auto n = stdr::fold_left(matrices | stdv::transform([](auto const &m) { return m.extent(0); }) | tl::to<std::vector>(), 0, std::plus<>());
42 auto U = nda::zeros<T>(n, n);
43 auto i = 0;
44 for (auto const &M : matrices) {
45 auto s = M.extent(0);
46 U(nda::range(i, i + s), nda::range(i, i + s)) = M;
47 i += s;
48 }
49 return U;
50 }
