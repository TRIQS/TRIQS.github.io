39 embedding make_embedding(local_space const &C_space, nda::array<std::vector<long>, 2> const &block_decomposition,
40 std::optional<std::vector<long>> const &atom_to_imp) {
41
42 long n_alpha = stdr::fold_left(block_decomposition(r_all, 0) | stdv::transform([](auto const &x) { return long(x.size()); }), 0, std::plus<>());
43
44 auto embed_bl_sizes = std::vector<long>{};
45 auto imp_decompositions = std::vector<std::vector<long>>{};
46 auto psi = nda::array<embedding::imp_block_t, 2>(n_alpha, C_space.n_sigma());
47
48 auto n_atoms = C_space.atoms_block_decomposition().shape()[0];
49
50 auto const &atom_imp_map = (atom_to_imp) ? *atom_to_imp : range(n_atoms) | tl::to<std::vector>();
51
52 for (auto atom : range(n_atoms)) {
53 for (auto &&[idx, bl_size] : enumerate(block_decomposition(atom, 0))) {
54 for (auto sigma : range(C_space.n_sigma())) { psi(embed_bl_sizes.size(), sigma) = {atom_imp_map[atom], idx, sigma}; }
55 embed_bl_sizes.emplace_back(bl_size);
56 }
57 auto build_impurity = C_space.first_shell_of_its_equiv_cls(atom) == atom;
58 if (build_impurity || !atom_to_imp) imp_decompositions.push_back(std::move(block_decomposition(atom, 0)));
59 }
60 return embedding{std::move(embed_bl_sizes), std::move(imp_decompositions), std::move(psi), C_space.sigma_names()};
61 }

64 embedding make_embedding_with_equivalences(local_space const &C_space) {
65
66 // cls_to_imp[alpha_to_cls[alpha]]
67 std::unordered_map<long, long> atom_to_cls, cls_to_imp;
68 long imp_idx = 0;
69 for (auto const &[ish, sh] : enumerate(C_space.atomic_shells())) {
70 atom_to_cls[ish] = sh.cls_idx;
71 if (not cls_to_imp.contains(sh.cls_idx)) cls_to_imp[sh.cls_idx] = imp_idx++;
72 }
73 auto atom_to_imp_idx = range(C_space.atomic_shells().size()) | stdv::transform([&](auto const &atom) { return cls_to_imp[atom_to_cls[atom]]; })
74 | tl::to<std::vector>();
75
76 auto decomp = C_space.atoms_block_decomposition();
77
78 return make_embedding(C_space, decomp, atom_to_imp_idx);

173 embedding embedding::drop(long imp_idx) const {
174 auto new_imp_decomps = this->imp_decomps;
175 auto new_psi = this->psi;
176 new_imp_decomps.erase(begin(new_imp_decomps) + imp_idx);
177 new_psi = nda::map([&](imp_block_t const &b) -> imp_block_t {
178 if (b.imp_idx < imp_idx) return {b.imp_idx, b.gamma, b.tau};
179 if (b.imp_idx > imp_idx) return {b.imp_idx - 1, b.gamma, b.tau};
180 return {-1, 0, 0}; // imp_idx == b.n_imp : we connect to nothing now
181 })(new_psi);
182 return {this->sigma_embed_decomp, new_imp_decomps, new_psi, this->_sigma_names};
183 }

201 embedding embedding::split(long imp_idx, std::function<bool(long)> p) const {
202 // auto res = *this; // copy. We will modify the impurities_gf_struct and the table
203 auto new_psi = this->psi;
204 auto new_imp_decomps = this->imp_decomps;
205 auto const &igs = new_imp_decomps[imp_idx];
206 auto indices = nda::range(long(igs.size())) | tl::to<std::vector>(); // {0,1,2, ... n_gamma}
207
208 // separate the block indices according to p. We put the indices in regard to the blocks and partition them
209 auto mid = std::stable_partition(begin(indices), end(indices), p);
210
211 // make 2 new gf_struct from the indices partition
212 auto get_igs = [&](int i) { return igs[i]; };
213 auto stru1 = stdr::subrange{begin(indices), mid} | stdv::transform(get_igs) | tl::to<std::vector>();
214 auto stru2 = stdr::subrange{mid, end(indices)} | stdv::transform(get_igs) | tl::to<std::vector>();
215 long L1 = long(stru1.size());
216
217 // Store the new gf_struct in result at the position of the old impurity model
218 new_imp_decomps[imp_idx] = std::move(stru1);
219 new_imp_decomps.insert(begin(new_imp_decomps) + imp_idx + 1, std::move(stru2));
220
221 // compute the inverse table of indices
222 auto indices_inv = indices;
223 for (auto i : range(indices.size())) indices_inv[indices[i]] = i;
224
225 new_psi = nda::map([&](imp_block_t const &b) -> imp_block_t {
226 if (b.imp_idx < imp_idx) return b;
227 if (b.imp_idx > imp_idx) return {b.imp_idx + 1, b.gamma, b.tau};
228 long new_gamma = indices_inv[b.gamma];
229 return (new_gamma < L1 ? imp_block_t{b.imp_idx, new_gamma, b.tau} : imp_block_t{b.imp_idx + 1, new_gamma - L1, b.tau});
230 })(new_psi);
231 return {this->sigma_embed_decomp, new_imp_decomps, new_psi, this->_sigma_names};
232 }

236 embedding embedding::split(long imp_idx, std::vector<long> const &block_list) const {
237 if (not stdr::all_of(block_list, [&](auto i) { return (i >= 0) and (i < n_gamma(imp_idx)); }))
238 throw std::runtime_error(fmt::format("[embedding::split] The list of block indices {} is incorrect. Indices i should all be 0<= i < N_γ = {}",
239 block_list, n_gamma(imp_idx)));
240 return split(imp_idx, [&](long idx) { return stdr::find(block_list, idx) != block_list.end(); });

256 std::vector<std::vector<nda::matrix<dcomplex>>> embedding::extract(nda::array<nda::matrix<dcomplex>, 2> const &matrix_C) const {
257 auto imp_gf_stru_list = imp_block_shape();
258
259 auto [_, n_sigma] = matrix_C.shape();
260 auto matrix_E = nda::array<nda::matrix<dcomplex>, 2>{sigma_embed_decomp.size(), n_sigma};
261 for (auto [alpha, Ralpha] : enumerated_sub_slices(sigma_embed_decomp)) {
262 for (auto sigma : range(n_sigma)) { matrix_E(alpha, sigma) = matrix_C(0, sigma)(Ralpha, Ralpha); }
263 }
264 auto extract_one_imp = [&](long n_imp) {
265 auto h_imp = std::vector<nda::matrix<dcomplex>>{};
266 for (auto [bl, bl_size] : imp_gf_stru_list[n_imp]) { h_imp.emplace_back(bl_size, bl_size); }
267 auto const &rpsi = reverse_psi[n_imp];
268 for (auto [gamma, tau] : rpsi.indices()) {
269 auto [alpha, sigma] = rpsi(gamma, tau)[0];
270 h_imp[gamma + n_gamma(n_imp) * tau] = matrix_E(alpha, sigma);
271 }
272 return h_imp;
273 };
274 return range(n_impurities()) | stdv::transform(extract_one_imp) | tl::to<std::vector>();
275 }
