23 one_body_elements_tb one_body_elements_from_wannier90(std::string const &wannier_file_path, spin_kind_e spin_kind,
24 std::vector<atomic_orbs> atomic_shells) {
25 if (spin_kind == spin_kind_e::Polarized) {
26 throw std::runtime_error("If performing a spin-polarized calculation, you need to supply two Wannier file paths for up and down channels.\n");
27 }
28 // call the wannier90 loader and set up tb_hamiltonian
29 // FIXME check if tb or hr exists -- write a lambda that checks which exists and returns just one of the two
30 auto [R, HR, _] = read_wannier90_tb_data(wannier_file_path);
31 std::vector<tb_hamiltonian> tb_H;
32 tb_H.emplace_back(R, HR);
33 // put the same H a second time for two spin channels
34 if (spin_kind == spin_kind_e::NonPolarized) { tb_H.emplace_back(R, HR); }
35
36 return make_obe_from_tb(std::move(tb_H), spin_kind, std::move(atomic_shells));

39 one_body_elements_tb one_body_elements_from_wannier90(std::string const &wannier_file_path_up, std::string const &wannier_file_path_dn,
40 spin_kind_e spin_kind, std::vector<atomic_orbs> atomic_shells) {
41 if (spin_kind != spin_kind_e::Polarized) {
42 throw std::runtime_error("For a non-spin polarized calculation, you should specify only one Wannier Hamiltonian.\n");
43 }
44
45 // call the wannier90 loader and set up tb_hamiltonian list
46 std::vector<tb_hamiltonian> tb_H;
47 for (auto file : {wannier_file_path_up, wannier_file_path_dn}) {
48 auto [R, HR, _] = read_wannier90_tb_data(file);
49 tb_H.emplace_back(R, HR);
50 }
51 if (tb_H[0].n_orbitals() != tb_H[1].n_orbitals()) {
52 throw std::runtime_error(
53 "Cannot construct a one_body_elements "
54 "using up and down H_k that have a different number of orbtials.");
55 }
56 return make_obe_from_tb(std::move(tb_H), spin_kind, std::move(atomic_shells));
57 }

61 nda::array<nda::matrix<dcomplex>, 2> Hloc(std::vector<tb_hamiltonian> const &H_sigma, std::vector<atomic_orbs> const &atomic_shells) {
62
63 // group the shells into atom indices
64 // makes a vector containing the dim of each atomic shell...
65 //auto shell_decomposition = atomic_shells | stdv::transform([](auto const &s) { return s.dim; });
66
67 // return Hloc with shape [n_atoms, nsigma]
68 nda::array<nda::matrix<dcomplex>, 2> Hloc_result(atomic_shells.size(), H_sigma.size());
69
70 for (auto [isigma, H] : enumerate(H_sigma)) {
71
72 // FIXME : REFACTOR decomposition from atomic shells
73 // check that HR and atomic shells list have the same total size
74 long n_orb = 0;
75 for (auto shell : atomic_shells) { n_orb += shell.dim; }
76 if (H.n_orbitals() != n_orb) {
77 throw std::runtime_error("Wannier Hamiltonian does not have the same number of orbitals as the provided atomic shells: HR "
78 + std::to_string(H.n_orbitals()) + " , atomic_shells total " + std::to_string(n_orb));
79 }
80
81 // find the home cell of the TB file to get H0
82 auto iR0 = H.get_R_idx(std::array<long, 3>{0, 0, 0});
83
84 // Hloc needs to have dimensions [n_atomic_shells, nspin].
85 // NOTE: we cannot use enumerated_subslice here because we do not have a C_space!
86 // this function computes Hloc so we can determine the C_space in the first place.
87 long start_orb = 0;
88 for (auto &&[ishell, shell] : enumerate(atomic_shells)) {
89 auto Hloc0_ab = nda::zeros<dcomplex>(shell.dim, shell.dim);
90 // need to create a matrix of dim x dim for each
91 for (auto iorb : nda::range(shell.dim)) {
92 for (auto jorb : nda::range(shell.dim)) { Hloc0_ab(iorb, jorb) = H.hoppings()[iR0](start_orb + iorb, start_orb + jorb); }
93 start_orb += shell.dim;
94 }
95 Hloc_result(ishell, isigma) = Hloc0_ab;
96 }
97 }
98 return Hloc_result;
99 }

101 nda::array<nda::matrix<dcomplex>, 2> impurity_levels(one_body_elements_tb const &obe) {
102
103 auto n_atoms = obe.C_space.n_atoms();
104 auto n_sigma = obe.C_space.n_sigma();
105 nda::array<nda::matrix<dcomplex>, 2> Hloc_result(n_atoms, n_sigma);
106
107 for (auto sigma : range(n_sigma)) {
108
109 // find the home cell of the TB file to get H0
110 auto iR0 = obe.H[sigma].get_R_idx({0, 0, 0});
111
112 for (auto const &[atom, R_atom] : enumerated_sub_slices(obe.C_space.atomic_decomposition())) {
113 Hloc_result(atom, sigma) = obe.H[sigma].hoppings()[iR0](R_atom, R_atom);
114 }
115 }
116 return Hloc_result;
117 }

121 one_body_elements_tb make_obe_from_tb(std::vector<tb_hamiltonian> H_sigma, spin_kind_e spin_kind, std::vector<atomic_orbs> atomic_shells) {
122
123 // calculate Hloc using helper function -- Hloc here is dim [nshells, nsigma]
124 nda::array<nda::matrix<dcomplex>, 2> hloc = Hloc(H_sigma, atomic_shells);
125
126 // construct block structure using Hloc
127 double block_threshold = 1e-6;
128 bool diagonalize_hloc = false; // CHECK this would mean looking at band energies at k = 0?
129 auto [decomposition, U] = discover_symmetries(hloc, atomic_shells, block_threshold, diagonalize_hloc);
130
131 // call constructor for local space, for now ignoring rotation_from_spherical_to_dft_basis
132 auto LS = local_space(spin_kind, atomic_shells, decomposition, U, {});
133
134 // construct and return obe_tb
135 return one_body_elements_tb{.C_space = std::move(LS), .H = std::move(H_sigma)};
136 }

140 one_body_elements_tb fold(lattice::superlattice const &sl, one_body_elements_tb const &obe) {
141 auto new_H = obe.H | stdv::transform([&](auto x) { return fold(sl, x); }) | tl::to<std::vector>();
142 auto sh = obe.C_space.atomic_shells();
143 decltype(sh) new_atomic_shells;
144 auto const &dec = obe.C_space.atoms_block_decomposition();
145 nda::array<std::vector<long>, 2> new_dec(dec.extent(0) * sl.n_cluster_sites(), dec.extent(1));
146 for (auto i : nda::range(sl.n_cluster_sites())) {
147 for (auto &&[j, shell] : enumerate(sh)) {
148 new_atomic_shells.emplace_back(shell);
149 new_dec(i * sh.size() + j, r_all) = dec(j, r_all);
150 }
151 }
152 return {.C_space = local_space{obe.C_space.spin_kind(), std::move(new_atomic_shells), new_dec, {}, {}}, .H = std::move(new_H)};
153 }
